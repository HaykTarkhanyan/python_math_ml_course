\section{Loss and Risk}

\begin{frame}{From point-wise loss to risk}
  In L01 we saw the $L_2$ loss for one point. The \textbf{loss function} measures the per-point error:
  \[
    L : \Yspace \times \R^g \to \R
  \]
  \vfill
  Common regression losses:
  \begin{itemize}
    \item \textbf{$L_2$ (quadratic)}: $\Lxy = (y - \fx)^2$
    \item \textbf{$L_1$ (absolute)}: $\Lxy = |y - \fx|$
  \end{itemize}
  \vfill
  We want a single number per model, not per point. Two ways:
  \begin{itemize}
    \item \textbf{Theoretical risk}: average over the true (unknown) distribution
    \item \textbf{Empirical risk}: average over the training data
  \end{itemize}
\end{frame}

\begin{frame}{Theoretical risk}
  Average loss when using $f$ on data from $\mathbb{P}_{xy}$:
  \[
    \mathcal{R}(f) := \E_{xy}[\Lxy] = \int \Lxy \, d\mathbb{P}_{xy}
  \]
  \vfill
  \begin{itemize}
    \item Goal of ML: find $f \in \Hspace$ that minimizes $\mathcal{R}(f)$
    \item \textbf{Problem}: $\mathbb{P}_{xy}$ is unknown, so we cannot compute the integral
    \item Estimating $\mathbb{P}_{xy}$ directly (e.g., kernel density estimation) doesn't scale to high dimensions
  \end{itemize}
\end{frame}

\begin{frame}{Empirical risk}
  We have $n$ i.i.d.\ samples — replace the integral with an average:
  \[
    \riske(f) = \sumin \Lxyi
  \]
  \vfill
  Since $f$ is parameterized by $\thetav$:
  \[
    \risket = \sumin L\!\left(\yi, f(\xv^{(i)} \mid \thetav)\right),
    \qquad \riske : \R^d \to \R
  \]
  \vfill
  One quality score per parameter vector $\thetav$. Now the problem is a numerical optimization.
\end{frame}

\begin{frame}{Empirical Risk Minimization (ERM)}
  \begin{columns}
    \column{0.55\textwidth}
    Pick the parameter vector with the smallest empirical risk:
    \[
      \boxed{\;\thetah \;=\; \argmin_{\thetav \in \Theta}\; \risket\;}
    \]
    \vfill
    \begin{itemize}\small
      \item Finite $\Hspace$: tabulate exhaustively (rare)
      \item Continuous $\thetav$: use \textbf{numerical optimization}
      \item ERM reduces ``learning'' to ``find the min of a function''
    \end{itemize}
    \vfill
    \textit{Caveat}: optimizing training error too well $\Rightarrow$ overfitting (L03–L04).
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/ml-basic-riskmin-error-surface.png}
  \end{columns}
\end{frame}

\section{Optimization: Gradient Descent}

\begin{frame}{The error surface}
  \begin{columns}
    \column{0.55\textwidth}
    Treat $\risket$ as a function over the parameter space.
    \begin{itemize}\small
      \item Find $\thetah$ by traversing the surface downhill
      \item For continuous $\thetav$: \emph{local} search from a start point
      \item Best optimizer depends on problem structure
    \end{itemize}
    \vfill
    Local vs global:
    \begin{itemize}\small
      \item \textbf{Global}: $\riske(\thetah) \le \riske(\thetav)\;\forall \thetav$
      \item \textbf{Local}: holds only in a neighborhood
      \item Every global min is local. Finding a local min is much easier.
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/err_surf.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Stationary points}
  If $\riske$ is continuously differentiable, a \textbf{sufficient condition} for a local minimum:
  \begin{itemize}
    \item Gradient is zero: $\quad \dfrac{d\riske}{d\thetav}(\thetah) = 0$
    \item Hessian at $\thetah$ is positive definite
  \end{itemize}
  \vfill
  \begin{itemize}
    \item \emph{Negative} gradient points in the direction of steepest descent
    \item Hessian measures local curvature
  \end{itemize}
\end{frame}

\begin{frame}{Gradient descent}
  Start from some $\thetav^{[0]}$. Iteratively move in the direction of steepest descent:
  \[
    \thetav^{[t+1]} = \thetav^{[t]} - \alpha \, \frac{d \riske}{d\thetav}(\thetav^{[t]})
  \]
  \vfill
  \begin{itemize}
    \item $\alpha$: \textbf{step size} / \textbf{learning rate}
    \item Negative gradient is direction of steepest local decrease
    \item Magnitude $\| -\nabla \riske \|$ tells us only about \emph{local} steepness — we have no guarantee against overshooting
  \end{itemize}
\end{frame}

\begin{frame}{Learning rate sensitivity}
  Step size $\alpha$ is the dominant hyperparameter.
  \vfill
  \begin{center}
    \begin{minipage}{0.3\textwidth}
      \centering\includegraphics[width=\textwidth]{figures/grad_desc_alpha1.pdf}\\
      {\scriptsize good convergence ($\alpha_1$)}
    \end{minipage}\hfill
    \begin{minipage}{0.3\textwidth}
      \centering\includegraphics[width=\textwidth]{figures/grad_desc_alpha2.pdf}\\
      {\scriptsize slow ($\alpha_2 < \alpha_1$)}
    \end{minipage}\hfill
    \begin{minipage}{0.3\textwidth}
      \centering\includegraphics[width=\textwidth]{figures/grad_desc_alpha3.pdf}\\
      {\scriptsize no convergence ($\alpha_3 > \alpha_1$)}
    \end{minipage}
  \end{center}
  \vfill
  Practical: constant $\alpha$ tuned by trial; decay schedules $\alpha_t = \alpha_0 / (1 + \gamma t)$; modern adaptive optimizers (Adam — see L18).
\end{frame}

\begin{frame}{Beyond vanilla GD}
  \begin{itemize}
    \item A few models (e.g., linear regression with $L_2$) admit a \textbf{closed-form} solution. Most don't.
    \item GD is a \textbf{first-order} method. \textbf{Second-order} methods (Newton–Raphson) use the Hessian for faster convergence but cost more per step.
    \item Many improvements: \textbf{momentum}, \textbf{Adam}, \textbf{RMSprop} — adaptive learning rates, escape saddle points, mimic curvature without computing the Hessian.
    \item \textbf{Stochastic gradient descent (SGD)}: compute the gradient on a small random batch, not the full dataset. Much more efficient on large data.
  \end{itemize}
\end{frame}

\section{\texorpdfstring{$L_1$}{L1} vs \texorpdfstring{$L_2$}{L2} Loss}

\begin{frame}{Why consider $L_1$?}
  \begin{itemize}
    \item $L_2$ minimizes squared residuals — wouldn't \emph{absolute} residuals seem more natural?
    \item \textbf{$L_1$ loss / Least Absolute Deviation (LAD)}:
      \[
        \Lxy = |y - \fx|
      \]
    \item Also called \textbf{median regression} — at the optimum the model predicts the conditional median of $y$, while $L_2$ predicts the conditional mean.
  \end{itemize}
\end{frame}

\begin{frame}{$L_1$ vs $L_2$: optimization difficulty}
  Loss surfaces look very different:
  \begin{itemize}
    \item \textbf{$L_2$}: smooth, convex, differentiable everywhere $\Rightarrow$ closed-form OLS
    \item \textbf{$L_1$}: convex, but \textbf{not differentiable} at $y = \fx$
      \begin{itemize}
        \item No closed-form solution
        \item Use subgradient methods or linear programming
        \item In practice: slower to fit than OLS
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{$L_1$ vs $L_2$: robustness}
  \begin{itemize}
    \item $L_2$ is quadratic in residuals $\Rightarrow$ outlying points carry massive weight
    \item Example: $3\times$ residual $\Rightarrow$ $9\times$ loss contribution
    \item $L_1$ scales linearly $\Rightarrow$ \textbf{more robust} to outliers
  \end{itemize}
  \vfill
  \begin{center}
    \includegraphics[width=0.85\textwidth]{figures/reg_l1_comparison_outlier.pdf}
  \end{center}
\end{frame}

\begin{frame}{$L_1$ vs $L_2$: when to use which}
  \textbf{Prefer $L_1$ when:}
  \begin{itemize}
    \item Data has heavy-tailed noise or suspected outliers
    \item You care about the median behavior, not the mean
  \end{itemize}
  \vfill
  \textbf{Prefer $L_2$ when:}
  \begin{itemize}
    \item Closed-form solution wanted
    \item Noise is roughly Gaussian
    \item Speed matters (e.g., inside a CV loop)
  \end{itemize}
\end{frame}

\begin{frame}{Cost in practice}
  Toy weather problem (predict mean temperature):
  \vfill
  \begin{itemize}
    \item Coefficients fitted under $L_1$ and $L_2$ differ a little, not a lot
    \item $L_1$ achieves slightly lower total $L_1$-loss; $L_2$ slightly lower $L_2$-loss
    \item $L_1$ runs noticeably \textbf{slower} as dataset size grows
  \end{itemize}
  \vfill
  Takeaway: choice of loss matters for \emph{robustness}, more than for typical-case parameter estimates on clean data.
\end{frame}

\section{Polynomial Regression}

\begin{frame}{Linear is more flexible than it looks}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item ``Linear regression'' is linear \emph{in the parameters}, not necessarily in features
      \item Generalization with basis functions $\textcolor{armblue}{\phi_j}$:
      \[
        \fx = \theta_0 + \sumjp \theta_j \, \textcolor{armblue}{\phi_j}(x_j)
      \]
      \item Vanilla LM: $\phi_j(x) = x$ for all $j$
      \item Solution via normal equations, just with the transformed $\Xmat$
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/reg_poly_yx3.pdf}\\[1mm]
    {\scriptsize $y$ vs $x$ looks nonlinear, but $y$ vs $x^3$ looks pretty linear.}
  \end{columns}
\end{frame}

\begin{frame}{Is $f(x) = \theta_0 + \theta_1 x^2$ really ``linear''?}
  \begin{itemize}
    \item \textbf{Not} in the input $x$: $f(ax + bx_*) \neq a f(x) + b f(x_*)$
    \item But \textbf{yes} in the parameters $\thetav$:
      \[
        a \, f(x, \textcolor{armblue}{\thetav}) + b \, f(x, \textcolor{armorange}{\thetav^*})
        = f(x,\; \textcolor{armred}{a\thetav + b\thetav^*})
      \]
  \end{itemize}
  \vfill
  This is the key property that lets us re-use OLS, gradient descent, all the linear-model machinery, on highly nonlinear functions.
\end{frame}

\begin{frame}{Polynomial basis functions}
  Map $x_j$ to its first $d$ powers:
  \[
    \phi^{(d)} : \R \to \R^d, \quad x \mapsto (x, x^2, \dots, x^d)
  \]
  Univariate model:
  \[
    \fx = \theta_0 + \theta_1 x + \theta_2 x^2 + \dots + \theta_d x^d
  \]
  \vfill
  \begin{center}
    \includegraphics[width=0.85\textwidth]{figures/reg_poly_basis.pdf}
  \end{center}
\end{frame}

\begin{frame}{Higher $d$ $\Rightarrow$ more flexible}
  \begin{itemize}
    \item Larger $d$ $=$ richer hypothesis space $=$ higher \textbf{capacity}
    \item Tempting: just let $d \to \infty$? \textbf{No.}
    \item Data has random \textbf{noise} that is not part of the true relationship
    \item Too much capacity fits the noise $\Rightarrow$ \textbf{overfitting} $\Rightarrow$ poor generalization (L04)
    \item Also: high-degree polynomials oscillate at edges (\textbf{Runge's phenomenon})
  \end{itemize}
  \vfill
  \begin{center}
    \includegraphics[width=0.85\textwidth]{figures/reg_poly_univ_4.pdf}
  \end{center}
\end{frame}

\begin{frame}{Worked example: bike rentals}
  OpenML \texttt{dailybike}: predict daily rentals from weather features.
  \[
    \fx = \sum_{k=1}^d \theta_{\text{temp}, k}\, x_{\text{temp}}^k
         + \theta_{\text{season}} x_{\text{season}}
         + \theta_{\text{hum}} x_{\text{hum}}
  \]
  \vfill
  Cross-validated test error (more on CV in L05):
  \begin{itemize}
    \item $d=1$ (linear in temperature): test error 121.9
    \item $d=3$ (cubic): test error 117.6  $\Leftarrow$ best
    \item Going higher: test error rises again (overfit)
  \end{itemize}
  \vfill
  Conclusion: a small dose of flexibility helps. Too much hurts. \textbf{Choosing $d$ is a tuning problem} (L05).
\end{frame}

\begin{frame}{What we covered}
  \begin{enumerate}
    \item Defined loss, theoretical risk, empirical risk
    \item ERM: training as numerical optimization of $\risket$
    \item Gradient descent: iterative downhill, sensitive to $\alpha$
    \item $L_1$ vs $L_2$ tradeoffs: optimization cost vs robustness
    \item Polynomial regression: linear-in-parameters $\neq$ linear-in-features
  \end{enumerate}
  \vfill
  Next (L03): when ERM picks too-complex models, what do we do? $\rightarrow$ regularization.
\end{frame}
