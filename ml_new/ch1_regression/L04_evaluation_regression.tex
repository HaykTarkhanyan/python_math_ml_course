\section{Training Error}

\begin{frame}{Training error}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Simplest evaluation: plug training data back into the trained model
      \item Aka \textbf{apparent error} or \textbf{resubstitution error}
      \item Often \emph{very} optimistic — model has already seen these points
    \end{itemize}
    \vfill
    \begin{itemize}\small
      \item 1-NN: training error is always 0 (each point is its own nearest neighbor)
      \item Any model storing training data $+$ replaying labels achieves zero training error
      \item Interpolating splines / Gaussian processes can perfectly match noise too
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/train_error.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Why training error is unreliable}
  \begin{columns}
    \column{0.55\textwidth}
    Polynomial regression on $y = 0.5 + 0.4 \sin(2\pi x) + \eps$:
    \begin{itemize}\small
      \item $d=1$: train MSE = 0.036 (underfit)
      \item $d=3$: train MSE = 0.003 (good)
      \item $d=9$: train MSE = 0.001 (overfit)
    \end{itemize}
    \vfill
    Training error \emph{monotonically decreases} with complexity. Used alone, it always picks the most complex model — \textbf{wrong}.
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/eval_train_2.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Classical GoF measures also use training data}
  \begin{itemize}
    \item $R^2$, log-likelihood, AIC, BIC, deviance — all computed on training data
    \item Work \emph{sometimes}: restricted-capacity models, plenty of data, distributional assumptions hold
    \item Hard to gauge \emph{when they break}, especially for high-dim or complex data
    \item No clean way to compare them across model families (e.g., LM vs random forest)
  \end{itemize}
  \vfill
  \textbf{Out-of-sample testing is the safer default.}
\end{frame}

\section{Test Error and Hold-Out}

\begin{frame}{Hold-out splitting}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Simulate ``new data'' by partitioning $\D$ into $\Dtrain$ and $\Dtest$
      \item Train on $\Dtrain$, evaluate on $\Dtest$
      \item Common split: 2/3 train, 1/3 test
      \item Aka the \textbf{untouched test set} principle — never train on, never tune on, never even peek at, the test set
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/test_error.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Same polynomial example, now with test set}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/eval_test_2.pdf}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item $d=1$: test MSE = 0.038 (underfit)
      \item $d=3$: test MSE = 0.002 (good)
      \item $d=9$: test MSE = \textbf{0.046} (overfit)
    \end{itemize}
    \vfill
    Test error \emph{exposes} the overfitting that training error hid.
  \end{columns}
\end{frame}

\begin{frame}{The U-shape of test error}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/eval_test_3.pdf}
    \column{0.45\textwidth}
    Increasing model complexity:
    \begin{itemize}\small
      \item Training error: \textbf{monotonically decreases}
      \item Test error: \textbf{U-shape}\\(underfit $\to$ sweet spot $\to$ overfit)
    \end{itemize}
    \vfill
    The U's minimum is the ``sweet spot.'' Finding it is what tuning (L05) is for.
  \end{columns}
\end{frame}

\begin{frame}{Bias-variance tradeoff of hold-out itself}
  Choice of split ratio matters:
  \begin{itemize}
    \item \textbf{Small training set} $\to$ poor fit, \emph{pessimistic} bias in estimated GE
    \item \textbf{Small test set} $\to$ high variance in the estimate (one bad test point swings the metric)
    \item Empirically: train/test $\approx$ 2/3 ratio minimizes MSE of the holdout estimator
  \end{itemize}
  \vfill
  \begin{itemize}
    \item Single split estimates fluctuate; \emph{repeated} holdout or \emph{cross-validation} reduce variance (L05)
    \item Small datasets: holdout is risky $\Rightarrow$ use CV
  \end{itemize}
\end{frame}

\section{Generalization Error}

\begin{frame}{Generalization error}
  Our actual quantity of interest:
  \[
    GE(f) = \E_{(\xv, y) \sim \mathbb{P}_{xy}}\!\left[ L(y, \fx) \right]
  \]
  Average loss on a fresh sample from the true distribution.
  \vfill
  \begin{itemize}
    \item We never observe $GE(f)$ directly (we don't have $\mathbb{P}_{xy}$)
    \item Test error is our \emph{estimator} of $GE(f)$
    \item As $|\Dtest| \to \infty$, test error $\to GE(f)$ (assuming i.i.d.)
    \item Whole field of ML evaluation: build good estimators of $GE(f)$ from finite data
  \end{itemize}
\end{frame}

\section{Overfitting and Underfitting}

\begin{frame}{Underfitting vs.\ appropriate fit}
  \begin{columns}
    \column{0.55\textwidth}
    \textbf{Underfitting}:
    \begin{itemize}\small
      \item Model can't capture the true relationship
      \item High train error \emph{and} high test error
      \item Hard to detect — we don't know the Bayes error
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=0.49\textwidth]{figures/eval_ofit_1u.pdf}
    \includegraphics[width=0.49\textwidth]{figures/eval_ofit_1a.pdf}\\
    {\scriptsize underfit \hfill appropriate}
  \end{columns}
\end{frame}

\begin{frame}{Overfitting vs.\ appropriate fit}
  \begin{columns}
    \column{0.55\textwidth}
    \textbf{Overfitting}:
    \begin{itemize}\small
      \item Model fits noise / artifacts in training data
      \item \emph{Small} train error, \emph{large} test error
      \item Predictions on new data can't be trusted
      \item Proper evaluation workflow makes it visible
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=0.49\textwidth]{figures/eval_ofit_1o.pdf}
    \includegraphics[width=0.49\textwidth]{figures/eval_ofit_1a.pdf}\\
    {\scriptsize overfit \hfill appropriate}
  \end{columns}
\end{frame}

\begin{frame}{Complexity vs.\ generalization error}
  \begin{center}
    \includegraphics[width=0.7\textwidth]{figures/eval_ofit_3.pdf}
  \end{center}
  \vfill
  \begin{itemize}
    \item Common U-shape as complexity (or training rounds) grows
    \item Optimal complexity: simplest model whose GE isn't significantly worse than more complex alternatives (\emph{Occam's razor})
  \end{itemize}
\end{frame}

\begin{frame}{What drives overfitting}
  More potential for overfitting when:
  \begin{itemize}
    \item Hypothesis space $\Hspace$ is complex
    \item Training data is small
    \item Feature space is high-dimensional
    \item Irreducible noise is large
  \end{itemize}
  \vfill
  Implications:
  \begin{itemize}
    \item Bigger $\Hspace$ $\Rightarrow$ need more data to identify good models
    \item Less data $\Rightarrow$ stick with constrained $\Hspace$
    \item Even linear models can overfit when $p$ is comparable to or larger than $n$
  \end{itemize}
\end{frame}

\section{Regression Metrics}

\begin{frame}{Mean Squared Error (MSE)}
  \begin{columns}
    \column{0.55\textwidth}
    \[
      \text{MSE}(\yv, \hat\yv) = \frac{1}{m}\sum_{i=1}^m (\yi - \yih)^2
    \]
    \vfill
    \begin{itemize}\small
      \item $\to$ $L_2$ loss averaged over the evaluation set
      \item \textbf{Outliers dominate} — squared residuals
      \item Variants: \textbf{SSE} (no division), \textbf{RMSE} (square root brings units back to $y$'s scale)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/plot_quad_loss.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Mean Absolute Error (MAE)}
  \begin{columns}
    \column{0.55\textwidth}
    \[
      \text{MAE}(\yv, \hat\yv) = \frac{1}{m}\sum_{i=1}^m |\yi - \yih|
    \]
    \vfill
    \begin{itemize}\small
      \item $\to$ $L_1$ loss averaged
      \item More \textbf{robust} to outliers
      \item Intuitive: ``on average we're off by X units''
      \item Variant: \textbf{Median absolute error} (even more robust)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/plot_abs_loss.pdf}
  \end{columns}
\end{frame}

\begin{frame}{MAPE — relative error}
  \[
    \text{MAPE}(\yv, \hat\yv) = \frac{1}{m}\sum_{i=1}^m \left|\frac{\yi - \yih}{\yi}\right|
  \]
  \vfill
  \begin{itemize}
    \item Useful when targets span orders of magnitude (sales, rentals)
    \item Small $|\yi|$ dominates — undefined at $\yi = 0$
    \item Variants: MASE, sMAPE (symmetric)
  \end{itemize}
\end{frame}

\begin{frame}{$R^2$: fraction of variance explained}
  \[
    R^2(\yv, \hat\yv) = 1 - \frac{\sum_i (\yi - \yih)^2}{\sum_i (\yi - \bar y)^2}
    = 1 - \frac{\text{SSE}_{\text{model}}}{\text{SSE}_{\text{constant baseline}}}
  \]
  \vfill
  \begin{itemize}
    \item $R^2 = 1$: perfect predictions
    \item $R^2 = 0$: model no better than the constant $\bar y$
    \item $R^2 < 0$ (on test data!): \emph{worse} than the constant — possible when $\bar y_{\text{train}} \neq \bar y_{\text{test}}$, or under overfitting
    \item Classic statistics-style measure. Often computed on training data, where it's $\in [0, 1]$ for any model better than the intercept.
  \end{itemize}
\end{frame}

\begin{frame}{$R^2$ vs MSE: don't be fooled}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/eval_mse_r2.pdf}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item Same DGP, fit on (a) half the data (black), (b) full data (black + red)
      \item Fit quality is essentially the same
      \item But $R^2$ goes \emph{up} when you add data with more variance in $x$
      \item Because $R^2$ depends on the spread of $y$, not just the residuals
    \end{itemize}
    \vfill
    {\small $R^2$ is comparable \emph{within} a fixed dataset, not across.}
  \end{columns}
\end{frame}

\begin{frame}{Spearman's $\rho$: rank correlation}
  \[
    \rho_{\text{Spearman}}(\yv, \hat\yv)
    = \frac{\text{cov}(\text{rg}(\yv), \text{rg}(\hat\yv))}
           {\sqrt{\text{var}(\text{rg}(\yv))}\sqrt{\text{var}(\text{rg}(\hat\yv))}}
    \in [-1, 1]
  \]
  \vfill
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Very robust to outliers
      \item Invariant to monotone transformations of $\hat\yv$
      \item $\pm 1$: predictions and labels are in the same (or reverse) rank order
      \item Use when you care about \emph{ranking} (recommender system, ordinal regression)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/eval_spearman.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Choosing a metric}
  Quick decision guide:
  \begin{itemize}
    \item \textbf{MSE / RMSE}: default. Good when outliers are real signal.
    \item \textbf{MAE}: more robust; interpret in original units.
    \item \textbf{MAPE}: when relative error matters (sales, demand).
    \item \textbf{$R^2$}: for communicating ``variance explained'' to non-technical audiences. Beware its caveats.
    \item \textbf{Spearman's $\rho$}: when only ranks matter.
  \end{itemize}
  \vfill
  Always evaluate on held-out / cross-validated data, not training data.
\end{frame}

\begin{frame}{Avoiding overfitting (preview)}
  \begin{itemize}
    \item Use more / better data (when feasible)
    \item Constrain $\Hspace$: simpler model class
    \item Hyperparameter tuning to find the sweet spot (L05)
    \item Regularization (L03)
    \item Early stopping (L15)
  \end{itemize}
  \vfill
  Next: L05 — beyond hold-out. Cross-validation, hyperparameter tuning, the full evaluation toolkit.
\end{frame}
