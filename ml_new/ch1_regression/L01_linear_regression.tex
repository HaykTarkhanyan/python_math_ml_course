\section{What is Machine Learning?}

\begin{frame}{ML is changing our world}
  \begin{itemize}
    \item Search engines learn your preferences
    \item Recommender systems learn your taste in books, music, movies
    \item Algorithms do automatic stock trading
    \item Translation between many languages
    \item DeepMind beats humans at Go
    \item Personalized medicine
    \item LLMs revolutionize many fields (especially coding)
    \item Data-driven discoveries in physics, biology, genetics, astronomy, neurology
    \item \dots
  \end{itemize}
\end{frame}

\begin{frame}{AI, ML, and DL}
  \begin{itemize}
    \item \textbf{Artificial Intelligence} — broad term; machines doing tasks that ``require intelligence.'' Includes ML, NLP, vision, robotics, planning, search.
    \item \textbf{Machine Learning} — mathematically well-defined; build predictive models \emph{from data} instead of programming them.
    \item \textbf{Deep Learning} — subfield of ML focused on neural networks. Excellent on images, text, signals.
  \end{itemize}
  \vfill
  \textit{``A computer program is said to learn from experience E with respect to some task T and performance measure P, if its performance on T, as measured by P, improves with experience E.''}\\
  \textcolor{gray}{\small — Tom Mitchell, Carnegie Mellon, 1998}
\end{frame}

\begin{frame}{ML vs.\ Statistics}
  \begin{itemize}
    \item Historically different fields, but methods and concepts heavily overlap
    \item \textbf{ML}: emphasis on accurate prediction with complex models
    \item \textbf{Stats}: emphasis on interpreting relationships and inference
    \item In practice: same problems, same tools, different vocabulary
    \item One useful framing: \emph{ML $\approx$ nonparametric statistics + efficient numerical optimization}
  \end{itemize}
\end{frame}

\begin{frame}{Three main directions of ML}
  \begin{itemize}
    \item \textbf{Supervised learning} — data has labels $y$. Learn $f: \xv \mapsto y$. Examples: predict house price, classify image. \textcolor{popblue}{Focus of this course.}
    \item \textbf{Unsupervised learning} — no labels. Find structure in $\xv$. Examples: clustering, dimensionality reduction, anomaly detection.
    \item \textbf{Reinforcement learning} — agent interacts with environment. Receives reward signal. Learns to choose actions that maximize future reward.
  \end{itemize}
\end{frame}

\section{Supervised Tasks: Regression vs.\ Classification}

\begin{frame}{Regression vs.\ Classification}
  Supervised learning splits by the type of the target $y$:
  \vfill
  \begin{itemize}
    \item \textbf{Regression}: $\Yspace \subseteq \R$ — predict a number.
      \begin{itemize}
        \item House prices, temperature, length of hospital stay
      \end{itemize}
    \item \textbf{Classification}: $\Yspace = \{C_1, \dots, C_g\}$ — predict a category.
      \begin{itemize}
        \item Spam vs.\ not spam, image of cat/dog/bird, risk category
      \end{itemize}
  \end{itemize}
  \vfill
  Both share the same machinery: features $\xv$, labels $y$, a model $f$, a loss.
\end{frame}

\begin{frame}{Two reasons to learn a relationship}
  \begin{itemize}
    \item \textbf{Learning to predict}: we want accurate predictions on new data. Internal structure of $f$ may not matter.
    \item \textbf{Learning to explain}: we use $f$ as a window into the data-generating process. Predictive accuracy still matters, but interpretability is the goal.
  \end{itemize}
  \vfill
  ML historically emphasized prediction; classical statistics emphasized explanation. Most modern work cares about both, to varying degrees.
\end{frame}

\section{Data}

\begin{frame}{Tabular data in supervised learning}
  $n$ observations of $p$ features plus a target:
  \vfill
  \begin{itemize}
    \item \textbf{Target} ($y$) — the variable we want to predict
    \item \textbf{Features} ($\xv$) — variables that describe each object
  \end{itemize}
  \vfill
  \textbf{Iris example} (Fisher, 1936): 150 flowers, 4 features (sepal length/width, petal length/width), 3 species. Tiny, clean, very easy to classify — \emph{not} representative of real data.
\end{frame}

\begin{frame}{Data types}
  Features and target may be of different types:
  \begin{itemize}
    \item \textbf{Numerical}: $\in \R$
    \item \textbf{Integer}: $\in \mathds{Z}$
    \item \textbf{Categorical}: $\in \{C_1, \dots, C_g\}$
    \item \textbf{Binary}: $\in \{0, 1\}$
  \end{itemize}
  \vfill
  \begin{itemize}
    \item Target type determines the task: regression vs.\ classification
    \item Most learning algorithms need \textbf{numerical} features. Categoricals are usually \textbf{encoded} first (trees are a notable exception).
  \end{itemize}
\end{frame}

\begin{frame}{Encoding categorical features}
  A categorical $x$ with $k$ values becomes a length-$\tilde{k}$ indicator vector:
  \vfill
  \begin{itemize}
    \item \textbf{One-hot encoding}: $\tilde{k} = k$ dummies; exactly one is 1.\\
      Example: $x \in \{a, b, c\}$, $x = c \;\mapsto\; (0, 0, 1)$
    \item \textbf{Dummy encoding}: $\tilde{k} = k - 1$ dummies; at most one is 1. Removes redundancy (needed when the design matrix must be non-singular, e.g., linear regression).
  \end{itemize}
  \vfill
  Ordinal features (small $<$ medium $<$ large) get an order-preserving encoding (e.g., $0, 1, 2$).
\end{frame}

\begin{frame}{Notation for data}
  A dataset of size $n$:
  \[
    \D \;=\; \left\{ \xyi[1], \dots, \xyi[n] \right\} \;\in\; (\Xspace \times \Yspace)^n
  \]
  where
  \begin{itemize}
    \item $\Xspace \subseteq \R^p$ — input (feature) space
    \item $\Yspace$ — output (target) space
    \item $\xyi$ — $i$-th observation
    \item $\xv^{(i)}_j$ — $j$-th feature of $i$-th observation
  \end{itemize}
  \vfill
  $\Rightarrow$ $n$ observations, each described by $p$ features.
\end{frame}

\begin{frame}{Data-generating process}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Data $\D$ generated by some distribution $\mathbb{P}_{xy}$ over $\Xspace \times \Yspace$
      \item $\mathbb{P}_{xy}$ is \textbf{unknown} — learning its structure is what ML is about
      \item Assume \textbf{i.i.d.}: samples drawn independently from the same distribution
      \item Strong assumption — fails for time series, spatial data. We focus on the i.i.d.\ case.
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/sample-dgp-2d.pdf}
  \end{columns}
\end{frame}

\section{Models and Parameters}

\begin{frame}{What is a model?}
  A \textbf{model} (or \textbf{hypothesis}) is a function
  \[
    f: \Xspace \to \R^g
  \]
  mapping feature vectors to predicted target values.
  \vfill
  \begin{itemize}
    \item Regression: $g = 1$
    \item Classification: $g$ = number of classes; outputs are scores or probabilities
    \item Models range from very simple (linear, tree stump) to very complex (deep networks)
    \item ML requires \textbf{constraining} $f$ to a chosen family of functions
  \end{itemize}
\end{frame}

\begin{frame}{Hypothesis spaces}
  \begin{itemize}
    \item Without restricting the function family, finding a ``good'' $f$ is hopeless
    \item We declare the model class \emph{a priori}; this is a \textbf{structural prior}
    \item The \textbf{hypothesis space} $\Hspace$ is the set of all admissible functions:
    \[
      \Hspace = \{f : f \text{ belongs to a chosen functional family}\}
    \]
  \end{itemize}
\end{frame}

\begin{frame}{Parameterization}
  \begin{itemize}
    \item Models in $\Hspace$ usually share a structure parameterized by $\thetav$
    \item Parameter vector $\thetav = (\theta_1, \dots, \theta_d) \in \Theta$
    \item Setting $\thetav$ \emph{fixes} a specific function from the family
  \end{itemize}
  \vfill
  \[
    \Hspace = \left\{ f_{\thetav} : f_{\thetav} \text{ in chosen family, parameterized by } \thetav \in \Theta \right\}
  \]
  \vfill
  \textbf{Finding the optimal model} $\Leftrightarrow$ \textbf{finding the optimal parameters} — search on a $d$-dimensional surface.
\end{frame}

\begin{frame}{Example: univariate linear functions}
  \[
    \Hspace = \left\{ f : f(x) = \textcolor{armorange}{\theta_0} + \textcolor{armblue}{\theta_1} x, \; \thetav \in \R^2 \right\}
  \]
  \vfill
  Two parameters: intercept $\textcolor{armorange}{\theta_0}$ and slope $\textcolor{armblue}{\theta_1}$. Each choice of $(\theta_0, \theta_1)$ picks one line from the family.
  \vfill
  \begin{center}
    \includegraphics[width=0.65\textwidth]{figures/hs-lin-functions.pdf}
  \end{center}
\end{frame}

\begin{frame}{Example: bivariate quadratic functions}
  \[
    \Hspace = \big\{ f : f(\xv) = \textcolor{armorange}{\theta_0} + \textcolor{armblue}{\theta_1} x_1 + \textcolor{armblue}{\theta_2} x_2 + \textcolor{armred}{\theta_3} x_1^2 + \textcolor{armred}{\theta_4} x_2^2 + \textcolor{armred}{\theta_5} x_1 x_2, \; \thetav \in \R^6 \big\}
  \]
  \vfill
  6 parameters. Adding terms (the squared and cross terms) lets the surface curve. More parameters $\Rightarrow$ richer hypothesis space, but also more risk of overfitting (next lectures).
\end{frame}

\section{Linear Regression with \texorpdfstring{$L_2$}{L2} Loss}

\begin{frame}{Linear regression}
  \begin{columns}
    \column{0.55\textwidth}
    Predict $y \in \R$ as a \textbf{linear} combination of features:
    \[
      \yh = \fx = \thx = \theta_0 + \theta_1 x_1 + \dots + \theta_p x_p
    \]
    \vfill
    \begin{itemize}\small
      \item $\Hspace = \{f : \fx = \thx,\; \thetav \in \R^{p+1} \}$
      \item Find loss-optimal $\thetav$ describing $y \mid \xv$
      \item Special case of a linear model in arbitrary \emph{basis functions} (polynomials — see L02)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/reg_l2_basic_lm.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Design matrix}
  \begin{itemize}
    \item Mismatch: $\thetav \in \R^{p+1}$ vs.\ $\xv \in \R^p$ because of the intercept
    \item \textbf{Trick}: pad each $\xv$ with a leading 1:
      \[
        \xv \mapsto (1, x_1, \dots, x_p)^\top
        \quad\Rightarrow\quad
        \thx = \theta_0 \cdot 1 + \theta_1 x_1 + \dots + \theta_p x_p
      \]
    \item Stack all observations into the \textbf{design matrix} $\Xmat \in \R^{n \times (p+1)}$:
      \[
        \hat{\yv} = \Xmat \thetav
        = \begin{pmatrix}
          1 & x_1^{(1)} & \cdots & x_p^{(1)} \\
          1 & x_1^{(2)} & \cdots & x_p^{(2)} \\
          \vdots & \vdots & & \vdots \\
          1 & x_1^{(n)} & \cdots & x_p^{(n)}
        \end{pmatrix}
        \begin{pmatrix} \theta_0 \\ \theta_1 \\ \vdots \\ \theta_p \end{pmatrix}
      \]
  \end{itemize}
\end{frame}

\begin{frame}{Effect interpretation}
  \begin{itemize}
    \item Big advantage of the linear model: \textbf{interpretable} feature effects.
    \item ``Increasing $x_j$ by 1 unit, holding all others fixed, increases $\yh$ by $\theta_j$ units.''\\
      $\rightarrow$ \emph{ceteris paribus} reading.
    \item Watch out: this interpretation breaks if features are strongly correlated, or if you transform features (e.g., log, squares).
  \end{itemize}
\end{frame}

\begin{frame}{Model fit: the $L_2$ loss}
  \begin{columns}
    \column{0.55\textwidth}
    Pick the ``best'' $\thetav$ by defining a \textbf{loss} and minimizing.
    \vfill
    Most popular loss for regression — the \textbf{$L_2$ / quadratic / squared error}:
    \[
      L\!\left(y, \fx\right) = (y - \fx)^2
    \]
    \vfill
    Why quadratic in residuals $r = y - \fx$?
    \begin{itemize}\small
      \item Convex and differentiable
      \item Closed-form solution for LMs
      \item Connects to classical stats (Gaussian noise)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/reg_l2_residual.pdf}\\[2mm]
    \includegraphics[width=\textwidth]{figures/reg_l2_lossplot_quad.pdf}
  \end{columns}
\end{frame}

\begin{frame}{From loss to empirical risk}
  The \textbf{empirical risk} sums the loss over the training set:
  \[
    \risket
    = \sumin \Lxyi
    = \sumin \left( \yi - \thetav^\top \xv^{(i)} \right)^2
  \]
  Also called \textbf{sum of squared errors (SSE)}.
  \vfill
  Optimization:
  \[
    \thetah \in \argmin_{\thetav}\; \risket
    \;=\; \argmin_{\thetav}\; \| \yv - \Xmat \thetav \|_2^2
  \]
\end{frame}

\begin{frame}{Analytical optimization: the normal equations}
  Setting the gradient to zero,
  \[
    \frac{\partial \risket}{\partial \thetav} = 0,
  \]
  gives the \textbf{normal equations}. Solving them yields the \textbf{ordinary least-squares (OLS)} estimator:
  \[
    \boxed{\;\thetah = (\Xmat^\top \Xmat)^{-1} \Xmat^\top \yv\;}
  \]
  \vfill
  \begin{itemize}
    \item Exists when $\Xmat^\top \Xmat$ is invertible (i.e., features are linearly independent)
    \item For most ML losses no such closed form exists $\Rightarrow$ we use gradient descent (next lecture)
  \end{itemize}
\end{frame}

\begin{frame}{Connection to classical statistics}
  Linear regression with $L_2$ loss = the classical statistics linear model under:
  \begin{itemize}
    \item Observations $\xv^{(i)}$ are i.i.d.
    \item \textbf{Homoskedastic} Gaussian errors:
      \[
        \yv = \Xmat \thetav + \bm{\eps},
        \quad \bm{\eps} \sim \mathcal{N}(0, \sigma^2 \id)
      \]
      so $\yv \mid \Xmat \sim \mathcal{N}(\Xmat \thetav, \sigma^2 \id)$
    \item Uncorrelated features (else: multicollinearity)
  \end{itemize}
  \vfill
  Under these, classical inference applies: $p$-values, $t$-tests, confidence intervals, $R^2$.
\end{frame}

\begin{frame}{Template: this is what most of ML looks like}
  We just saw the basic ML recipe:
  \begin{enumerate}
    \item Choose a hypothesis space $\Hspace = \{f_{\thetav}\}$
    \item Choose a loss $L(y, \fx)$
    \item Define the empirical risk $\riske(\thetav) = \sum_i L(\yi, f(\xv^{(i)} \mid \thetav))$
    \item Find $\thetah = \argmin_{\thetav} \riske(\thetav)$ — by closed form, gradient descent, or other
  \end{enumerate}
  \vfill
  L02 — alternative losses (L1, robust). L03 — what happens when we want fewer parameters (regularization). L04/L05 — evaluating and tuning.
\end{frame}
