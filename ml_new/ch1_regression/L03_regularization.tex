\section{Why Regularize?}

\begin{frame}{Recap: overfitting}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Overfitting: model fits noise / artifacts in the training data
      \item Symptom: small \emph{training} error, large \emph{test} error
      \item L02 example: degree-15 polynomial on a tiny dataset
      \item Three high-level fixes:
        \begin{enumerate}
          \item Collect more data (often infeasible)
          \item Reduce hypothesis space (binary feature in/out — coarse)
          \item Penalize complexity \emph{continuously} $\Rightarrow$ \textbf{regularization}
        \end{enumerate}
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/complexity-vs-fit2.png}
  \end{columns}
\end{frame}

\begin{frame}{What is regularization?}
  Methods that inject an \textbf{inductive bias} (typically ``low complexity'') into the model to reduce overfitting and improve the bias-variance tradeoff.
  \vfill
  \begin{itemize}
    \item \textbf{Explicit} regularization: add a complexity penalty to ERM ($L_1$, $L_2$).
    \item \textbf{Implicit} regularization: early stopping, data augmentation, parameter sharing, dropout, ensembling.
    \item \textbf{Structured} regularization: prior knowledge over groups of parameters (e.g., group lasso).
  \end{itemize}
  \vfill
  This lecture focuses on \textbf{explicit} $L_1$ and $L_2$ penalties on the coefficients.
\end{frame}

\begin{frame}{Regularized empirical risk minimization}
  Trade off two competing goals:
  \begin{itemize}
    \item Maximize \textbf{fit} (minimize training loss)
    \item Minimize \textbf{complexity} of the model
  \end{itemize}
  \vfill
  Combine into a single objective:
  \[
    \boxed{\;\riskr(\thetav) \;=\; \risket \;+\; \lambda \cdot J(\thetav)\;}
  \]
  \begin{itemize}\small
    \item $J(\thetav)$: \textbf{complexity penalty} / regularizer
    \item $\lambda \ge 0$: \textbf{regularization strength}
    \item $\lambda = 0$: plain ERM
    \item $\lambda \to \infty$: model becomes as ``simple'' as possible (often 0 coefficients)
    \item Choosing $\lambda$ is a tuning problem — usually via cross-validation (L05)
  \end{itemize}
\end{frame}

\section{Bias-Variance Tradeoff}

\begin{frame}{Bias and variance, intuitively}
  Two sources of error when an estimated model differs from the truth:
  \vfill
  \begin{itemize}
    \item \textbf{Bias}: how far the \emph{average} estimate is from the truth
      \begin{itemize}
        \item High bias $=$ model class too restrictive to capture truth (underfit)
      \end{itemize}
    \item \textbf{Variance}: how much the estimate \emph{wiggles} across different training sets
      \begin{itemize}
        \item High variance $=$ model class so flexible it chases noise (overfit)
      \end{itemize}
  \end{itemize}
  \vfill
  Regularization \textbf{increases bias} but \textbf{reduces variance}. Net effect on test error is often favorable.
\end{frame}

\begin{frame}{Visualizing the model spaces}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item $\Hspace_0$: ``all'' functions; ideal Bayes-optimal $f^*_0 \in \Hspace_0$
      \item $\Hspace$: our chosen model class; best-in-class $f^* \in \Hspace$
      \item $\Hspace_R \subset \Hspace$: regularized class
      \item $d(f^*_0, f^*) =$ \textbf{model bias}
      \item $d(f^*, f^*_R) =$ \textbf{estimation bias} from regularization
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/bv_anim_5.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Estimation variance shrinks under regularization}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Estimated model $\fh$ depends on the random training sample
      \item \textbf{Estimation variance}: spread of $\fh$ across resampled training sets
      \item Restricting to $\Hspace_R$ \emph{reduces} this spread
      \item Tradeoff: higher bias (we're constrained), lower variance
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/bv_anim_3.pdf}\\[2mm]
    \includegraphics[width=\textwidth]{figures/bv_anim_2.pdf}
  \end{columns}
\end{frame}

\section{Ridge Regression (\texorpdfstring{$L_2$}{L2} Penalty)}

\begin{frame}{Ridge regression}
  Measure complexity as deviation from the origin. Shrink coefficients toward 0:
  \begin{align*}
    \thetah_{\text{ridge}}
    &= \argmin_{\thetav}\; \sumin (\yi - \thetav^\top \xv^{(i)})^2 \;+\; \lambda \sum_{j=1}^p \theta_j^2 \\
    &= \argmin_{\thetav}\; \| \yv - \Xmat \thetav \|_2^2 + \lambda \|\thetav\|_2^2
  \end{align*}
  \vfill
  Still has a \textbf{closed form}:
  \[
    \boxed{\;\thetah_{\text{ridge}} = (\Xmat^\top \Xmat + \lambda \id)^{-1} \Xmat^\top \yv\;}
  \]
  \vfill
  Name comes from adding positive entries along the diagonal ``ridge'' of $\Xmat^\top \Xmat$. Bonus: solves the singular-$\Xmat^\top \Xmat$ problem (multicollinearity).
\end{frame}

\begin{frame}{Ridge as constrained optimization}
  \begin{columns}
    \column{0.5\textwidth}
    Equivalent constrained form (Lagrange duality):
    \[
      \min_{\thetav}\; \sumin (\yi - \thetav^\top \xv^{(i)})^2
      \quad \text{s.t.}\quad \|\thetav\|_2^2 \le t
    \]
    \vfill
    \begin{itemize}\small
      \item Feasible region is an $L_2$ ball of radius $\sqrt{t}$
      \item Bijection: large $\lambda$ $\Leftrightarrow$ small $t$
      \item Solution lies on the \emph{boundary} of the ball (where it touches the smallest possible loss contour)
    \end{itemize}
    \column{0.5\textwidth}
    \includegraphics[width=\textwidth]{figures/ridge_perspectives_03.png}
  \end{columns}
\end{frame}

\begin{frame}{Ridge solution path}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/lin_model_regu_02.png}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item Toy DGP: $y = 3 x_1 - 2 x_2 + \eps$. True optimum $\thetav^* = (3, -2)$.
      \item As $\lambda \uparrow$, $\thetah_{\text{ridge}}$ is pulled toward the origin
      \item Contours $=$ unregularized objective; red points $=$ ridge solutions at increasing $\lambda$
      \item Path is \textbf{smooth and non-sparse}
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Why ridge helps: polynomial example}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/poly_ridge_02.png}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item True $f(x) = 5 + 2x + 10 x^2 - 2 x^3$ (red)
      \item Fit a degree-10 polynomial — overfits at $\lambda = 0$
      \item As $\lambda$ grows: coefficients shrink, model gets smoother, generalization improves
      \item Lets us pick \emph{$d$ too large} on purpose and let $\lambda$ control complexity (continuous, not discrete)
    \end{itemize}
  \end{columns}
\end{frame}

\section{Lasso Regression (\texorpdfstring{$L_1$}{L1} Penalty)}

\begin{frame}{Lasso regression}
  Penalize the $L_1$ norm of the coefficients instead:
  \[
    \thetah_{\text{lasso}}
    = \argmin_{\thetav}\; \| \yv - \Xmat \thetav \|_2^2 + \lambda \|\thetav\|_1
    = \argmin_{\thetav}\; \sumin (\yi - \thetav^\top \xv^{(i)})^2 + \lambda \sum_{j=1}^p |\theta_j|
  \]
  \vfill
  \begin{itemize}
    \item ``Least Absolute Shrinkage and Selection Operator''
    \item Still convex, but \textbf{not differentiable} at $\theta_j = 0$ $\Rightarrow$ no closed form
    \item Solve via coordinate descent, LARS, or proximal methods
    \item Key property: \textbf{sparsity} — many coefficients become \emph{exactly} 0
  \end{itemize}
\end{frame}

\begin{frame}{Lasso solution path}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/lin_model_regu_01.png}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item Same toy DGP ($y = 3 x_1 - 2 x_2 + \eps$)
      \item Lasso path takes a \emph{different route} than ridge
      \item Coefficients can become \textbf{exactly zero} for large enough $\lambda$
      \item Side effect: built-in \textbf{feature selection}
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Lasso vs ridge: geometric intuition}
  \begin{columns}
    \column{0.6\textwidth}
    \includegraphics[width=\textwidth]{figures/lasso_contour_cases.png}
    \column{0.4\textwidth}
    \begin{itemize}\small
      \item Ridge constraint: $L_2$ ball (smooth)
      \item Lasso constraint: $L_1$ \textbf{diamond} with kinks on the axes
      \item Loss contours hit the diamond's \textbf{corners} first — corners lie on coordinate axes — so some $\theta_j$ become exactly 0
      \item Ridge's round ball is tangent to contours \emph{between} the axes, so no sparsity
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Soft thresholding (orthonormal $\Xmat$)}
  In the special case $\Xmat^\top \Xmat = \id$, closed forms exist:
  \vfill
  \begin{itemize}
    \item \textbf{Lasso}: $\thetah_{\text{lasso}} = \text{sign}(\thetah_{\text{OLS}}) \cdot (|\thetah_{\text{OLS}}| - \lambda)_+$
      \begin{itemize}
        \item ``Soft thresholding'': small coefficients $\to$ 0, large ones shrunk by $\lambda$
      \end{itemize}
    \item \textbf{Ridge}: $\thetah_{\text{ridge}} = \thetah_{\text{OLS}} / (1 + \lambda)$
      \begin{itemize}
        \item Uniform downscaling, no sparsity
      \end{itemize}
  \end{itemize}
  \vfill
  \begin{center}
    \includegraphics[width=0.45\textwidth]{figures/soft_thresholding.png}
  \end{center}
\end{frame}

\section{Comparing \texorpdfstring{$L_1$}{L1} vs \texorpdfstring{$L_2$}{L2}}

\begin{frame}{Solution paths side-by-side}
  \begin{center}
    \includegraphics[width=0.85\textwidth]{figures/solution_paths_01.png}
  \end{center}
  \vfill
  \begin{itemize}
    \item \textbf{Ridge}: smooth, non-sparse path; all features stay in the model
    \item \textbf{Lasso}: piecewise-linear path; coefficients hit 0 one by one as $\lambda$ grows
  \end{itemize}
\end{frame}

\begin{frame}{When to use which}
  \textbf{Ridge ($L_2$)}:
  \begin{itemize}
    \item Many small-to-moderate effects expected
    \item Multicollinearity present (helps numerically)
    \item Closed-form fit is desirable (cheap)
    \item You don't need exact zeros
  \end{itemize}
  \vfill
  \textbf{Lasso ($L_1$)}:
  \begin{itemize}
    \item You suspect only a few features matter (sparse truth)
    \item You want automatic feature selection
    \item Interpretability matters
  \end{itemize}
  \vfill
  \textbf{Elastic Net} = $\alpha \|\thetav\|_1 + (1-\alpha) \|\thetav\|_2^2$: blends both. Good default when in doubt.
\end{frame}

\begin{frame}{Wrap-up}
  \begin{enumerate}
    \item Regularization adds an inductive bias toward ``simple'' models
    \item Common pattern: $\riskr(\thetav) = \risket + \lambda J(\thetav)$
    \item Ridge ($L_2$): smooth shrinkage, closed-form, no sparsity
    \item Lasso ($L_1$): kinked, sparse — does feature selection for free
    \item $\lambda$ controls the bias-variance tradeoff; pick it via CV (L05)
  \end{enumerate}
  \vfill

\end{frame}
