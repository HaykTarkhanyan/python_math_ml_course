\section{Why Resample?}

\begin{frame}{Recap: hold-out has problems}
  \begin{itemize}\small
    \item Small training set $\to$ pessimistic bias
    \item Small test set $\to$ high variance in the estimate
    \item One unlucky split can mislead
  \end{itemize}
  \vfill
  \textbf{Resampling}: repeatedly split into train/test, average the test errors.
  \begin{itemize}\small
    \item Use \emph{most} of the data for training each time (low bias)
    \item Average across repetitions (low variance)
    \item Costs more compute, but pays for itself in reliability
  \end{itemize}
  \vfill
  \begin{center}
    \includegraphics[width=0.45\textwidth]{figures/resampling_error.pdf}
  \end{center}
\end{frame}

\section{$k$-Fold Cross-Validation}

\begin{frame}{$k$-fold CV}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Split $\D$ into $k$ roughly equal folds
      \item Each fold serves as test set \emph{exactly once}; the other $k-1$ as training
      \item Get $k$ test errors; report the mean (often the std too)
      \item Fraction $(k-1)/k$ is used for training — e.g., 90\% with 10-fold CV
    \end{itemize}
    \vfill
    \begin{itemize}\small
      \item $k = 5$ or $k = 10$ are standard defaults
      \item $k = n$ is \textbf{leave-one-out (LOO) CV}
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/crossvalidation.png}\\[2mm]
    {\scriptsize 3-fold CV example.}
  \end{columns}
\end{frame}

\begin{frame}{Choosing $k$: bias vs variance}
  \begin{itemize}
    \item \textbf{Larger $k$}: each fold uses more data for training $\Rightarrow$ less pessimistic bias
    \item \textbf{LOO ($k = n$)}: nearly unbiased, but high variance — the $n$ training sets overlap almost completely so estimates are correlated
    \item \textbf{Smaller $k$}: cheaper, more independent splits, but bigger bias
  \end{itemize}
  \vfill
  Practical defaults:
  \begin{itemize}
    \item Most cases: \textbf{5- or 10-fold CV}
    \item Small datasets: \textbf{repeated CV} (e.g., $10\times$ 5-fold) — averages out fold-luck
    \item Avoid LOO unless you really need it (e.g., very small $n$ with cheap learner)
  \end{itemize}
\end{frame}

\begin{frame}{Stratified CV}
  When classes (or values) are imbalanced:
  \begin{itemize}
    \item A random fold might miss a rare class entirely
    \item \textbf{Stratified CV}: keep the target distribution \emph{within} each fold close to the overall distribution
    \item For classification: split each class separately, then join the per-class splits into folds
    \item Default in scikit-learn's \texttt{StratifiedKFold}
  \end{itemize}
\end{frame}

\begin{frame}{Bootstrap and subsampling}
  \textbf{Subsampling} (Monte Carlo CV):
  \begin{itemize}\small
    \item Repeated hold-out: draw a random train/test split many times, average
    \item Typical training fraction: 4/5 or 9/10
    \item More reps $\to$ less variance; smaller training fraction $\to$ more pessimistic bias
  \end{itemize}
  \vfill
  \textbf{Bootstrap}:
  \begin{itemize}\small
    \item Draw $B$ training sets of size $n$ \emph{with replacement} from $\D$
    \item Test on \textbf{out-of-bag} (OOB) points $\Dtest^b = \D \setminus \Dtrain^b$
    \item Each bootstrap training set contains $\approx 63.2\%$ of unique points
    \item Replicated points can cause artifacts; B632 / B632+ are variance-reducing extensions
  \end{itemize}
\end{frame}

\begin{frame}{Leave-one-object-out}
  When data is grouped (multiple obs per person, hospital, image, etc.):
  \begin{itemize}
    \item Standard i.i.d.\ assumption breaks
    \item Obs from the same object should be \emph{either} all in train \emph{or} all in test — never split
    \item Otherwise: the test set leaks information from training
    \item \textbf{Group $k$-fold} or \textbf{leave-one-object-out}: CV on objects, not individual rows
  \end{itemize}
\end{frame}

\section{Hyperparameters and Tuning}

\begin{frame}{Parameters vs.\ hyperparameters}
  \begin{itemize}
    \item \textbf{Model parameters} $\thetav$ — optimized \emph{during} training; an \emph{output} of training
      \begin{itemize}
        \item Linear regression: coefficients $\thetav$
        \item Decision tree: splits and leaf values
      \end{itemize}
    \item \textbf{Hyperparameters} $\lamv$ — fixed \emph{before} training; an \emph{input} to training
      \begin{itemize}
        \item Tree depth, $k$ in $k$-NN, regularization $\lambda$, learning rate $\alpha$
        \item Often control model \emph{complexity}
      \end{itemize}
  \end{itemize}
  \vfill
  Finding good hyperparameters is called \textbf{tuning}.
\end{frame}

\begin{frame}{Why tuning matters}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item Many learners are very sensitive to HP values
      \item Same algorithm, different HPs $\Rightarrow$ vastly different generalization performance
      \item Defaults rarely optimal for your data
    \end{itemize}
    \column{0.5\textwidth}
    \includegraphics[width=\textwidth]{figures/tuning_importance.png}\\
    {\scriptsize \texttt{mlbench.spiral} — different HPs yield big gaps.}
  \end{columns}
\end{frame}

\begin{frame}{The tuning problem (bilevel optimization)}
  Two nested optimizations:
  \begin{enumerate}
    \item \textbf{Inner loop}: given $\lamv$, train the model to find $\thetah$
    \item \textbf{Outer loop}: search $\lamv$ to minimize the estimated generalization error
  \end{enumerate}
  \vfill
  \begin{center}
    \includegraphics[width=0.45\textwidth]{figures/riskmin_bilevel3.png}
  \end{center}
\end{frame}

\begin{frame}{HP types}
  Hyperparameters can be mixed:
  \begin{itemize}
    \item \textbf{Real-valued}: learning rate, regularization $\lambda$
    \item \textbf{Integer}: tree depth, $k$ in $k$-NN, \texttt{mtry} in RF
    \item \textbf{Categorical}: split criterion (Gini / entropy), distance metric, optimizer choice
  \end{itemize}
  \vfill
  Often \textbf{hierarchically dependent}: e.g., \emph{if} kernel is RBF, what is its bandwidth? — irrelevant when kernel is linear.
\end{frame}

\section{Grid Search and Random Search}

\begin{frame}{Grid search}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Predefine candidate values for each HP
      \item Try \emph{all combinations}
      \item Pick the best by CV score
    \end{itemize}
    \vfill
    \textbf{Pros}:
    \begin{itemize}\small
      \item Dead simple to implement
      \item All HP types supported
      \item Embarrassingly parallel
    \end{itemize}
    \textbf{Cons}:
    \begin{itemize}\small
      \item \textbf{Combinatorial explosion} in high-dim HP spaces
      \item Wastes evaluations on uninteresting regions
      \item Discretization is arbitrary
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/cart_tuning_balgos_1.pdf}\\
    {\scriptsize 10$\times$10 grid.}
  \end{columns}
\end{frame}

\begin{frame}{Random search}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\small
      \item Sample HP configurations uniformly from the region of interest
      \item Pick the best by CV score
    \end{itemize}
    \vfill
    \textbf{Pros}:
    \begin{itemize}\small
      \item Easy to implement, parallelizable
      \item \textbf{Anytime} — stop whenever budget runs out
      \item Each HP gets a unique value every trial — better coverage than grid in the few dimensions that matter
    \end{itemize}
    \textbf{Cons}:
    \begin{itemize}\small
      \item Still inefficient in high dim
      \item No use of information from past trials (cf.\ Bayesian opt)
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{figures/cart_tuning_balgos_2.pdf}\\
    {\scriptsize 100 random samples.}
  \end{columns}
\end{frame}

\begin{frame}{Why random often beats grid}
  \begin{columns}
    \column{0.55\textwidth}
    \includegraphics[width=\textwidth]{figures/rs_gs_simulation_plot.png}
    \column{0.45\textwidth}
    \begin{itemize}\small
      \item Suppose the target depends mostly on $x_1$ (irrelevant $x_2$)
      \item Grid: only 5 distinct $x_1$ values are tried (the rest are wasted on $x_2$)
      \item Random: each trial samples a fresh $x_1$ $\Rightarrow$ much better coverage of the relevant axis
      \item \textbf{Bergstra \& Bengio (2012)} popularized this argument
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Beyond grid and random}
  \begin{itemize}
    \item \textbf{Bayesian optimization} (BO / SMBO): build a surrogate model of the HP$\to$score function; pick the next config to maximize expected improvement. Much more sample-efficient.
    \item \textbf{Hyperband}: spend less compute on bad configs early; promising configs run longer. Good when training is expensive.
    \item \textbf{BOHB}: combines BO with Hyperband.
    \item Libraries: \texttt{optuna}, \texttt{Ray Tune}, \texttt{Hyperopt}, \texttt{scikit-optimize}.
  \end{itemize}
\end{frame}

\section{Workflow Summary}

\begin{frame}{Putting it all together}
  Typical workflow for an honest evaluation \emph{with} tuning:
  \begin{enumerate}
    \item Hold out a final test set, set it aside — never touch it
    \item On the remaining data, run CV-based tuning: search HP space, score by inner-CV mean
    \item Refit the best HPs on the full inner data
    \item Evaluate \emph{once} on the held-out test set $\to$ unbiased estimate of GE
  \end{enumerate}
  \vfill
  \textbf{Why not just CV the chosen HPs?}\\
  Because tuning peeks at the CV score, which \emph{also} acts as test data $\Rightarrow$ optimistic bias. The fix: \textbf{nested resampling} (L05b).
\end{frame}

\begin{frame}{Wrap-up}
  \begin{enumerate}
    \item Resampling beats single hold-out: lower variance, lower bias
    \item $k$-fold CV ($k = 5$ or $10$) is the default; stratify for imbalanced data
    \item HPs are inputs to training; tuning is a bilevel optimization
    \item Grid search: simple. Random search: usually better. BO / Hyperband: even better when expensive.
    \item Tuning on CV makes the same CV score \emph{biased} $\Rightarrow$ nested resampling (L05b).
  \end{enumerate}
\end{frame}
