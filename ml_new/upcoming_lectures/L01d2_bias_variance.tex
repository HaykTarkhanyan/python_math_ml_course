\section{The two failure modes}

\begin{frame}{Underfit vs.\ overfit}
  Models fail in two opposite ways:
  \vskip 4pt
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Underfitting.} Model class too rigid to capture the signal.
    \begin{itemize}\small
      \item High training error AND high test error.
      \item Adding data won't help.
      \item Fix: more flexible model, more features, less regularization.
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Overfitting.} Model so flexible it memorizes noise.
    \begin{itemize}\small
      \item Low training error, high test error.
      \item More data WILL help.
      \item Fix: simpler model, fewer features, more regularization, more data.
    \end{itemize}
  \end{columns}

\begin{frame}{Predict-first: where does the error come from?}
  Same data, three fits. The training error tells us nothing useful by itself.
  \vfill
  \pause
  \textbf{Predict:} for each fit above, decompose the error into ``the model class can't represent the truth'' vs.\ ``the model overreacted to noise in this specific sample''.
  \pause
  \vfill
  \begin{itemize}\small
    \item \textbf{Degree 1} (underfit): error is mostly from \emph{model class}. A straight line cannot bend. Different training data $\to$ similar wrong fit. Low variance, high systematic error.
    \item \textbf{Degree 9} (overfit): error is mostly from \emph{specific training sample}. The wiggles chase noise. Different training data $\to$ wildly different fits. High variance, low systematic error.
    \item \textbf{Degree 3} (sweet spot): small contribution from each. The model is flexible enough to capture the truth without chasing noise.
  \end{itemize}
  \vfill
  \emph{Systematic error} $=$ \textbf{bias}. \emph{Sensitivity to the specific training sample} $=$ \textbf{variance}.
\end{frame}

\section{The decomposition}

\begin{frame}{Bias and variance: definitions}
  Setup: true relationship $y = f(\xv) + \varepsilon$ with $\E[\varepsilon] = 0$, $\Var(\varepsilon) = \sigma^2$. A training set $D$ is randomly drawn; you train a model $\fD$ on it.
  \vfill
  Imagine repeating training many times on independent $D$'s. Define the \textbf{average model} at a fixed test point:
  \[
    \fbar(\xv) \;:=\; \E_D\!\left[\fDx\right]
  \]
  Then:
  \begin{itemize}
    \item \textbf{Bias} $= \fbar(\xv) - f(\xv)$. How far off the \emph{average} fit is from the truth. A property of the \emph{model class}, not of any one training set.
    \item \textbf{Variance} $= \E_D\!\left[\left(\fDx - \fbar(\xv)\right)^2\right]$. How much the fit \emph{wobbles} from one $D$ to another. A property of \emph{sensitivity to training data}.
    \item \textbf{Irreducible error} $= \sigma^2$. The noise floor. No model can do better than this.
  \end{itemize}
  \vfill
  \textbf{Main result:} for a fixed test point $\xv$,
  \[
    \boxed{\;\E_{D,\varepsilon}\!\left[(y - \fDx)^2\right] \;=\; \Bias^2[\fDx] \;+\; \Var_D[\fDx] \;+\; \sigma^2\;}
  \]
\end{frame}

\begin{frame}{The full derivation}
  \footnotesize
  \textbf{Plan.} Add zero twice. Step 1 adds $\varepsilon$ on one side (exposes $\sigma^2$). Step 2 adds $\fbar(\xv)$ (splits the rest into bias$^2$ and variance). Cross terms vanish both times.
  \vskip 4pt
  \textbf{Setup.} $y = f(\xv) + \varepsilon$, $\E[\varepsilon] = 0$, $\Var(\varepsilon) = \sigma^2$, $\varepsilon \perp D$. Let $\fbar(\xv) := \E_D[\fDx]$. We expand $\E[(y - \fDx)^2]$ (expectation over both $D$ and $\varepsilon$, conditional on $\xv$).
  \vfill
  \textbf{Step 1: separate the noise.} Substitute $y = f(\xv) + \varepsilon$:
  \begin{align*}
    \E[(y - \fDx)^2]
    &= \E\!\left[\bigl(f(\xv) - \fDx + \varepsilon\bigr)^2\right] \\[2pt]
    &= \E\!\left[(f - \fDx)^2\right]
       + \;2\,\underbrace{\E\!\left[\varepsilon\,(f - \fDx)\right]}_{=\;0\;\text{since}\;\varepsilon \perp D,\;\E[\varepsilon]=0}
       + \;\underbrace{\E[\varepsilon^2]}_{=\;\sigma^2}
  \end{align*}
  \vfill
  \textbf{Step 2: add and subtract the average model $\fbar(\xv)$.} Inside the model-error term:
  \begin{align*}
    \E\!\left[(f - \fDx)^2\right]
    &= \E\!\left[\bigl((f - \fbar) + (\fbar - \fDx)\bigr)^2\right] \\[2pt]
    &= \underbrace{(f - \fbar)^2}_{\text{Bias}^2,\;\text{deterministic}}
       + \;2(f - \fbar)\,\underbrace{\E_D[\fbar - \fDx]}_{=\;\fbar - \fbar = 0}
       + \underbrace{\E_D\!\left[(\fbar - \fDx)^2\right]}_{=\;\Var_D[\fDx]}
  \end{align*}
  \vfill
  \textbf{Combine:}
  \[
    \boxed{\;\E_{D,\varepsilon}[(y - \fDx)^2 \mid \xv]
      \;=\; \underbrace{(f(\xv) - \fbar(\xv))^2}_{\Bias^2}
      \;+\; \underbrace{\E_D[(\fDx - \fbar(\xv))^2]}_{\Var}
      \;+\; \underbrace{\sigma^2}_{\text{noise}}\;}
  \]
\end{frame}

\begin{frame}{Bayes risk + the four sources of error}
  Bias-variance is \emph{pointwise} (at fixed $\xv$). A broader view, used in learning theory, decomposes the \emph{risk} $R(f) = \E_{(\xv,y)}[L(y, f(\xv))]$ into four levels.
  \vfill
  Let $f^\star$ $=$ best predictor over \emph{all} functions; $f_H^\star$ $=$ best within your hypothesis class $H$; $\fD$ $=$ what you actually trained.
  \[
    \underbrace{R(\fD) - R(f^\star)}_{\text{excess risk}}
    \;=\;
    \underbrace{[R(f_H^\star) - R(f^\star)]}_{\text{approximation}}
    \;+\;
    \underbrace{[R(\fD) - R(f_H^\star)]}_{\text{estimation} + \text{optimization}}
  \]
  \vfill
  \footnotesize
  \begin{itemize}\footnotesize
    \item \textbf{Bayes risk} $R(f^\star) = \sigma^2$ for squared loss. The noise floor. \emph{No model can do better}; you'd have to change the problem (collect cleaner labels) to reduce it.
    \item \textbf{Approximation error.} Cost of restricting to $H$. ``A linear model cannot fit a cubic.'' $\approx$ bias.
    \item \textbf{Estimation error.} Cost of finite data $D$. ``Same model class, but parameters differ from the population optimum.'' $\approx$ variance.
    \item \textbf{Optimization error.} Cost of incomplete convergence. ``Gradient descent stopped before reaching the minimum.'' \emph{Zero for OLS / Ridge / Lasso} (closed form or convex), often nonzero for NNs.
  \end{itemize}
  \vfill
  Bias-variance assumes the optimizer is perfect. For everything in L01-L05 (linear models) this holds. For NNs the optimization term reappears - and double descent (frame 18) lives in this regime.
\end{frame}

\section{Levers and the tradeoff}

\begin{frame}{What controls bias and variance?}
  \begin{center}\small
    \begin{tabular}{p{3.8cm} c c p{5.5cm}}
      \textbf{Lever (increase)} & \textbf{Bias} & \textbf{Var} & \textbf{Why} \\ \hline
      Model complexity & $\downarrow$ & $\uparrow$ & more wiggle room; fits both signal AND noise \\
      Training data $n$ & $=$ & $\downarrow$ ($\propto 1/n$) & average over more samples is more stable \\
      Regularization $\lambda$ & $\uparrow$ & $\downarrow$ & shrinks coefficients $\to$ less flexible, less sensitive \\
      Features $p$ & $\downarrow$ & $\uparrow$ & more degrees of freedom (bias if signal, variance always) \\
      Noise $\sigma^2$ & $=$ & $=$ & raises the irreducible floor only \\
      Ensembling (bagging) & $=$ & $\downarrow$ & averaging $B$ models reduces variance by $\sim 1/B$ \\
    \end{tabular}
  \end{center}
  \vfill
  \textbf{The fundamental tension:} all the ``classical'' levers (complexity, regularization, features) trade one for the other. \emph{Only data and ensembling reduce variance without raising bias.} Hence: ``the easiest way to improve a model is to get more data.''
\end{frame}

\begin{frame}{The classical tradeoff curve}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    As model complexity grows:
    \begin{itemize}\small
      \item Bias$^2$ \textbf{falls} (more flexibility, can match truth).
      \item Variance \textbf{rises} (more flexibility, fits noise too).
      \item Total test error follows a \textbf{U-shape}.
      \item The minimum is the \emph{sweet spot}.
    \end{itemize}
    \vskip 6pt
    Cross-validation is just an empirical procedure for finding this sweet spot without knowing the curve.
    \column{0.5\textwidth}
    \begin{center}

\begin{frame}{Predict-first: what does more data do to bias and variance?}
  You have $n = 100$ samples and a fixed-complexity model. You double to $n = 200$.
  \vfill
  \pause
  \textbf{Predict:} what happens to
  \begin{itemize}
    \item bias?
    \item variance?
    \item irreducible noise?
  \end{itemize}
  \pause
  \vfill
  Answers:
  \begin{itemize}
    \item \textbf{Bias: unchanged.} Bias is a property of the \emph{model class}, not the data size. A linear model still can't bend, regardless of how many points you give it.
    \item \textbf{Variance: roughly halves} ($\propto 1/n$ for many estimators). Each fit gets closer to the average fit when there's more data to average over.
    \item \textbf{$\sigma^2$: unchanged.} The noise in $y \mid \xv$ doesn't shrink when you collect more rows.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.96\textwidth}{\small
    \textbf{The operational test.} Train error \emph{low} + test error \emph{high} $\Rightarrow$ \textbf{get more data} (it's a variance problem). Train error \emph{also high} $\Rightarrow$ \textbf{change the model} (more data won't help; it's a bias problem). This single rule resolves most ``my model isn't working'' questions.
  }}
\end{frame}

\section{Fixing each problem}

\begin{frame}{Fixing underfitting (high bias)}
  The model class can't capture the signal. Make it more flexible. Try in order of effort:
  \vfill
  \begin{enumerate}
    \item \textbf{Reduce regularization first.} Smallest, fastest change. Lower $\lambda$ in Ridge/Lasso (L01b, L03); raise \texttt{max\_depth} on trees. If this helps, you were over-regularized.
    \item \textbf{Add domain features.} Ratios, differences, interactions (L01g). Often the highest-impact change for tabular data.
    \item \textbf{Polynomial expansion.} Use $\phi(\xv) = (1, x, x^2, \ldots, x^d)$ (L01b). Cheap second-order signal.
    \item \textbf{Move to a more flexible model class.} Linear $\to$ tree $\to$ forest $\to$ GBM. Qualitative jump.
    \item \textbf{Train longer.} For GD-style methods (NN, GBM with early stopping). If train loss is still falling, you're still learning.
    \item \textbf{Remove constraints.} Larger \texttt{max\_depth}, wider hidden layers, etc.
  \end{enumerate}
  \vfill
  \footnotesize Adding data does \emph{not} fix underfitting. Bias is a property of the model class.
\end{frame}

\begin{frame}{Fixing overfitting (high variance)}
  The model is too flexible. Constrain it or drown it in data. Ranked by reliability:
  \vfill
  \begin{enumerate}
    \item \textbf{More data.} The single most reliable fix. Variance scales as $1/n$. If you can collect or synthesize more, do that first.
    \item \textbf{Regularization.} Ridge, Lasso, ElasticNet (L01b, L03). For trees: \texttt{min\_samples\_leaf}, \texttt{max\_depth}, pruning. Cheapest knob if data is fixed.
    \item \textbf{Drop features.} Feature selection (L01h). Helps especially when $p > n$.
    \item \textbf{Simpler model class.} Forest $\to$ shallow tree, polynomial degree 9 $\to$ 3. Qualitative reduction.
    \item \textbf{Ensembling (bagging).} Average over bootstrapped models. Cuts variance without changing bias.
    \item \textbf{Early stopping.} For iterative methods, halt when val score plateaus.
    \item \textbf{CV-tune the hyperparameters} you control with all of the above (L01d, L01e).
  \end{enumerate}
\end{frame}

\begin{frame}{Regularization as the bias-variance lever}
  Regularization $\lambda$ is the cleanest knob. With Ridge regression:
  \[
    \thetah_{\text{Ridge}} = \argmin_{\thetav}\; \|\yv - \Xmat\thetav\|^2 + \lambda\,\|\thetav\|^2
  \]
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \begin{itemize}\footnotesize
      \item $\lambda = 0$: pure OLS. Lowest bias, highest variance.
      \item $\lambda \to \infty$: $\thetah \to 0$. Highest bias (predicts $\bar y$), zero variance.
      \item Intermediate $\lambda$: trades bias against variance smoothly.
    \end{itemize}
    \vskip 4pt
    Pick $\lambda$ at the minimum of CV error.\\[2pt]
    \textbf{Same U as complexity}, x-axis flipped: $\lambda$ is the \emph{inverse} of model capacity. Two knobs, one tradeoff.
    \column{0.5\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4cm,
                     xlabel={$\log \lambda$ (regularization $\to$)},
                     ylabel={error}, xlabel style={font=\scriptsize},
                     ylabel style={font=\scriptsize}, ytick=\empty,
                     xtick=\empty, axis line style={gray!60}, ymin=0, ymax=1.2,
                     legend pos=north east, legend style={font=\tiny},
                     grid=major, grid style={dashed,gray!30}]
          \addplot[armblue, very thick, smooth, domain=-3:3, samples=50] {0.1 + 0.07*x*x};
          \addlegendentry{Bias$^2$}
          \addplot[armred, very thick, smooth, domain=-3:3, samples=50] {0.05 + 0.6*exp(-x)};
          \addlegendentry{Variance}
          \addplot[paramgreen, ultra thick, smooth, domain=-3:3, samples=50] {0.25 + 0.07*x*x + 0.6*exp(-x)};
          \addlegendentry{Total}
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Cross-validation as a bias-variance manager}
  CV is itself subject to bias-variance considerations. Choosing $k$ trades them:
  \vfill
  \begin{itemize}\small
    \item \textbf{$k = n$ (LOO).} Train sets are nearly identical $\to$ correlated fold scores $\to$ high \emph{variance} of the CV estimate. Bias of the CV estimate is low (uses $n-1$ samples per fit, almost the whole training set).
    \item \textbf{$k = 5$ or $10$.} Train sets share less data $\to$ less correlated $\to$ lower variance. Each fit uses $80\%$-$90\%$ of the data $\to$ small bias.
    \item \textbf{$k = 2$.} Each fit uses only half the data $\to$ \emph{biased} (CV scores look worse than the model trained on full data) but low variance.
  \end{itemize}
  \vfill
  Standard $k = 5$ is the sweet spot for the CV estimate's own bias-variance tradeoff (the same U-shape, at a different level).
  \vfill
  \textbf{Takeaway:} you use CV to find the bias-variance sweet spot \emph{for the model}; meanwhile your choice of $k$ trades bias and variance \emph{for the CV estimate itself}. Bias-variance is fractal.
\end{frame}

\section{Modern wrinkle: double descent}

\begin{frame}{Double descent}
  Classical theory: test error has a single U-shape vs.\ complexity. \emph{Modern} models (deep nets, very-wide-feature linear models) show a \textbf{second descent}.
  \vfill
  \begin{columns}[T]
    \column{0.45\textwidth}
    \footnotesize
    \begin{itemize}\footnotesize
      \item Up to the \textbf{interpolation threshold} ($p \approx n$): test error follows the classical U.
      \item At $p \approx n$: test error \emph{peaks}. The model can just barely fit the training data, and does so with huge coefficients.
      \item Beyond ($p \gg n$): test error \emph{falls again}. There are many ways to fit the training data exactly; gradient descent happens to prefer low-norm solutions, and those generalize. (The technical name is ``implicit regularization''.)
    \end{itemize}
    \vskip 4pt
    \textbf{Why students care:} this is why deep nets work even though they're massively over-parameterized. Classical bias-variance is the small-$p$ half of a bigger picture.
    \column{0.55\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4.5cm,
                     xlabel={model size / complexity}, ylabel={test error},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     xtick=\empty, ytick=\empty, axis line style={gray!60},
                     xmin=0, xmax=10, ymin=0, ymax=1.5,
                     grid=major, grid style={dashed,gray!30}]
          % classical U from 0 to interpolation
          \addplot[armred, very thick, smooth, domain=0.5:4.5, samples=80]
            {1.0/x^1.5 + 0.05*x^1.5};
          % peak around interpolation threshold
          \addplot[armred, very thick, smooth, domain=4.5:5.5, samples=80]
            {0.42 + 1.2*exp(-((x-5)*(x-5))/0.08)};
          % modern descent
          \addplot[armred, very thick, smooth, domain=5.5:10, samples=80]
            {0.5 - 0.04*(x-5.5) + 0.01*sin(deg(2*x))};
          \draw[gray!50, dashed] (axis cs:5, 0) -- (axis cs:5, 1.5);
          \node[font=\tiny, gray, anchor=south west] at (axis cs:5, 1.45) {interpolation};
          \node[font=\tiny, gray, anchor=south west] at (axis cs:5, 1.30) {threshold ($p = n$)};
          \node[font=\tiny, paramgreen, anchor=south] at (axis cs:2.5, 0.05) {classical U};
          \node[font=\tiny, paramgreen, anchor=south] at (axis cs:7.5, 0.05) {modern regime};
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\end{frame}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Two failure modes:} underfit (high bias) vs.\ overfit (high variance).
    \item \textbf{The decomposition:} $\E[(y - \fDx)^2] = \Bias^2 + \Var + \sigma^2$, derived in 4 lines via two ``add-and-subtract'' tricks.
    \item \textbf{Levers:} complexity, regularization, features all trade bias for variance. Only \emph{data} and \emph{ensembling} reduce variance without raising bias.
    \item \textbf{Diagnosis:} train vs.\ val error gap. Both high $=$ underfit; gap big $=$ overfit.
    \item \textbf{CV} finds the bias-variance sweet spot empirically. The choice of $k$ trades bias and variance \emph{of the CV estimate itself} - typically $k = 5$.
    \item \textbf{Modern wrinkle:} beyond the interpolation threshold, test error descends again (deep nets). Classical bias-variance is the small-$n$ half of the picture.
    \item \textbf{When to stop.} Train-val gap is small AND CV error has plateaued AND the operational test (frame 11) says ``no easy fix left''. Then ship.
  \end{itemize}
\end{frame}

\begin{frame}{HW01d2 - empirical bias-variance}
  \footnotesize Estimated time: 90-120 minutes. Tasks 1-4 graded; bonuses optional.
  \begin{enumerate}\small
    \item Generate $n = 30$ noisy samples from $y = \sin(\pi x) + \mathcal{N}(0, 0.2^2)$ on $x \in [0,1]$. Fit polynomial regression for degrees $d = 1, 2, \ldots, 15$. Plot training MSE and test MSE vs.\ $d$. Identify the sweet spot.
    \item \textbf{Empirical bias-variance.} Repeat $M = 500$ times: regenerate training data, refit each degree. At a held-out test grid, compute mean prediction $\bar{f}_M$ and variance $\Var_M$. Estimate empirical $\Bias^2$ using $f(x) = \sin(\pi x)$. Plot $\widehat{\Bias^2}$, $\widehat{\Var}$, $\widehat{\Bias^2} + \widehat{\Var} + \sigma^2$ vs.\ $d$. Verify the third curve overlaps your test MSE from Task 1.
    \item Vary $n \in \{10, 30, 100, 300\}$. Re-run Task 2 for each. Plot bias and variance vs.\ $n$. Confirm: bias roughly constant, variance $\propto 1/n$.
    \item Fit Ridge with degree-15 features and $\lambda \in \{10^{-4}, 10^{-2}, 1, 10^2, 10^4\}$. Empirically decompose bias-variance for each $\lambda$. Where is the Ridge sweet spot?
    \item \emph{Bonus 1:} Simulate double descent. Use a degree-50 polynomial on $n = 30$ samples with no regularization. Plot test error as you add features one at a time. Look for the spike near $p = n$.
    \item \emph{Bonus 2:} Pair Task 2 with bagging: average predictions from 50 Ridge models trained on bootstrap samples. Show variance drops, bias roughly constant.
    \item \emph{Bonus 3:} Repeat Task 2 with a tree of varying \texttt{max\_depth}. Compare the bias-variance trajectory to polynomial.
  \end{enumerate}
\end{frame}
