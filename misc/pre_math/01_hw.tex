\section*{1. Limits}

\begin{enumerate}
    \item $\displaystyle \lim_{x \to 2} (3x + 1)$
        
    \item $\displaystyle \lim_{x \to \infty} \frac{1}{x}$
    
    \item $\displaystyle \lim_{x \to 0} (x^2)$
    
    \item $\displaystyle \lim_{x \to \infty} (2x + 5)$
\end{enumerate}


\section*{2. Monotonicity (Increasing or Decreasing)}

\begin{enumerate}    
    \item $f(x) = -x^2 + 4x$ \\
    Find where the function increases and decreases.
    
    \item $f(x) = x^3$ \\
    Check if the function is always increasing, decreasing, or both.
    
    \item $f(x) = 2x - 5$ \\
    Is it increasing or decreasing? Explain using the derivative.
\end{enumerate}


\section*{3. Mini Challenge}

$f(x) = x^2 - 4x + 3$
\begin{enumerate}
    \item Find $f'(x)$. 
    \item Find where $f(x)$ is increasing.
    \item Find where $f(x)$ is decreasing.
\end{enumerate}
