\section{What Is Claude Code?}

\begin{frame}{What Is a Coding Assistant?}
A coding assistant uses language models to tackle complex programming tasks.

\bigskip
\textbf{How it works (same as a human developer):}
\begin{enumerate}
  \item \textbf{\color{popblue}Gather context} --- read files, understand errors, find relevant code
  \item \textbf{\color{paramgreen}Formulate a plan} --- decide on a fix or implementation strategy
  \item \textbf{\color{orange1}Take action} --- edit files, run commands, verify results
\end{enumerate}

\bigskip
Steps 1 and 3 require the model to \emph{interact with the outside world}.

But language models can only process text and return text\ldots
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Claude Code vs.\ GitHub Copilot}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{\color{popblue}Claude Code}
\begin{itemize}
  \item \textbf{Agentic orchestration}: plans and executes entire tasks autonomously
  \item 1M token context window --- full repo understanding
  \item 300+ MCP integrations (Slack, Sentry, DBs, \ldots)
  \item Git-native: stages, commits, creates PRs
  \item Best for \textbf{delegating entire tasks}
\end{itemize}

\column{0.48\textwidth}
\textbf{\color{sampred}GitHub Copilot}
\begin{itemize}
  \item \textbf{Multi-model wrapper}: GPT, Claude, Gemini behind one UI
  \item Developer-directed: you guide each step
  \item Deep GitHub ecosystem (Actions, security, org controls)
  \item Agent mode for multi-file, but iterative
  \item Best for \textbf{speeding up known work}
\end{itemize}
\end{columns}

\bigskip
\begin{block}{Why orchestration matters (SWE-bench)}
Same Claude 3.5 Sonnet: \textbf{49\%} with basic scaffold $\to$ \textbf{64\%} with agentic scaffolding (+30\% relative). \\
Orchestration intelligence matters as much as model intelligence for complex coding tasks.
\end{block}
\end{frame}

\section{Getting Hands On}

\begin{frame}[fragile]{Setup \& Project Initialization}
\textbf{Installation:}
\begin{lstlisting}[language=bash]
npm install -g @anthropic-ai/claude-code
cd your-project
claude          # launches Claude Code
\end{lstlisting}

\bigskip
\textbf{The \texttt{/init} command:}
\begin{itemize}
  \item Analyzes your entire codebase
  \item Creates a \texttt{CLAUDE.md} summary (architecture, commands, patterns)
  \item Included in \emph{every} request --- persistent system prompt
\end{itemize}

\bigskip
\begin{alertblock}{Tip}
Press \textbf{Shift+Tab} to cycle permission modes (e.g.\ auto-accept edits for the session).
\end{alertblock}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}[fragile]{Permissions}
\textbf{Three tool approval tiers:}
\begin{table}
\centering\small
\begin{tabular}{lll}
\toprule
\textbf{Tool type} & \textbf{Example} & \textbf{Approval} \\
\midrule
Read-only & File reads, Grep & Automatic \\
Bash commands & Shell execution & Per command \\
File modification & Edit, Write & Per session \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Permission modes} (cycle with \textbf{Shift+Tab}):
\begin{center}
\begin{tikzpicture}[
  node distance=0.3cm and 0.4cm,
  box/.style={rectangle, rounded corners, draw=popblue, fill=lightbg,
              text width=2.2cm, minimum height=0.65cm, align=center, font=\footnotesize},
]
  \node[box] (def) {\texttt{default}\\ask each tool};
  \node[box, right=of def] (edit) {\texttt{acceptEdits}\\auto-approve edits};
  \node[box, right=of edit] (plan) {\texttt{plan}\\read-only};
  \node[box, right=of plan] (auto) {\texttt{auto}\\smart auto-approve};
  \node[box, right=of auto] (bypass) {\texttt{bypass}\\skip all prompts};
\end{tikzpicture}
\end{center}

\smallskip
\textbf{Pre-approve in \texttt{settings.json}:}
\begin{lstlisting}
"permissions": { "allow": ["Bash(npm run *)"],
                  "deny":  ["Read(.env)"] }
\end{lstlisting}
\vspace{-0.3em}
\footnotesize Evaluation order: \textbf{deny} $>$ ask $>$ allow (deny always wins).
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}[fragile]{Adding Context: CLAUDE.md}
\textbf{Three locations, three scopes:}
\begin{table}
\centering\small
\begin{tabular}{lll}
\toprule
\textbf{File} & \textbf{Scope} & \textbf{Shared?} \\
\midrule
\texttt{CLAUDE.md} & Project & Yes (committed) \\
\texttt{CLAUDE.local.md} & Project & No (personal) \\
\texttt{\textasciitilde/.claude/CLAUDE.md} & All projects & No (global) \\
\bottomrule
\end{tabular}
\end{table}

\bigskip
\textbf{Memory mode with \texttt{\#}:}
\begin{lstlisting}
# Use comments sparingly. Only comment complex code.
\end{lstlisting}
Claude merges this instruction into your \texttt{CLAUDE.md} automatically.

\bigskip
\textbf{File mentions with \texttt{@}:}
\begin{lstlisting}
How does the auth system work? @auth
\end{lstlisting}
Includes the file's contents in the request. Also works inside \texttt{CLAUDE.md}.
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Making Changes}
\textbf{Key techniques:}
\begin{itemize}
  \item \textbf{Screenshots} --- paste images directly; Claude reads error messages, UI issues
  \item \textbf{Planning mode} --- one of the \texttt{Shift+Tab} modes; Claude outlines steps before executing
  \item \textbf{Adaptive thinking} --- Claude adjusts reasoning depth; tune with \texttt{/effort}
  \item \textbf{Auto mode} --- classifier auto-approves safe operations, blocks risky ones
\end{itemize}

\bigskip
\begin{exampleblock}{Workflow}
\begin{enumerate}
  \item Describe the problem or feature
  \item Let Claude explore, plan, and implement
  \item Review the diff, approve or redirect
\end{enumerate}
\end{exampleblock}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Thinking \& Effort Levels}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{\color{popblue}Adaptive Thinking} (Opus 4.6 / Sonnet 4.6)
\begin{itemize}
  \item Claude \textbf{dynamically allocates} reasoning depth
  \item Controlled via \texttt{/effort} command:
\end{itemize}

\smallskip
{\small
\begin{tabular}{ll}
\texttt{low} & Fast, cheap --- straightforward tasks \\
\texttt{medium} & Default balanced mode \\
\texttt{high} & Deep reasoning --- complex problems \\
\texttt{max} & No token limit (Opus 4.6 only) \\
\end{tabular}}

\bigskip
\textbf{\color{warnred}Magic keyword:} include \texttt{ultrathink} anywhere in your prompt $\to$ sets effort to \texttt{high} for that one turn.

\column{0.4\textwidth}
\textbf{\color{paramgreen}Controls}
\begin{itemize}
  \item \texttt{/effort high} --- session-wide
  \item \texttt{--effort high} --- at startup
  \item \texttt{Alt+T} --- toggle thinking on/off
  \item \texttt{Ctrl+O} --- show thinking output
\end{itemize}

\bigskip
\begin{alertblock}{Common mistake}
``think harder'' in your prompt does \textbf{nothing}. Only \texttt{ultrathink} and \texttt{/effort} actually change reasoning depth.
\end{alertblock}
\end{columns}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Controlling Context}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{\color{popblue}Interrupt \& Redirect}
\begin{itemize}
  \item \textbf{Escape} --- stop mid-response, redirect
  \item \textbf{Escape $\times$ 2} --- rewind to earlier point
  \item Combine with \texttt{\#} to fix repeating errors
\end{itemize}

\column{0.5\textwidth}
\textbf{\color{paramgreen}Commands}
\begin{itemize}
  \item \texttt{/compact} --- summarize history, keep knowledge
  \item \texttt{/clear} --- fresh start, no prior context
  \item Auto-compaction at $\sim$95\% context capacity
\end{itemize}
\end{columns}

\bigskip
\begin{alertblock}{When to use what}
\texttt{/compact} when switching to a \emph{related} task (preserves context). \\
\texttt{/clear} when switching to an \emph{unrelated} task (avoids confusion).
\end{alertblock}
\end{frame}

\end{itemize}

\end{itemize}

\bigskip
\textbf{\color{violet1}Session management}

\end{itemize}

\begin{frame}[fragile]{MCP in Practice: Context7}
\textbf{Problem:} Claude's training data lags behind library releases $\to$ hallucinated APIs.

\textbf{Solution:} \textbf{Context7} MCP fetches \emph{current} official documentation on demand.

\smallskip
\textbf{Setup (one command):}
\begin{lstlisting}[language=bash]
claude mcp add context7 -- npx -y @upstash/context7-mcp@latest
\end{lstlisting}

\textbf{Usage --- just say ``use context7'':}
\begin{lstlisting}[]
> How does Next.js after() work? use context7
\end{lstlisting}

\begin{columns}[T]
\column{0.5\textwidth}
\textbf{\color{popblue}How it works:}
\begin{enumerate}
  \item \texttt{resolve-library-id} $\to$ finds the library
  \item \texttt{query-docs} $\to$ fetches version-specific docs
  \item Claude answers with \emph{current} API
\end{enumerate}

\column{0.5\textwidth}
\textbf{\color{paramgreen}Why it matters:}
\begin{itemize}
  \item No hallucinated deprecated methods
  \item Working code on first try
  \item Version-specific examples
  \item Free, open-source (Upstash)
\end{itemize}
\end{columns}
\end{frame}

\begin{columns}[T]
\column{0.5\textwidth}
\textbf{\color{warnred}PreToolUse}
\begin{itemize}
  \item Runs \emph{before} tool call
  \item Can \textbf{block or rewrite} the operation
  \item Access control, validation
\end{itemize}

\column{0.5\textwidth}
\textbf{\color{paramgreen}PostToolUse}
\begin{itemize}
  \item Runs \emph{after} tool call
  \item Cannot block (already done)
  \item Formatting, testing, logging
\end{itemize}
\end{columns}

\bigskip
\small
\textbf{Hook types:} \texttt{"command"} (shell) $|$ \texttt{"http"} (POST JSON to URL)\\
\textbf{Config locations:} \texttt{\textasciitilde/.claude/settings.json} (global) $|$
\texttt{.claude/settings.json} (team) $|$
\texttt{.claude/settings.local.json} (personal)
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}[fragile]{Hook Configuration}
\textbf{PreToolUse example} --- guard specific tool calls:
\begin{lstlisting}[]
"PreToolUse": [{
  "matcher": "Read|Grep",
  "hooks": [{
    "type": "command",
    "command": "node ./hooks/read_hook.js"
  }]
}]
\end{lstlisting}
\smallskip
\textbf{PostToolUse example} --- run formatter after edits:
\begin{lstlisting}[]
"PostToolUse": [{
  "matcher": "Write|Edit|MultiEdit",
  "hooks": [{
    "type": "command",
    "command": "node ./hooks/format_hook.js"
  }]
}]
\end{lstlisting}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}[fragile]{Implementing a Hook: Protect \texttt{.env}}
\begin{lstlisting}[]
async function main() {
  const chunks = [];
  for await (const chunk of process.stdin) {
    chunks.push(chunk);
  }
  const toolArgs = JSON.parse(
    Buffer.concat(chunks).toString()
  );

  const readPath =
    toolArgs.tool_input?.file_path
    || toolArgs.tool_input?.path || "";

  if (readPath.includes('.env')) {
    console.error("You cannot read the .env file");
    process.exit(2);  // exit code 2 = block
  }
}

main();
\end{lstlisting}

\footnotesize
Hook receives tool call data via \texttt{stdin} as JSON. Exit code \texttt{2} blocks the operation with an error message back to Claude.
\end{frame}

\begin{frame}{Hook Gotchas \& Practical Recipes}
\begin{alertblock}{Common Gotchas}
\begin{itemize}\setlength\itemsep{0em}
  \item Use \textbf{absolute paths} in hook commands
  \item Restart Claude Code after changing hooks
  \item Test hooks with simple cases first
\end{itemize}
\end{alertblock}

\smallskip
\small
\textbf{Useful hook ideas:}
\begin{itemize}\setlength\itemsep{0em}
  \item \textbf{TypeScript type-checking} --- run \texttt{tsc} after edits, feed errors back
  \item \textbf{Auto-formatting} --- run Prettier/Black after writes
  \item \textbf{Query duplication prevention} --- detect duplicate SQL/API calls
  \item \textbf{Test runner} --- auto-run tests on changed files
  \item \textbf{Sensitive file protection} --- block access to credentials
\end{itemize}

\smallskip
\footnotesize
\textbf{Other hook events:} Notification, Stop, SubagentStop, PreCompact, UserPromptSubmit, SessionStart, SessionEnd.
\end{frame}

\begin{frame}[fragile]{The Claude Code SDK}
\textbf{Claude Agent SDK} --- run Claude Code programmatically (TypeScript, Python, CLI).

\begin{lstlisting}[]
import { query } from "@anthropic-ai/claude-code";

const prompt = "Look for duplicate queries in ./src/queries";

for await (const message of query({ prompt })) {
  console.log(JSON.stringify(message, null, 2));
}
\end{lstlisting}

\textbf{Permissions:} read-only by default. Enable writes with:
\begin{lstlisting}[]
for await (const message of query({
  prompt,
  options: { allowedTools: ["Edit"] }
})) { /* ... */ }
\end{lstlisting}

\smallskip
\textbf{Use cases:} Git hooks for code review, CI/CD quality checks, automated documentation, build optimization scripts.
\end{frame}

\begin{frame}{Workflow Tips}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{\color{popblue}1.\ Plan before you code}
\begin{itemize}
  \item Use \textbf{plan mode} (\texttt{Shift+Tab}) for unfamiliar code
  \item Ask Claude to research first: ``google how X works, then implement''
  \item For big features: have Claude interview you, write a spec, then start fresh
\end{itemize}

\bigskip
\textbf{\color{paramgreen}2.\ Keep memory fresh}
\begin{itemize}
  \item Periodically: ``update your memory with what you learned''
  \item \texttt{/memory} to browse and prune stale entries
  \item Keep \texttt{CLAUDE.md} under 200 lines --- specificity $>$ length
\end{itemize}

\column{0.5\textwidth}
\textbf{\color{orange1}3.\ Ask Claude to improve your workflow}
\begin{itemize}
  \item ``What workflow improvements would you suggest for this project?''
  \item ``Review my CLAUDE.md --- what's redundant or missing?''
  \item \texttt{/insights} for session analytics
\end{itemize}

\bigskip
\textbf{\color{violet1}4.\ Manage context aggressively}
\begin{itemize}
  \item \texttt{/clear} between unrelated tasks
  \item \texttt{/compact} when switching related tasks
  \item Use subagents for long explorations
  \item \texttt{/btw} for quick side questions
\end{itemize}
\end{columns}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Common Pitfalls}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{alertblock}{Kitchen sink sessions}
Mixing unrelated tasks fills context with noise. \\
\textbf{Fix:} \texttt{/clear} between unrelated tasks.
\end{alertblock}

\smallskip
\begin{alertblock}{Correcting the same bug 3$\times$}
Failed attempts accumulate, confusing Claude. \\
\textbf{Fix:} After 2 fails, \texttt{/clear} and write a better prompt.
\end{alertblock}

\smallskip
\begin{alertblock}{Vague prompts}
``fix the login bug'' gives Claude nothing to work with. \\
\textbf{Fix:} Scope + symptoms + files: ``login fails after timeout, check \texttt{src/auth/}''.
\end{alertblock}

\column{0.5\textwidth}
\begin{exampleblock}{Good habits}
\begin{itemize}
  \item Provide \textbf{verification}: tests, screenshots, expected output
  \item Use \texttt{@file\#5-10} to point at specific code
  \item Name sessions: \texttt{claude -n "feature-auth"}
  \item Start in plan mode for unfamiliar codebases
  \item Paste images of errors --- Claude reads screenshots
  \item Scope exploration: ``investigate token refresh in \texttt{src/auth/} only''
\end{itemize}
\end{exampleblock}
\end{columns}
\end{frame}
