\section{The Big Three LLMs --- Education Modes}

% ──────────────────────────────────────────────
\begin{frame}{ChatGPT Study Mode}

\textbf{\color{popblue}Launched:} July 2025 \quad|\quad
\textbf{\color{popblue}Access:} Free tier (select ``study and learn'' in tools)

\bigskip
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{How it works:}
\begin{itemize}
  \item Guides you step-by-step instead of giving answers
  \item \textbf{Refuses} direct answers until you engage with the material
  \item Asks questions to test your understanding
  \item Calibrates to your skill level
\end{itemize}

\bigskip
\textbf{Interactive Learning:}
\begin{itemize}
  \item 70+ math \& science topics
  \item Pythagorean theorem, ideal gas law, lens equations\ldots
  \item Visualizations for complex concepts
\end{itemize}

\column{0.42\textwidth}
\begin{block}{Example interaction}
\ttfamily\scriptsize
\textbf{You:} How do I find the derivative\\
of $f(x) = x^3 + 2x$?\\[4pt]
\textbf{ChatGPT:} Before I show you,\\
let me ask --- do you remember\\
the power rule? What happens\\
when you differentiate $x^n$?\\[4pt]
\textbf{You:} $nx^{n-1}$?\\[4pt]
\textbf{ChatGPT:} Exactly! Now try\\
applying that to each term\ldots
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Claude: Learning \& Explanatory Modes}

\textbf{\color{violet1}Launched:} August 2025 \quad|\quad
\textbf{\color{violet1}Stat:} 76\% of users reported improved understanding

\bigskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Learning Mode}
\textbf{Socratic pair-programming}
\begin{itemize}
  \item Pauses and inserts \texttt{\#TODO} for you to write code
  \item You complete 5--10 lines yourself
  \item Claude reviews your work and guides further
  \item Learn by doing, not by reading
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{Explanatory Mode}
\textbf{Senior dev mentoring}
\begin{itemize}
  \item Narrates every decision as it works
  \item Explains architecture choices, tradeoffs
  \item Shares best practices in real-time
  \item Like looking ``under the hood''
\end{itemize}
\end{block}
\end{columns}

\bigskip
\textbf{\color{violet1}Claude for Education} (since April 2025):
deployed at Northeastern (50K users), LSE, Champlain College.
\textit{``AI should make students think harder, not less.''}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{Claude: Example Interaction}

\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Learning Mode example}
\ttfamily\scriptsize
\textbf{You:} Help me write a function\\
that reverses a linked list.\\[4pt]
\textbf{Claude:} Let's build this together.\\
I'll set up the structure ---\\
you write the core logic.\\[4pt]
def reverse\_list(head):\\
\quad prev = None\\
\quad current = head\\
\quad \# TODO: Write a while loop that\\
\quad \# iterates through the list and\\
\quad \# reverses the pointers.\\
\quad \# (about 5 lines)\\
\quad return prev
\end{block}

\column{0.48\textwidth}
\begin{block}{Explanatory Mode example}
\ttfamily\scriptsize
\textbf{You:} Explain binary search.\\[4pt]
\textbf{Claude:} Binary search works on\\
\textbf{sorted} arrays. The key insight\\
is: by checking the middle element,\\
we eliminate \textbf{half} the remaining\\
candidates each step.\\[4pt]
Why sorted? Because if arr[mid]\\
is too large, everything to its\\
right is also too large. This\\
guarantee is what makes the\\
O(log n) complexity possible.\\[4pt]
Think of it like a phone book\ldots
\end{block}
\end{columns}

\bigskip
\begin{center}
\small\color{violet1}
Demo: Ask Claude to teach you a concept in Learning or Explanatory mode
\end{center}
\end{frame}

\section{LLMs and the Armenian Language}

% ──────────────────────────────────────────────
\begin{frame}{ArmBench-LLM: Which Models Speak Armenian?}
\small
\textbf{\color{popblue}ArmBench-LLM 1.0} --- first comprehensive benchmark for Armenian language tasks.\\
Created by \textbf{Metric AI Lab} (April 2026). Fully open-sourced.

\smallskip
\textbf{7 task categories:}
text classification, multiple choice QA, grammar correction, space fixing (OCR),
summarization, translation, reading comprehension.

\smallskip
\begin{center}
\footnotesize
\begin{tabular}{@{}r l c r@{}}
\toprule
\textbf{Rank} & \textbf{Model} & \textbf{Score} & \textbf{Cost} \\
\midrule
1 & \textbf{\color{paramgreen}Gemini 3 Flash} & 0.635 & \$3.28 \\
2 & GPT-5.2 Pro & 0.617 & \$160.20 \\
3 & Grok 4 Fast & 0.604 & \$2.71 \\
4 & Claude 3.7 Sonnet & 0.607 & \$16.49 \\
5 & \textbf{\color{violet1}Qwen 3.5 27B} {\scriptsize(open-source)} & 0.577 & free \\
\bottomrule
\end{tabular}
\end{center}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{exampleblock}{Key insight}
Global rankings $\neq$ Armenian proficiency!\\
Gemini Flash \textbf{beats} Gemini Pro on Armenian (opposite of global ranking).
\end{exampleblock}

\column{0.48\textwidth}
\begin{exampleblock}{Open-source win}
Qwen 3.5 27B beats 600B+ models (GLM-5, Mistral-Large). Runs on a \textbf{single GPU}.
\end{exampleblock}
\end{columns}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{ArmBench: Task-Specific Results}
\footnotesize
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{\color{popblue}Best models by task:}

\smallskip
\begin{tabular}{@{}l l@{}}
\toprule
\textbf{Task} & \textbf{Best model} \\
\midrule
Translation & Gemini 3 Flash \\
Summarization & Gemini 3 Flash \\
Armenian Literature & Gemini 3 Flash \\
Armenian History & Gemini 3 Pro \\
Math (university exams) & Grok 4 Fast \\
Named Entity Recog.\ & Grok 4 Fast \\
\bottomrule
\end{tabular}

\smallskip
\small\textbf{\color{warnred}Cost matters:}
\footnotesize
\begin{itemize}\setlength\itemsep{0em}
  \item GPT-5.2 Pro costs \textbf{50$\times$} more than Gemini Flash for \emph{lower} accuracy
  \item Armenian tokenizers vary wildly --- same text = different cost
  \item Reasoning models burn tokens on chain-of-thought
\end{itemize}

\column{0.48\textwidth}
\small\textbf{\color{paramgreen}Practical recommendations:}
\footnotesize

\begin{block}{Best overall}
\textbf{Gemini 3 Flash} --- highest score, lowest cost. Best for translation, summarization, literature.
\end{block}

\begin{block}{Best budget}
\textbf{Grok 4 Fast} --- nearly as good, cheapest (\$2.71). Best for math, NER.
\end{block}

\begin{block}{Best open-source}
\textbf{Qwen 3.5 27B} --- run locally, no API costs, beats much larger models.
\end{block}

\begin{frame}{NotebookLM: Your Personal Study Assistant}
\small
\textbf{\color{popblue}What:} Upload YOUR materials $\to$ AI assistant grounded in YOUR sources

\smallskip
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{How it's different from chatbots:}
\begin{itemize}\setlength\itemsep{0em}
  \item Regular LLMs answer from training data (generic internet)

  \item Source-grounded = fewer hallucinations
  \item Every answer references your materials
\end{itemize}

\smallskip
\textbf{What you can upload:}
\begin{itemize}\setlength\itemsep{0em}

  \item Lecture notes, textbook chapters
  \item Research papers, YouTube videos, audio

% ──────────────────────────────────────────────
\begin{frame}{The Wrong Way}

\begin{center}
\begin{tikzpicture}[
  box/.style={draw=warnred, fill=warnred!8, rounded corners=5pt,

  \node[box] (a) at (8,0) {Get\\answer};
  \node[box] (s) at (12,0) {Submit};

  \draw[arr] (q) -- (ai);
  \draw[arr] (ai) -- (a);
  \draw[arr] (a) -- (s);
\end{tikzpicture}
\end{center}

\bigskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{alertblock}{What happens}
\begin{itemize}
  \item Zero effort $\to$ zero learning
  \item False confidence on exams
  \item Fragile knowledge that crumbles under pressure
  \item Can't explain your own work
\end{itemize}
\end{alertblock}

\column{0.48\textwidth}
\begin{alertblock}{The exam problem}
\begin{itemize}
  \item You can't bring AI to the exam
  \item If you never struggled with the material, you never learned it
  \item \textbf{Struggle is where learning happens}
\end{itemize}
\end{alertblock}
\end{columns}
\end{frame}

% ──────────────────────────────────────────────
\begin{frame}{The Right Way: AI as Tutor, Not Calculator}

\begin{center}
\begin{tikzpicture}[
  box/.style={draw=paramgreen, fill=paramgreen!8, rounded corners=5pt,
              minimum width=2.2cm, minimum height=1.4cm, align=center, font=\small},
  arr/.style={-{Stealth[length=6pt]}, thick, paramgreen}
]

\begin{columns}[T]

\begin{exampleblock}{The key difference}
\begin{itemize}
  \item ``Help me understand X'' \checkmark
  \item ``Solve X for me'' $\times$
  \item You do the thinking --- AI guides
  \item Learning modes enforce this!

\begin{frame}{Prompting for Learning (Part 1)}
\small
\textbf{\color{popblue}Prompts that maximize learning:}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Guide, don't solve}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Don't give me the answer. Guide me step by step.''}
  \item \textit{``I think the answer is X. Am I right? Where did I go wrong?''}
  \item \textit{``What's wrong with my approach?''}
\end{itemize}
\end{block}

\smallskip
\begin{block}{Explain \& understand}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Explain this using an analogy.''}
  \item \textit{``What's the intuition behind [formula]?''}
  \item \textit{``Why does this work? Not how --- why.''}
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{Test yourself}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Quiz me on [topic] --- start easy, then harder.''}
  \item \textit{``Give me 5 practice problems similar to this.''}
  \item \textit{``I studied X. Check if I really understand.''}
\end{itemize}
\end{block}

\smallskip
\begin{block}{Go deeper}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``What concept am I missing?''}
  \item \textit{``What are common misconceptions here?''}
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Prompting for Learning (Part 2)}
\small
\textbf{\color{popblue}More advanced techniques:}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Level adjustment}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Explain like I'm a complete beginner.''}
  \item \textit{``Now explain at a graduate level.''}
  \item \textit{``Use only concepts from Calculus 1.''}
\end{itemize}
\end{block}

\smallskip
\begin{block}{Role-based learning}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Act as a [subject] professor --- give me a 5-min lecture on [topic].''}
  \item \textit{``Be my study partner. Let's challenge each other.''}
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{Connections \& synthesis}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``How does [A] relate to [B]?''}
  \item \textit{``Real-world example of [concept].''}
  \item \textit{``Compare and contrast [X] and [Y].''}
\end{itemize}
\end{block}

\smallskip
\begin{block}{Exam prep}
\begin{itemize}\setlength\itemsep{0em}
  \item \textit{``Study plan for [topic], 2h/day, 5 days.''}
  \item \textit{``Practice exam: 10 questions on ch.\ 5--8.''}
  \item \textit{``Most likely exam questions on [topic]?''}
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Custom Instructions: Set the Rules Once}
\small
\textbf{\color{popblue}All major LLMs let you set persistent instructions that apply to every conversation.}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{What are custom instructions?}
\begin{itemize}\setlength\itemsep{0em}
  \item Rules the AI follows in \textbf{every} conversation
  \item Set once, applies automatically
  \item Control tone, depth, format, behavior
\end{itemize}
\end{block}

\smallskip
\begin{exampleblock}{Where to find them}
\begin{itemize}\setlength\itemsep{0em}
  \item \textbf{ChatGPT:} Settings $\to$ Custom Instructions
  \item \textbf{Claude:} Settings $\to$ Profile
  \item \textbf{Gemini:} Settings $\to$ Gems
\end{itemize}
\end{exampleblock}

\column{0.48\textwidth}
\begin{block}{Example instructions}
\ttfamily\scriptsize
I'm a 2nd-year CS student.\\[2pt]
When I ask about code, don't give\\
me the answer directly --- guide me\\
step by step.\\[2pt]
Use simple analogies.\\[2pt]
Quiz me after explaining a concept.\\[2pt]
If I ask in Armenian, respond in\\
Armenian.
\end{block}

\smallskip
\begin{alertblock}{Pro tip}
Update your instructions as you progress. What works for a beginner won't work 6 months later.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Passive Language Learning in Every Chat}
\small
\textbf{\color{popblue}Learn vocabulary without extra effort --- just by chatting about anything.}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{The idea}
Ask the LLM to replace 1--2 common words in every response with translations in your target language. You absorb vocabulary \textbf{passively} while doing your normal work.
\end{block}

\smallskip
\begin{exampleblock}{Example prompt}
\textit{``In every response, replace 1--2 common words with their German translation (in parentheses).''}
\end{exampleblock}

\column{0.48\textwidth}
\begin{block}{What it looks like}
\ttfamily\scriptsize
\textbf{You:} Explain how a hash map works.\\[4pt]
\textbf{AI:} A hash map stores Schl\"ussel (keys) and their associated values.
When you insert data, a Funktion (function) converts the key into an index\ldots
\end{block}

\smallskip
\begin{alertblock}{Power move}
Put this in your \textbf{custom instructions} and it happens in every conversation automatically --- zero extra effort.
\end{alertblock}
\end{columns}

\medskip
\begin{center}
\small\color{paramgreen}
Works for \textbf{any} language pair. Great for German, French, Spanish, or even Armenian!
\end{center}
\end{frame}

\begin{frame}{Voice Mode: Practice Job Interviews with AI}
\small
\textbf{\color{popblue}Use voice mode to simulate real interviews --- out loud, in real time.}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{How it works}
\begin{enumerate}\setlength\itemsep{0em}
  \item Find a job posting on LinkedIn
  \item Paste it into ChatGPT / Claude / Gemini
  \item Say: \textit{``Interview me for this role. Ask me technical and behavioral questions one at a time.''}
  \item Switch to \textbf{voice mode} and answer out loud
  \item Ask for feedback after each answer
\end{enumerate}
\end{block}

\column{0.48\textwidth}
\begin{block}{Example prompt}
\ttfamily\scriptsize
Here's a job posting for a Junior\\
Data Analyst at [company]:\\[2pt]
[paste the LinkedIn post]\\[2pt]
Interview me for this position.\\
Ask one question at a time. After\\
I answer, give brief feedback and\\
then ask the next question. Start\\
with a behavioral question, then\\
move to technical.
\end{block}

\smallskip
\begin{exampleblock}{Why voice matters}
\begin{itemize}\setlength\itemsep{0em}
  \item Real interviews are \textbf{spoken}, not typed
  \item Practice articulating under pressure
  \item Builds confidence before the real thing
\end{itemize}
\end{exampleblock}
\end{columns}
\end{frame}

\begin{frame}{Creative LLM Learning Hacks}
\footnotesize
\textbf{\color{popblue}Unconventional ways to squeeze more learning from every interaction:}

\smallskip
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Passive language immersion}
\textit{``Replace 3--5 words in every response with their French equivalent (in parentheses).''} Absorb vocabulary \textbf{without dedicated study time}.
\end{block}

\begin{block}{Daily drip --- scheduled learning}
Set a daily reminder to prompt the LLM:
\textit{``Teach me one new concept about [topic] in 3 sentences.''} Micro-learning compounds over weeks.
\end{block}

\begin{block}{``Always teach me something new''}
\textit{``End every response with one surprising fact about what we discussed.''} Turns every chat into discovery.
\end{block}

\column{0.48\textwidth}
\begin{block}{Teach-back method}
Explain a concept \textbf{to the LLM}, then ask:
\textit{``Find the holes in my explanation.''} Teaching is the best way to learn.
\end{block}

\begin{block}{``Debug my thinking''}
\textit{``Here's my mental model of [X]. Where am I wrong?''} Surfaces blind spots you didn't know you had.
\end{block}

\begin{block}{Counterfactual exploration}
\textit{``What if Newton never discovered calculus?''} ``What if'' questions build deeper intuition by exploring \textbf{why} things are the way they are.
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Hallucinations: AI Makes Stuff Up}
\small
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{\color{warnred}What are hallucinations?}
\begin{itemize}\setlength\itemsep{0em}
  \item LLMs generate plausible-sounding but \textbf{wrong} info
  \item They don't ``know'' facts --- they predict likely text
  \item Can invent citations, formulas, historical events
  \item Sounds confident even when completely wrong
\end{itemize}

\smallskip
\textbf{\color{warnred}Educational risks:}
\begin{itemize}\setlength\itemsep{0em}
  \item Learning wrong formulas or definitions
  \item Citing non-existent papers
  \item Building understanding on incorrect foundations
\end{itemize}

\column{0.42\textwidth}
\begin{alertblock}{How to protect yourself}
\begin{enumerate}\setlength\itemsep{0em}
  \item \textbf{Always verify} with textbook/professor
  \item Use \textbf{NotebookLM} for source-grounded answers
  \item \textbf{Cross-reference} multiple tools
  \item Be skeptical of confident answers
\end{enumerate}
\end{alertblock}

\smallskip
\begin{exampleblock}{Rule of thumb}
If AI says something surprising, verify before relying on it.
\end{exampleblock}
\end{columns}
\end{frame}
