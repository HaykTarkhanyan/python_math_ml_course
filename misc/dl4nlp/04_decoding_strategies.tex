% ============================================================
% WHY LOG-PROBABILITIES
% ============================================================
\begin{frame}
\frametitle{Why log-probabilities?}

\vspace{0.1cm}
A sequence's probability is a \textbf{product} of per-token probabilities --- take the log to turn it into a sum:
\begin{center}
$\displaystyle P(w_1, \ldots, w_L) = \prod_{t=1}^{L} P(w_t \mid w_{<t}) \qquad\Longrightarrow\qquad \log P = \sum_{t=1}^{L} \log P(w_t \mid w_{<t})$
\end{center}

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}[
  rbox/.style={draw, rounded corners=6pt, text width=3.7cm, align=left, inner sep=8pt, font=\small, minimum height=2.6cm}
]
  \node[rbox, fill=sampred!6, draw=sampred] at (-4.3, 0) {
    \textbf{\textcolor{sampred}{1. No underflow}}\\[3pt]
    {\footnotesize $0.1^{200}\!\approx\! 10^{-200}$ rounds to $0$ in floating point. A \emph{sum} of logs stays representable.}
  };
  \node[rbox, fill=popblue!7, draw=popblue] at (0, 0) {
    \textbf{\textcolor{popblue}{2. Products $\to$ sums}}\\[3pt]
    {\footnotesize $\log\!\big(\prod P\big) = \sum \log P$. Adding is cheaper and more stable than multiplying.}
  };
  \node[rbox, fill=paramgreen!8, draw=paramgreen] at (4.3, 0) {
    \textbf{\textcolor{paramgreen}{3. Same ranking}}\\[3pt]
    {\footnotesize $\log$ is monotonic, so the $\arg\max$ and the ordering are unchanged.}
  };
\end{tikzpicture}
\end{center}

\vspace{0.15cm}
\begin{center}
\fcolorbox{gray}{gray!6}{\parbox{11.5cm}{\centering\small Since $P \le 1$, every $\log P \le 0$ --- so scores are \textbf{negative}, and ``higher'' means closer to $0$.}}
\end{center}
\end{frame}

\end{center}
\end{frame}

% BEAM SEARCH: LENGTH NORMALIZATION
% ============================================================
\begin{frame}
\frametitle{Beam search: why normalize by length}

\begin{center}
\small Each step adds a $\log P \le 0$, so the running score only \textbf{goes down}. A longer sequence scores lower --- even when it reads better.
\end{center}

\vspace{0.1cm}
\begin{center}

\begin{center}
{\footnotesize A sequence's log-prob $= \sum_t \log P(w_t \mid \cdot)$; each term is negative (probabilities $< 1$), so the total is negative --- near $0$ = high-probability text, more negative = rarer. The $-12$ to $-2$ scale is schematic.}

    {\footnotesize --- Holtzman et al., ``The Curious Case of Neural Text Degeneration'' (2019)}
  }%
}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Temperature: worked example}

\vspace{0.1cm}
\begin{center}
\small Three tokens (\textit{the}, \textit{a}, \textit{his}) with logits $z = (2,\ 1,\ 0)$. Plug into $p_i = \dfrac{e^{z_i/T}}{\sum_j e^{z_j/T}}$:
\end{center}

\vspace{0.15cm}
\textcolor{popblue}{\textbf{$T = 1$ (unchanged):}}
\vspace{-0.2cm}
$$p = \frac{\big(e^{2},\, e^{1},\, e^{0}\big)}{e^{2}+e^{1}+e^{0}} = \frac{(7.39,\ 2.72,\ 1.00)}{11.11} = (\textbf{.66},\ .24,\ .09)$$

\textcolor{orange1}{\textbf{$T = 10$ (hot} --- shrink the gaps\textbf{):}}
\vspace{-0.2cm}
$$p = \frac{\big(e^{0.2},\, e^{0.1},\, e^{0}\big)}{e^{0.2}+e^{0.1}+e^{0}} = \frac{(1.22,\ 1.11,\ 1.00)}{3.33} = (\textbf{.37},\ .33,\ .30) \approx \text{uniform}$$

\textcolor{sampred}{\textbf{$T \to 0$ (greedy):}}
\vspace{-0.2cm}
$$\frac{z_i}{T} \to \infty\text{-sized gaps} \;\Rightarrow\; p = (\textbf{1},\ 0,\ 0) \quad(\text{all mass on } \arg\max)$$

\vspace{0.1cm}
\begin{center}
\fcolorbox{paramgreen}{paramgreen!5}{%
  \parbox{12cm}{\centering\small
  Lower $T$ sharpens toward the top token; higher $T$ flattens toward uniform. $T$ is the same softmax --- only the logit gaps are rescaled.}%
}
\end{center}
\end{frame}

% CONTRASTIVE SEARCH
% ============================================================
\sectiontransition{Contrastive search \& combining knobs}{Kill repetition without killing coherence.}
\begin{frame}
\frametitle{Contrastive search: the idea}

\textbf{The dilemma so far:}
\vspace{0.1cm}
\begin{itemize}\setlength{\itemsep}{4pt}
  \item \textcolor{popblue}{\textbf{Greedy / beam}} --- maximize probability $\Rightarrow$ coherent but \textbf{repetitive} (``the hill $\to$ the road $\to$ the hill \ldots'').
  \item \textcolor{orange1}{\textbf{Sampling}} --- inject randomness $\Rightarrow$ diverse but sometimes \textbf{incoherent}.
\end{itemize}

\vspace{0.2cm}
\begin{center}
\fcolorbox{paramgreen}{paramgreen!6}{%
  \parbox{12cm}{\centering\small
    \textbf{Contrastive search:} get \emph{both}, \emph{deterministically}. At each step prefer a token that is
    \textbf{likely} under the model \emph{but} \textbf{not too similar} to what was already generated.}%
}
\end{center}

\vspace{0.3cm}
\begin{center}
\begin{tikzpicture}[scale=0.95, transform shape]
  \node[font=\scriptsize] at (1.9, 1.5) {hidden-state space};
  % context cluster
  \foreach \p in {(0,0),(0.45,0.3),(-0.3,0.28),(0.25,-0.3),(-0.22,-0.12)} \fill[gray!55] \p circle (2.6pt);
  \draw[gray!40, dashed] (0.02,0.02) circle (0.75);
  \node[font=\scriptsize, gray] at (0, -1.15) {past tokens};
  % repeat candidate near
  \fill[sampred] (1.05, 0.55) circle (3.2pt);
  \node[font=\scriptsize, sampred, anchor=west] at (1.3, 0.55) {``hill'' --- near $\Rightarrow$ penalized};
  % fresh candidate far
  \fill[paramgreen] (3.9, -0.5) circle (3.2pt);
  \node[font=\scriptsize, paramgreen, anchor=west] at (4.15, -0.5) {``ocean'' --- far $\Rightarrow$ preferred};
\end{tikzpicture}
\end{center}
\end{frame}

% ============================================================
% CONTRASTIVE SEARCH: THE SCORE
% ============================================================
\begin{frame}
\frametitle{Contrastive search: the score}

\vspace{0.1cm}
\begin{center}
$\displaystyle x_t = \arg\max_{v \in V^{(k)}} \Big\{\, \underbrace{(1-\alpha)\,\textcolor{popblue}{p_\theta(v \mid x_{<t})}}_{\text{model confidence}} \;-\; \underbrace{\alpha\,\textcolor{sampred}{\max_{j<t}\, s(h_v, h_{x_j})}}_{\text{degeneration penalty}} \,\Big\}$
\end{center}

\vspace{0.25cm}
\begin{center}
\begin{tikzpicture}[
  box/.style={draw, rounded corners=6pt, minimum height=2.7cm, text width=5.3cm, align=left, inner sep=9pt, font=\small}
]
  \node[box, fill=popblue!8, draw=popblue] at (-3.3, 0) {
    \textbf{\textcolor{popblue}{Model confidence}}\\[3pt]
    $p_\theta(v \mid x_{<t})$ over the \textbf{top-$k$} candidates $V^{(k)}$.\\[3pt]
    Keeps the token likely, on-topic, grammatical.
  };
  \node[box, fill=sampred!7, draw=sampred] at (3.3, 0) {
    \textbf{\textcolor{sampred}{Degeneration penalty}}\\[3pt]
    $\max_j s(h_v, h_{x_j})$: largest \textbf{cosine similarity} of $v$'s hidden state to any past token's.\\[3pt]
    High = echoes the context $\Rightarrow$ punished.
  };
  \node[font=\Large\bfseries, paramgreen] at (0, 0) {$-$};
\end{tikzpicture}
\end{center}

\vspace{0.15cm}
\begin{center}
{\small $\alpha$ balances the two terms: $\alpha = 0$ recovers \textbf{greedy}; typical $\alpha \in \{0.4, 0.5, 0.6\}$, $k \in \{5, \ldots, 15\}$.}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Speculative decoding: how verification works}

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}[scale=0.92, transform shape,
  tok/.style={draw, rounded corners=3pt, minimum width=1.05cm, minimum height=0.6cm, font=\small, align=center}]
  \foreach \i/\w/\f in {0/the/paramgreen, 1/cat/paramgreen, 2/sat/paramgreen, 3/on/paramgreen, 4/moon/sampred} {
    \node[tok, fill=\f!14, draw=\f] (t\i) at (\i*1.4, 0) {\w};
  }

    node[midway, above=6pt, font=\small, popblue]{target scores all $k$ draft tokens in \textbf{one} forward pass};
  \foreach \i in {0,1,2,3} \draw[paramgreen, line width=1.3pt] (\i*1.4-0.13,-0.72) -- (\i*1.4-0.02,-0.86) -- (\i*1.4+0.17,-0.56);
  \draw[sampred, line width=1.3pt] (5.6-0.14,-0.56)--(5.6+0.14,-0.86);
  \draw[sampred, line width=1.3pt] (5.6-0.14,-0.86)--(5.6+0.14,-0.56);
  \node[font=\scriptsize, sampred, anchor=west] at (6.05,-0.72) {first mismatch: stop, resample here};
\end{tikzpicture}
\end{center}

\vspace{0.25cm}
{\small
\textbf{\textcolor{popblue}{One parallel pass.}} Causal attention gives the target's next-token distribution at \emph{every} position at once --- the expensive step is paid once, not $k$ times.\\[6pt]
\textbf{\textcolor{sampred}{Accept / reject, left to right.}} Greedy target: accept if the draft token $=$ target's $\arg\max$. Sampling target: accept $x$ with probability $\min\!\big(1,\, p_{\text{target}}(x)/p_{\text{draft}}(x)\big)$. Stop at the first reject, keep the matching prefix, and let the target fill the corrected token (plus a free bonus token if all $k$ pass).
}

\vspace{0.15cm}
\begin{center}
\fcolorbox{paramgreen}{paramgreen!6}{\parbox{12.4cm}{\centering\small
  That accept/reject rule is exactly \textbf{rejection sampling} for the target distribution $\Rightarrow$ the output is distributed \emph{identically} to sampling the target alone. That is why it is \textbf{lossless}.}}
\end{center}
\end{frame}

% COMMON QUESTIONS
% ============================================================
\begin{frame}
\frametitle{Common questions}

{\small
\textcolor{popblue}{\textbf{Q. Is $T = 0$ exactly greedy?}}\\
Yes --- the softmax collapses onto the $\arg\max$. Libraries special-case $T{=}0$ so they don't divide by zero.\\[6pt]

\textcolor{popblue}{\textbf{Q. If I set both top-$k$ and top-$p$, which wins?}}\\
Both apply: you sample from their \emph{intersection}, after temperature has reshaped the distribution.\\[6pt]

\textcolor{popblue}{\textbf{Q. Is temperature applied before or after top-$k$/top-$p$?}}\\
Before. Temperature rescales the logits; truncation then cuts the tail of the reshaped distribution.\\[6pt]

\textcolor{popblue}{\textbf{Q. Is contrastive search a kind of sampling?}}\\
No --- it is deterministic. It takes the $\arg\max$ of a confidence-minus-penalty score, not a random draw.\\[6pt]

\textcolor{popblue}{\textbf{Q. Does speculative decoding change the output?}}\\
No. It is lossless: identical to sampling the target model alone, only faster.
}
\end{frame}
