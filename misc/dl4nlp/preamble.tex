% Section-transition slide: a [plain] divider with a popblue title + one
% motivation line. Also registers the \section for the Outline TOC, so call
% this once per section (instead of a bare \section) where dividers are wanted.
%   \sectiontransition{Section title}{one-line motivation}
\newcommand{\sectiontransition}[2]{%
  \section{#1}%
  \begin{frame}[plain]
  \centering
  \vfill
  {\color{popblue}\huge\bfseries #1\par}
  \vspace{0.55cm}
  {\large\color{black!65} #2\par}
  \vfill
  \end{frame}%
}
