nums = [1, 100, 3, -123, 4, 509, 7]

num_len = len(nums)

num_operators = 0

for iter_num in range(num_len):
    swap = False
    for i in range(num_len - 1 - iter_num):
        # print(f"{nums[i]}, {nums[i+1]}")
        num_operators += 1

        if nums[i] > nums[i+1]:
            swap = True
            # print("swapping")
            nums[i], nums[i+1] = nums[i+1], nums[i]
    
    if not swap:
        print("Done")
        break
    print(nums)
