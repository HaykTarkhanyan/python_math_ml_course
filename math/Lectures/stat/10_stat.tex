\begin{frame}
\frametitle{Nested Models: Simple vs Complex}

\begin{center}
\begin{tikzpicture}[
  mbox/.style={draw, rounded corners=8pt, text width=4.8cm, align=center, inner sep=10pt, font=\small, minimum height=2.5cm}
]
  \node[mbox, fill=black!5, draw=black!40] (m0) at (-4, 0) {
    {\large\textbf{$H_0$: Restricted model}}\\[6pt]
    Parameter space $\Theta_0$\\[3pt]
    ``Nothing special happening''\\[3pt]
    Fewer free parameters
  };
  \node[mbox, fill=paramgreen!8, draw=paramgreen!40] (m1) at (4, 0) {
    {\large\textbf{$H_1$: Full model}}\\[6pt]
    Parameter space $\Theta \supset \Theta_0$\\[3pt]
    ``Something is going on''\\[3pt]
    More free parameters
  };
  \draw[->, very thick, popblue] (m0) -- (m1) node[midway, above, font=\small\bfseries] {$\Theta_0 \subset \Theta$};
\end{tikzpicture}
\end{center}

\vspace{0.2cm}
\small
\textbf{Every test asks:} does the full model fit \textit{significantly} better than the restricted model?

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}[
  ebox/.style={draw=popblue!40, fill=popblue!3, rounded corners=4pt, text width=12cm, align=left, inner sep=6pt, font=\small}

  \node[ebox] at (0, 0.0) {$H_0\!: \mu_1 = \mu_2$ \;vs\; $H_1\!: \mu_1 \neq \mu_2$ \quad (1 mean vs 2 $\Rightarrow$ $k = 1$)};
  \node[ebox] at (0, -0.8) {$H_0\!:$ rows \& columns independent \;vs\; $H_1\!:$ associated \quad ($k = (r{-}1)(c{-}1)$)};
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}
\frametitle{The Likelihood Ratio Test (LRT)}

\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=12pt] {
    {\Large $\Lambda = \dfrac{\sup_{\theta \in \Theta_0} L(\theta)}{\sup_{\theta \in \Theta} L(\theta)} = \dfrac{L(\hat\theta_0)}{L(\hat\theta)}$}\\[10pt]
    \small
    \textbf{Numerator:} best fit \textit{with} constraint ($H_0$). \quad
    \textbf{Denominator:} best fit \textit{without} constraint.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\begin{center}
\begin{tikzpicture}[
  ebox/.style={draw, rounded corners=5pt, text width=5.5cm, align=left, inner sep=7pt, font=\small}
]
  \node[ebox, fill=popblue!8] at (-4.2, 0) {
    \textbf{$\Lambda$ close to 1}\\[3pt]
    $H_0$ fits almost as well as $H_1$.\\
    No evidence against $H_0$.\\
    $\Rightarrow$ fail to reject.
  };
  \node[ebox, fill=warnred!8] at (4.2, 0) {
    \textbf{$\Lambda$ close to 0}\\[3pt]
    $H_0$ fits much worse than $H_1$.\\
    Strong evidence against $H_0$.\\
    $\Rightarrow$ reject $H_0$.
  };
\end{tikzpicture}
\end{center}

\vspace{0.15cm}
\begin{center}
\small In practice, work with $-2\log\Lambda$ (large = evidence against $H_0$).
\end{center}
\end{frame}

\begin{frame}
\frametitle{Wilks' Theorem: $-2\log\Lambda \to \chi^2$}

\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=11cm, align=center, inner sep=12pt] {
    {\Large $-2\log\Lambda \;\xrightarrow{\;d\;}\; \chi^2_k \quad \text{as } n \to \infty$}\\[10pt]
    \small where $k = \dim(\Theta) - \dim(\Theta_0)$ = number of constraints imposed by $H_0$.
  };
\end{tikzpicture}
\end{center}

\vspace{0.2cm}
\begin{center}
\begin{tikzpicture}[
  kbox/.style={draw=popblue!50, fill=popblue!5, rounded corners=5pt, text width=3.5cm, align=center, inner sep=6pt, font=\small, minimum height=1.4cm}
]
  \node[kbox] at (-5, 0) {Test one mean\\$k = 1$\\$-2\log\Lambda \sim \chi^2_1$};
  \node[kbox] at (0, 0) {Two means equal\\$k = 1$\\$-2\log\Lambda \sim \chi^2_1$};
  \node[kbox] at (5, 0) {$r \times c$ independence\\$k = (r{-}1)(c{-}1)$\\$-2\log\Lambda \sim \chi^2_{(r-1)(c-1)}$};
\end{tikzpicture}
\end{center}

\vspace{0.15cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{10cm}{\centering\small
  \textbf{One theorem produces all the classical tests.}\\
  The $z$-test, $t$-test, and $\chi^2$-test are all special cases of the LRT.\\[2pt]
  \textbf{Caveat:} Wilks' theorem is \textbf{asymptotic} ($n \to \infty$). Poor approximation for small $n$ or boundary hypotheses.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{LRT Worked Example: Testing a Normal Mean}

\small
\textbf{Setup:} $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$ (known $\sigma^2$). \; $H_0\!: \mu = \mu_0$ \;vs\; $H_1\!: \mu \neq \mu_0$.

\pause
\vspace{0.1cm}
\textbf{Numerator} (best fit under $H_0$): \; $L(\mu_0) = \prod \frac{1}{\sigma\sqrt{2\pi}} e^{-(X_i - \mu_0)^2 / 2\sigma^2}$

\textbf{Denominator} (best fit overall): \; $L(\hat\mu) = L(\bar{X})$ \quad (MLE is $\bar{X}$)

\pause
\vspace{0.1cm}
\textbf{Log-likelihood ratio:}
$$-2\log\Lambda = -2\bigl[\ell(\mu_0) - \ell(\bar{X})\bigr] = \frac{n(\bar{X} - \mu_0)^2}{\sigma^2} = \left(\frac{\bar{X} - \mu_0}{\sigma/\sqrt{n}}\right)^{\!2} = Z^2$$

\pause
\vspace{-0.1cm}
\begin{center}
\fcolorbox{violet1}{violet1!5}{\parbox{11cm}{\centering\small
  The LRT statistic is just the \textbf{$z$-statistic squared}: $\;-2\log\Lambda = Z^2 \sim \chi^2_1$.\\[2pt]
  By Wilks' theorem, $k = 1$ (one constraint: $\mu = \mu_0$). Everything checks out!\\[2pt]
  Replace $\sigma$ with $S$ $\Rightarrow$ you get the \textbf{$t$-test}. Same idea, different distribution.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{One-Sample $t$-Test}

\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=10cm, align=center, inner sep=10pt] {
    $H_0\!: \mu = \mu_0$ \quad vs \quad $H_1\!: \mu \neq \mu_0$\\[8pt]
    {\Large $T = \dfrac{\bar{X} - \mu_0}{S / \sqrt{n}} \;\sim\; t_{n-1}$} \quad under $H_0$
  };
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
\small
\textbf{Example:} Coffee shop claims cups contain $\mu_0 = 350$\,mL. We measure $n = 25$: $\bar{X} = 345$, $S = 10$.

$$T = \frac{345 - 350}{10/\sqrt{25}} = \frac{-5}{2} = -2.50$$

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    width=10cm, height=2.8cm,
    domain=-4:4, samples=100,
    axis lines=middle, ticks=none,
    ymin=0, ymax=0.42,
    every axis plot/.style={thick, no markers},
    clip=false
  ]
    \addplot[popblue, thick] {(1 + x^2/24)^(-12.5) * 0.3989};
    \addplot[fill=warnred!25, draw=none, domain=-4:-2.064] {(1 + x^2/24)^(-12.5) * 0.3989} \closedcycle;
    \addplot[fill=warnred!25, draw=none, domain=2.064:4] {(1 + x^2/24)^(-12.5) * 0.3989} \closedcycle;
    \draw[thick, dashed, sampred] (axis cs:-2.50, 0) -- (axis cs:-2.50, 0.35);
    \node[font=\footnotesize\bfseries, sampred, above] at (axis cs:-2.50, 0.35) {$T = -2.50$};
    \node[font=\tiny, warnred] at (axis cs:-3, 0.06) {Reject};
    \node[font=\tiny, warnred] at (axis cs:3, 0.06) {Reject};
    \node[font=\tiny] at (axis cs:-2.064, -0.04) {$-2.064$};
    \node[font=\tiny] at (axis cs:2.064, -0.04) {$2.064$};
  \end{axis}
\end{tikzpicture}
\end{center}

\vspace{0.05cm}
$t_{24, 0.025} = 2.064$. Since $|T| = 2.50 > 2.064$: \textbf{reject} $H_0$. The cups are underfilled.

\vspace{0.05cm}
{\footnotesize\textcolor{gray}{\textbf{Assumptions:} (1) Data approximately Normal (or $n \geq 30$), (2) observations independent. If non-Normal with small $n$: use Wilcoxon signed-rank (Part~IV).}}
\end{frame}

\begin{frame}
\frametitle{Paired $t$-Test: Before vs After}

\small
If you have paired observations $(X_i, Y_i)$, compute differences $D_i = X_i - Y_i$ and apply a one-sample $t$-test on $D_i$.

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=8cm, align=center, inner sep=10pt] {
    $H_0\!: \mu_D = 0$ \quad vs \quad $H_1\!: \mu_D \neq 0$\\[8pt]
    {\Large $T = \dfrac{\bar{D}}{S_D / \sqrt{n}} \;\sim\; t_{n-1}$}
  };
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
\textbf{Example:} Blood pressure before/after drug, $n = 8$ patients.

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[font=\footnotesize\ttfamily, inner sep=0pt] {
    \begin{tabular}{l|cccccccc}
      Before & 148 & 142 & 136 & 134 & 138 & 140 & 132 & 144 \\
      After  & 140 & 138 & 132 & 130 & 135 & 137 & 130 & 140 \\
      \hline
      \rowcolor{popblue!10}
      $D_i$  & 8 & 4 & 4 & 4 & 3 & 3 & 2 & 4 \\
    \end{tabular}
  };
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
$\bar{D} = 4.0$, \;$S_D = 1.77$, \;$T = \dfrac{4.0}{1.77/\sqrt{8}} = \dfrac{4.0}{0.626} = 6.39$.

\vspace{0.1cm}
$t_{7, 0.025} = 2.365$. Since $|T| = 6.39 \gg 2.365$: \textbf{reject} $H_0$. The drug works. \; Cohen's $d = \bar{D}/S_D = 4.0/1.77 = 2.26$ (very large).
\end{frame}

\begin{frame}
\frametitle{Two-Sample $t$-Test: Comparing Two Groups}
\framesubtitle{Two independent groups: $n_1$ observations with $(\bar{X}_1, S_1)$ and $n_2$ with $(\bar{X}_2, S_2)$}

\vspace{-0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=8pt, text width=12cm, align=center, inner sep=7pt] {
    $H_0\!: \mu_1 = \mu_2$ \quad vs \quad $H_1\!: \mu_1 \neq \mu_2$\\[5pt]
    \textbf{Equal variances assumed:}\quad
    $T = \dfrac{\bar{X}_1 - \bar{X}_2}{S_p\sqrt{1/n_1 + 1/n_2}} \;\sim\; t_{n_1+n_2-2}$\\[5pt]
    \small Pooled SD: $\;S_p = \sqrt{\dfrac{(n_1{-}1)S_1^2 + (n_2{-}1)S_2^2}{n_1 + n_2 - 2}}$
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=8pt, text width=12cm, align=center, inner sep=7pt] {
    \textbf{Welch's $t$-test (no equal-variance assumption --- default choice):}\\[5pt]
    $T = \dfrac{\bar{X}_1 - \bar{X}_2}{\sqrt{S_1^2/n_1 + S_2^2/n_2}}$ \qquad
    $\text{df} = \dfrac{(S_1^2/n_1 + S_2^2/n_2)^2}{\frac{(S_1^2/n_1)^2}{n_1-1} + \frac{(S_2^2/n_2)^2}{n_2-1}}$
  };
\end{tikzpicture}
\end{center}

\vspace{0.05cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{10cm}{\centering\footnotesize
  \textbf{Always default to Welch's.} Correct whether variances are equal or not.
  The pooled version only gains a tiny bit of power when $\sigma_1 = \sigma_2$.\\[2pt]
  \textbf{R and Python use Welch by default.} Don't ``test for equal variances first, then decide'' --- this two-step procedure distorts your Type~I error.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Two-Sample Example: Teaching Methods}

\small
Method A ($n_1 = 10$): $\bar{X}_1 = 82.3$, $S_1 = 8.5$. \quad
Method B ($n_2 = 12$): $\bar{X}_2 = 74.1$, $S_2 = 10.2$.

\vspace{0.15cm}
\textbf{Welch's $t$-test:}
$$T = \frac{82.3 - 74.1}{\sqrt{8.5^2/10 + 10.2^2/12}} = \frac{8.2}{\sqrt{7.225 + 8.67}} = \frac{8.2}{3.99} = 2.06$$

$\text{df} \approx 19.8$, \;$t_{19.8,\, 0.025} \approx 2.09$.

\vspace{0.1cm}
$|T| = 2.06 < 2.09$: \textbf{fail to reject} at $\alpha = 0.05$ (just barely!).

\vspace{0.15cm}

\begin{frame}
\frametitle{Test for a Proportion}

\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=10cm, align=center, inner sep=10pt] {
    $H_0\!: p = p_0$ \quad vs \quad $H_1\!: p \neq p_0$\\[8pt]
    {\Large $Z = \dfrac{\hat{p} - p_0}{\sqrt{p_0(1-p_0)/n}} \;\sim\; N(0,1)$}\\[6pt]
    \small (Use $p_0$ in SE, not $\hat{p}$!)
  };
\end{tikzpicture}
\end{center}

\vspace{0.15cm}
\small
\textbf{Example:} Is a coin fair? $n = 200$ flips, $k = 115$ heads, $\hat{p} = 0.575$.

$$Z = \frac{0.575 - 0.5}{\sqrt{0.5 \times 0.5/200}} = \frac{0.075}{0.0354} = 2.12$$

$p\text{-value} = 2(1 - \Phi(2.12)) = 2 \times 0.017 = 0.034 < 0.05$: \textbf{reject} $H_0$. Evidence the coin is biased.

\vspace{0.15cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{10cm}{\centering\small
  \textbf{Rule of thumb:} use this test when $np_0 \geq 10$ and $n(1 - p_0) \geq 10$.\\
  For two proportions: $Z = \frac{\hat{p}_1 - \hat{p}_2}{\sqrt{\hat{p}(1-\hat{p})(1/n_1 + 1/n_2)}}$
  where $\hat{p} = \frac{k_1 + k_2}{n_1 + n_2}$ (pooled under $H_0\!: p_1 = p_2$).
}}
\end{center}
\end{frame}

\section{Chi-Squared Tests}

\begin{frame}
\begin{center}
  \vspace{1.5cm}
  {\Huge\bfseries\textcolor{popblue}{Part III: Chi-Squared Tests}}

  \vspace{0.8cm}
  {\large When the data is counts, not measurements}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Goodness-of-Fit: Does the Data Match a Distribution?}

\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=11cm, align=center, inner sep=10pt] {
    Observed counts $O_1, \ldots, O_k$ across $k$ categories, total $n$.\\
    Expected under $H_0$: $E_i = n \cdot p_i^0$.\\[8pt]
    {\Large $X^2 = \displaystyle\sum_{i=1}^{k} \frac{(O_i - E_i)^2}{E_i} \;\dot{\sim}\; \chi^2_{k-1}$}
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\small
\textbf{Example: Is a die fair?} Roll $n = 120$ times. $H_0\!: p_i = 1/6$ $\Rightarrow$ $E_i = 20$.

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    width=9cm, height=3cm,
    ybar,
    bar width=0.5cm,
    xlabel={Face}, ylabel={Count},
    symbolic x coords={1,2,3,4,5,6},
    xtick=data,
    ymin=0, ymax=30,
    nodes near coords,
    nodes near coords style={font=\tiny},
    every axis plot/.style={fill=popblue!50, draw=popblue!70},
    clip=false
  ]
    \addplot coordinates {(1,25) (2,17) (3,15) (4,23) (5,24) (6,16)};
    \draw[thick, dashed, sampred] (axis cs:1, 20) -- (axis cs:6, 20);
    \node[font=\scriptsize, sampred, right] at (axis cs:6, 20) {$E = 20$};
  \end{axis}
\end{tikzpicture}
\end{center}

$X^2 = \frac{25}{20} + \frac{9}{20} + \frac{25}{20} + \frac{9}{20} + \frac{16}{20} + \frac{16}{20} = 5.0$. \quad
df $= 5$. \quad $\chi^2_{5,\,0.05} = 11.07$.\\
$5.0 < 11.07$: \textbf{fail to reject}. No evidence the die is unfair.

\vspace{0.05cm}
{\footnotesize\textcolor{gray}{\textbf{Assumption:} All $E_i \geq 5$. If some expected counts are $< 5$, merge categories or use an exact test.}}
\end{frame}

\begin{frame}
\frametitle{Test of Independence: Two Categorical Variables}

\small
\textbf{Setup:} Contingency table with $r$ rows and $c$ columns. $H_0$: variables are independent.

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=10cm, align=center, inner sep=10pt] {
    Under $H_0$:\; $E_{ij} = \dfrac{(\text{row } i \text{ total}) \times (\text{col } j \text{ total})}{n}$\\[10pt]
    {\Large $X^2 = \displaystyle\sum_{i,j} \frac{(O_{ij} - E_{ij})^2}{E_{ij}} \;\dot{\sim}\; \chi^2_{(r-1)(c-1)}$}
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.15cm}
\textbf{Example: Smoking and lung cancer} ($n = 200$).

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[font=\small, inner sep=0pt] {
    \begin{tabular}{l|cc|c}
      & Cancer & No cancer & Total \\
      \hline
      Smoker & \textbf{40} \;\footnotesize{($E{=}25$)} & \textbf{60} \;\footnotesize{($E{=}75$)} & 100 \\
      Non-smoker & \textbf{10} \;\footnotesize{($E{=}25$)} & \textbf{90} \;\footnotesize{($E{=}75$)} & 100 \\
      \hline
      Total & 50 & 150 & 200 \\
    \end{tabular}
  };
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
$X^2 = \frac{(40{-}25)^2}{25} + \frac{(60{-}75)^2}{75} + \frac{(10{-}25)^2}{25} + \frac{(90{-}75)^2}{75} = 9 + 3 + 9 + 3 = \mathbf{24.0}$

\vspace{0.1cm}
df $= (2{-}1)(2{-}1) = 1$. \;$\chi^2_{1,\,0.05} = 3.84$. \;$24.0 \gg 3.84$: \textbf{reject}. Strong association.

\vspace{0.05cm}
{\footnotesize\textcolor{gray}{Effect size: Cram\'er's $V = \sqrt{X^2/(n \cdot \min(r{-}1,c{-}1))} = \sqrt{24/200} = 0.35$ (medium). \;\textbf{Small expected counts ($E_{ij} < 5$)?} Use Fisher's exact test (\texttt{scipy.stats.fisher\_exact}).\\
\textbf{Common mistake:} $\chi^2$ tests association, not causation. Smoking is associated with cancer --- this table alone doesn't prove causation.}}
\end{frame}

\begin{frame}
\frametitle{Mann--Whitney $U$ Test}
\framesubtitle{Comparing two independent groups --- nonparametric alternative to two-sample $t$}

\small
\textbf{Idea:} Replace values with their \textit{ranks}, then compare rank sums.

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}
  % Group A values
  \node[font=\small\bfseries, popblue] at (-1.5, 1.8) {A:};
  \foreach \x/\v in {0/3, 1/7, 2/5, 3/9, 4/2} {
    \node[circle, fill=popblue!60, inner sep=2pt, font=\scriptsize, text=white, minimum size=15pt] at (\x, 1.8) {\v};
  }
  % Group B values
  \node[font=\small\bfseries, sampred] at (-1.5, 0.8) {B:};
  \foreach \x/\v in {0/8, 1/6, 2/1, 3/4, 4/10} {
    \node[circle, fill=sampred!60, inner sep=2pt, font=\scriptsize, text=white, minimum size=15pt] at (\x, 0.8) {\v};
  }

  % Combined ranked
  \draw[->, thick, orange1] (4.6, 1.3) -- (5.2, 1.3) node[midway, above, font=\scriptsize] {Rank};

  \node[font=\scriptsize\bfseries] at (5.2, 2.2) {Sorted:};
  \foreach \x/\v/\r/\col in {5.8/1/1/sampred, 6.5/2/2/popblue, 7.2/3/3/popblue, 7.9/4/4/sampred, 8.6/5/5/popblue, 9.3/6/6/sampred, 10/7/7/popblue, 10.7/8/8/sampred, 11.4/9/9/popblue, 12.1/10/10/sampred} {
    \node[circle, fill=\col!60, inner sep=1pt, font=\tiny, text=white, minimum size=12pt] at (\x, 1.8) {\v};
    \node[font=\tiny, below] at (\x, 1.5) {$r{=}\r$};
  }
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=popblue!50, fill=popblue!3, rounded corners=5pt, text width=12cm, align=left, inner sep=7pt, font=\small] {
    \textbf{A's ranks:} $2 + 3 + 5 + 7 + 9 = 26$. \quad \textbf{B's ranks:} $1 + 4 + 6 + 8 + 10 = 29$.\\[3pt]
    If groups are identical, rank sums should be similar.\\[3pt]
    \textbf{When to use:} skewed data, outliers, ordinal data, small $n$ with non-Normal distributions.
  };
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Wilcoxon Signed-Rank Test}
\framesubtitle{Paired nonparametric alternative to the paired $t$-test}

\small
Compute differences $D_i$, rank the $|D_i|$, then compare the sum of positive vs negative ranks.

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    width=9cm, height=3.5cm,
    ybar,
    bar width=0.4cm,
    xlabel={Subject},
    ylabel={$D_i$ (Before $-$ After)},
    symbolic x coords={1,2,3,4,5,6,7,8},
    xtick=data,
    ymin=-3, ymax=10,
    axis lines=left,
    clip=false
  ]
    \addplot[fill=paramgreen!60, draw=paramgreen!80] coordinates {
      (1,8) (2,4) (3,4) (4,4) (5,3) (6,3) (7,2) (8,4)
    };
    \draw[thick, dashed, black!50] (axis cs:1, 0) -- (axis cs:8, 0);
  \end{axis}
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=5pt, text width=5.5cm, align=left, inner sep=7pt, font=\small}
]
  \node[cbox, fill=paramgreen!8] at (-4.2, 0) {
    All $D_i > 0$: all positive ranks.\\
    $W^+ = 1+2+\cdots+8 = 36$\\
    $W^- = 0$\\[3pt]
    $W = \min(W^+, W^-) = 0$\\
    Very small $\Rightarrow$ reject $H_0$.
  };
  \node[cbox, fill=orange1!8] at (4.2, 0) {
    \textbf{When to use:}\\[3pt]
    Paired data, but not Normal.\\
    Ordinal measurements.\\
    Small samples with outliers.\\[3pt]
    \textbf{Trade-off:} less power than $t$-test\\
    when data actually is Normal.
  };
\end{tikzpicture}
\end{center}

\vspace{0.05cm}
{\footnotesize\textcolor{gray}{\textbf{For 3+ groups:} Kruskal--Wallis test (nonparametric ANOVA). See Lecture~11.}}
\end{frame}

\section{Summary}

\begin{frame}
\frametitle{Summary: Classical Tests \& LRT}
\begin{center}
\begin{tikzpicture}[
  sbox/.style={draw=popblue!60, fill=popblue!5, rounded corners=4pt, text width=12cm, minimum height=0.5cm, align=left, inner sep=5pt, font=\small}
]
  \node[sbox] at (0, 3.2) {\textbf{LRT:} Compare best fit under $H_0$ vs overall. $-2\log\Lambda \sim \chi^2_k$ (Wilks' theorem).};
  \node[sbox] at (0, 2.4) {\textbf{One-sample $t$:} $T = (\bar{X} - \mu_0)/(S/\sqrt{n}) \sim t_{n-1}$. The workhorse of statistics.};
  \node[sbox] at (0, 1.6) {\textbf{Paired $t$:} Compute differences first, then one-sample $t$. More powerful than unpaired.};
  \node[sbox] at (0, 0.8) {\textbf{Welch's $t$:} Two independent means. No equal-variance assumption. Default choice.};
  \node[sbox] at (0, 0.0) {\textbf{Proportions:} $Z$-test with SE $= \sqrt{p_0(1-p_0)/n}$. Use when $np \geq 10$.};
  \node[sbox] at (0, -0.8) {\textbf{$\chi^2$ GoF:} Do observed counts match expected? $\sum(O-E)^2/E \sim \chi^2_{k-1}$.};
  \node[sbox] at (0, -1.6) {\textbf{$\chi^2$ independence:} Are two categorical variables associated? df $= (r{-}1)(c{-}1)$.};
  \node[sbox] at (0, -2.4) {\textbf{Nonparametric:} Mann--Whitney (independent), Wilcoxon (paired) when normality fails.};
\end{tikzpicture}
\end{center}
\end{frame}
