\begin{frame}
\frametitle{Before the Formula: An Example}

\small
\textbf{A surgeon performs 3 heart surgeries --- all 3 are successful.}\\[4pt]
What is their true success rate $p$?

\pause
\vspace{0.3cm}
\begin{center}
\begin{tikzpicture}[

  };

  };
  \draw[-{Stealth}, thick] (-3, 0) -- (-1.8, 0);
  \node[font=\Large] at (-2.4, 0.5) {$+$};
  \draw[-{Stealth}, thick] (1.8, 0) -- (3, 0);
  \node[font=\Large] at (2.4, 0.5) {$=$};
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\begin{center}
\fcolorbox{violet1}{violet1!5}{\parbox{11cm}{\centering\small
  By combining \textbf{prior knowledge} with data, we get a more sensible estimate.\\[2pt]
  Bayes' theorem tells us \textit{exactly} how to combine them. Let's see the formula.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Bayes' Theorem for Parameters}
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\underbrace{P(\theta \mid \text{data})}_{\text{posterior}} \;=\; \frac{\overbrace{P(\text{data} \mid \theta)}^{\text{likelihood}} \;\cdot\; \overbrace{P(\theta)}^{\text{prior}}}{\underbrace{P(\text{data})}_{\text{evidence}}}$
    }
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\centering\small
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{@{}r@{\; --- \;}l@{}}
\textcolor{popblue}{\textbf{Prior} $P(\theta)$} & what you believed about $\theta$ \textit{before} any data\\
\textcolor{sampred}{\textbf{Likelihood} $P(\text{data} \mid \theta)$} & how probable is the data \textit{if} $\theta$ were the true value\\
\textcolor{violet1}{\textbf{Posterior} $P(\theta \mid \text{data})$} & your \textit{updated} belief after combining both\\
\textcolor{gray}{\textbf{Evidence} $P(\text{data})$} & normalizing constant (same for all $\theta$, so we can drop it)\\
\end{tabular}

\pause
\vspace{0.15cm}

\small Or simply: \quad $\text{posterior} \;\propto\; \text{likelihood} \;\times\; \text{prior}$\\[3pt]
{\footnotesize ($\propto$ means ``proportional to'' --- equal up to a constant. We drop $P(\text{data})$ since it doesn't change $\arg\max_\theta$.)}

\end{frame}

\begin{frame}
\frametitle{The Beta Distribution: A Prior for Probabilities}

\small
We need a prior distribution for $p \in [0,1]$. The \textbf{Beta distribution} is the natural choice:

\vspace{0.1cm}
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{11cm}{\centering
  $\text{Beta}(\alpha, \beta): \quad f(p) = \frac{p^{\alpha-1}(1-p)^{\beta-1}}{B(\alpha,\beta)}, \quad 0 \le p \le 1$\\[6pt]
  \small Mean $= \alpha/(\alpha{+}\beta)$. \quad Higher $\alpha + \beta$ $\Rightarrow$ more concentrated.
}}
\end{center}

\pause
\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    width=12cm, height=4.5cm,
    xlabel={$p$}, ylabel={Density},
    xmin=0, xmax=1, ymin=0, ymax=4.5,
    axis lines=left,
    legend style={at={(0.98,0.95)}, anchor=north east, font=\scriptsize},
    grid=major, grid style={gray!15},
    every axis plot/.append style={line width=1.5pt, samples=100}
  ]
    \addplot[gray, dashed, domain=0.01:0.99] {1};
    \addlegendentry{Beta(1,1) = Uniform}
    \addplot[popblue, domain=0.01:0.99] {6*x*(1-x)};
    \addlegendentry{Beta(2,2): weak prior}
    \addplot[violet1, domain=0.01:0.99] {630*x^4*(1-x)^4};
    \addlegendentry{Beta(5,5): moderate}
  \end{axis}
\end{tikzpicture}
\end{center}

\pause
\small
\textbf{Why Beta?} (1)~Defined on $[0,1]$, matching probabilities. (2)~Very flexible shape. (3)~\textbf{Conjugate} to the Binomial --- posterior updates are trivial.
\end{frame}

\begin{frame}
\frametitle{How Did We Compute the Posterior?}

\small
\textbf{Prior:} $p \sim \text{Beta}(5,5)$: \; $P(p) \propto p^{\alpha-1}(1-p)^{\beta-1} = p^{4}(1-p)^{4}$

\pause
\vspace{0.2cm}
\textbf{Likelihood:} 7 heads in 10 flips: \; $P(\text{data} \mid p) \propto p^{7}(1-p)^{3}$

\pause
\vspace{0.2cm}
\textbf{Posterior} $= \text{likelihood} \times \text{prior}$:
$$P(p \mid \text{data}) \;\propto\; p^{7}(1-p)^{3} \cdot p^{4}(1-p)^{4} \;=\; p^{7+4}(1-p)^{3+4} \;=\; p^{11}(1-p)^{7}$$

\pause
\vspace{-0.2cm}
This is a $\text{Beta}(\textcolor{popblue}{12}, \textcolor{sampred}{8})$ distribution: just \textbf{add the exponents}!

\vspace{0.2cm}
\begin{center}
\fcolorbox{paramgreen}{paramgreen!5}{\parbox{10cm}{\centering\small
  $\text{Beta}(\underbrace{5}_{\alpha} {+} \underbrace{7}_{k},\;\; \underbrace{5}_{\beta} {+} \underbrace{3}_{n-k}) = \text{Beta}(12, 8)$\\[6pt]
  The prior parameters $(\alpha, \beta)$ simply absorb the data counts $(k, n{-}k)$.\\[3pt]
  This ``addition rule'' is why Beta--Binomial is called a \textbf{conjugate pair}.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Conjugate Priors as Pseudo-Observations}

\small
\textbf{Conjugate prior} = a prior whose posterior is in the \textit{same distribution family}.\\[4pt]
Beta is conjugate to Binomial: prior is Beta $\Rightarrow$ posterior is also Beta. No integrals needed!

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=6pt, minimum height=1.2cm, align=center, inner sep=8pt, font=\normalsize}
]
  \node[cbox, fill=popblue!10] (prior) at (-4.5, 0) {Prior: $\text{Beta}(\alpha, \beta)$};
  \node[font=\Large] at (-2, 0) {$+$};
  \node[cbox, fill=sampred!10] (data) at (0, 0) {Data: $k$ H, $n{-}k$ T};
  \node[font=\Large] at (2, 0) {$=$};
  \node[cbox, fill=violet1!10] (post) at (5, 0) {Posterior: $\text{Beta}(\alpha{+}k,\; \beta{+}n{-}k)$};
\end{tikzpicture}
\end{center}

\pause
\vspace{0.15cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{12cm}{\centering
  \textbf{The prior acts like ``fake data'' you've already seen:}\\[4pt]
  \small Beta$(\alpha, \beta)$ $\;=\;$ pretend you already observed $\alpha{-}1$ heads and $\beta{-}1$ tails.\\[3pt]
  Beta$(5,5)$: ``I've seen 4H and 4T'' (8 pseudo-observations).\\[3pt]
  After 7H, 3T (10 real obs): posterior = Beta$(12, 8)$ $\;=\;$ ``11H, 7T out of 18 total.''
}}
\end{center}

\vspace{0.15cm}
\small As $n \to \infty$, the pseudo-observations become negligible $\Rightarrow$ posterior $\to$ likelihood $\Rightarrow$ MAP $\to$ MLE.
\end{frame}

\begin{frame}
\frametitle{Common Conjugate Pairs}

\centering
\small
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{}lccc@{}}
\hline
\textbf{Likelihood} & \textbf{Conjugate Prior} & \textbf{Posterior} & \textbf{Pseudo-data}\\
\hline
$\text{Binomial}(n,p)$ & $\text{Beta}(\alpha,\beta)$ & $\text{Beta}(\alpha{+}k,\;\beta{+}n{-}k)$ & $\alpha{-}1$ H, $\beta{-}1$ T\\
$\text{Poisson}(\lambda)$ & $\text{Gamma}(a,b)$ & $\text{Gamma}(a{+}\sum x_i,\; b{+}n)$ & $a$ events in $b$ time\\
$N(\mu,\sigma_0^2)$ & $N(m, \tau^2)$ & $N\!\big(\frac{n\bar{x}/\sigma_0^2 + m/\tau^2}{n/\sigma_0^2 + 1/\tau^2},\;\frac{1}{n/\sigma_0^2 + 1/\tau^2}\big)$ & 1 obs of value $m$\\
$\text{Exp}(\lambda)$ & $\text{Gamma}(a,b)$ & $\text{Gamma}(a{+}n,\; b{+}\sum x_i)$ & $a$ events in $b$ time\\
\hline
\end{tabular}

\vspace{0.4cm}
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{10cm}{\centering\small
  \textbf{Pattern:} if the likelihood is in the exponential family, there is always a conjugate prior.\\[2pt]
  The posterior update just adds the sufficient statistics.
}}
\end{center}
\end{frame}

\section{MAP Estimation}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{MAP Estimation}}\\[0.3cm]
  {\large \textbf{M}aximum \textbf{A} \textbf{P}osteriori}

  \vspace{0.5cm}
  {\large The mode of the posterior}
\end{center}
\end{frame}

\begin{frame}
\frametitle{MAP = Mode of the Posterior}
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\hat{\theta}_{\text{MAP}} = \arg\max_\theta \; P(\theta \mid \text{data}) = \arg\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$
    }\\[12pt]
    \small Maximize: \quad log-likelihood $+$ log-prior
  };
\end{tikzpicture}
\end{center}

\vspace{0.5cm}
\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=4pt, minimum height=0.8cm, align=center, font=\normalsize, inner sep=6pt}
]
  \node[cbox, fill=popblue!10] at (-3, 0) {MLE: $\arg\max_\theta \; \ell(\theta)$};
  \node[cbox, fill=violet1!10] at (4, 0) {MAP: $\arg\max_\theta \; \ell(\theta) + \log P(\theta)$};
  \draw[-{Stealth}, thick] (0.3, 0) -- (1, 0);
  \node[font=\small] at (0.65, 0.5) {$+$ prior};
\end{tikzpicture}
\end{center}

\vspace{0.3cm}
\pause
\small MAP = MLE with an extra penalty/bonus term from the prior.
\end{frame}

\begin{frame}
\frametitle{MAP Step by Step: Finding the Mode}

\small
We computed earlier that the posterior is $\text{Beta}(12, 8)$ for our coin example.\\[4pt]
\textbf{MAP} = the \textbf{mode} (peak) of this posterior.

\pause
\vspace{0.2cm}
\textbf{Step 1: Write the log-posterior.}
\vspace{-0.1cm}
$$\log P(p \mid \text{data}) = 11\log p + 7\log(1-p) + \text{const}$$

\pause
\vspace{-0.2cm}
\textbf{Step 2: Differentiate and set to zero.}
\vspace{-0.1cm}
$$\frac{d}{dp}\log P = \frac{11}{p} - \frac{7}{1-p} = 0 \quad\Rightarrow\quad 11(1-p) = 7p \quad\Rightarrow\quad p = \frac{11}{18}$$

\pause
\vspace{-0.1cm}
\textbf{General formula for Beta$(\alpha', \beta')$:}
$$\boxed{\hat{p}_{\text{MAP}} = \frac{\alpha' - 1}{\alpha' + \beta' - 2}}$$

\pause
\vspace{-0.1cm}
For our example: $\hat{p}_{\text{MAP}} = \frac{12 - 1}{12 + 8 - 2} = \frac{11}{18} \approx \textcolor{violet1}{0.611}$

\vspace{0.1cm}
\centering Compare: \; $\hat{p}_{\text{MLE}} = \frac{7}{10} = \textcolor{sampred}{0.700}$ \quad (data alone, no prior)
\end{frame}

\begin{frame}
\frametitle{MAP Under Different Priors}

\small
Same data (7 heads in 10 flips), but different priors:

\vspace{0.2cm}
\centering
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}lccc@{}}
\hline
\textbf{Prior} & \textbf{Posterior} & \textbf{MAP estimate} & \textbf{Pull toward 0.5}\\
\hline
Beta$(1,1)$ = Uniform & Beta$(8, 4)$ & $7/10 = \textcolor{sampred}{0.700}$ & None (= MLE)\\
Beta$(2,2)$ & Beta$(9, 5)$ & $8/12 = 0.667$ & Weak\\
Beta$(5,5)$ & Beta$(12, 8)$ & $11/18 = \textcolor{violet1}{0.611}$ & Moderate\\
Beta$(50,50)$ & Beta$(57, 53)$ & $56/108 = 0.519$ & Strong\\
\hline
\end{tabular}

\pause
\vspace{0.3cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{Stronger prior} (higher $\alpha + \beta$) $\Rightarrow$ more pull toward the prior mean.\\[3pt]
  \textbf{More data} ($n \uparrow$) $\Rightarrow$ data overwhelms the prior $\Rightarrow$ MAP $\to$ MLE.\\[3pt]
  A \textbf{flat prior} (Uniform) gives MAP $=$ MLE: no prior $=$ no regularization.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{MAP for a Normal Mean}

\small
\textbf{Model:} $X_1, \ldots, X_n \sim N(\mu, \sigma_0^2)$ (known $\sigma_0^2$), \; prior $\mu \sim N(m, \tau^2)$.

\pause
\vspace{0.15cm}
\textbf{Log-posterior:}
\begin{align*}
\log P(\mu \mid \text{data}) &= \underbrace{-\frac{1}{2\sigma_0^2}\sum_{i=1}^n(X_i - \mu)^2}_{\text{log-likelihood}} \;\;+\;\; \underbrace{\left(-\frac{(\mu - m)^2}{2\tau^2}\right)}_{\text{log-prior}} + \text{const}
\end{align*}

\pause
\textbf{Take derivative, set to 0:}
$$\frac{n(\bar{X} - \mu)}{\sigma_0^2} - \frac{\mu - m}{\tau^2} = 0$$

\pause
\textbf{Solve:}
$$\boxed{\hat{\mu}_{\text{MAP}} = \frac{\frac{n}{\sigma_0^2}}{\frac{n}{\sigma_0^2} + \frac{1}{\tau^2}} \cdot \bar{X} \;+\; \frac{\frac{1}{\tau^2}}{\frac{n}{\sigma_0^2} + \frac{1}{\tau^2}} \cdot m = \textcolor{violet1}{w \cdot \bar{X} + (1-w) \cdot m}}$$

\vspace{0.1cm}
\centering\small
A \textbf{weighted average} of the sample mean $\bar{X}$ and the prior mean $m$.\\
More data ($n \uparrow$) $\Rightarrow$ $w \to 1$ $\Rightarrow$ MAP $\to$ MLE. \; Stronger prior ($\tau^2 \downarrow$) $\Rightarrow$ $w \to 0$ $\Rightarrow$ MAP $\to m$.
\end{frame}

\section{Regularization as MAP}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{Regularization as MAP}}

  \vspace{0.8cm}
  {\large Priors are penalties in disguise}
\end{center}
\end{frame}

\begin{frame}
\frametitle{What Is Regularization?}

\small
In machine learning, models can \textbf{overfit}: they memorize noise in the training data.

\pause
\vspace{0.2cm}
\textbf{Regularization} adds a \textbf{penalty} on the model parameters to keep them small:
$$\hat{\theta} = \arg\min_\theta \bigl[\underbrace{\text{Loss}(\theta)}_{\text{data fit}} + \underbrace{\lambda \cdot \text{Penalty}(\theta)}_{\text{keep } \theta \text{ small}}\bigr]$$

\pause
\vspace{0.1cm}
Two common penalties:

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}[
  rbox/.style={draw, rounded corners=5pt, text width=5.2cm, minimum height=1.8cm, align=center, inner sep=8pt, font=\small}
]
  \node[rbox, fill=popblue!8] at (-4, 0) {
    \textbf{L2 (Ridge)}\\[4pt]
    $\text{Penalty} = \|\theta\|_2^2 = \sum \theta_j^2$\\[4pt]
    Shrinks all coefficients toward 0
  };
  \node[rbox, fill=paramgreen!8] at (4, 0) {
    \textbf{L1 (Lasso)}\\[4pt]
    $\text{Penalty} = \|\theta\|_1 = \sum |\theta_j|$\\[4pt]
    Drives some coefficients to \textbf{exactly 0}
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\centering\small
$\lambda$ controls the trade-off: $\lambda = 0$ means no penalty (= MLE); large $\lambda$ means heavy penalty.\\[3pt]
\textbf{Key insight:} these penalties are secretly \textbf{priors in disguise}. Let's see why.
\end{frame}

\begin{frame}
\frametitle{The Key Connection: Regularization = MAP}
\begin{center}
\begin{tikzpicture}
  \node[draw=orange1, fill=orange1!5, rounded corners=10pt, text width=12.5cm, align=center, inner sep=12pt] {
    {\large MAP: $\;\hat{\theta} = \arg\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$}\\[8pt]
    is the same as\\[8pt]
    {\large Regularization: $\;\hat{\theta} = \arg\min_\theta \bigl[-\ell(\theta) + \lambda \cdot \text{penalty}(\theta)\bigr]$}\\[10pt]
    \small The log-prior acts as a \textbf{penalty on the parameters}.
  };
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}
\frametitle{From Prior to Penalty: The Algebra}

\small
\textbf{Assume:} $y_i = \mathbf{x}_i^\top\theta + \varepsilon_i$, \; $\varepsilon_i \sim N(0, \sigma^2)$, \; prior $\theta_j \sim N(0, \tau^2)$.

\pause
\vspace{0.05cm}
\textbf{MAP objective:} $\;\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$

\pause
\vspace{-0.05cm}
\begin{align*}
\ell(\theta) &= -\frac{1}{2\sigma^2}\textstyle\sum_{i=1}^n(y_i - \mathbf{x}_i^\top\theta)^2 + \text{const}\\[2pt]
\pause
\log P(\theta) &= -\frac{\|\theta\|_2^2}{2\tau^2} + \text{const}
\end{align*}

\vspace{-0.2cm}
\pause
\textbf{Combine and flip sign} (max $\to$ min, drop constants):
\vspace{-0.1cm}
$$\hat{\theta}_{\text{MAP}} = \arg\min_\theta \left[\underbrace{\frac{1}{2\sigma^2}\sum_{i=1}^n(y_i - \mathbf{x}_i^\top\theta)^2}_{\text{data fit (= MSE loss)}} + \underbrace{\frac{1}{2\tau^2}\|\theta\|_2^2}_{\text{from the prior}}\right]$$

\vspace{-0.2cm}
\pause
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{10cm}{\centering\small
  Set $\lambda = \sigma^2/\tau^2$: \; \textbf{small $\tau^2$} (tight prior) $\Rightarrow$ \textbf{large $\lambda$} (strong penalty).\\[2pt]
  This is exactly \textbf{Ridge regression}.
}}
\end{center}
\end{frame}

\vspace{0.1cm}

\begin{frame}
\frametitle{Two Perspectives: Bayesian vs Frequentist}

\small
The Regularization Map reveals a deep duality between two schools of statistics:

\vspace{0.2cm}
\begin{center}
\begin{tikzpicture}[
  vbox/.style={draw, rounded corners=6pt, text width=5.5cm, minimum height=3.5cm, align=left, inner sep=8pt, font=\small}

  \node[vbox, fill=popblue!8] at (-4, 0) {
    \textbf{\textcolor{popblue}{\large Bayesian view}}\\[6pt]
    Parameters have \textit{distributions}.\\[3pt]
    Start with a \textbf{prior} belief,\\
    update with data via Bayes' rule.\\[3pt]
    ``$\theta$ is probably near 0''\\
    $\to$ Gaussian prior $\mathcal{N}(0, \tau^2)$\\[3pt]
    {\footnotesize Result: posterior distribution.}
  };

    add \textbf{penalty} to avoid overfitting.\\[3pt]
    ``Keep coefficients small''\\
    $\to$ Ridge penalty $\lambda\|\theta\|_2^2$\\[3pt]
    {\footnotesize Result: point estimate.}
  };
\end{tikzpicture}
\end{center}

}}
\end{center}

\section{Summary}

\begin{frame}
\frametitle{Summary: From MLE to MAP}
\begin{center}
\begin{tikzpicture}[
  sbox/.style={draw=popblue!60, fill=popblue!5, rounded corners=4pt, minimum width=12.5cm, minimum height=0.5cm, align=left, inner sep=5pt, font=\small}
]
  \node[sbox] at (0, 2.8) {\textbf{Bayes' rule:} $\text{posterior} \propto \text{likelihood} \times \text{prior}$. Update beliefs with data.};
  \node[sbox] at (0, 2.0) {\textbf{Conjugacy:} Beta--Binomial, Normal--Normal. Prior acts like pseudo-observations.};
  \node[sbox] at (0, 1.2) {\textbf{MAP:} $\hat\theta = \arg\max[\ell(\theta) + \log P(\theta)]$. MLE with a prior bonus.};
  \node[sbox] at (0, 0.4) {\textbf{Normal MAP:} Weighted average $w\bar{X} + (1{-}w)m$. More data $\Rightarrow$ MAP $\to$ MLE.};
  \node[sbox] at (0, -0.4) {\textbf{Gaussian prior $\to$ Ridge:} $-\log P(\theta) \propto \|\theta\|_2^2$. Shrinks toward 0.};
  \node[sbox] at (0, -1.2) {\textbf{Laplace prior $\to$ Lasso:} $-\log P(\theta) \propto \|\theta\|_1$. Drives coefficients to exactly 0.};
  \node[sbox] at (0, -2.0) {\textbf{Flat prior $\to$ no penalty:} MAP reduces to MLE. No prior = no regularization.};
  \node[sbox] at (0, -2.8) {\textbf{Bayesian = Frequentist:} Gaussian prior $\Leftrightarrow$ Ridge, Laplace $\Leftrightarrow$ Lasso. Two views, same math.};
\end{tikzpicture}
\end{center}
\end{frame}
