\section{Bayes' Theorem for Parameters}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{Bayes' Theorem\\[0.3cm] for Parameters}}

  \vspace{0.8cm}
  {\large From likelihood to posterior}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Bayes' Theorem for Parameters}
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\underbrace{P(\theta \mid \text{data})}_{\text{posterior}} \;=\; \frac{\overbrace{P(\text{data} \mid \theta)}^{\text{likelihood}} \;\cdot\; \overbrace{P(\theta)}^{\text{prior}}}{\underbrace{P(\text{data})}_{\text{evidence}}}$
    }\\[15pt]
    \pause
    \small Or simply: \quad $\text{posterior} \;\propto\; \text{likelihood} \;\times\; \text{prior}$
  };
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Conjugate Priors as Pseudo-Observations}

\small
Why did the math work out so neatly? Because Beta is \textbf{conjugate} to Binomial:

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=6pt, minimum height=1.2cm, align=center, inner sep=8pt, font=\normalsize}
]
  \node[cbox, fill=popblue!10] (prior) at (-4.5, 0) {Prior: $\text{Beta}(\alpha, \beta)$};
  \node[font=\Large] at (-2, 0) {$+$};
  \node[cbox, fill=sampred!10] (data) at (0, 0) {Data: $k$ H, $n{-}k$ T};
  \node[font=\Large] at (2, 0) {$=$};
  \node[cbox, fill=violet1!10] (post) at (5, 0) {Posterior: $\text{Beta}(\alpha{+}k,\; \beta{+}n{-}k)$};
\end{tikzpicture}
\end{center}

\pause
\vspace{0.15cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{12cm}{\centering
  \textbf{The prior acts like ``fake data'' you've already seen:}\\[4pt]
  \small Beta$(\alpha, \beta)$ $\;=\;$ pretend you already observed $\alpha{-}1$ heads and $\beta{-}1$ tails.\\[3pt]
  Beta$(5,5)$: ``I've seen 4H and 4T'' (8 pseudo-observations).\\[3pt]
  After 7H, 3T (10 real obs): posterior = Beta$(12, 8)$ $\;=\;$ ``11H, 7T out of 18 total.''
}}
\end{center}

\vspace{0.15cm}
\small As $n \to \infty$, the pseudo-observations become negligible $\Rightarrow$ posterior $\to$ likelihood $\Rightarrow$ MAP $\to$ MLE.
\end{frame}

\begin{frame}
\frametitle{Common Conjugate Pairs}

\centering
\small
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{@{}lccc@{}}
\hline
\textbf{Likelihood} & \textbf{Conjugate Prior} & \textbf{Posterior} & \textbf{Pseudo-data}\\
\hline
$\text{Binomial}(n,p)$ & $\text{Beta}(\alpha,\beta)$ & $\text{Beta}(\alpha{+}k,\;\beta{+}n{-}k)$ & $\alpha{-}1$ H, $\beta{-}1$ T\\
$\text{Poisson}(\lambda)$ & $\text{Gamma}(a,b)$ & $\text{Gamma}(a{+}\sum x_i,\; b{+}n)$ & $a$ events in $b$ time\\
$N(\mu,\sigma_0^2)$ & $N(m, \tau^2)$ & $N\!\big(\frac{n\bar{x}/\sigma_0^2 + m/\tau^2}{n/\sigma_0^2 + 1/\tau^2},\;\ldots\big)$ & 1 obs of value $m$\\
$\text{Exp}(\lambda)$ & $\text{Gamma}(a,b)$ & $\text{Gamma}(a{+}n,\; b{+}\sum x_i)$ & $a$ events in $b$ time\\
\hline
\end{tabular}

\vspace{0.4cm}
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{10cm}{\centering\small
  \textbf{Pattern:} if the likelihood is in the exponential family, there is always a conjugate prior.\\[2pt]
  The posterior update just adds the sufficient statistics.
}}
\end{center}
\end{frame}

\begin{frame}

  \vspace{0.8cm}
  {\large The mode of the posterior}
\end{center}

\begin{frame}
\frametitle{MAP = Mode of the Posterior}
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\hat{\theta}_{\text{MAP}} = \arg\max_\theta \; P(\theta \mid \text{data}) = \arg\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$
    }\\[12pt]
    \small Maximize: \quad log-likelihood $+$ log-prior
  };
\end{tikzpicture}
\end{center}

\vspace{0.5cm}
\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=4pt, minimum height=0.8cm, align=center, font=\normalsize, inner sep=6pt}
]
  \node[cbox, fill=popblue!10] at (-3, 0) {MLE: $\arg\max_\theta \; \ell(\theta)$};
  \node[cbox, fill=violet1!10] at (4, 0) {MAP: $\arg\max_\theta \; \ell(\theta) + \log P(\theta)$};
  \draw[-{Stealth}, thick] (0.3, 0) -- (1, 0);
  \node[font=\small] at (0.65, 0.5) {$+$ prior};
\end{tikzpicture}
\end{center}

\vspace{0.3cm}
\pause
\small MAP = MLE with an extra penalty/bonus term from the prior.
\end{frame}

\begin{frame}
\frametitle{MAP Step by Step: Coin Bias}

\small
\textbf{Setup:} Prior $p \sim \text{Beta}(5,5)$, data: 7 heads in 10 flips.

\pause
\vspace{0.1cm}
\textbf{Step 1: Write the posterior.}
\vspace{-0.2cm}
\begin{align*}
P(p \mid \text{data}) &\propto \underbrace{p^7(1-p)^3}_{\text{likelihood}} \cdot \underbrace{p^{4}(1-p)^{4}}_{\text{Beta}(5,5)\text{ prior}} = p^{11}(1-p)^{7}
\end{align*}

\vspace{-0.3cm}
\pause
This is $\text{Beta}(\textcolor{popblue}{12}, \textcolor{sampred}{8})$ --- conjugate update: $\alpha' = 5+7$, $\beta' = 5+3$.

\pause
\vspace{0.1cm}
\textbf{Step 2: Find the mode.}
\vspace{-0.1cm}
$$\hat{p}_{\text{MAP}} = \frac{\alpha' - 1}{\alpha' + \beta' - 2} = \frac{12 - 1}{12 + 8 - 2} = \frac{11}{18} \approx \textcolor{violet1}{0.611}$$

\vspace{-0.2cm}
\pause
\textbf{Compare with MLE:} \; $\hat{p}_{\text{MLE}} = \frac{7}{10} = \textcolor{sampred}{0.700}$

\vspace{0.1cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{10cm}{\centering\small
  The prior pulled the estimate from $\textcolor{sampred}{0.70}$ toward $0.50$ (the prior mean), landing at $\textcolor{violet1}{0.61}$.\\
  With more data, this effect shrinks.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{MAP for a Normal Mean}

\small
\textbf{Model:} $X_1, \ldots, X_n \sim N(\mu, \sigma_0^2)$ (known $\sigma_0^2$), \; prior $\mu \sim N(m, \tau^2)$.

\pause
\vspace{0.15cm}
\textbf{Log-posterior:}
\begin{align*}
\log P(\mu \mid \text{data}) &= \underbrace{-\frac{1}{2\sigma_0^2}\sum_{i=1}^n(X_i - \mu)^2}_{\text{log-likelihood}} \;\;+\;\; \underbrace{\left(-\frac{(\mu - m)^2}{2\tau^2}\right)}_{\text{log-prior}} + \text{const}
\end{align*}

\pause
\textbf{Take derivative, set to 0:}
$$\frac{n(\bar{X} - \mu)}{\sigma_0^2} - \frac{\mu - m}{\tau^2} = 0$$

\pause
\textbf{Solve:}
$$\boxed{\hat{\mu}_{\text{MAP}} = \frac{\frac{n}{\sigma_0^2}}{\frac{n}{\sigma_0^2} + \frac{1}{\tau^2}} \cdot \bar{X} \;+\; \frac{\frac{1}{\tau^2}}{\frac{n}{\sigma_0^2} + \frac{1}{\tau^2}} \cdot m = \textcolor{violet1}{w \cdot \bar{X} + (1-w) \cdot m}}$$

\vspace{0.1cm}
\centering\small
A \textbf{weighted average} of the sample mean $\bar{X}$ and the prior mean $m$.\\
More data ($n \uparrow$) $\Rightarrow$ $w \to 1$ $\Rightarrow$ MAP $\to$ MLE. \; Stronger prior ($\tau^2 \downarrow$) $\Rightarrow$ $w \to 0$ $\Rightarrow$ MAP $\to m$.
\end{frame}

\section{Regularization as MAP}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{Regularization as MAP}}

  \vspace{0.8cm}
  {\large Priors are penalties in disguise}
\end{center}
\end{frame}

\begin{frame}
\frametitle{The Key Connection: Regularization = MAP}
\begin{center}
\begin{tikzpicture}
  \node[draw=orange1, fill=orange1!5, rounded corners=10pt, text width=12.5cm, align=center, inner sep=12pt] {
    {\large MAP: $\;\hat{\theta} = \arg\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$}\\[8pt]
    is the same as\\[8pt]
    {\large Regularization: $\;\hat{\theta} = \arg\min_\theta \bigl[-\ell(\theta) + \lambda \cdot \text{penalty}(\theta)\bigr]$}\\[10pt]
    \small The log-prior acts as a \textbf{penalty on the parameters}.
  };
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}
\frametitle{From Prior to Penalty: The Algebra}

\small
\textbf{Assume:} $y_i = \mathbf{x}_i^\top\theta + \varepsilon_i$, \; $\varepsilon_i \sim N(0, \sigma^2)$, \; prior $\theta_j \sim N(0, \tau^2)$.

\pause
\vspace{0.05cm}
\textbf{MAP objective:} $\;\max_\theta \bigl[\ell(\theta) + \log P(\theta)\bigr]$

\pause
\vspace{-0.05cm}
\begin{align*}
\ell(\theta) &= -\frac{1}{2\sigma^2}\textstyle\sum_{i=1}^n(y_i - \mathbf{x}_i^\top\theta)^2 + \text{const}\\[2pt]
\pause
\log P(\theta) &= -\frac{\|\theta\|_2^2}{2\tau^2} + \text{const}
\end{align*}

\vspace{-0.2cm}
\pause
\textbf{Combine and flip sign} (max $\to$ min, drop constants):
\vspace{-0.1cm}
$$\hat{\theta}_{\text{MAP}} = \arg\min_\theta \left[\underbrace{\frac{1}{2\sigma^2}\sum_{i=1}^n(y_i - \mathbf{x}_i^\top\theta)^2}_{\text{data fit (= MSE loss)}} + \underbrace{\frac{1}{2\tau^2}\|\theta\|_2^2}_{\text{from the prior}}\right]$$

\vspace{-0.2cm}
\pause
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{10cm}{\centering\small
  Set $\lambda = \sigma^2/\tau^2$: \; \textbf{small $\tau^2$} (tight prior) $\Rightarrow$ \textbf{large $\lambda$} (strong penalty).\\[2pt]
  This is exactly \textbf{Ridge regression}.
}}
\end{center}
\end{frame}

\begin{center}

  \node[pbox, fill=sampred!10] at (-5, 0) {
    \textbf{\large MLE}\\[6pt]
    \small Mode of likelihood\\[4pt]
    No prior info used\\[4pt]
    Risk: overfitting\\with small $n$\\[4pt]
    \footnotesize $\hat{\theta} = \arg\max \ell(\theta)$
  };
  \node[pbox, fill=violet1!10] at (0, 0) {
    \textbf{\large MAP}\\[6pt]
    \small Mode of posterior\\[4pt]
    Uses prior beliefs\\[4pt]
    Still a point estimate\\[4pt]
    \footnotesize $\hat{\theta} = \arg\max [\ell + \log P]$
  };
  \node[pbox, fill=paramgreen!10] at (5, 0) {
    \textbf{\large Full Bayesian}\\[6pt]
    \small Entire posterior\\[4pt]
    Full uncertainty\\[4pt]
    Computationally\\expensive (MCMC)\\[4pt]
    \footnotesize $P(\theta \mid \text{data})$
  };

  \draw[-{Stealth}, thick] (-3, -0.3) -- (-2, -0.3) node[midway, below, font=\footnotesize] {add prior};
  \draw[-{Stealth}, thick] (2, -0.3) -- (3, -0.3) node[midway, below, font=\footnotesize] {keep full dist.};
\end{tikzpicture}
\end{center}

\vspace{0.1cm}
\pause
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{Empirical Bayes} (a hybrid): estimate the prior \textit{from the data itself}.\\[2pt]
  Use MLE to pick hyperparameters $(\alpha, \beta, \tau^2, \ldots)$, then do Bayesian inference.\\[2pt]
  Frequentist spirit + Bayesian machinery. Popular in genomics, NLP, and James--Stein shrinkage.
}}
\end{center}
\end{frame}

\section{Summary}

\begin{frame}
\frametitle{Summary: From MLE to MAP}
\begin{center}
\begin{tikzpicture}[
  sbox/.style={draw=popblue!60, fill=popblue!5, rounded corners=4pt, minimum width=12.5cm, minimum height=0.5cm, align=left, inner sep=5pt, font=\small}
]
  \node[sbox] at (0, 2.8) {\textbf{Bayes' rule:} $\text{posterior} \propto \text{likelihood} \times \text{prior}$. Update beliefs with data.};
  \node[sbox] at (0, 2.0) {\textbf{Conjugacy:} Beta--Binomial, Normal--Normal. Prior acts like pseudo-observations.};
  \node[sbox] at (0, 1.2) {\textbf{MAP:} $\hat\theta = \arg\max[\ell(\theta) + \log P(\theta)]$. MLE with a prior bonus.};
  \node[sbox] at (0, 0.4) {\textbf{Normal MAP:} Weighted average $w\bar{X} + (1{-}w)m$. More data $\Rightarrow$ MAP $\to$ MLE.};
  \node[sbox] at (0, -0.4) {\textbf{Gaussian prior $\to$ Ridge:} $-\log P(\theta) \propto \|\theta\|_2^2$. Shrinks toward 0.};
  \node[sbox] at (0, -1.2) {\textbf{Laplace prior $\to$ Lasso:} $-\log P(\theta) \propto \|\theta\|_1$. Drives coefficients to exactly 0.};
  \node[sbox] at (0, -2.0) {\textbf{Flat prior $\to$ no penalty:} MAP reduces to MLE. No prior = no regularization.};
  \node[sbox] at (0, -2.8) {\textbf{Full Bayesian:} Use the entire posterior, not just its mode. Gold standard but expensive.};
\end{tikzpicture}
\end{center}
\end{frame}
