\pause
\vspace{0.2cm}
\begin{center}

  \textbf{Key insight:} $\hat\theta$ is a \textit{function of random data}, so $\hat\theta$ itself is a random variable.\\[2pt]

}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{Example: Sample Mean from $N(5, 4)$}
\framesubtitle{Population: $X_i \sim N(\mu = 5, \sigma^2 = 4)$, \; $n = 10$, \; estimator: $\hat\mu = \bar{X}$}

\vspace{-0.2cm}
\begin{center}
\begin{tikzpicture}
  % Left: population

  \begin{scope}[xshift=4.5cm]

\end{center}

\centering\small
The sampling distribution is \textbf{much narrower} than the population --- averaging reduces the spread by $\sqrt{n}$.
\end{frame}

\section{The Central Limit Theorem}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{The Central Limit Theorem}}

  \vspace{0.8cm}
  {\large Why everything looks Normal}
\end{center}
\end{frame}

\begin{frame}
\frametitle{CLT: The Most Important Theorem in Statistics}

\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    \textbf{Central Limit Theorem:} If $X_1, \ldots, X_n$ are i.i.d.\ with mean $\mu$ and variance $\sigma^2 < \infty$, then
    \\[10pt]
    {\Large $\frac{\bar{X} - \mu}{\sigma / \sqrt{n}} \;\xrightarrow{d}\; N(0, 1) \quad \text{as } n \to \infty$}
    \\[10pt]
    \small Equivalently: $\;\bar{X} \;\dot\sim\; N\!\left(\mu, \;\frac{\sigma^2}{n}\right)$ \; for large $n$.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.3cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{The magic:} It doesn't matter what the population looks like ---\\
  uniform, exponential, Bernoulli, bimodal, anything.\\[2pt]
  As long as the variance is finite, $\bar{X}$ becomes Normal.
}}
\end{center}
\end{frame}

\section{Standard Error}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{Standard Error}}

  \vspace{0.8cm}
  {\large The most important number after the estimate itself}
\end{center}
\end{frame}

\begin{frame}
\frametitle{SD $\neq$ SE: The Most Confused Pair in Statistics}

\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=6pt, minimum height=3.5cm, text width=5.5cm, align=center, inner sep=8pt}
]
  \node[cbox, fill=popblue!8] at (-4.2, 0) {
    \textbf{\large\textcolor{popblue}{Standard Deviation (SD)}}\\[8pt]
    \small
    Spread of the \textbf{data}\\[4pt]
    $\sigma = \sqrt{E[(X - \mu)^2]}$\\[6pt]
    ``How variable are individual\\observations?''\\[6pt]
    \footnotesize\textbf{Does not shrink with $n$.}\\
    More data doesn't make\\individuals less variable.
  };

  \node[cbox, fill=violet1!8] at (4.2, 0) {
    \textbf{\large\textcolor{violet1}{Standard Error (SE)}}\\[8pt]
    \small
    Spread of the \textbf{estimator}\\[4pt]
    $\text{SE}(\bar{X}) = \sigma / \sqrt{n}$\\[6pt]
    ``How variable is $\hat\theta$ across\\repeated experiments?''\\[6pt]
    \footnotesize\textbf{Shrinks with $n$.}\\
    More data $\Rightarrow$ more\\precise estimate.
  };

  \draw[thick, gray, dashed] (0, -2.5) -- (0, 2.5);
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\centering\small
\textbf{Example:} Heights of adults: SD $\approx$ 7 cm (people vary).\\
Average of 100 heights: SE $= 7/\sqrt{100} = 0.7$ cm (the average is precise).
\end{frame}

\begin{frame}
\frametitle{Standard Error of the Sample Mean}

\begin{center}
\begin{tikzpicture}
  \node[draw=orange1, fill=orange1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\text{SE}(\bar{X}) = \frac{\sigma}{\sqrt{n}}$
    }\\[12pt]
    \small The \textbf{standard error} is the standard deviation of the sampling distribution.\\[4pt]
    It measures how much $\bar{X}$ would vary if we repeated the experiment.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\small\textbf{Derivation:}
\vspace{-0.1cm}
$$\text{Var}(\bar{X}) = \text{Var}\!\left(\frac{1}{n}\sum_{i=1}^n X_i\right) = \frac{1}{n^2}\sum_{i=1}^n \text{Var}(X_i) = \frac{1}{n^2} \cdot n\sigma^2 = \frac{\sigma^2}{n}$$

\pause
\vspace{-0.15cm}
$$\Rightarrow \quad \text{SE}(\bar{X}) = \sqrt{\text{Var}(\bar{X})} = \frac{\sigma}{\sqrt{n}}$$

\vspace{0.1cm}
\centering\small
Uses i.i.d.\ assumption: observations are independent, so variances \textbf{add}.
\end{frame}

\begin{frame}
\frametitle{The $\sqrt{n}$ Law: Precision Is Expensive}

\begin{center}

    width=12cm, height=5cm,

  \end{axis}
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\begin{center}
\fcolorbox{warnred}{warnred!5}{\parbox{10cm}{\centering\small
  To \textbf{halve} the SE, you must \textbf{quadruple} the sample size.\\[2pt]
  Going from $n = 100$ to $n = 10{,}000$ only reduces SE by $10\times$.\\[2pt]
  This is why sample size planning matters!
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{SE for Any MLE: The General Formula}

\small
From Lecture~4 (Cram\'er--Rao) and Lecture~5 (MLE asymptotics):

\vspace{0.15cm}
\begin{center}
\begin{tikzpicture}
  \node[draw=popblue, fill=popblue!5, rounded corners=10pt, text width=12cm, align=center, inner sep=12pt] {
    {\Large $\text{SE}(\hat\theta_{\text{MLE}}) \;\approx\; \frac{1}{\sqrt{n \cdot I(\theta)}}$}\\[10pt]
    \small The Fisher information $I(\theta)$ determines the precision of the MLE.\\[4pt]
    More informative data (larger $I$) $\Rightarrow$ smaller SE.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\textbf{Problem:} This formula contains the \textit{unknown} $\theta$!

\pause
\vspace{0.15cm}
\textbf{Solution: Plug-in.} Replace $\theta$ with $\hat\theta$:
\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=8pt, text width=8cm, align=center, inner sep=8pt] {
    {\large $\widehat{\text{SE}}(\hat\theta) = \frac{1}{\sqrt{n \cdot I(\hat\theta)}}$}\\[6pt]
    \small Estimated standard error --- this is what we report in practice.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\centering\small
For large $n$, $\hat\theta \approx \theta$, so the plug-in is accurate.
\end{frame}

\begin{frame}
\frametitle{Known vs Estimated SE}

\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=6pt, minimum height=3.0cm, text width=5.5cm, align=center, inner sep=8pt, font=\small}
]
  \node[cbox, fill=popblue!8] at (-4, 0) {
    \textbf{If $\sigma$ is known:}\\[6pt]
    $\text{SE}(\bar{X}) = \dfrac{\sigma}{\sqrt{n}}$\\[8pt]
    \footnotesize Rare in practice.\\
    Use the $z$-distribution (Normal).
  };

  \pause
  \node[cbox, fill=violet1!8] at (4, 0) {
    \textbf{If $\sigma$ is unknown (typical):}\\[6pt]
    $\widehat{\text{SE}}(\bar{X}) = \dfrac{S}{\sqrt{n}}$\\[8pt]
    \footnotesize Replace $\sigma$ with $S = \sqrt{\frac{1}{n-1}\sum(X_i - \bar{X})^2}$.\\
    Use the $t$-distribution.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.3cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{10cm}{\centering\small
  \textbf{The $t$-distribution} has heavier tails than the Normal --- it accounts for the\\
  extra uncertainty from estimating $\sigma$. As $n \to \infty$, $t_{n-1} \to N(0,1)$.
}}
\end{center}
\end{frame}
