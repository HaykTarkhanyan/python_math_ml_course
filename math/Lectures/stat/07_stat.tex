\pause
\vspace{0.2cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{Key insight:} $\hat\theta$ is a \textit{function of random data}, so $\hat\theta$ itself is a random variable.\\[2pt]
  Its distribution is called the \textbf{sampling distribution}. Understanding it is the foundation of all inference.
}}
\end{center}
\end{frame}

\frametitle{Example: Sample Mean from $N(5, 4)$}
\framesubtitle{Population: $X_i \sim N(\mu = 5, \sigma^2 = 4)$, \; $n = 10$, \; estimator: $\hat\mu = \bar{X}$}

\vspace{-0.2cm}
\begin{center}
\begin{tikzpicture}
  % Left: population

  \end{scope}

  % Right: sampling distribution
  \begin{scope}[xshift=4.5cm]
    \node[font=\small\bfseries, violet1] at (2, 2.6) {Sampling dist.\ of $\bar{X}$};
    \begin{axis}[

\end{tikzpicture}
\end{center}

\centering\small
The sampling distribution is \textbf{much narrower} than the population --- averaging reduces the spread by $\sqrt{n}$.
\end{frame}

\section{The Central Limit Theorem}

\begin{frame}
\begin{center}
  \vspace{2cm}
  {\Huge\bfseries\textcolor{popblue}{The Central Limit Theorem}}

  \vspace{0.8cm}
  {\large Why everything looks Normal}
\end{center}
\end{frame}

\begin{frame}
\frametitle{CLT: The Most Important Theorem in Statistics}

\begin{center}
\begin{tikzpicture}
  \node[draw=violet1, fill=violet1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    \textbf{Central Limit Theorem:} If $X_1, \ldots, X_n$ are i.i.d.\ with mean $\mu$ and variance $\sigma^2 < \infty$, then
    \\[10pt]
    {\Large $\frac{\bar{X} - \mu}{\sigma / \sqrt{n}} \;\xrightarrow{d}\; N(0, 1) \quad \text{as } n \to \infty$}
    \\[10pt]
    \small Equivalently: $\;\bar{X} \;\dot\sim\; N\!\left(\mu, \;\frac{\sigma^2}{n}\right)$ \; for large $n$.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.3cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{The magic:} It doesn't matter what the population looks like ---\\
  uniform, exponential, Bernoulli, bimodal, anything.\\[2pt]
  As long as the variance is finite, $\bar{X}$ becomes Normal.
}}
\end{center}
\end{frame}

\frametitle{CLT: What It Does \textbf{Not} Say}

\small
The CLT is powerful, but often over-applied. Three common misconceptions:

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}[
  mbox/.style={draw=warnred!60, fill=warnred!5, rounded corners=4pt, text width=12cm, align=left, inner sep=6pt, font=\small}

  };
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}

  {\large The most important number after the estimate itself}

\begin{frame}
\frametitle{SD $\neq$ SE: The Most Confused Pair in Statistics}

\begin{center}
\begin{tikzpicture}[
  cbox/.style={draw, rounded corners=6pt, minimum height=3.5cm, text width=5.5cm, align=center, inner sep=8pt}
]
  \node[cbox, fill=popblue!8] at (-4.2, 0) {
    \textbf{\large\textcolor{popblue}{Standard Deviation (SD)}}\\[8pt]
    \small
    Spread of the \textbf{data}\\[4pt]
    $\sigma = \sqrt{E[(X - \mu)^2]}$\\[6pt]
    ``How variable are individual\\observations?''\\[6pt]
    \footnotesize\textbf{Does not shrink with $n$.}\\
    More data doesn't make\\individuals less variable.
  };

  \node[cbox, fill=violet1!8] at (4.2, 0) {
    \textbf{\large\textcolor{violet1}{Standard Error (SE)}}\\[8pt]
    \small
    Spread of the \textbf{estimator}\\[4pt]
    $\text{SE}(\bar{X}) = \sigma / \sqrt{n}$\\[6pt]
    ``How variable is $\hat\theta$ across\\repeated experiments?''\\[6pt]
    \footnotesize\textbf{Shrinks with $n$.}\\
    More data $\Rightarrow$ more\\precise estimate.
  };

  \draw[thick, gray, dashed] (0, -2.5) -- (0, 2.5);
\end{tikzpicture}
\end{center}

\pause
\vspace{0.1cm}
\centering\small
\textbf{Example:} Heights of adults: SD $\approx$ 7\,cm (people vary).\\
Average of 100 heights: SE $= 7/\sqrt{100} = 0.7$\,cm (the average is precise).
\end{frame}

\begin{frame}
\frametitle{Standard Error of the Sample Mean}

\begin{center}
\begin{tikzpicture}
  \node[draw=orange1, fill=orange1!5, rounded corners=10pt, text width=12cm, align=center, inner sep=15pt] {
    {\Large
    $\text{SE}(\bar{X}) = \frac{\sigma}{\sqrt{n}}$
    }\\[12pt]
    \small The \textbf{standard error} is the standard deviation of the sampling distribution.\\[4pt]
    It measures how much $\bar{X}$ would vary if we repeated the experiment.
  };
\end{tikzpicture}
\end{center}

\pause
\vspace{0.2cm}
\small\textbf{Derivation:}
\vspace{-0.1cm}
$$\text{Var}(\bar{X}) = \text{Var}\!\left(\frac{1}{n}\sum_{i=1}^n X_i\right) = \frac{1}{n^2}\sum_{i=1}^n \text{Var}(X_i) = \frac{1}{n^2} \cdot n\sigma^2 = \frac{\sigma^2}{n}$$

\pause
\vspace{-0.15cm}
$$\Rightarrow \quad \text{SE}(\bar{X}) = \sqrt{\text{Var}(\bar{X})} = \frac{\sigma}{\sqrt{n}}$$

\vspace{0.1cm}
\centering\small
Uses i.i.d.\ assumption: observations are independent, so variances \textbf{add}.
\end{frame}

\begin{frame}
\frametitle{SE in Practice: A Polling Example}

\small
\textbf{Scenario:} A poll of $n = 1{,}000$ voters finds $\hat{p} = 0.52$ support for a candidate.

\pause
\vspace{0.15cm}
\textbf{Step 1:} Compute the estimated SE.
$$\widehat{\text{SE}}(\hat{p}) = \sqrt{\frac{\hat{p}(1-\hat{p})}{n}} = \sqrt{\frac{0.52 \times 0.48}{1000}} \approx 0.016$$

\pause
\vspace{-0.2cm}
\textbf{Step 2:} Interpret. The estimate is $\hat{p} = 0.52 \pm 0.016$.

\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \draw[thick, ->] (0, 0) -- (12, 0) node[right, font=\small] {$p$};
  % SE region
  \fill[violet1!15] (4.3, 0) rectangle (7.7, 1.2);
  \draw[very thick, violet1] (6, 0) -- (6, 1.3);
  \node[above, font=\small\bfseries, violet1] at (6, 1.3) {$\hat{p} = 0.52$};
  \draw[{Stealth}-{Stealth}, thick, orange1] (4.3, 0.6) -- (7.7, 0.6) node[midway, above, font=\scriptsize] {$\pm\, 2 \times \text{SE} \approx \pm\, 0.032$};
  \draw[dashed, sampred, thick] (5.3, -0.3) -- (5.3, 1.0);

  \node[font=\scriptsize] at (6, -0.6) {0.50 is within $\pm 2$SE --- race is too close to call!};
\end{tikzpicture}
\end{center}

\pause
\vspace{-0.1cm}
\begin{center}
\fcolorbox{popblue}{popblue!5}{\parbox{10cm}{\centering\small
  \textbf{Rule of thumb:} always report $\hat\theta \pm \text{SE}$. An estimate without a standard error is like a measurement without units --- it tells you nothing about reliability.
}}
\end{center}
\end{frame}

\begin{frame}
\frametitle{The $t$-Distribution: What It Is and Why It Exists}

\small
\textbf{The problem:} When $\sigma$ is known, $\frac{\bar{X} - \mu}{\sigma/\sqrt{n}} \sim N(0,1)$ exactly (for Normal data).

\vspace{0.1cm}
But when we \textbf{replace $\sigma$ with $S$}, the ratio is \textit{no longer} Normal --- it has \textbf{heavier tails}:
$$T = \frac{\bar{X} - \mu}{S/\sqrt{n}} \;\sim\; t_{n-1}$$

\pause
\vspace{-0.1cm}
\textbf{Why heavier tails?} Sometimes $S$ underestimates $\sigma$, making $T$ too large. Sometimes it overestimates, making $T$ too small. This \textbf{extra randomness from $S$} inflates the tails.

\pause
\vspace{0.1cm}
\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    width=12cm, height=4.5cm,
    xlabel={$t$},
    ylabel={Density},
    xmin=-5, xmax=5,
    ymin=0, ymax=0.42,
    axis lines=left,
    legend style={at={(0.98,0.95)}, anchor=north east, font=\small},
    grid=major, grid style={gray!15},
    every axis plot/.append style={line width=1.8pt, samples=100}
  ]
    % t with df=3
    \addplot[sampred, domain=-5:5] {0.3676 * (1 + x^2/3)^(-2)};
    \addlegendentry{$t_3$ (heavy tails)}

    % t with df=10
    \addplot[orange1, domain=-5:5] {0.3891 * (1 + x^2/10)^(-5.5)};
    \addlegendentry{$t_{10}$ (moderate)}

    % Normal
    \addplot[popblue, dashed, domain=-5:5] {exp(-x^2/2)/sqrt(2*pi)};
    \addlegendentry{$N(0,1)$ (the limit)}
  \end{axis}
\end{tikzpicture}
\end{center}

\vspace{-0.15cm}
\centering\small
Fewer observations ($n$ small) $\Rightarrow$ $S$ is noisy $\Rightarrow$ $t_{n-1}$ has fat tails.\\
More observations ($n$ large) $\Rightarrow$ $S \approx \sigma$ $\Rightarrow$ $t_{n-1} \to N(0,1)$. \; By $n \geq 30$, nearly identical.
\end{frame}

\begin{frame}
\frametitle{The $t$-Distribution: Key Facts}

\begin{center}
\begin{tikzpicture}[
  fbox/.style={draw=popblue!60, fill=popblue!5, rounded corners=4pt, minimum width=12.5cm, align=left, inner sep=6pt, font=\small}
]
  \node[fbox] at (0, 2.4) {\textbf{Shape:} Symmetric, bell-shaped, like a Normal but with \textbf{heavier tails}.};
  \node[fbox] at (0, 1.6) {\textbf{Parameter:} Degrees of freedom $\nu = n - 1$. Smaller $\nu$ $\Rightarrow$ heavier tails.};
  \node[fbox] at (0, 0.8) {\textbf{Mean:} $0$ (for $\nu > 1$). \;\; \textbf{Variance:} $\nu/(\nu-2)$ (for $\nu > 2$), slightly $> 1$.};
  \node[fbox] at (0, 0.0) {\textbf{Convergence:} As $\nu \to \infty$, $\;t_\nu \to N(0,1)$. For $\nu \geq 30$, the difference is tiny.};
  \node[fbox] at (0, -0.8) {\textbf{When to use:} Normal data, unknown $\sigma$. This is the standard setting in practice.};
\end{tikzpicture}
\end{center}

\vspace{0.2cm}
\begin{center}
\fcolorbox{orange1}{orange1!5}{\parbox{11cm}{\centering\small
  \textbf{Why ``degrees of freedom'' $= n - 1$?}\\[2pt]
  We estimate $\sigma$ using $S^2 = \frac{1}{n-1}\sum(X_i - \bar{X})^2$. The deviations from $\bar{X}$ must sum to $0$, so only $n-1$ of them are ``free.'' This lost degree of freedom is exactly what makes $T$ not Normal.
}}
\end{center}
\end{frame}
