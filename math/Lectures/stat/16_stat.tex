\begin{frame}
\frametitle{Trick \#2: The 3D Pictograph}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\vspace{0.3cm}
\textbf{The problem with 3D:}
\begin{itemize}
  \item Perspective makes back slices look \textbf{smaller}
  \item Front slices appear \textbf{larger}
  \item Tilting distorts all proportions
  \item Looks fancy $\Rightarrow$ harder to read
\end{itemize}

\vspace{0.3cm}
\textbf{The pictograph trick:}
\begin{itemize}
  \item Double the \textit{height} of an icon
  \item But it also doubles the \textit{width}
  \item $2\times$ difference $\to$ looks $4\times$ bigger (area)
  \item In 3D: looks $8\times$ bigger (volume!)
\end{itemize}
\end{column}
\begin{column}{0.45\textwidth}
\begin{center}
\vspace{0.2cm}
{\small \textbf{``Company B is twice as big''}}
\vspace{0.2cm}

\begin{tikzpicture}[scale=0.7]
  % Company A - small bag
  \node[font=\scriptsize, above] at (0.6, 2.4) {Company A};
  \node[font=\scriptsize, below] at (0.6, -0.1) {\$100M};
  \fill[orange1!60] (0.2,0) -- (0,2) -- (1.2,2) -- (1.0,0) -- cycle;

\begin{frame}
\frametitle{Why Does This Happen?}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
\vspace{0.2cm}
\textbf{The multiple comparisons trap:}

\vspace{0.2cm}
\begin{itemize}
  \item Tyler Vigen tested $\sim$25{,}000 variable pairs
  \item At $\alpha = 0.05$: expect $\sim$1{,}250 ``significant'' results \textbf{by chance alone}
  \item Only show the best ones $\Rightarrow$ instant ``discovery''
\end{itemize}

\vspace{0.3cm}
\textbf{The formula for nonsense:}

\begin{tikzpicture}
\node[draw=warnred, fill=warnred!5, rounded corners=5pt, inner sep=6pt, font=\small] {
  $P(\text{at least 1 false positive}) = 1 - (1-\alpha)^m$
};
\end{tikzpicture}

\vspace{0.2cm}
With $m = 100$ tests at $\alpha = 0.05$: \\
$P = 1 - 0.95^{100} = 0.994$ \quad (\textcolor{sampred}{99.4\%!})

\begin{frame}
\frametitle{The Prosecutor's Fallacy}

\begin{columns}[T]
\begin{column}{0.5\textwidth}
\vspace{0.2cm}
\textbf{Sally Clark case (1999, UK):}
\begin{itemize}
  \item Two of her babies died (SIDS)
  \item Expert witness: ``The probability of two SIDS deaths is 1 in 73 million''
  \item She was \textbf{convicted of murder}
\end{itemize}

\vspace{0.2cm}
\textbf{The error:}
\begin{itemize}
  \item $P(\text{evidence}|\text{innocent})$ is small
  \item But that's NOT $P(\text{innocent}|\text{evidence})$!
  \item Double murder of own babies is \textit{also} extremely rare
\end{itemize}

\vspace{0.1cm}
{\small She was \textcolor{paramgreen}{\textbf{exonerated}} in 2003. \\
The expert was found guilty of misconduct.}
\end{column}
\begin{column}{0.45\textwidth}
\vspace{0.3cm}
\begin{tikzpicture}
\node[draw=warnred, fill=warnred!5, rounded corners=8pt, text width=5.5cm, inner sep=8pt] {
  \textbf{\textcolor{warnred}{The Fallacy:}} \\[6pt]
  \begin{tabular}{@{}l@{}}
  $P(\text{data}|\text{innocent})$  \\[3pt]
  \hspace{1cm}$\neq$ \\[3pt]
  $P(\text{innocent}|\text{data})$
  \end{tabular} \\[8pt]
  This is exactly \textbf{Bayes' theorem}: \\[3pt]
  $P(H|\text{data}) = \dfrac{P(\text{data}|H)\,P(H)}{P(\text{data})}$ \\[8pt]
  You must consider the \textbf{prior} $P(H)$ --- how common is double murder?
};
\end{tikzpicture}
\end{column}

  \item \textbf{Mean vs.\ median salary:} A company reports ``average employee compensation is \$120K.'' You discover that 2 executives earn \$500K and 48 other employees earn \$X each, and the median salary is \$65K. Find \$X. Which average should the company report and why?

  \item \textbf{The base rate problem:} An AI system detects shoplifters with 99.5\% accuracy and a 1\% false positive rate. In a mall with 10{,}000 visitors/day, where 0.1\% actually shoplift, how many innocent people get flagged? Should the mall use this system?
\end{enumerate}
