\section{Why we need impurity}
\sectiontransition{Why we need impurity}{To grow a tree we must score a split: how much purer did the node get?}

\begin{frame}{Feature order matters}
  We started with Outlook. What if we had started with \textbf{Temperature}?
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Outlook first} (our tree)
    \begin{itemize}\small
      \item Today $\to$ Sunny $\to$ High $\to$ \textcolor{armred}{No}.
      \item Leaf backed by \textbf{3 rows}.
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Temperature first} (another tree)
    \begin{itemize}\small
      \item Today $\to$ Cool $\to$ Sunny $\to \dots$ only \textbf{1 matching row}
        (D9, which was Normal/Weak) $\to$ \textcolor{paramgreen}{Yes}.
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    Same data, \textbf{different feature order $\Rightarrow$ different tree $\Rightarrow$
    different prediction} for today (No vs Yes). Which one should we trust?}}
\end{frame}

\begin{frame}{Predict-first: which tree is better?}
  One tree predicts \textbf{No} from a leaf resting on \textbf{3} weekends; the other
  predicts \textbf{Yes} from a leaf resting on \textbf{1} weekend (that also differed in
  humidity and wind).
  \vfill
  \textbf{Which prediction would you trust?}
  \pause
  \vfill
  The one backed by \textbf{more rows}. A leaf covering many consistent examples
  generalizes; a leaf covering one example is memorizing.
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Smaller trees usually generalize better} -- their leaves rest on more rows
    (higher coverage). A good split order reaches \emph{pure, well-covered} leaves
    \emph{sooner}. (We will make this precise as overfitting in Section 5.)}}
\end{frame}

\begin{frame}{So: split on whatever increases purity the most}
  We want pure leaves \emph{sooner} $\Rightarrow$ at each node, pick the feature whose
  split makes the children \textbf{as pure as possible}. We need a number for how
  \textbf{impure} (mixed) a leaf is -- \emph{zero} when it is all one class, \emph{large}
  when it is 50/50:
  \vfill
  \begin{center}
    \includegraphics[width=0.66\linewidth]{fig/impurity_pair.pdf}
  \end{center}
  \vfill
  Two standard measures do exactly this: \textbf{Gini} and \textbf{entropy}.
\end{frame}

\section{Impurity measures}
\sectiontransition{Impurity measures}{Gini and entropy: two ways to put a number on how mixed a node is.}

\begin{frame}{Gini impurity (and its famous cousin)}
  \begin{columns}[T]
    \column{0.54\textwidth}
    \[
      \text{Gini} = 1 - \sum_k p_k^2 \quad(\text{binary: } 2p(1-p)).
    \]
    \textbf{Reading it:} draw \emph{two} points at random (with replacement); Gini $=$
    P(they are \textbf{different} classes).
    \vskip 2pt
    50/50: P(differ) $= 0.5{\cdot}0.5 + 0.5{\cdot}0.5 = 0.5$ (red-then-blue or
    blue-then-red) $= 1-\sum_k p_k^2$.
    \begin{itemize}\small
      \item pure: the two draws always match $\Rightarrow$ Gini $0$.
      \item 50/50: $0.5$ (maximally mixed); \; 80/20: $0.32$.
    \end{itemize}
    \column{0.44\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=3.7cm,
          xlabel={$p$ (fraction positive)}, ylabel={Gini},
          xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
          xmin=0, xmax=1, ymin=0, ymax=0.55, xtick={0,0.5,1}, ytick={0,0.25,0.5},
          axis line style={gray!60}]
          \addplot[armblue, very thick, domain=0:1, samples=60] {2*x*(1-x)};
          \draw[dashed, gray] (axis cs:0.5,0) -- (axis cs:0.5,0.5);
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vskip 2pt
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{The economic cousin.} Named after Corrado Gini (1912). The economics
    \emph{Gini index} of income inequality is a sibling measure -- both ask
    \emph{``how different are two random members of a population?''} (here, their
    class; in economics, their income). Same idea, different formula.}}
\end{frame}

\begin{frame}{Entropy and information gain}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Entropy} (in bits) is another impurity measure:
    \[
      H = -p\log_2 p - (1-p)\log_2(1-p).
    \]
    The split's value is the \textbf{information gain} -- impurity removed:
    \[
      \mathrm{IG} = H_{\text{parent}} - \!\!\sum_{\text{child } c} \frac{n_c}{n}\, H_c .
    \]
    \footnotesize Same idea for Gini (``Gini gain''). Pick the split with the biggest
    drop.
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4.2cm,
          xlabel={$p$}, ylabel={impurity},
          xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
          xmin=0, xmax=1, ymin=0, ymax=1.05, xtick={0,0.5,1}, ytick={0,0.5,1},
          axis line style={gray!60},
          legend style={font=\tiny, at={(0.5,0.28)}, anchor=north}, legend cell align=left]
          \addplot[armred, very thick, domain=0.001:0.999, samples=80] {-(x*ln(x)/ln(2)) - ((1-x)*ln(1-x)/ln(2))};
          \addlegendentry{entropy (max 1)}
          \addplot[armblue, very thick, domain=0:1, samples=60] {2*x*(1-x)};
          \addlegendentry{Gini (max 0.5)}
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vskip 2pt
  \footnotesize Both peak at $p=0.5$ and vanish at the pure ends -- they mostly agree.
\end{frame}

\begin{frame}{Akinator: 20 questions is information gain in disguise}
  \begin{columns}[T]
    \column{0.52\textwidth}
    The web genie \textbf{Akinator} guesses your character from yes/no questions -- it keeps a
    \textbf{candidate set} and asks whatever \textbf{shrinks it fastest}, i.e.\ maximizes
    information gain.
    \vskip 4pt
    Pool of 8 Armenians (Aznavour, Komitas, Petrosyan, Aivazovsky, \dots),
    $H_{\text{parent}} = \log_2 8 = 3$ bits:
    \begin{itemize}\small
      \item \textbf{``Musician?''} $4/4$: children $2$ bits each, $\mathrm{IG} = \mathbf{1}$ bit.
      \item \textbf{``Is it Aivazovsky?''} $1/7$: $\mathrm{IG} = 3 - 2.46 = \mathbf{0.54}$ bit.
    \end{itemize}
    A \textbf{balanced} question buys a whole bit; a narrow one barely helps.
    \column{0.46\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/akinator_ig.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\footnotesize
    A tree splits the \emph{same} way -- pick the question that removes the most uncertainty.
    \textbf{20 balanced questions} separate $2^{20} \approx 1{,}000{,}000$ characters: why
    \emph{informative} splits build \emph{short} trees.}}
\end{frame}

\begin{frame}{Worked by hand: Gini gain of the Outlook split}
  \textbf{Parent} node (9 Yes, 5 No):
  \[
    \text{Gini} = 1 - \left(\tfrac{9}{14}\right)^2 - \left(\tfrac{5}{14}\right)^2
      = 1 - 0.413 - 0.128 = \mathbf{0.459}.
  \]
  \textbf{Children} after splitting on Outlook:
  \vskip -4pt
  \begin{center}\small
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{l c c}
      child & counts & Gini \\
      \hline
      Sunny & 2 Yes, 3 No & $1-(\tfrac{2}{5})^2-(\tfrac{3}{5})^2 = 0.48$ \\
      Overcast & 4 Yes, 0 No & $1-1-0 = 0$ \;(pure!) \\
      Rain & 3 Yes, 2 No & $1-(\tfrac{3}{5})^2-(\tfrac{2}{5})^2 = 0.48$ \\
      \hline
    \end{tabular}
  \end{center}
  \vskip 2pt
  \textbf{Weighted child impurity} (weight $=$ child rows $/14$ -- a leaf covering more
  rows counts more; this is the \emph{expected} impurity after the split), then the gain:
  \[
    \tfrac{5}{14}(0.48) + \tfrac{4}{14}(0) + \tfrac{5}{14}(0.48) = 0.343
    \;\Rightarrow\; \text{gain} = 0.459 - 0.343 = \mathbf{0.116}.
  \]
  \footnotesize Overcast is a \emph{free pure leaf} ($\text{Gini}=0$) -- that is why
  Outlook scores so well. Do this for every feature; keep the biggest gain.
\end{frame}

\begin{frame}{Information gain, visually: uncertainty removed}
  A split's \textbf{information gain} $=$ the drop in entropy from the parent to its
  (row-weighted) leaves. Splitting the 14 weekends on \texttt{Humidity}:
  \vskip 2pt
  \begin{center}
    \includegraphics[width=0.82\linewidth]{fig/ig_split.pdf}
  \end{center}
  \vskip -2pt
  \footnotesize Parent uncertainty $0.94$ bits $\to$ weighted leaves $0.79$ bits, so
  $\mathrm{IG} = 0.94 - 0.79 = 0.15$. CART scores \emph{every} feature this way and keeps
  the biggest gain -- the next slide does exactly that.
\end{frame}

\begin{frame}{Worked split: which feature is the best root?}
  Parent entropy (9 Yes, 5 No): $H = -\tfrac{9}{14}\log_2\tfrac{9}{14} - \tfrac{5}{14}\log_2\tfrac{5}{14} = 0.940$ bits. Information gain of each feature:
  \vfill
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l c}
      split on & information gain \\
      \hline
      \textbf{Outlook} & \textbf{0.247} \quad$\leftarrow$ winner (Overcast is pure) \\
      Humidity & 0.152 \\
      Wind & 0.048 \\
      Temperature & 0.029 \\
      \hline
    \end{tabular}
  \end{center}
  \vfill
  \textbf{Outlook} removes the most impurity, so CART makes it the root -- exactly the
  tree we drew by hand.
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\scriptsize
    These numbers use \textbf{entropy / information gain}. sklearn defaults to
    \textbf{Gini} -- it picks the same winner here and gives a near-identical tree.
    Do not be surprised the criterion silently changes between this slide and the
    sklearn one.}}
\end{frame}

\begin{frame}{The trap: information gain loves many-valued features}
  \textbf{Predict-first:} add a \texttt{Day} column (14 distinct dates, one per row) to the
  weather data. Which feature does information gain pick as the root?
  \pause
  \vfill
  \textbf{Day wins} - and it is useless. Splitting on Day gives 14 pure single-row leaves, so its
  children have zero impurity:
  \[
    \mathrm{IG}(\texttt{Day}) = H_{\text{parent}} - 0 = 0.940 \text{ bits} \;>\;
    \mathrm{IG}(\texttt{Outlook}) = 0.247.
  \]
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    That 14-way split is the \textbf{ID3} picture. \textbf{sklearn's CART splits only
    binary}, so it never carves 14 leaves at once -- \emph{but} a many-valued feature still
    offers \textbf{many candidate cut-points}, so raw gain stays \textbf{biased toward
    high-cardinality features} (even a useless ID), just milder. Fixes: \textbf{gain ratio}
    (C4.5) normalizes by the split's own entropy; and high-cardinality categoricals need
    care (CatBoost's target encoding, [20]).}}
\end{frame}

\begin{frame}{Regression trees: same idea, numeric target}
  Predict a \emph{number} (apartment rent from area). Split to reduce \textbf{spread}
  (variance / SSE) instead of class impurity; each leaf predicts the \textbf{mean} of
  its rows.
  \vskip -2pt
  \begin{columns}[T]
    \column{0.46\textwidth}
    \textbf{Predict-first:} what does the tree predict for an area \emph{between} two
    training points?
    \pause
    \vskip 2pt
    The leaf \textbf{mean} -- a \textbf{flat step}, not a smooth line. A tree's
    prediction is piecewise-constant.
    \vskip 4pt
    \footnotesize Deeper tree $\Rightarrow$ more steps $\Rightarrow$ closer fit (and,
    eventually, overfitting). Seeds the gradient-boosting residual idea in [19].
    \column{0.5\textwidth}

\begin{frame}{Regression split: reduce the spread, predict the mean}
  No Gini for a number -- a split is scored by how much it \textbf{reduces spread}
  (variance), and each \textbf{leaf predicts the mean} of its rows.
  \vskip 2pt
  \begin{center}

\begin{frame}{The CART algorithm}
  \begin{columns}[T]
    \column{0.54\textwidth}
    \fcolorbox{armblue}{armblue!6}{\parbox{0.95\linewidth}{\small
    \textbf{Grow(node):}
    \begin{enumerate}\small
      \item If pure or a stop rule fires $\to$ make a leaf (majority class / mean).
      \item Else try every (feature, split) and score by impurity drop.
      \item Keep the best split; send rows to the children.
      \item \textbf{Recurse} on each child.
    \end{enumerate}}}
    \column{0.42\textwidth}
    \textbf{Numeric features} can't split ``by value'' -- they need a \textbf{threshold}
    (\texttt{age > 28}?). How that threshold is chosen is the next two slides.
    \vskip 4pt
    \footnotesize Play Tennis is all-categorical, so the threshold mechanic only shows
    up on numeric data (e.g.\ Titanic \texttt{age} / \texttt{fare}).
  \end{columns}
  \vfill
  \footnotesize CART = \emph{Classification And Regression Trees}: same recursion, only
  the impurity (Gini/entropy vs variance) and the leaf output (class vs mean) change.
\end{frame}

\begin{frame}{Choosing a split on a numeric feature}
  A numeric feature needs a \textbf{threshold}. The exact recipe:
  \begin{enumerate}\small
    \item Sort the rows by the feature.
    \item Candidate thresholds $=$ \textbf{midpoints} between adjacent values.
    \item Score each by impurity drop; keep the best.
  \end{enumerate}
  \vskip 2pt
  \begin{columns}[T]
    \column{0.54\textwidth}
    \small Example -- feature \texttt{age}, label survived (1) / died (0), sorted:
    \vskip 3pt
    \begin{tabular}{l c c c c c}
      \texttt{age} & 20 & 25 & 30 & 40 & 50 \\
      $y$ & 0 & 0 & 1 & 1 & 1 \\
    \end{tabular}
    \vskip 5pt
    candidate midpoints: $22.5,\ \mathbf{27.5},\ 35,\ 45$.
    \column{0.43\textwidth}
    \small Gini gain (parent Gini $= 0.48$):
    \vskip 3pt
    \renewcommand{\arraystretch}{1.15}
    \begin{tabular}{c c}
      threshold & gain \\
      \hline
      22.5 & 0.18 \\
      \textbf{27.5} & \textbf{0.48} \\
      35 & 0.21 \\
      45 & 0.08 \\
      \hline
    \end{tabular}
  \end{columns}
  \vfill
  \footnotesize \texttt{age > 27.5} splits it perfectly (two pure leaves).
  \emph{Shortcut:} only thresholds where the \emph{label changes} can ever win, so you
  skip the rest.
\end{frame}

\begin{frame}{Watch one split get chosen}
  \only<1>{Ten students, one feature (\texttt{hours studied}); the whole node is
    \textbf{mixed}. A split sends points \textbf{left / right} into two child leaves -- which
    threshold makes those leaves \textbf{purest}?}%
  \only<2-6>{Each threshold splits the node into a \textbf{left leaf} and a \textbf{right leaf}.
    \textbf{Gini gain} $=$ parent $-$ the row-weighted average of the two leaf Ginis.}%
  \only<7>{\textbf{hours $\leq 5.5$ wins}: a \emph{pure} left leaf (Gini $0$) and a mostly-pass
    right leaf give the biggest gain. CART keeps this split, then \textbf{recurses} on each
    leaf.}%
  \vskip 2pt
  \begin{center}
    \only<1>{\includegraphics[width=0.82\linewidth]{fig/split_anim_1.pdf}}%
    \only<2>{\includegraphics[width=0.82\linewidth]{fig/split_anim_2.pdf}}%
    \only<3>{\includegraphics[width=0.82\linewidth]{fig/split_anim_3.pdf}}%
    \only<4>{\includegraphics[width=0.82\linewidth]{fig/split_anim_4.pdf}}%
    \only<5>{\includegraphics[width=0.82\linewidth]{fig/split_anim_5.pdf}}%
    \only<6>{\includegraphics[width=0.82\linewidth]{fig/split_anim_6.pdf}}%
    \only<7>{\includegraphics[width=0.82\linewidth]{fig/split_anim_7.pdf}}%

    \textcolor{armblue}{$\bullet$}~passed}

\begin{frame}{Scaling up: quantile and histogram splits}
  Exact search sorts every feature and tests $\sim n$ thresholds \emph{at every node} --
  fine for thousands of rows, painful for millions. Two faster approximations:
  \vfill
  \begin{itemize}\small
    \item \textbf{Quantile / percentile splits.} Instead of all midpoints, only test
      thresholds at the feature's quantiles (e.g.\ the 32 or 256 percentile cut-points).
      Far fewer candidates, near-identical splits. (XGBoost's ``approx'' / weighted
      quantile sketch.)
    \item \textbf{Histogram-based splits.} \emph{Bin} each feature into a fixed number of
      bins (often 255) \textbf{once}; a split then only scans bin edges and sums per-bin
      counts. Cost drops from $O(n)$ to $O(\#\text{bins})$, and the bins are reused at
      every node. This is how \textbf{LightGBM}, \texttt{HistGradientBoosting}, and
      XGBoost (\texttt{hist}) get their speed (callback: the LightGBM frame).
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Same idea as exact search -- just \emph{fewer} candidate thresholds. A tiny accuracy
    cost (coarser cut-points) buys a big speed-up on large data; for small data, exact
    is fine.}}
\end{frame}

\begin{frame}{Greedy, not optimal}
  \begin{columns}[T]
    \column{0.5\textwidth}
    CART is \textbf{greedy}: at each node it takes the split that looks best \emph{right
    now} and never reconsiders.
    \begin{itemize}\small
      \item Finding the \emph{smallest} tree that fits is NP-hard -- we cannot search all trees.
      \item ``Increase purity most'' is a cheap \textbf{heuristic} for ``end up small''.
      \item It can \textbf{miss a split that only pays off later} -- like XOR (right): one
        split gains \emph{nothing}, so a greedy ``stop when no split helps'' rule would quit.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/tree_greedy_xor.pdf}\\[-2pt]
      \scriptsize Depth 1 gains nothing ($\approx$ coin flip); grow deeper and CART does
      separate XOR. The useless-looking first split is the real greedy caveat.
    \end{center}
  \end{columns}
\end{frame}

\section{Overfitting and pruning}
\sectiontransition{Overfitting and pruning}{A tree left unchecked memorizes the training set; pruning buys back generalization.}

\begin{frame}{Predict-first: does an unrestricted tree overfit?}
  \begin{columns}[T]
    \column{0.45\textwidth}
    Grow with \textbf{no depth limit}. What happens to train vs test accuracy?
    \pause
    \vskip 4pt
    On Titanic the unrestricted tree grows to \textbf{depth 24, 316 leaves}:
    \begin{itemize}\small
      \item train accuracy \textbf{96.5\%} -- near-perfect, it memorizes.
      \item test (5-fold CV) only \textbf{76\%} -- \emph{worse} than the depth-1
        ``\texttt{gender}'' stump (78\%).
    \end{itemize}
    A 20-point train--test gap: textbook overfitting.
    \column{0.53\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/titanic_overfit.pdf}\\[-2pt]
      \scriptsize Depth is the complexity knob (callback: cross-validation). Train keeps rising;
      test peaks early, then falls.
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\footnotesize
    \textbf{Only unrestricted depth overfits.} A \emph{shallow} tree (depth 2--3,
    $\approx$80\% here) is the healthy baseline -- the depth-2 Titanic tree we draw later
    is \emph{not} overfit. Depth is a dial: too deep memorizes, too shallow underfits.}}
\end{frame}

\begin{frame}{Pre-pruning: stop growing early}
  Set limits \emph{before/while} growing -- refuse splits that go too far:
  \vfill
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l p{0.58\linewidth}}
      \textbf{sklearn parameter} & \textbf{stops a split when\dots} \\
      \hline
      \texttt{max\_depth} & the tree is already this deep \\
      \texttt{min\_samples\_split} & the node has too few rows to split \\
      \texttt{min\_samples\_leaf} & a child would be too small \\
      \texttt{max\_leaf\_nodes} & the tree already has enough leaves \\
      \texttt{min\_impurity\_decrease} & the split barely helps \\
      \hline
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize Cheap, needs no extra data. Risk: too aggressive $\Rightarrow$
  underfit; it may stop before a good \emph{later} split.
\end{frame}

\begin{frame}{Post-pruning: grow full, then cut back}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Grow a big tree, then \textbf{remove} subtrees that do not earn their keep.
    \begin{itemize}\small
      \item \textbf{Reduced-error pruning}: cut a subtree if it does not help on a
        hold-out set.
      \item \textbf{Cost-complexity pruning} (\texttt{ccp\_alpha}): minimize
        $\text{error} + \alpha\,(\#\text{leaves})$. Larger $\alpha \Rightarrow$ smaller
        tree.
    \end{itemize}
    \vskip 2pt
    \footnotesize On Titanic, pruning lifts \textbf{test accuracy to 83.5\%} at the best
    \texttt{ccp\_alpha} -- selective cuts beat a uniform depth cap.
    \column{0.52\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/titanic_pruning.pdf}\\[-2pt]
      \scriptsize Test peaks at the right $\alpha$; over-pruning then underfits.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Pre- vs post-pruning, and how to choose}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Pre-pruning}
    \begin{itemize}\small
      \item Fast; no extra data.
      \item Can stop too early (misses good later splits).
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Post-pruning}
    \begin{itemize}\small
      \item Slower; needs data.
      \item Can undo bad greedy splits.
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Either way, pick the knob by cross-validation}. Sweep
    \texttt{max\_depth} or \texttt{ccp\_alpha}, keep the value with the best CV score.
    The depth U-curve is the same picture as the cross-validation lecture's complexity-knob plot.}}
\end{frame}

\begin{frame}{Key hyperparameters and their impact}
  Almost every tree knob controls \textbf{one thing -- complexity}. Here they are
  together, with which way each pushes the over/underfit dial. \textbf{Tune by CV}.
  \vskip 2pt
  \begin{center}\footnotesize
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{l l l}
      hyperparameter & what it controls & raising it $\Rightarrow$ \\
      \hline
      \texttt{max\_depth} & longest question chain & more complex (\textcolor{armred}{overfit}) \\
      \texttt{min\_samples\_leaf} & smallest allowed leaf & simpler (\textcolor{armblue}{regularizes}) \\
      \texttt{min\_samples\_split} & rows needed to split & simpler \\
      \texttt{max\_leaf\_nodes} & total \# of leaves & more complex \\
      \texttt{min\_impurity\_decrease} & min gain to allow a split & simpler \\
      \texttt{ccp\_alpha} & post-pruning strength & simpler (more pruned) \\
      \texttt{criterion} & \texttt{gini} / \texttt{entropy} & $\approx$ no accuracy effect (gini faster) \\
      \texttt{max\_features} & features tried per split & $<$ all adds randomness (mainly RF, [18]) \\
      \hline
    \end{tabular}
  \end{center}
  \vskip 2pt
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{In practice:} tune \texttt{max\_depth} (or \texttt{ccp\_alpha}) and
    \texttt{min\_samples\_leaf} by CV -- those two do most of the work.
    \texttt{criterion} barely matters; \texttt{class\_weight="balanced"} handles
    imbalance (a separate concern, not complexity).}}
\end{frame}

\section{In practice and bridge}
\sectiontransition{In practice and bridge}{Using trees in sklearn - and why one tree is never the end of the story.}

\begin{frame}[fragile]{Decision trees in sklearn (the default tool)}
\begin{lstlisting}
from sklearn.tree import DecisionTreeClassifier, plot_tree

tree = DecisionTreeClassifier(max_depth=3, ccp_alpha=0.0,
                              random_state=509)
tree.fit(X_train, y_train)        # Titanic features
plot_tree(tree, feature_names=cols, filled=True)
\end{lstlisting}
  \vfill
  On Titanic the tree usually asks \texttt{gender} first, then an \texttt{age} threshold --
  readable, you can show it to a stakeholder.
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{No scaling needed} -- trees only compare thresholds, so monotone rescaling
    changes nothing (callback: preprocessing). \emph{But} sklearn trees still need
    \textbf{numeric input}: encode categoricals yourself (they are \emph{not}
    auto-handled, despite folklore).}}
\end{frame}

\begin{frame}{CART speaks binary: what one-hot does to the tree}
  The hand-built tree split \texttt{Outlook} \textbf{three ways} at once (Sunny / Overcast / Rain) -
  that is ID3. sklearn's CART only ever asks \textbf{one yes/no question} per node.
  \vfill
  So you one-hot encode first, and CART recovers the \emph{same decisions} as a chain of binary
  splits:
  \[
    \texttt{Outlook\_Overcast} \le 0.5 \;?\quad\to\quad
    \texttt{Outlook\_Sunny} \le 0.5 \;?\quad\to\quad \dots
  \]
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{Same tree, different shape.} Do not be surprised when the tree you \emph{run} looks
    deeper and asks \texttt{Outlook\_Overcast <= 0.5} instead of a tidy 3-way Outlook node - the
    splits and predictions match, only the drawing differs.}}
\end{frame}

  \begin{columns}[T]

\begin{frame}{Axis-aligned means staircases}
  A tree can only cut \textbf{straight, axis-aligned} lines. A slanted boundary is
  approximated by a \textbf{staircase} -- finer (and more overfit) as depth grows.
  \begin{center}

  \end{center}
  \vskip -2pt
  \footnotesize The diagonal (red) is trivial for a line but awkward for a tree -- deeper
  trees chase it with more, smaller boxes. A hint of why we sometimes reach for other models.
\end{frame}

\begin{frame}{Trees cannot extrapolate}
  \begin{columns}[T]
    \column{0.46\textwidth}
    A leaf predicts the \textbf{mean of its training rows} -- a constant. Outside the
    training range there is no new leaf, so the prediction \textbf{flatlines}.
    \vskip 4pt
    A linear model keeps trending; a tree \emph{and every tree ensemble} does not. It
    matters whenever you must predict \emph{beyond} what you have seen (prices, demand, time).
    \column{0.54\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/tree_extrapolation.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Strengths and weaknesses}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Strengths}
    \begin{itemize}\small
      \item Interpretable -- read the rules.
      \item No scaling; mixed feature types.
      \item Captures interactions \& non-linearities.
      \item Automatic feature selection; fast.
      \item Missing values handled natively (no imputation; sklearn $\ge$ 1.3).
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Weaknesses}
    \begin{itemize}\small
      \item \textbf{High variance / unstable}: change a few rows $\to$ a very different
        tree.
      \item Greedy, not optimal.
      \item \textbf{Axis-aligned only}: a diagonal boundary needs a staircase of splits.
      \item \textbf{Never smooth}; extrapolates poorly (flat beyond the data).
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\scriptsize
    \textbf{Instability example:} on data like the Wisconsin breast-cancer set,
    dropping a \emph{single} row can flip many predictions and produce a visibly
    different tree. This variance is the headline weakness -- and the reason ensembles
    exist.}}
\end{frame}

\begin{frame}{Feature importance (read with care)}
  \begin{columns}[T]
    \column{0.46\textwidth}
    A tree scores each feature by \textbf{how much impurity its splits removed}
    (summed over the tree). Quick way to see what drove the model.
    \vskip 6pt
    \footnotesize\fcolorbox{armred}{armred!8}{\parbox{0.93\linewidth}{\scriptsize
      \textbf{Caveat:} this impurity-based importance is \textbf{biased toward
      high-cardinality} features (many split points). Prefer \emph{permutation
      importance}; SHAP comes later (interpretability).}}

\begin{frame}[fragile]{The tree is the atom of this chapter}
  A single tree is weak (high variance). Everything next is built from it.
\begin{lstlisting}

LGBMRegressor(n_estimators=1, learning_rate=1.0) # LightGBM

\section*{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item A decision tree = \textbf{nested yes/no questions}; a leaf gives the
      prediction (majority class or mean).
    \item Grow greedily (\textbf{CART}): at each node pick the split that drops
      \textbf{impurity} most (\textbf{Gini}/entropy for classification, variance for
      regression).
    \item Unlimited depth $\Rightarrow$ 0 training error but \textbf{overfit}; control
      with \textbf{pre-pruning} and \textbf{post-pruning} (\texttt{ccp\_alpha}), chosen
      by \textbf{CV}.
    \item Strengths: interpretable, no scaling. Weakness: \textbf{high variance}.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Workflow:} encode categoricals $\to$ fit a shallow tree as an interpretable
    baseline $\to$ tune \texttt{max\_depth}/\texttt{ccp\_alpha} by CV $\to$ read the
    rules / importances $\to$ if you need accuracy over interpretability, reach for
    ensembles ([18]--[20]).}}
\end{frame}
