\section{XGBoost: the regularized objective and the structure score}
\modeltransition{XGBoost}{Extreme gradient boosting: a regularized objective $+$ a closed-form
  structure score.}{2016}{Chen \& Guestrin (Univ.\ of Washington / DMLC)}{dmlc/xgboost}

\begin{frame}{XGBoost: extreme gradient boosting}
  A scalable tree-boosting system (Chen \& Guestrin, 2016):
  \begin{itemize}
    \item a \textbf{regularized objective} - penalties baked into the loss;
    \item \textbf{approximate / histogram} split finding for large data;
    \item row $+$ column \textbf{subsampling}; \textbf{sparsity-aware} splits;
    \item parallel split search and \textbf{GPU} support.
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    Often the top performer on tabular benchmarks - \textbf{if properly tuned}. Many knobs, and
    the defaults are rarely optimal.}}
\end{frame}

\begin{frame}{The plan, in plain words}
  Before any algebra, here is the whole idea:
  \begin{enumerate}
    \item \textbf{Score} how good a tree is by how much it would cut the loss - call it the
      \emph{structure score}.
    \item To get that score, \textbf{approximate the loss as a parabola} at the current prediction
      (a Newton step: use the slope \emph{and} the curvature).
    \item \textbf{Solve} that parabola for the best value in each leaf.
    \item \textbf{Grow greedily}: keep a split only if it raises the score by more than the cost of
      adding a leaf.
  \end{enumerate}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    The next slides just make each step precise. Hold on to this picture:
    \textbf{score a tree, then split to improve the score}.}}
\end{frame}

\begin{frame}{A regularized objective}
  At step $m$ we add one tree $b$ with leaf values $c_1,\dots,c_T$. XGBoost minimizes the loss
  \emph{plus} complexity penalties:
  \[
    \mathrm{obj} = \sum_{i=1}^{n} L\big(y_i,\, f^{[m-1]}(x_i) + b(x_i)\big)
      \;+\; \underbrace{\gamma\,T}_{\#\,\text{leaves}}
      \;+\; \underbrace{\tfrac12 \lambda \|c\|_2^2}_{L_2}
      \;+\; \underbrace{\alpha \|c\|_1}_{L_1}.
  \]
  \begin{itemize}\small
    \item $\gamma$ penalizes the \textbf{number of leaves} (tree size);
    \item $\lambda$ ($L_2$) and $\alpha$ ($L_1$) shrink the \textbf{leaf values}.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Contrast [19]: plain gradient boosting put \emph{no} penalty inside the objective. Here
    regularization is part of what every tree optimizes. (Software: \texttt{gamma},
    \texttt{lambda}, \texttt{alpha}.)}}
\end{frame}

\begin{frame}{Second-order Taylor of the loss}
  Expand the loss around the current prediction $f^{[m-1]}$, to \textbf{second order}:
  \[
    L\big(y,\, f^{[m-1]} + b\big) \;\approx\;
    L\big(y,\, f^{[m-1]}\big) + g\, b + \tfrac12 h\, b^2,
  \]
  with gradient and Hessian
  \[
    g = \frac{\partial L}{\partial f^{[m-1]}}, \qquad
    h = \frac{\partial^2 L}{\partial (f^{[m-1]})^2}.
  \]
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    $g$ is exactly the \textbf{negative pseudo-residual} from [19]. XGBoost's one real addition is
    the \textbf{curvature} $h$. \emph{Intuition:} picture a parabola hugging the loss - slope $g$
    points downhill, bend $h$ says how big a step to trust (a Newton step).}}
\end{frame}

\begin{frame}{Group the objective by leaf}
  Drop the constant $L(y,f^{[m-1]})$ and use that every point in leaf $t$ shares one value $c_t$:
  \[
    \mathrm{obj} \approx \sum_{t=1}^{T}\Big[\, G_t\, c_t + \tfrac12 (H_t + \lambda)\, c_t^2 \,\Big] + \gamma\,T,
  \]
  \[
    G_t = \!\!\sum_{i \in \text{leaf } t}\!\! g_i, \qquad H_t = \!\!\sum_{i \in \text{leaf } t}\!\! h_i .
  \]
  \vfill
  \emph{Intuition:} every point in a leaf gets the \textbf{same} correction $c_t$, so a leaf is
  just \textbf{one number to choose} - and the objective is now a simple quadratic in it.
\end{frame}

\begin{frame}{Optimal leaf value $=$ the ``structure score''}
  Minimizing each quadratic $G_t c_t + \tfrac12(H_t+\lambda)c_t^2$ gives the best leaf value
  \[
    c_t^{*} = -\,\frac{G_t}{H_t + \lambda}.
  \]
  Plugging back, the best objective for a \emph{fixed tree shape} is the \textbf{structure score}
  \[
    \mathrm{obj}^{*} = -\tfrac12 \sum_{t=1}^{T} \frac{G_t^2}{H_t + \lambda} + \gamma\,T .
  \]
  {\footnotesize \emph{Intuition:} a leaf's best step $=$ total error to fix $(G_t)$ per unit of
  confidence $(H_t{+}\lambda)$; the score adds up each leaf's payoff - \textbf{lower is a better tree}.}
  \vskip 3pt
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{Numbers:} $G_t={-3},\ H_t=4,\ \lambda=1 \;\Rightarrow\; c_t^{*} = -({-3})/(4{+}1) = 0.6$.}}
\end{frame}

\begin{frame}{So how does the parabola become a tree?}
  \begin{columns}[T]
    \column{0.5\textwidth}
    The Taylor step gave \textbf{each point} its own parabola in the correction $b$; its ideal
    step is the Newton point $-g_i/h_i$ (the blue dots).
    \vskip 3pt
    But a tree \textbf{cannot} give each point its own value -- everyone in a leaf shares
    \emph{one} number $c_t$. So we \textbf{add the leaf's parabolas} and jump to the bottom of
    the sum:
    \[
      c_t^{*} = -\frac{G_t}{H_t + \lambda},
    \]
    the curvature-weighted \textbf{compromise} of what its points want (here $0.75$).
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/leaf_parabola.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    So what do the \textbf{splits} do? They \textbf{group points that want similar steps into the
    same leaf}, so each leaf's compromise is tight. The split-gain score (next) rewards exactly
    that grouping -- the tree structure just decides \emph{who shares a parabola}.}}
\end{frame}

\begin{frame}{Worked bridge: the leaf value is [19]'s residual mean, braked}
  In [19] the round-1 \textbf{left leaf} just used the residual mean, $-17.1$. XGBoost's
  $c^{*} = -G_t/(H_t+\lambda)$ turns out to give the \emph{same} leaf, gently shrunk by $\lambda$.
  \vfill
  For \textbf{squared loss} the Hessian is $h_i = 1$, so a leaf with $n$ rows has
  \[
    G_t = \sum_{\text{leaf}} -(y_i - F) = -\!\sum r_i, \qquad H_t = n
    \;\;\Rightarrow\;\;
    c_t^{*} = \frac{\sum r_i}{n + \lambda} = \underbrace{\bar r}_{\text{[19]'s leaf}}\cdot\frac{n}{n+\lambda}.
  \]
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Left leaf: $n=41$, $\bar r = -17.1$, so $c^{*} = -17.1\times\tfrac{41}{42} = \mathbf{-16.7}$ at
    $\lambda=1$ (barely braked), or $-13.8$ at $\lambda=10$. \textbf{The structure score is [19]'s
    by-hand step plus a built-in $\lambda$ brake.}}}
\end{frame}

\begin{frame}{The split-gain criterion}
  \begin{columns}[T]
    \column{0.57\textwidth}
    How good is a split? By how much it \textbf{improves the structure score}:
    \[
      \mathrm{gain} = \tfrac12\!\left[ \frac{G_L^2}{H_L+\lambda} + \frac{G_R^2}{H_R+\lambda}
        - \frac{G^2}{H+\lambda} \right] - \gamma .
    \]
    \begin{itemize}\small
      \item \emph{Intuition:} do the two children together score better than the parent, by more
        than the cost $\gamma$ of a new leaf?
      \item Take the split only if $\mathrm{gain} > 0$; $\gamma$ is the \textbf{minimum gain to keep it}.
      \item This - not Gini/variance - is what XGBoost maximizes at each node.
    \end{itemize}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\scriptsize
      \textbf{Numbers:} parent $(G{=}{-3},H{=}4)$, $\lambda{=}1$, split into $L({-}3,2)$, $R(0,2)$:
      $\mathrm{gain} = \tfrac12[\tfrac{9}{3}+\tfrac{0}{3}-\tfrac{9}{5}] - \gamma = 0.6-\gamma$
      (keep only if $\gamma < 0.6$); then $c_L^{*}{=}1.0,\ c_R^{*}{=}0$.}}
    \column{0.43\textwidth}
    \begin{center}
      \begin{tikzpicture}[scale=0.9, every node/.style={font=\small}]
        \node[draw, rounded corners, fill=popblue!15] (t) at (0,2.2) {$t:\ G,\,H$};
        \node[draw, rounded corners, fill=armblue!15] (l) at (-1.35,0.5) {$L:\ G_L,H_L$};
        \node[draw, rounded corners, fill=armred!15] (r) at (1.35,0.5) {$R:\ G_R,H_R$};
        \draw[->,thick] (t) -- (l);
        \draw[->,thick] (t) -- (r);
        \node[font=\scriptsize] at (0,-0.35) {$G=G_L+G_R,\ \ H=H_L+H_R$};
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\section{Engineering: why it is fast}
\sectiontransition{Why it is fast}{The same gradient boosting, engineered for speed and scale.}

\begin{frame}{Tree growing and pruning}
  \begin{itemize}
    \item Grow each tree \textbf{level by level}, up to \texttt{max\_depth}.
    \item A split is kept only if its \textbf{gain} (previous slide) is positive.
    \item After growing, \textbf{prune} back any split with $\mathrm{gain} < 0$ - the $\gamma$ threshold.
  \end{itemize}
  \vfill
  \footnotesize Depth-limited growth $+$ post-pruning by gain keeps trees shallow and regularized.
\end{frame}

\begin{frame}{Approximate (histogram) split finding}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Testing every possible split is slow. Instead, bucket each feature into ${\sim}\,\ell$
    \textbf{percentile bins} and evaluate only the \textbf{bin edges}.
    \begin{itemize}\small
      \item \textbf{Global}: propose the bins once per tree.
      \item \textbf{Local}: re-propose at every node (more accurate, more work).
    \end{itemize}
    \footnotesize ``Histogram-based gradient boosting'' - XGBoost \texttt{tree\_method="hist"};
    the default in LightGBM and sklearn \texttt{HistGradientBoosting}.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{fig/12_xgb_histogram_split.pdf}\\[1pt]
      {\scriptsize Candidates sit at percentile edges (denser where data is dense).
      Source: Chen \& Guestrin (2016).}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Sparsity-aware splits}
  \begin{columns}[T]
    \column{0.56\textwidth}
    Real data is full of \textbf{missing values} and \textbf{zeros} (one-hot columns).
    \vskip 4pt
    XGBoost gives every split a learned \textbf{default direction}: rows missing (or zero) on the
    split feature all go that way.
    \begin{itemize}\small
      \item Handles missing values \textbf{natively} - no imputation.
      \item Skips absent entries during split search, so sparse data is \textbf{faster}.
    \end{itemize}
    \column{0.44\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize]
        \node[draw, rounded corners, fill=armblue!12] (n) at (0,0) {split on $f$};
        \node[draw, rounded corners, fill=paramgreen!12] (l) at (-1.3,-1.4) {$f \le t$};
        \node[draw, rounded corners, fill=paramgreen!12] (r) at (1.3,-1.4) {$f > t$};
        \draw[-{Latex}] (n) -- (l);
        \draw[-{Latex}] (n) -- (r);
        \node[armred] (m) at (0,-1.4) {miss / 0};
        \draw[-{Latex}, armred, dashed, thick] (m) -- (r)
          node[midway, above, font=\tiny] {default};
      \end{tikzpicture}\\[2pt]
      \scriptsize A learned default branch for absent values.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Subsampling, parallelism, GPU}
  \begin{itemize}
    \item \textbf{Row subsampling} (\texttt{subsample}) - stochastic gradient boosting.
    \item \textbf{Column subsampling} (\texttt{colsample\_bytree/bylevel/bynode}) - like a random
      forest's \texttt{mtry}; faster and more diverse trees.
    \item Boosting is \textbf{sequential}, but the \textbf{split search inside one tree} parallelizes
      over sorted feature blocks - and moves well to \textbf{GPU}.
  \end{itemize}
  \vfill
  \footnotesize Subsampling is both a speed-up and a regularizer.
\end{frame}

\begin{frame}{DART: dropout for trees}
  \textbf{DART} (\emph{Dropouts meet Multiple Additive Regression Trees}) borrows \textbf{dropout}
  from deep learning. When adding a new tree, compute the update against a
  \textbf{random subset} of the existing trees (drop the rest), then rescale so the ensemble does
  not overshoot.
  \vskip 2pt
  \begin{center}
    \begin{tikzpicture}
      \foreach \i/\c in {1/armblue,2/gray!35,3/armblue,4/armblue,5/gray!35} {
        \fill[\c] (\i*0.85,0) -- (\i*0.85+0.32,0.6) -- (\i*0.85+0.64,0) -- cycle;}
      \node[font=\tiny, gray!60!black] at (3.05,-0.28)
        {grey $=$ dropped when the next tree is added};
    \end{tikzpicture}
  \end{center}
  \vskip 2pt
  \begin{itemize}\small
    \item $p_{\text{drop}} = 0$: ordinary gradient boosting.
    \item $p_{\text{drop}} = 1$: trees trained independently, equally weighted $\approx$ a random forest.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    $p_{\text{drop}}$ is a smooth dial from \textbf{boosting} to \textbf{random forest}
    (XGBoost \texttt{booster="dart"}).}}
\end{frame}

\section{The modern trio: LightGBM and CatBoost}

\modeltransition{LightGBM~\textcolor{armred}{$\heartsuit$}}{Leaf-wise growth, gradient sampling

\begin{frame}{LightGBM~\textcolor{armred}{$\heartsuit$}}
  \begin{columns}[T]
    \column{0.6\textwidth}
    For everyday tabular work, a great first thing to reach for:
    \begin{itemize}
      \item \textbf{very fast} and \textbf{low memory} (histogram $+$ leaf-wise);
      \item excellent \textbf{out-of-the-box defaults};
      \item \textbf{native categorical} support;
      \item scales smoothly to \textbf{large} data.
    \end{itemize}
    \column{0.4\textwidth}
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      A solid \textbf{default} first model on tabular data -- the next frames are why.}}
  \end{columns}
\end{frame}

\begin{frame}{Predict-first: leaf-wise vs level-wise}
  Give both the \textbf{same budget} of leaves. Level-wise grows a full depth level at a time;
  leaf-wise always splits the \emph{highest-loss} leaf. Which \textbf{overfits sooner}?
  \pause
  \vfill
  \textbf{Leaf-wise.} Spending the whole leaf budget on the worst region makes \emph{deeper,
  narrower} trees that can chase noise -- which is exactly why LightGBM needs \texttt{num\_leaves}
  (and \texttt{max\_depth}) kept in check.
\end{frame}

\begin{frame}{Leaf-wise (best-first) growth}
  \begin{columns}[T]
    \column{0.54\textwidth}
    XGBoost grows \textbf{level-wise by default} (a whole depth level at a time). LightGBM instead
    splits the \textbf{single leaf with the largest gain}, anywhere in the tree.
    \begin{itemize}\small
      \item Unbalanced, deeper trees; \textbf{converges in fewer rounds}.
      \item Can \textbf{overfit} - so the main complexity knob is \texttt{num\_leaves}
        (keep it $< 2^{\texttt{max\_depth}}$), plus \texttt{min\_data\_in\_leaf}.
    \end{itemize}
    {\footnotesize \emph{Intuition:} spend each split where it helps most, instead of splitting
    every leaf just because it is at the same depth.}
    \vskip 2pt
    \hfill {\scriptsize Source: LightGBM docs.}

\begin{frame}{Predict-first: GOSS throws away rows}
  \textbf{Gradient-based One-Side Sampling (GOSS)} keeps all \textbf{large-gradient} rows but
  \textbf{randomly drops most small-gradient} ones (the points the model already fits well).
  How much accuracy do you lose?
  \pause
  \vfill
  \textbf{Almost none.} Well-fit rows carry little new signal, so dropping most of them (with a
  reweight to stay unbiased) speeds training a lot at negligible cost -- that is the whole point.
\end{frame}

\begin{frame}{GOSS, part 1: what ``gradient'' means here}
  \begin{columns}[T]
    \column{0.52\textwidth}
    The ``gradient'' is nothing new -- it is the \textbf{per-row gradient}
    $g_i = \partial L / \partial F$ from [19], at the current model:
    \begin{itemize}\small
      \item $|g_i|$ $=$ \textbf{how wrong the model still is} on row $i$.
      \item For \textbf{squared loss} $g_i = -(y_i - F_i)$, so $|g_i|$ is just the
        \textbf{residual size}; for log-loss it is $|p_i - y_i|$.
      \item \textbf{Big} $|g_i|$: a still-misfit, \emph{informative} row. \textbf{Small}
        $|g_i|$: already learned, little left to teach.
    \end{itemize}
    \vskip 3pt
    So sample rows \emph{by} $|g_i|$ instead of uniformly -- keep the hard ones.
    \column{0.48\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/goss.pdf}\\[1pt]
      {\scriptsize Most rows have a small $|$gradient$|$ (already fit); a few are still hard.}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{GOSS, part 2: the recipe}
  \textbf{Gradient-based One-Side Sampling} keeps the informative rows and thins the rest:
  \begin{enumerate}\small
    \item \textbf{Keep} the top $a\!\cdot\!n$ rows with the \textbf{largest} $|g_i|$ (the
      still-hard cases).
    \item \textbf{Randomly sample} $b\!\cdot\!n$ of the small-gradient rest.
    \item \textbf{Up-weight} that sample by $\tfrac{1-a}{b}$ so the split-gain estimate stays
      \textbf{unbiased} (the kept easy rows now stand in for all of them).
  \end{enumerate}
  \vskip 2pt
  {\small Defaults $a = 0.2,\ b = 0.1$: e.g.\ \textbf{1000 rows} $\to$ keep the 200 hardest
  $+$ a random 80 of the other 800 $=$ train on ${\sim}280$.}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \emph{Why it works:} rows the model already gets right carry little new signal -- spend the
    budget on the hard rows, and reweight the easy sample so the data still looks unbiased.
    Big speed-up, negligible accuracy loss.} \hfill {\scriptsize Ke et al.\ (2017).}}
\end{frame}

\begin{frame}{EFB: exclusive feature bundling}
  \begin{columns}[T]
    \column{0.58\textwidth}
    In sparse, high-dimensional data many features are \textbf{mutually exclusive} - never
    nonzero at the same time (one-hot columns).
    \vskip 3pt
    \textbf{Bundle} them into one feature with a \textbf{single histogram} - nearly lossless.
    \begin{itemize}\small
      \item Histogram cost $\mathcal{O}(n\,p) \to \mathcal{O}(n \cdot \#\text{bundles})$.
      \item Optimal bundling is NP-hard; a \textbf{greedy} grouping works well.
    \end{itemize}
    \column{0.42\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\tiny]
        \foreach \c/\fr/\col in {0/2/armblue, 1/0/armred, 2/3/paramgreen} {
          \foreach \r in {0,1,2,3} {\draw (\c*0.48,-\r*0.4) rectangle (\c*0.48+0.4,-\r*0.4-0.4);}
          \fill[\col] (\c*0.48,-\fr*0.4) rectangle (\c*0.48+0.4,-\fr*0.4-0.4);}
        \node at (0.68,-1.95) {3 sparse cols};
        \draw[-{Latex}, thick] (1.7,-0.8) -- (2.5,-0.8) node[midway, above] {bundle};
        \foreach \r in {0,1,2,3} {\draw (2.8,-\r*0.4) rectangle (3.2,-\r*0.4-0.4);}
        \fill[armred] (2.8,0) rectangle (3.2,-0.4);
        \fill[armblue] (2.8,-0.8) rectangle (3.2,-1.2);
        \fill[paramgreen] (2.8,-1.2) rectangle (3.2,-1.6);
        \node at (3.0,-1.95) {1 col};
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vskip 2pt
  {\footnotesize \emph{Intuition:} one-hot columns rarely fire together - pack them into one and
  lose almost nothing while doing far less work.} \hfill {\scriptsize Ke et al.\ (2017).}
\end{frame}

\begin{frame}{EFB by example: how two features actually merge}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Two sparse features never nonzero together (two one-hot columns), $A \in \{0\ldots10\}$,
    $B \in \{0\ldots5\}$. \textbf{Shift $B$ by an offset $=\max(A)=10$}:
    \[
      \text{bundle} =
      \begin{cases}
        A & A > 0\\ B + 10 & B > 0\\ 0 & \text{both } 0
      \end{cases}
    \]
    \begin{center}\small
      \renewcommand{\arraystretch}{1.0}
      \begin{tabular}{ccc}
        \toprule $A$ & $B$ & bundle \\ \midrule
        3 & 0 & 3 \\ 0 & 2 & \textbf{12} \\ 0 & 0 & 0 \\ 0 & 4 & \textbf{14} \\ \bottomrule
      \end{tabular}
    \end{center}
    \column{0.5\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize]
        \draw[armblue, line width=6pt] (0,0) -- (2.4,0);
        \draw[armred, line width=6pt] (2.5,0) -- (3.7,0);
        \node[below] at (0,-0.08) {\tiny 0};
        \node[below] at (2.4,-0.08) {\tiny 10};
        \node[below] at (3.7,-0.08) {\tiny 15};
        \node[armblue] at (1.2,0.33) {$A$'s bins};
        \node[armred] at (3.1,0.33) {$B{+}10$};
      \end{tikzpicture}
    \end{center}
    \textbf{Read it back:} bundle $\le 10 \Rightarrow$ $A$'s value; bundle $> 10 \Rightarrow$
    $B$'s value ($-10$).
    \vskip 3pt
    One histogram now holds \emph{both} in \textbf{disjoint bins}, so a split on the bundle
    \emph{is} a split on one original feature -- \textbf{nothing lost}.
  \end{columns}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Exclusivity is the trick:} $A$, $B$ never both nonzero, so each value decodes
    unambiguously. (Real EFB allows a \emph{few} conflicts to bundle more.)}}
\end{frame}

\modeltransition{CatBoost}{Ordered target statistics and ordered boosting -- categorical
  features without target leakage.}{2018}{Prokhorenkova et al.\ (Yandex)}{catboost/catboost}

\begin{frame}{CatBoost: the categorical specialist}
  \begin{columns}[T]
    \column{0.6\textwidth}
    Built around \textbf{categorical} features:
    \begin{itemize}\small
      \item \textbf{Ordered target statistics} - encode a category by its target mean on
        \emph{past} rows only, so \textbf{no target leakage};
      \item \textbf{Ordered boosting} - a permutation trick that removes the
        \textbf{prediction-shift} bias;
      \item \textbf{Symmetric (oblivious) trees} - one split per level: fast, regularized.
    \end{itemize}
    \column{0.4\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\tiny]
        \foreach \i in {1,2,3,5} {\node[draw, minimum width=1cm, minimum height=0.3cm]
          (r\i) at (0,-\i*0.45) {row \i};}
        \node[draw, thick, armred, fill=armred!8, minimum width=1cm, minimum height=0.3cm]
          (r4) at (0,-4*0.45) {row 4};
        \draw[-{Latex}, armblue, thick] (0.55,-2*0.45) to[out=0,in=0] (0.55,-4*0.45);
        \node[armblue, align=left] at (1.75,-3*0.45) {encode from\\rows above};
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vskip 2pt
  {\footnotesize \emph{Intuition:} the whole-dataset average peeks at the answer (leakage); a
  running average over \emph{earlier} rows is honest - that is what ``ordered'' means.}
\end{frame}

\begin{frame}{CatBoost's oblivious (symmetric) trees}
  \begin{columns}[T]
    \column{0.48\textwidth}
    An \textbf{oblivious} tree uses the \textbf{same split on every node at a given depth} -- the
    whole level tests one $(\text{feature}, \text{threshold})$.
    \begin{itemize}\small
      \item A depth-$d$ tree is just \textbf{$d$ conditions}; its $2^d$ leaves are indexed by the
        $d$ yes/no answers (a bit-vector).
      \item \textbf{Very fast} prediction: compute $d$ tests $\to$ look the leaf up in an array
        (no node-by-node traversal).
      \item The forced symmetry is a \textbf{regularizer} -- balanced trees, less overfitting.
    \end{itemize}
    \footnotesize A normal CART tree may split each node on a \emph{different} feature; oblivious
    trees give that up for speed and regularity.
    \column{0.52\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize,
          q/.style={draw, rounded corners, fill=popblue!15, inner sep=2.5pt},
          lf/.style={draw, rounded corners, fill=paramgreen!18, inner sep=2.5pt, minimum width=6mm}]
        \node[q] (r)  at (0,2.1)   {age $>30$?};
        \node[q] (a)  at (-1.75,1.0){hours $>40$?};
        \node[q] (b)  at (1.75,1.0) {hours $>40$?};
        \node[armred, font=\tiny] at (0,1.0) {same split};
        \node[lf] (l1) at (-2.55,-0.1){$v_1$};
        \node[lf] (l2) at (-0.95,-0.1){$v_2$};
        \node[lf] (l3) at (0.95,-0.1) {$v_3$};
        \node[lf] (l4) at (2.55,-0.1) {$v_4$};
        \draw[-{Latex}] (r)--(a) node[midway,above left,font=\tiny]{no};
        \draw[-{Latex}] (r)--(b) node[midway,above right,font=\tiny]{yes};
        \draw[-{Latex}] (a)--(l1); \draw[-{Latex}] (a)--(l2);
        \draw[-{Latex}] (b)--(l3); \draw[-{Latex}] (b)--(l4);
      \end{tikzpicture}\\[2pt]
      \scriptsize Both level-2 nodes ask the \emph{same} question.
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Leaf $=$ a lookup by $(\text{age}{>}30,\ \text{hours}{>}40)$ -- four cells, like a tiny
    decision \textbf{table}. That table structure is what makes CatBoost inference so fast.}}
\end{frame}

\begin{frame}{Which one, when?}
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l c c c}
      \toprule
                          & \textbf{XGBoost}   & \textbf{LightGBM}    & \textbf{CatBoost} \\
      \midrule
      Tree growth         & level-wise         & leaf-wise            & symmetric \\
      Row sampling (option) & random           & random or GOSS       & random \\
      Native categoricals & no$^{*}$           & yes                  & yes (best) \\
      Speed on large $n$  & good               & \textbf{fastest}     & good \\
      Reach for it when   & solid baseline     & default for tabular  & many categoricals \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize $^{*}$ XGBoost can also grow leaf-wise and gradient-subsample, and has experimental
  categorical support. The table shows \emph{defaults} - except row sampling, which is \emph{off} by
  default in all three (cells name the available strategy). \textbf{All three are the same gradient
  boosting} - the differences are engineering and categorical handling.
\end{frame}

\section{Using them well}
\sectiontransition{Using them well}{Tuning order, domain constraints, and reading the model.}

\begin{frame}{Hyperparameters that matter}
  \begin{center}\scriptsize
    \renewcommand{\arraystretch}{1.35}
    \begin{tabular}{l p{4.0cm} p{4.4cm}}
      \toprule
      \textbf{Group} & \textbf{XGBoost} & \textbf{LightGBM} \\
      \midrule
      Learning    & \texttt{eta}, \texttt{n\_estimators} ($+$ early stop) & \texttt{learning\_rate}, \texttt{n\_estimators} \\
      Complexity  & \texttt{max\_depth}, \texttt{min\_child\_weight}      & \texttt{num\_leaves}, \texttt{min\_data\_in\_leaf} \\
      Regularize  & \texttt{gamma}, \texttt{lambda}, \texttt{alpha}       & \texttt{min\_gain\_to\_split}, \texttt{lambda\_l1/l2} \\
      Randomness  & \texttt{subsample}, \texttt{colsample\_by*}           & \texttt{bagging\_fraction}, \texttt{feature\_fraction} \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{The knob differs:} XGBoost sets tree size with \texttt{max\_depth}; LightGBM with
    \texttt{num\_leaves} (default 31). Defaults: XGB \texttt{eta} 0.3 / depth 6; LGBM lr 0.1.}}
\end{frame}

\begin{frame}{The tuning order that matters}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \begin{enumerate}\small
      \item Set \texttt{n\_estimators} \emph{large} and use \textbf{early stopping}.
      \item Trade \textbf{learning rate} against the number of trees.
      \item \textbf{Tree complexity}: \texttt{max\_depth} (XGB) / \texttt{num\_leaves} (LGBM).
      \item \textbf{Regularization}: \texttt{gamma}, \texttt{lambda}, \texttt{alpha}.
      \item \textbf{Subsampling}: rows $+$ columns.
    \end{enumerate}
    \footnotesize Search with random or Bayesian optimization (see the hyperparameter-tuning deck).
    \column{0.48\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{fig/12_xgb_earlystop.pdf}\\[1pt]
      {\scriptsize Early stopping takes the round before validation stalls - step 1.}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Inject domain knowledge: constraints}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Sometimes you \emph{know} something the data should obey:
    \begin{itemize}\small
      \item \textbf{Monotonic constraints} (\texttt{monotone\_constraints}): force a feature's
        effect to go only one way. E.g.\ \emph{bigger apartment $\Rightarrow$ predicted rent
        must not drop}; \emph{higher income $\Rightarrow$ credit score must not drop}.
      \item \textbf{Interaction constraints}: choose which features may combine in a tree. E.g.\
        let \texttt{area} and \texttt{district} interact (a big flat \emph{downtown} is
        special), but make \texttt{floor} act on its own.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/xgb_monotonic.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Constraints add \textbf{trust} and act as \textbf{free regularization} - use them when the
    direction or structure is known a priori.}}
\end{frame}

\begin{frame}{Are they black boxes? Feature importance}
  \begin{columns}[T]
    \column{0.52\textwidth}
    Boosted models are \textbf{not} opaque:
    \begin{itemize}\small
      \item built-in importances: \textbf{gain} (loss reduction), cover, frequency;
      \item but \textbf{split-count importance is biased} toward high-cardinality features -
        prefer \textbf{permutation importance}.
    \end{itemize}
    \footnotesize For per-prediction attribution there is \textbf{SHAP} (fast \emph{TreeSHAP} for
    trees) - we will cover it in its own session.
    \column{0.48\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{fig/12_xgb_importance.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{In code: LightGBM and XGBoost, side by side}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{center}{\scriptsize LightGBM (the favorite)}\end{center}
\begin{lstlisting}[basicstyle=\ttfamily\scriptsize]
from lightgbm import LGBMClassifier
from lightgbm import early_stopping
m = LGBMClassifier(
    n_estimators=2000,
    learning_rate=0.05,
    num_leaves=31)
m.fit(X_tr, y_tr,
  eval_set=[(X_val, y_val)],
  callbacks=[early_stopping(50)])
\end{lstlisting}
    \column{0.5\textwidth}
    \begin{center}{\scriptsize XGBoost (near-identical API)}\end{center}
\begin{lstlisting}[basicstyle=\ttfamily\scriptsize]

    n_estimators=2000,
    learning_rate=0.05,
    max_depth=6,
    early_stopping_rounds=50)

  eval_set=[(X_val, y_val)])
\end{lstlisting}
  \end{columns}
  \vskip 4pt
  \footnotesize No install? \texttt{sklearn.ensemble.HistGradientBoostingClassifier} mirrors both.
\end{frame}

\begin{frame}{Beyond binary classification}
  The same $g, h$ machinery powers many tasks - just change the \textbf{objective}:
  \begin{itemize}\small
    \item \textbf{Regression} (\texttt{reg:squarederror}), \textbf{multiclass} (softmax),
      \textbf{ranking} (\texttt{rank:*} / LambdaMART), Poisson, quantile - even \textbf{custom}
      losses (supply your own $g$ and $h$).
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Imbalanced classes?} Reweight the rare class with \texttt{scale\_pos\_weight} (XGBoost)
    or \texttt{is\_unbalance} / \texttt{class\_weight} (LightGBM), and judge with \textbf{AUC / PR},
    not accuracy. Full playbook: the imbalanced-learning lecture ([15]).}}
\end{frame}

\begin{frame}{When boosting struggles}
  Powerful, but not magic:
  \begin{itemize}\small
    \item \textbf{No extrapolation} - a tree predicts a constant outside the training range; it
      cannot trend beyond what it has seen.
    \item \textbf{Needs tuning} - strong defaults exist, but top results need real HP search
      (a random forest is more forgiving).
    \item \textbf{Sequential} - trees are built one after another; less parallel than bagging.
    \item \textbf{No online learning} - batch-trained; you cannot cheaply update it
      example-by-example (as SGD models can). New data usually means \textbf{retraining}.
    \item \textbf{Not for raw unstructured data} - images, audio, long text $\to$ neural nets;
      very sparse high-dimensional signals can favor linear models.
    \item \textbf{Miscalibrated scores} - boosted \texttt{predict\_proba} is often over-confident;
      calibrate before you trust it (calibration lecture, [14]).
  \end{itemize}
  \vfill
  \footnotesize \emph{Rule of thumb:} reach for boosting on \textbf{tabular} data; reach elsewhere
  when the data is not tabular.
\end{frame}

\section{Stacking: combining different models}
\sectiontransition{Stacking}{The third ensemble idea: let a meta-learner blend different model types - and close the chapter.}

\begin{frame}{Stacking: a meta-learner over base models}
  \begin{columns}[T]
    \column{0.54\textwidth}
    Bagging and boosting combine \emph{one} kind of model. \textbf{Stacking} combines
    \textbf{different} ones:
    \begin{itemize}\small
      \item train several diverse \textbf{base learners} (level 0) - e.g.\ LightGBM, a random
        forest, a linear model;
      \item feed their predictions to a \textbf{meta-learner} (level 1) that learns how best to
        \emph{combine} them.
    \end{itemize}
    {\footnotesize \emph{Intuition:} different models make different mistakes; the meta-learner
    learns whom to trust, and when.}
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}[scale=0.8, font=\scriptsize,
          bx/.style={draw,rounded corners,minimum width=11mm,minimum height=5.5mm,inner sep=2pt}]
        \node[bx,fill=popblue!12] (x) at (0,3) {data $X$};
        \node[bx,fill=armblue!12] (m1) at (-1.7,1.7) {LGBM};
        \node[bx,fill=armblue!12] (m2) at (0,1.7) {RF};
        \node[bx,fill=armblue!12] (m3) at (1.7,1.7) {lin.};
        \node[bx,fill=paramgreen!15,minimum width=22mm] (meta) at (0,0.5) {meta-learner};
        \node[bx,fill=armorange!20] (out) at (0,-0.7) {prediction};
        \draw[->] (x)--(m1); \draw[->] (x)--(m2); \draw[->] (x)--(m3);
        \draw[->] (m1)--(meta); \draw[->] (m2)--(meta); \draw[->] (m3)--(meta);
        \draw[->] (meta)--(out);
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Stacking done right: out-of-fold predictions}
  \textbf{The trap:} if the meta-learner trains on base predictions of the \emph{same} rows the
  base models were fit on, it sees \textbf{leaked}, over-optimistic inputs.
  \vskip 3pt
  \textbf{The fix:} feed it \textbf{out-of-fold} (cross-validated) predictions - each row's base
  prediction comes from a model that did \emph{not} train on it. \texttt{StackingClassifier} does
  this for you via \texttt{cv}.
  \vskip 4pt
\begin{lstlisting}[basicstyle=\ttfamily\scriptsize]
from sklearn.ensemble import StackingClassifier
from sklearn.linear_model import LogisticRegression

base = [("lgbm", LGBMClassifier()), ("rf", RandomForestClassifier())]
clf = StackingClassifier(estimators=base,
                         final_estimator=LogisticRegression(), cv=5)
clf.fit(X_tr, y_tr)   # cv=5 -> honest out-of-fold base predictions
\end{lstlisting}
\end{frame}

\begin{frame}{Simpler cousins: voting and blending}
  \begin{itemize}
    \item \textbf{Voting} - no meta-learner, just combine base predictions directly: \textbf{hard}
      (majority class) or \textbf{soft} (average predicted probabilities, usually better).
      \texttt{VotingClassifier}.
    \item \textbf{Blending} - like stacking, but the meta-learner trains on a single \textbf{holdout}
      set instead of full cross-validation. Simpler and faster; uses a bit less data.
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    Stacking usually wins but costs the most (every base model $+$ the meta-learner, with CV).
    Start with one strong model; stack only to chase the last bit of accuracy.}}
\end{frame}

\begin{frame}{Does stacking actually pay off?}
  \begin{columns}[T]
    \column{0.44\textwidth}
    A quick receipt on one dataset: three diverse bases, then a logistic meta-learner over
    their out-of-fold predictions.
    \vskip 4pt
    Here the stack \textbf{ties} the best base (random forest) and does not beat it -- the weak
    logistic base added no usable diversity.
    \column{0.56\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/xgb_stacking.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    The honest lesson: stacking's gains are \textbf{real but small and not guaranteed}. Start with
    one strong model; reach for a stack only when the bases are \textbf{diverse and each genuinely
    good}.}}
\end{frame}

\section*{Wrap-up}

\begin{frame}{The ensemble landscape}
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l c c c}
      \toprule
                   & \textbf{Bagging} & \textbf{Boosting} & \textbf{Stacking} \\
                   & {\scriptsize (random forest)} & {\scriptsize (GB / XGB / LGBM)} & \\
      \midrule
      Base models  & same type, parallel & same type, sequential & different types \\
      Combine by   & averaging & weighted additive sum & a meta-learner \\
      Mainly cuts  & variance & bias & both \\
      Trains       & in parallel & sequentially & bases parallel, then meta \\
      Use when     & low-tuning baseline & top tabular accuracy & squeeze diverse models \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize Three ways to turn many so-so models into one strong one.
\end{frame}

\begin{frame}{Recap: the trees and ensembles chapter}
  \begin{itemize}\small
    \item \textbf{One tree} ([17]) - interpretable, but unstable.
    \item \textbf{Bagging $\to$ random forests} ([18]) - average many parallel trees, cut variance.
    \item \textbf{Boosting} ([19]) - add trees sequentially to fix errors, cut bias; gradient
      descent in function space.
    \item \textbf{The libraries} ([20]) - XGBoost (regularized objective $+$ structure score),
      \textbf{LightGBM} \textcolor{armred}{$\heartsuit$} (leaf-wise $+$ GOSS $+$ EFB), CatBoost
      (categoricals); tune in order, add constraints, read importance.
    \item \textbf{Stacking} - a meta-learner over diverse models, to squeeze out the last gains.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Trees $+$ ensembles dominate tabular data.} Next: models for data where trees
    struggle - images, text, sequences.}}
\end{frame}
