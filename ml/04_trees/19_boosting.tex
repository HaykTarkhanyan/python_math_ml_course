\section{The boosting idea}
\sectiontransition{The boosting idea}{Combine many weak learners, each fixing what the others miss.}

\begin{frame}{The theoretical question: weak $\Rightarrow$ strong?}
  \begin{quote}
    Can a set of \textbf{weak} learners -- each only \emph{slightly} better than a coin flip
    (say 51\% accurate) -- be combined into one \textbf{arbitrarily strong} learner?
  \end{quote}
  \hfill {\footnotesize (posed by Kearns \& Valiant, 1988.)}
  \vfill
  \textbf{Schapire (1990): yes -- and constructively.} ``Weak learnability'' and ``strong
  learnability'' turn out to be \textbf{equivalent}: if you can reliably do a hair better than
  chance, you can be boosted to \emph{any} accuracy. That was a genuine surprise -- and
  \textbf{boosting} is the algorithm that does it, each learner focusing on what the previous
  ones got wrong.
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    The ``weak learner'' is a \textbf{shallow tree}, often a \textbf{stump} -- the ``atom''
    from [17]. We add many, one at a time. (Next: AdaBoost, the first algorithm to pull it
    off.)}}
\end{frame}

\begin{frame}{Bagging vs boosting}
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l l l}
      & \textbf{Bagging / RF ([18])} & \textbf{Boosting (today)} \\
      \hline
      trees built & in \textbf{parallel}, independent & in \textbf{sequence}, dependent \\
      each tree sees & a bootstrap resample & the previous ones' \textbf{errors} \\
      combine by & averaging / voting & adding (a weighted sum) \\
      mainly cuts & \textcolor{armblue}{variance} & \textcolor{armred}{bias} \\
      more trees & safe (plateaus) & \textbf{can overfit} \\
      \hline
    \end{tabular}
  \end{center}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Mirror of [18].} RF bags \textbf{deep} (high-variance) trees and averages the
    variance away. Boosting boosts \textbf{shallow} (high-bias) trees -- each only nudges
    the fit, and the bias is removed \emph{across rounds}. Opposite ingredient, opposite
    target.}}
\end{frame}

\begin{frame}{AdaBoost: reweight the mistakes}
  The original boosting algorithm (Freund \& Schapire, 1995). Each round:
  \begin{enumerate}\small
    \item fit a \textbf{stump} on the current sample weights;
    \item find the points it got \textbf{wrong} (\textcolor{armred}{red rings} below);
    \item \textbf{up-weight} those, down-weight the rest, and refit.
  \end{enumerate}
  \begin{center}
    \includegraphics[width=0.86\linewidth]{fig/adaboost_rounds.pdf}
  \end{center}
  \vskip -2pt
  \footnotesize Each stump is forced to attend to what the previous ones kept missing -- the
  weights carry the ``leftover error'' from round to round.
\end{frame}

\begin{frame}{AdaBoost: many weak stumps, one strong vote}
  \begin{columns}[T]
    \column{0.44\textwidth}
    The prediction is a \textbf{vote weighted by each stump's accuracy}:
    \[
      H(\xv) = \operatorname{sign}\!\Big(\textstyle\sum_m \alpha_m\, h_m(\xv)\Big),
    \]
    with $\alpha_m = \tfrac12\ln\frac{1-\text{err}_m}{\text{err}_m}$ -- more accurate stumps
    get a \textbf{louder} vote (derived in the HW).
    \vskip 3pt
    \footnotesize A single stump is weak (one straight cut); \textbf{50} of them, weighted,
    bend into the true shape -- the weak-to-strong promise, delivered.
    \column{0.56\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/adaboost_combined.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\scriptsize
    \textbf{AdaBoost = boosting with the exponential loss.} Gradient boosting (next) generalizes
    it to \emph{any} loss -- both are the same ``forward stagewise additive model'' (AdaBoost
    just predates the gradient view).}}
\end{frame}

\section{Gradient boosting}
\sectiontransition{Gradient boosting}{Fit the leftover error, add a slice, repeat -- and watch it work.}

\begin{frame}{The idea: fit the residuals}
  Predict apartment rent from area. The recipe:
  \begin{enumerate}\small
    \item Start simple: $F_0 = \text{mean}(y)$.
    \item Compute \textbf{residuals} $r = y - F_0$ -- what is still wrong.
    \item Fit a \textbf{small tree} to the residuals.
    \item Add it, shrunk by a \textbf{learning rate} $\eta$: $F_1 = F_0 + \eta\,(\text{tree})$.
    \item Recompute residuals; repeat.
  \end{enumerate}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Each tree learns the \emph{previous} ensemble's mistakes. The fit creeps toward the
    data one correction at a time.}}
\end{frame}

\begin{frame}{By hand: round 1 (Yerevan rent vs area)}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \small
    \textbf{Init:} $F_0 = \text{mean}(y) = \mathbf{304}$ kAMD (a flat line).
    \vskip 3pt
    \textbf{Residuals:} $r_i = y_i - 304$.
    \vskip 3pt
    \textbf{First stump} asks ``area $\le 100$?'' and predicts each side's
    \emph{residual mean}:
    \[
      \text{left} = -17,\qquad \text{right} = +78 .
    \]
    \textbf{Update} with $\eta = 0.4$:
    \[
      F_1 = 304 + 0.4 \times
      \begin{cases} -17 & \text{area} \le 100 \\ +78 & \text{area} > 100 \end{cases}
    \]
    \[
      \Rightarrow\; F_1 = \begin{cases} \mathbf{297} \\ \mathbf{336} \end{cases}
    \]
    \column{0.46\textwidth}
    \vskip 12pt
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      One stump nudges the flat line into \textbf{two steps}. The big positive residuals
      at large area shrink first -- the model fixes its worst errors first.}}
    \vskip 6pt
    \footnotesize These are the real numbers behind round 1 of the next slide.
  \end{columns}
\end{frame}

\begin{frame}{Predict-first: does boosting overfit with more trees?}
  \begin{columns}[T]
    \column{0.46\textwidth}
    In [18], \emph{more trees never hurt} a random forest. Same for boosting?
    \pause
    \vskip 4pt
    \textbf{No -- the opposite.} Training error drives toward \textbf{0} (it can memorize);
    test error drops, then \textbf{U-turns} and rises.
    \vskip 4pt
    \footnotesize Boosting keeps fitting the \emph{previous} errors, eventually fitting
    noise. Controlled by a small $\eta$ + \textbf{early stopping}.
    \column{0.52\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/boost_overfit.pdf}\\[-2pt]
      \scriptsize A high-capacity config (depth-3 trees, 250 rounds) on a noisy toy --
      cranked up to \emph{show} the failure mode.
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\scriptsize
    \textbf{Contrast with random forests:} RF averages \emph{independent} trees $\to$ more is safe.
    Boosting adds \emph{dependent} trees that chase residuals $\to$ more can overfit.}}
\end{frame}

\begin{frame}{The reframe: you were doing gradient descent}
  Why is fitting residuals a good idea? Look at the squared loss
  $L = \tfrac12 (y - F)^2$:
  \[
    \frac{\partial L}{\partial F} = -(y - F)
    \quad\Longrightarrow\quad
    -\frac{\partial L}{\partial F} = y - F = \textbf{the residual}.
  \]
  (The $\tfrac12$ is the convention that makes this come out with no stray factor of 2.)
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    So the residual you have been fitting \textbf{is the negative gradient} of the loss.
    Each tree takes one step \textbf{downhill} -- you were doing gradient descent all
    along, just in \emph{function space} (adding functions) instead of parameter space.}}
\end{frame}

\begin{frame}{Function-space gradient descent (the general algorithm)}
  \fcolorbox{armblue}{armblue!6}{\parbox{0.96\linewidth}{\small
  For $m = 1, 2, \dots, M$:
  \begin{enumerate}\small
    \item \textbf{Pseudo-residuals} (the negative gradient at the current model):
      \[
        r_{im} = -\left[\frac{\partial L(y_i, F(\xv_i))}{\partial F(\xv_i)}\right]_{F = F_{m-1}}
      \]
    \item Fit a base learner $h_m$ to the pseudo-residuals $r_{im}$.
    \item \textbf{Line search} for the best step: $\gamma_m = \argmin_\gamma \sumin L(y_i, F_{m-1}(\xv_i) + \gamma\, h_m(\xv_i))$.
    \item Update: $F_m = F_{m-1} + \eta\,\gamma_m\, h_m$.
  \end{enumerate}}}
  \vskip 4pt
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\scriptsize
    \textbf{What sklearn actually does} (a refinement): it fits the tree, then sets
    \emph{each leaf} to its own optimal step (a per-leaf line search), and shrinks all
    leaves by $\eta$. For squared loss that per-leaf step is just the leaf's
    \textbf{residual mean} -- exactly the by-hand round you did two slides ago.}}
\end{frame}

\begin{frame}{So what do we actually fit each round?}
  \begin{columns}[T]
    \column{0.52\textwidth}
    Cut through the notation -- every round is the \emph{same} three moves:
    \begin{enumerate}\small
      \item \textbf{target} for each row $=$ the negative gradient at the current model (the
        ``pseudo-residual'' $r_{im}$);
      \item fit a \textbf{plain regression tree} to $(\xv,\ r)$ -- ordinary inputs,
        residual-like targets;
      \item add it, shrunk by $\eta$.
    \end{enumerate}
    \column{0.48\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize, node distance=4.5mm,
          bx/.style={draw, rounded corners, fill=armblue!10, align=center, inner sep=3pt}]
        \node[bx] (f) {current model $F_{m-1}$};
        \node[bx, below=of f] (r) {targets $r = -\nabla L$};
        \node[bx, below=of r] (t) {fit tree $h_m$ on $(\xv, r)$};
        \node[bx, below=of t] (u) {$F_m = F_{m-1} + \eta\, h_m$};
        \draw[-{Latex}] (f)--(r); \draw[-{Latex}] (r)--(t); \draw[-{Latex}] (t)--(u);
        \draw[-{Latex}] (u.east) to[out=0,in=0] (f.east);
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    ``Fitting the residuals'' \textbf{is} gradient descent -- one and the same procedure. For
    \textbf{squared loss} the target is literally $y - F$ (the ordinary residual); for
    \textbf{log-loss} it is $y - p$. Only the \emph{target formula} changes; the base learner is
    always a vanilla regression tree on $(\xv,\ \text{target})$.}}
\end{frame}

\begin{frame}{Any loss you like}
  Swap the loss $\Rightarrow$ swap the pseudo-residual. The whole machinery is unchanged;
  only ``what each tree chases'' changes.
  \vfill
  \begin{itemize}\small
    \item \textbf{Squared loss} $\to$ pseudo-residual $= y - F$ (the plain residual).
    \item \textbf{Absolute / Huber loss} $\to$ pseudo-residual $=$ the \emph{sign} or a
      \emph{clipped} residual $\to$ \textbf{robust to outliers} (one bad point can't drag
      the fit far).
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    This is boosting's superpower over AdaBoost: pick the loss that matches your problem
    (regression, classification, ranking, robust, quantile) -- the algorithm does not change.}}
\end{frame}

\begin{frame}{Gradient boosting for classification}
  \begin{columns}[T]
    \column{0.4\textwidth}
    Same machinery, \textbf{log-loss} instead of squared loss:
    \begin{itemize}\small
      \item each tree still fits the pseudo-residual, now $\mathbf{y - p}$ (with
        $p = \sigma(F)$);
      \item the trees sum into $F$, which lives in \textbf{log-odds} space (any real number);
      \item apply the \textbf{sigmoid once, at the very end} to read off a probability.
    \end{itemize}
    \footnotesize In the plot \textbf{each line follows one training point} (3 per class):
    $F$ climbs for class-1 points and falls for class-0 points; $\sigma(F)$ then heads to
    $1$ or $0$.
    \column{0.6\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/boost_gb_classification.pdf}
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\scriptsize
    Regression chases $y - F$ and predicts $F$ directly; classification chases $y - p$,
    accumulates $F$ in log-odds, and squashes with $\sigma$ at the end. \textbf{Same gradient
    descent} -- this is what the Titanic homework runs.}}
\end{frame}

\section{Regularization}
\sectiontransition{Regularization}{More power than you want -- three knobs hold it back.}

\begin{frame}{Three knobs}
  \begin{center}\footnotesize
    \renewcommand{\arraystretch}{1.25}
    \begin{tabular}{l p{0.62\linewidth}}
      knob & what it controls \\
      \hline
      \texttt{n\_estimators} ($M$) & number of trees / rounds. More $=$ lower bias, but
        eventually overfits $\Rightarrow$ use \textbf{early stopping}. \\
      \texttt{max\_depth} & per-tree depth $=$ \textbf{interaction order} (depth 1 $=$ main
        effects only; deeper $=$ feature interactions). Boosting likes \emph{shallow}. \\
      \texttt{learning\_rate} ($\eta$) & \textbf{shrinkage}: how big a step each tree takes.
        Smaller $=$ steadier, generalizes better. \\
      \hline
    \end{tabular}
  \end{center}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    The big one is the pair $(\eta, M)$: they trade off -- \textbf{smaller $\eta$ needs more
    trees}, but usually generalizes better. Standard recipe: \textbf{small $\eta$ + many trees
    + early stopping}.}}
\end{frame}

\begin{frame}{Stochastic gradient boosting}
  Add a dash of \textbf{bagging}: each round, fit the tree on a random \textbf{subsample}
  of the rows (and optionally columns), not all of them.
  \begin{itemize}\small
    \item Faster (smaller fits) and a \textbf{regularizer} (the randomness decorrelates
      the trees, callback [18]).
    \item A bagging $+$ boosting hybrid (Friedman, 2002).
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Heads-up:} this is NOT on by default -- XGBoost's \texttt{subsample} and
    LightGBM's \texttt{bagging\_fraction} both default to 1.0 (no subsampling). The
    libraries make it easy; \textbf{how they use it is [20]}.}}
\end{frame}

\section*{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Boosting} adds shallow trees \textbf{in sequence}, each fitting the
      previous ensemble's leftover error; it cuts \textbf{bias}.
    \item That ``leftover error'' is the \textbf{negative gradient} of the loss -- boosting
      is gradient descent in function space (any loss: $y - F$ regression, $y - p$
      classification).
    \item More trees \textbf{can overfit} (unlike RF) -- control with small $\eta$,
      shallow trees, and \textbf{early stopping}.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Workflow:} small $\eta$ + many trees + early stopping; keep trees shallow; add
    row/column subsampling if it helps. For the production versions, see \textbf{[20]}.}}
\end{frame}
