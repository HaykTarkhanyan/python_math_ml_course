\section{The cost of full fine-tuning}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\color{popblue}\Large\bfseries Why full fine-tuning does not scale}\\[4pt]
    \small And why existing efficient methods were not good enough.
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Full fine-tuning: as big as the original}
  Fine-tuning learns a task-specific increment $\Delta\Phi$ with $|\Delta\Phi|=|\Phi_0|$ --
  every parameter moves:
  \[
    \max_{\Phi}\sum_{(x,y)}\sum_{t}\log P_{\Phi}\big(y_t\mid x,y_{<t}\big)
  \]
  \begin{itemize}\small
    \item For GPT-2 / BERT this is an inconvenience. For GPT-3 175B it is a
      \textbf{deployment blocker}: a separate 175B model per task.
    \item You must store and \emph{serve} a full-size copy for every downstream use.
  \end{itemize}
  \vskip 4pt
  \keybox{Goal: encode the update with a \emph{tiny} parameter set $\Theta$, $|\Theta|\ll|\Phi_0|$
    -- as small as \textbf{0.01\%} of the model -- without giving up quality or adding
    latency.}
\end{frame}

\begin{frame}{Existing PEFT methods: not good enough}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Adapter layers}
    \begin{itemize}\small
      \item Insert small trainable layers \emph{in sequence} inside each block.
      \item Few parameters, but the extra depth is on the critical path.
      \item \textbf{Adds inference latency} -- worst for online, batch-1 serving (up to
        \textbf{+30\%}).
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Prefix / prompt tuning}
    \begin{itemize}\small
      \item Prepend trainable ``virtual tokens'' to the input.
      \item \textbf{Hard to optimize}; performance is non-monotonic in \# params.
      \item Prefix tokens \textbf{eat the sequence length} available for the actual task.
    \end{itemize}
  \end{columns}
  \vskip 6pt
  \warnbox{Both families often \emph{fail to match} full fine-tuning -- a real
    efficiency-vs-quality trade-off. LoRA aims to remove that trade-off entirely.}
\end{frame}

\section{The idea: intrinsic rank}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\color{popblue}\Large\bfseries The key hypothesis}\\[4pt]
    \small The \emph{update} during adaptation has a low intrinsic rank.
  \end{center}
  \vfill
\end{frame}

\begin{frame}{From intrinsic dimension to intrinsic rank}
  \textbf{Prior finding:} over-parameterized pretrained models live on a low
  \emph{intrinsic dimension} -- they can be fine-tuned in a tiny random subspace and still
  learn.
  \vskip 8pt
  \textbf{LoRA's hypothesis:} the \emph{change in weights} $\Delta W$ during adaptation also
  has a low \emph{``intrinsic rank.''} So even though $W_0$ is, say, $12{,}288\times12{,}288$,
  the update $\Delta W$ can be captured by a rank-$r$ factorization with $r$ as small as
  $1$ or $2$.
  \vskip 6pt
  \keybox{If $\Delta W$ is really low-rank, we should not need to learn a full
    $d\times k$ matrix -- just its two thin factors.}
\end{frame}

\section{The LoRA method}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\color{popblue}\Large\bfseries The method}\\[4pt]
    \small Freeze $W_0$; learn $\Delta W = BA$ in parallel.
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The reparameterization}
  \begin{columns}[T]
    \column{0.52\textwidth}
    Replace the update with a low-rank product:
    \[
      \Wz+\Delta W=\Wz+BA,\qquad
      \begin{aligned}
        &B\in\R^{d\times r},\ A\in\R^{r\times k}\\
        &r\ll\min(d,k)
      \end{aligned}
    \]
    Forward pass (with scaling $\alpha/r$):
    \[
      \boxed{\;h=\Wz\,x+\frac{\alpha}{r}\,BA\,x\;}
    \]
    \begin{itemize}\small
      \item $\Wz$ \textbf{frozen} (no gradients); train only $A,B$.
      \item Init $A\sim\mathcal{N}$, $B=0$ so $\Delta W=0$ at the start -- training begins
        exactly at the pretrained model.
      \item $\alpha$ acts like a learning-rate knob; set once, not tuned.
    \end{itemize}
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\footnotesize, node distance=0.5cm]
        \node (x) {$x$};
        \node[draw=armblue, very thick, fill=armblue!8, minimum width=2.0cm,
              minimum height=1.9cm, above left=1.1cm and -0.3cm of x, align=center] (w)
              {$\Wz$\\[2pt]\scriptsize $d\times k$\\ \scriptsize frozen};
        \node[draw=armorange, thick, fill=armorange!12, minimum width=1.5cm,
              above right=0.4cm and 0.2cm of x, align=center] (A)
              {$A$ \scriptsize $r\times k$};
        \node[draw=paramgreen, thick, fill=paramgreen!12, minimum width=1.5cm,
              above=0.55cm of A, align=center] (B) {$B$ \scriptsize $d\times r$};
        \node[circle, draw, above=3.3cm of x, inner sep=1pt] (plus) {$+$};
        \node[above=0.3cm of plus] (h) {$h$};
        \draw[-{Latex}] (x) -- (w);
        \draw[-{Latex}] (x) -| (A);
        \draw[-{Latex}] (A) -- (B) node[midway,right,font=\tiny]{rank $r$};
        \draw[-{Latex}] (w) |- (plus);
        \draw[-{Latex}] (B) |- (plus) node[near end,above,font=\tiny]{$\times\,\alpha/r$};
        \draw[-{Latex}] (plus) -- (h);
      \end{tikzpicture}
      \\[2pt]
      \scriptsize frozen $\Wz$ and trainable $BA$ run \emph{in parallel}.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Why it is (almost) free}
  \textbf{Parameter savings.} One matrix full update $=d\times k$. LoRA $=r(d+k)$, i.e.
  $2rd\ll d^2$. Total budget $|\Theta|=2\cdot\hat L\cdot d_{\text{model}}\cdot r$ over the
  $\hat L$ adapted matrices.
  \vskip 6pt
  \textbf{No inference latency.} Because $BA$ is added \emph{in parallel} (not in sequence),
  at deploy time you simply \textbf{merge}:
  \[
    W=\Wz+BA
  \]
  and run an ordinary layer -- \emph{zero} extra FLOPs. To switch tasks, subtract $BA$ and
  add a different $B'A'$.
  \vskip 4pt
  \watchbox{Merging removes latency but you lose \textbf{swap-ability}: a merged model can
    no longer batch \emph{different} tasks in one forward pass. Keep adapters un-merged when
    you need per-request task selection.}
\end{frame}

\section{Why does low rank work?}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\color{popblue}\Large\bfseries Why does such a low rank work?}\\[4pt]
    \small A look inside $\Delta W$.
  \end{center}
  \vfill
\end{frame}

\begin{frame}{$\Delta W$ amplifies directions already in $W$}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \includegraphics[width=\linewidth]{fig/amplification.pdf}
    \column{0.5\textwidth}
    Project the update $\Delta W$ onto the top singular directions of $W$ and compare
    magnitudes:
    \begin{itemize}\small
      \item $\Delta W$ overlaps directions that are present in $W$ but \emph{not emphasized}
        during pretraining.
      \item It \textbf{amplifies} them $\sim$\textbf{21$\times$} at $r{=}4$ (only $\sim$2$\times$
        at $r{=}64$).
      \item So adaptation surfaces a few task-specific features already latent in the model
        -- not new ones. That is why a tiny rank is enough.
    \end{itemize}
  \end{columns}
\end{frame}

\section{Recap}

\begin{frame}{Recap and practical notes}
  \begin{itemize}
    \item LoRA freezes $\Wz$ and learns $\Delta W=BA$ with rank $r\ll\min(d,k)$ --
      $h=\Wz x+\frac{\alpha}{r}BAx$.
    \item Init $B=0$: training starts exactly at the pretrained model.
    \item \textbf{10,000$\times$} fewer trainable params, \textbf{35 MB} checkpoints,
      \textbf{no} inference latency (merge $W=\Wz+BA$).
    \item Adapt $\{W_q,W_v\}$; rank as small as $1$--$8$ usually suffices.
    \item It works because $\Delta W$ has low intrinsic rank -- it amplifies latent
      task directions rather than inventing new ones.
    \item \textbf{Not} distillation / pruning / quantization: the base model is untouched
      and full-size; LoRA \emph{adds} a small trainable branch. Rank $r$ is a hyperparameter.
  \end{itemize}
  \vfill
  \nextbox{\textbf{Next:} \emph{LIMA} -- if fine-tuning is this cheap, how \textbf{much} data
    do we actually need? The answer is surprisingly little.}
\end{frame}
