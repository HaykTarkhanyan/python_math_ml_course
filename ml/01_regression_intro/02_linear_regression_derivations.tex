\section{Why a column of 1s?}

\begin{frame}{The intercept problem - can a line through the origin do it?}
  Suppose Yerevan studio rent vs.\ floor area. Even the smallest viable 25 m$^2$ studio doesn't cost (25 $\times$ per-m$^2$ price): there's a baseline (registration, condition, neighbourhood floor). A line forced through the origin can't absorb that baseline.
  \vfill
  \begin{center}
    \begin{tikzpicture}
      \begin{axis}[
        width=0.65\textwidth, height=4.5cm,
        xlabel={floor area $x$ (m$^2$)},
        ylabel={rent $y$ (kAMD)},
        xmin=0, xmax=80, ymin=0, ymax=400,
        legend pos=south east,
        legend style={font=\footnotesize},
        grid=major, grid style={dashed,gray!30},
      ]
        % data: y ≈ 100 + 3.5*x + noise
        \addplot[only marks, mark=*, armred, mark size=1.5pt] coordinates {
          (25,195) (30,205) (33,225) (38,228) (42,255)
          (47,270) (50,260) (55,295) (58,305) (62,320)
          (65,330) (70,345) (74,360) (78,370)
        };
        \addlegendentry{data}
        % through-origin: y = (avg y / avg x) * x  ~ 5.4 * x
        \addplot[armblue, very thick, dashed, domain=0:80] {5.4*x};
        \addlegendentry{$y = \theta_1 x$ (no intercept)}
        % with intercept: y ≈ 100 + 3.5x
        \addplot[paramgreen, very thick, domain=0:80] {100 + 3.5*x};
        \addlegendentry{$y = \theta_0 + \theta_1 x$ (with intercept)}
      \end{axis}
    \end{tikzpicture}
  \end{center}
\end{frame}

\begin{frame}{What we learn from the picture}
  \begin{itemize}
    \item The \textcolor{armblue}{dashed line} (no intercept) is forced through (0, 0). To get near the data cloud it has to be much steeper than the cloud actually slopes.
    \item Result: it under-shoots small apartments and over-shoots large ones.
    \item The \textcolor{paramgreen}{solid line} (with intercept) is free to start at $\theta_0 \approx 100$ kAMD and rise gently. Much smaller residuals.
  \end{itemize}
  \vfill
  Conclusion: we need an intercept term $\theta_0$. The next three frames make $\theta_0$ disappear into the same dot product as the other coefficients.
\end{frame}

\begin{frame}{Two parameters, one notation?}
  \textbf{Why we want this:} we're about to vectorize all $n$ predictions as a single matrix multiplication. A clean dot product makes that one line of NumPy code instead of a loop.
  \vfill
  We \emph{want} to write every prediction as:
  \[
    \fx = \thetav^\top \xv.
  \]
  But $\thetav = (\theta_0, \theta_1, \dots, \theta_p)^\top \in \R^{p+1}$ has \textbf{one more entry} than $\xv = (x_1, \dots, x_p)^\top \in \R^p$.
  \vfill
  Naive dot product:
  \[
    \thetav^\top \xv = \theta_1 x_1 + \dots + \theta_p x_p \quad \text{(missing $\theta_0$!)}
  \]
\end{frame}

\begin{frame}{The padding trick: prepend a 1}
  Replace every feature vector by an \textbf{augmented} version:
  \[
    \xv = (x_1, \dots, x_p)^\top
    \;\longmapsto\;
    \tilde{\xv} = (1, x_1, \dots, x_p)^\top \in \R^{p+1}.
  \]
  Then:
  \[
    \thetav^\top \tilde{\xv}
    = \theta_0 \cdot 1 + \theta_1 x_1 + \dots + \theta_p x_p
    = \theta_0 + \theta_1 x_1 + \dots + \theta_p x_p.
  \]
  \vfill
  \textbf{Now the intercept lives inside $\thetav$.} No special case in any formula.
  \vfill
  \emph{From now on we drop the tilde and just write $\xv \in \R^{p+1}$ with a leading 1.}
\end{frame}

\begin{frame}{What this buys us, concretely}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Without the trick:}
    \begin{itemize}\small
      \item Prediction: $\fx = \theta_0 + \sumjp \theta_j x_j$
      \item Two cases for the gradient (one for $\theta_0$, one for $\theta_j$)
      \item In code: carry $\theta_0$ as its own variable
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{With the trick:}
    \begin{itemize}\small
      \item Prediction: $\fx = \thetav^\top \xv$
      \item One gradient formula handles every parameter
      \item In code: \texttt{X = np.hstack([np.ones((n,1)), X])} once at the top, then everything is \texttt{X @ theta}
    \end{itemize}
  \end{columns}
  \vfill
  \textbf{Caveat:} regularization (L03) usually does \emph{not} penalize $\theta_0$. The penalty term acts on $\theta_{1:p}$ only - common gotcha when implementing Ridge by hand.
\end{frame}

\section{Design matrix and vectorized implementation}

\begin{frame}{One observation $\to$ matrix-vector product}
  For a single augmented input $\xv^{(i)} \in \R^{p+1}$:
  \[
    \yih = \thetav^\top \xv^{(i)}.
  \]
  Predictions for \emph{all} $n$ observations:
  \[
    \begin{pmatrix} \yh^{(1)} \\ \yh^{(2)} \\ \vdots \\ \yh^{(n)} \end{pmatrix}
    =
    \begin{pmatrix} \thetav^\top \xv^{(1)} \\ \thetav^\top \xv^{(2)} \\ \vdots \\ \thetav^\top \xv^{(n)} \end{pmatrix}.
  \]
  \vfill
  Each entry of the right-hand vector is the dot product of $\thetav$ with one input. We can collect all of them into a single matrix-vector product.
\end{frame}

\begin{frame}{The design matrix $\Xmat$}
  Stack the augmented inputs as the \textbf{rows} of $\Xmat$:
  \[
    \Xmat =
    \begin{pmatrix}
      1 & x_1^{(1)} & x_2^{(1)} & \cdots & x_p^{(1)} \\
      1 & x_1^{(2)} & x_2^{(2)} & \cdots & x_p^{(2)} \\
      \vdots & \vdots & \vdots & & \vdots \\
      1 & x_1^{(n)} & x_2^{(n)} & \cdots & x_p^{(n)}
    \end{pmatrix}
    \;\in\; \R^{n \times (p+1)}.
  \]
  \vfill
  Shape mnemonic: \textbf{\emph{rows are observations, columns are features.}} First column is always 1 (intercept).
  \vfill
  In NumPy: \texttt{X.shape == (n, p + 1)}.
\end{frame}

\begin{frame}{The full prediction in one expression}
  All $n$ predictions become a single matrix-vector product:
  \[
    \boxed{\;\cyh = \cX\,\cth\;}
    \quad\text{where}\quad
    \cyh \in \R^n,\;\; \Xmat \in \R^{n \times (p+1)},\;\; \cth \in \R^{p+1}.
  \]
  \vfill
  Shape check (do this every time you write a new matrix expression):
  \[
    \underbrace{(n \times (p+1))}_{\Xmat} \cdot \underbrace{((p+1) \times 1)}_{\thetav}
    = \underbrace{(n \times 1)}_{\hat{\yv}}.
  \]
  \vfill
  Residuals and SSE also vectorize:
  \[
    \cres = \cy - \cyh = \cy - \cX \cth,
    \qquad
    \riske = \sum_{i=1}^n r_i^2 = \cres^\top \cres = \| \cy - \cX \cth \|_2^2.
  \]
\end{frame}

\begin{frame}[fragile]{Vectorized vs.\ loop: code}
  \textbf{Slow (Python loop) - teaching example only, DO NOT use in real code:}
\begin{lstlisting}
yhat = np.zeros(n)
for i in range(n):                  # row by row
    s = 0.0
    for j in range(p + 1):          # element by element
        s += theta[j] * X[i, j]
    yhat[i] = s
\end{lstlisting}
  \vfill
  \textbf{Fast (vectorized NumPy) - the actual implementation:}
\begin{lstlisting}
yhat = X @ theta            # one matrix-vector product
residuals = y - yhat
sse = residuals @ residuals
\end{lstlisting}
  \vfill
  The \texttt{@} operator is matrix multiplication (covered in \texttt{python\_libs/02\_numpy}).
\end{frame}

\begin{frame}{Why vectorization wins}
  Two reasons the second version is dramatically faster:
  \begin{enumerate}
    \item \textbf{No per-element Python overhead.} Every iteration of the loop pays the cost of a Python bytecode dispatch. The vectorized version hands the whole job to a single call into compiled linear-algebra code (the BLAS library, written in C / Fortran).
    \item \textbf{The CPU can process many numbers in parallel.} BLAS routines stream contiguous chunks of memory through SIMD instructions, where one CPU instruction operates on 4-16 floats at once. The Python loop processes one number at a time.
  \end{enumerate}
  \vfill
  Rule of thumb on this laptop (Intel Iris Xe, 16 GB): for $n = 10^5$, $p = 50$, the loop takes seconds; \texttt{X @ theta} takes milliseconds.
  \vfill
  \textbf{Habit to build:} when you catch yourself writing a Python \texttt{for} loop over rows of an array, stop and ask ``what matrix op does this?''.
\end{frame}

\section{Deriving the normal equation}

\begin{frame}{The setup}
  We want the $\thetav$ that minimizes empirical risk:
  \[
    \cthh \in \argmin_{\thetav}\; \risket
    = \argmin_{\thetav}\; \sum_{i=1}^n \left( \yi - \thetav^\top \xv^{(i)} \right)^2
    = \argmin_{\thetav}\; \| \cy - \cX \cth \|_2^2.
  \]
  \vfill
  The function $\risket$ is a smooth (differentiable everywhere) function with no constraints on $\thetav$. So the strategy from calculus:
  \begin{enumerate}
    \item Expand the squared norm so we can differentiate.
    \item Take the gradient w.r.t.\ $\thetav$.
    \item Set the gradient to zero $\to$ this gives the \emph{normal equations}.
    \item Solve the resulting linear system for $\cthh$.
  \end{enumerate}
\end{frame}

\begin{frame}{Speedrun: the derivation in 4 lines}
  Before the careful version, here is the whole thing in 4 lines. Read these once, then we'll justify each step.
  \begin{align*}
    \risket &= \| \cy - \cX \cth \|_2^2
              = \cy^\top \cy - 2\,\cth^\top \cX^\top \cy + \cth^\top \cX^\top \cX\, \cth \\[2pt]
    \nabla_{\thetav} \risket &= -2\, \cX^\top \cy + 2\, \cX^\top \cX\, \cth \\[2pt]
    \text{set to 0:}\quad \cX^\top \cX\, \cthh &= \cX^\top \cy \quad \text{(\emph{normal equations})} \\[2pt]
    \boxed{\cthh = (\cX^\top \cX)^{-1} \cX^\top \cy} & \quad \text{(\emph{OLS estimator})}
  \end{align*}
  \vfill
  Two facts make this work:
  \begin{itemize}\small
    \item $\nabla_{\thetav}\, \mathbf{a}^\top \thetav = \mathbf{a}$ \quad and \quad $\nabla_{\thetav}\, \thetav^\top A\, \thetav = 2 A\, \thetav$ for symmetric $A$.
    \item $\cX^\top \cX$ is symmetric: $(\cX^\top \cX)^\top = \cX^\top (\cX^\top)^\top = \cX^\top \cX$. \checkmark
  \end{itemize}
\end{frame}

\begin{frame}{Step 1: expand the squared norm}
  Recall: $\| \mathbf{v} \|_2^2 = \mathbf{v}^\top \mathbf{v}$ for any vector $\mathbf{v}$ (the squared norm is the dot product with itself). Apply with $\mathbf{v} = \cy - \cX\cth$:
  \begin{align*}
    \risket
    &= (\cy - \cX\cth)^\top (\cy - \cX\cth) \\[2pt]
    &= \cy^\top \cy \;-\; \cy^\top \cX\cth \;-\; (\cX\cth)^\top \cy \;+\; (\cX\cth)^\top (\cX\cth).
  \end{align*}
  \vfill
  \textbf{Why the middle two terms are equal.} Each one is a scalar:
  \[
    \underbrace{\cy^\top}_{1 \times n}\,\underbrace{\cX}_{n \times (p+1)}\,\underbrace{\cth}_{(p+1) \times 1}
    \;\;\to\;\; 1 \times 1 \;\;\to\;\; \text{a number.}
  \]
  A scalar equals its own transpose, so $\cy^\top \cX\cth = (\cy^\top \cX\cth)^\top = \cth^\top \cX^\top \cy$. Combine:
  \[
    \risket = \cy^\top \cy \;-\; 2\, \cth^\top \cX^\top \cy \;+\; \cth^\top \cX^\top \cX\, \cth.
  \]
\end{frame}

\begin{frame}{Step 2: vector-calculus rules we need}
  \textbf{Two rules.} For constants $\mathbf{a} \in \R^d$ and symmetric $A \in \R^{d \times d}$:
  \[
    \boxed{\;\nabla_{\thetav}\, \mathbf{a}^\top \thetav = \mathbf{a}\;}
    \qquad
    \boxed{\;\nabla_{\thetav}\, \thetav^\top A\, \thetav = 2\,A\, \thetav\;}
  \]
  \vfill
  \textbf{Where rule 1 comes from} (one-line derivation):
  \[
    \mathbf{a}^\top \thetav = \sum_{j} a_j \theta_j
    \;\Rightarrow\;
    \frac{\partial}{\partial \theta_j}(\mathbf{a}^\top \thetav) = a_j
    \;\Rightarrow\;
    \nabla_{\thetav}(\mathbf{a}^\top \thetav) = \mathbf{a}.
  \]
  Rule 2 follows similarly (see math module 07 - multivariate calculus).
  \vfill
  \textbf{1D sanity check.} If $d = 1$ and $a, A$ are scalars,
  \[
    \tfrac{d}{d\theta}(a\theta) = a, \qquad \tfrac{d}{d\theta}(A\theta^2) = 2A\theta. \;\checkmark
  \]
  And $A = \cX^\top \cX$ is symmetric because $(\cX^\top \cX)^\top = \cX^\top \cX$, so rule 2 applies.
\end{frame}

\begin{frame}{Step 3: take the gradient}
  Differentiate $\risket = \cy^\top \cy - 2\,\cth^\top \cX^\top \cy + \cth^\top (\cX^\top \cX)\cth$ term-by-term:
  \begin{itemize}
    \item $\nabla_{\thetav}\, \cy^\top \cy = 0$ \quad (no $\cth$)
    \item $\nabla_{\thetav}\, \left(-2\, \cth^\top \cX^\top \cy\right) = -2\, \cX^\top \cy$ \quad (rule 1 with $\mathbf{a} = \cX^\top \cy$)
    \item $\nabla_{\thetav}\, \cth^\top (\cX^\top \cX)\, \cth = 2\, (\cX^\top \cX)\, \cth$ \quad (rule 2)
  \end{itemize}
  \vfill
  Sum:
  \[
    \nabla_{\thetav}\, \risket = -2\, \cX^\top \cy + 2\, \cX^\top \cX\, \cth.
  \]
\end{frame}

\begin{frame}{Step 4: set the gradient to zero}
  At a stationary point of $\risket$:
  \[
    -2\, \cX^\top \cy + 2\, \cX^\top \cX\, \cth = 0
    \quad\Longrightarrow\quad
    \boxed{\;\cX^\top \cX\, \cthh = \cX^\top \cy\;}
  \]
  \vfill
  These are the \textbf{normal equations} - a linear system in $\thetav$.
  \vfill
  \textbf{Why ``normal''?} The residual $\cres = \cy - \cX\cthh$ ends up \emph{orthogonal} (normal) to every column of $\cX$. The system above says exactly
  \[
    \cX^\top \cres = 0.
  \]
  Geometrically: each column of $\cX$ is a ``feature direction''. The optimal residual has no remaining correlation with any feature - if it did, we could reduce the loss by moving $\cth$ along that direction.
\end{frame}

\begin{frame}{Step 5: solve for $\cthh$}
  If $\cX^\top \cX$ is invertible, left-multiply both sides of $\cX^\top \cX\, \cthh = \cX^\top \cy$ by $(\cX^\top \cX)^{-1}$:
  \[
    \boxed{\;\cthh = (\cX^\top \cX)^{-1}\, \cX^\top \cy\;}
  \]
  This is the \textbf{ordinary least squares (OLS) estimator}.
  \vfill
  \textbf{Why this is the global minimum (not just a stationary point).} The Hessian of $\risket$ is
  \[
    \nabla^2_{\thetav}\, \risket = 2\, \cX^\top \cX.
  \]
  $\cX^\top \cX$ is positive semi-definite, meaning the loss surface is a \emph{bowl} (never curves down). A stationary point of a bowl is the bottom. So $\cthh$ is the global minimum.
  \vfill
  Bonus: closed-form $\to$ no iteration, no learning rate, no convergence questions. The price: $\cX^\top \cX$ has to be invertible (next section).
\end{frame}

\begin{frame}{Worked numerical example}
  Let $\cX = \begin{pmatrix} 1 & 1 \\ 1 & 2 \\ 1 & 3 \end{pmatrix}$, $\cy = \begin{pmatrix} 2 \\ 3 \\ 5 \end{pmatrix}$. Compute $\cthh$ by hand.
  \vfill
  \begin{columns}[T]
    \column{0.55\textwidth}
    \small
    \begin{align*}
      \cX^\top \cX
      &= \begin{pmatrix} 3 & 6 \\ 6 & 14 \end{pmatrix},
      \qquad
      \cX^\top \cy
      = \begin{pmatrix} 10 \\ 23 \end{pmatrix} \\[4pt]
      \det(\cX^\top \cX) &= 3 \cdot 14 - 6 \cdot 6 = 6 \\[4pt]
      (\cX^\top \cX)^{-1}
      &= \tfrac{1}{6}\!\begin{pmatrix} 14 & -6 \\ -6 & 3 \end{pmatrix} \\[4pt]
      \cthh
      &= \tfrac{1}{6}\!\begin{pmatrix} 14 & -6 \\ -6 & 3 \end{pmatrix}
        \!\begin{pmatrix} 10 \\ 23 \end{pmatrix}
      = \tfrac{1}{6}\!\begin{pmatrix} 2 \\ 9 \end{pmatrix}
      = \begin{pmatrix} 1/3 \\ 3/2 \end{pmatrix}.
    \end{align*}
    \column{0.45\textwidth}
    \small
    \textbf{Fitted line:} $\hat{y} = \tfrac{1}{3} + \tfrac{3}{2}\, x$.
    \vskip 4pt
    \textbf{Residuals.} $\cyh = \cX\cthh = (\tfrac{11}{6}, \tfrac{10}{3}, \tfrac{29}{6})$, so
    \[
      \cres = \cy - \cyh = (\tfrac{1}{6}, -\tfrac{2}{6}, \tfrac{1}{6}).
    \]
    \textbf{Orthogonality check:} $\cX^\top \cres$ should be $0$.
    \[
      \tfrac{1}{6} - \tfrac{2}{6} + \tfrac{1}{6} = 0 \;\checkmark
    \]
    \[
      1 \cdot \tfrac{1}{6} + 2 \cdot (-\tfrac{2}{6}) + 3 \cdot \tfrac{1}{6} = 0 \;\checkmark
    \]
    Both columns of $\cX$ are orthogonal to $\cres$, exactly as the normal equations require.
  \end{columns}
\end{frame}

\begin{frame}{Geometric picture}
  $\cyh = \cX \cthh$ is the \textbf{orthogonal projection} of $\cy$ onto the \textbf{column space} of $\cX$ (the set of all vectors $\cX\thetav$ as $\thetav$ ranges over $\R^{p+1}$ - the subspace reachable from our features).
  \vfill
  \begin{center}
    \begin{tikzpicture}[scale=1]
      \fill[armblue!10] (-2.6,-0.8) -- (2.6,-1.4) -- (3.0,0.8) -- (-2.2,1.4) -- cycle;
      \draw[armblue!50, thick] (-2.6,-0.8) -- (2.6,-1.4) -- (3.0,0.8) -- (-2.2,1.4) -- cycle;
      \node[armblue] at (-2.0, 1.0) {\small col($\Xmat$)};
      \draw[->, armred, very thick] (0,0) -- (1.4, 2.2) node[above right] {$\cy$};
      \draw[->, paramgreen, very thick] (0,0) -- (1.4, 0.2) node[below right=2pt] {$\cyh = \cX \cthh$};
      \draw[->, armorange, very thick] (1.4, 0.2) -- (1.4, 2.2);
      \node[armorange] at (2.1, 1.2) {$\cres$};
      \draw[armorange] (1.4, 0.2) -- (1.25, 0.2) -- (1.25, 0.35) -- (1.4, 0.35);
    \end{tikzpicture}
  \end{center}
  \textbf{OLS in one sentence:} pick $\cth$ so that the residual is perpendicular to every feature direction.
\end{frame}

\section{When OLS fails (and what to do)}

\begin{frame}{When does $\cX^\top \cX$ fail to be invertible?}
  \textbf{Fact (from linear algebra):} $\cX^\top \cX$ is invertible exactly when the $p+1$ columns of $\cX$ are linearly independent.
  \vfill
  Three common ways the columns become \emph{dependent}:
  \begin{itemize}
    \item \textbf{Multicollinearity.} Two features are essentially the same. Example: weight in kg and weight in pounds (one is a constant multiple of the other).
    \item \textbf{Redundant encoding (the dummy trap).} Including \emph{all} dummy columns for a categorical variable AND an intercept. The dummies always sum to the all-ones column.
    \item \textbf{$p \geq n$.} More features than observations. Columns live in $\R^n$ - you can't fit $p+1 > n$ linearly independent vectors in $\R^n$.
  \end{itemize}
\end{frame}

\begin{frame}{What to do when it fails}
  Options, in order of practicality:
  \begin{enumerate}
    \item \textbf{Drop the redundant column.} Cheap fix for the dummy trap and for obvious unit-of-measure duplicates.
    \item \textbf{Use the pseudo-inverse:} \texttt{theta = np.linalg.pinv(X) @ y}. Works via SVD; returns the minimum-norm solution when the system is under-determined.
    \item \textbf{Regularize.} Replace $\cX^\top \cX$ with $\cX^\top \cX + \lambda \id$. Always invertible for $\lambda > 0$. This is \textbf{Ridge regression} - L03.
    \item \textbf{Gradient descent.} Works even when the closed form doesn't, and scales to large $n, p$ - L02.
  \end{enumerate}
\end{frame}

\begin{frame}{Computational cost}
  Forming and solving the normal equations costs:
  \begin{itemize}
    \item Building $\cX^\top \cX$: $\mathcal{O}(n p^2)$ multiplications.
    \item Solving the $(p+1) \times (p+1)$ system (LU / Cholesky): $\mathcal{O}(p^3)$.
    \item Total: $\mathcal{O}(n p^2 + p^3)$.
  \end{itemize}
  \vfill
  Big-$\mathcal{O}$ implications:
  \begin{itemize}
    \item Fine for $p \lesssim 10^3$. A laptop solves it instantly.
    \item Bad for $p \gtrsim 10^4$. The $p^3$ term dominates. Use gradient descent (L02) or stochastic methods.
    \item Numerically: prefer \texttt{np.linalg.lstsq(X, y)} or \texttt{np.linalg.solve(X.T @ X, X.T @ y)} over an explicit \texttt{np.linalg.inv(...)} call. Same answer, much better conditioning.
  \end{itemize}
\end{frame}

\section{Polynomial regression}

\begin{frame}{Same model, different features}
  Linear regression is linear \textbf{in the parameters $\thetav$}, not in the inputs $\xv$. Nothing stops us from feeding it nonlinear functions of $x$.
  \vfill
  For a single feature $x \in \R$, the degree-$d$ polynomial model is
  \[
    \fx = \theta_0 + \theta_1 x + \theta_2 x^2 + \dots + \theta_d x^d.
  \]
  \vfill
  \textbf{``But $x^2$ is nonlinear!''} Yes - the \emph{fitted function} is nonlinear in $x$. But the \emph{optimization problem} is still linear regression: we treat $x^2$ as just another column of the design matrix. Define the basis expansion
  \[
    \phi(x) = (1, x, x^2, \dots, x^d)^\top \in \R^{d+1},
  \]
  then $\fx = \thetav^\top \phi(x)$ - linear in $\thetav$.
\end{frame}

\begin{frame}{Polynomial design matrix}
  Stack the basis-expanded inputs into a new design matrix:
  \[
    \Xmat_\phi =
    \begin{pmatrix}
      1 & x^{(1)} & (x^{(1)})^2 & \cdots & (x^{(1)})^d \\
      1 & x^{(2)} & (x^{(2)})^2 & \cdots & (x^{(2)})^d \\
      \vdots & \vdots & \vdots & & \vdots \\
      1 & x^{(n)} & (x^{(n)})^2 & \cdots & (x^{(n)})^d
    \end{pmatrix}
    \in \R^{n \times (d+1)}.
  \]
  \vfill
  \textbf{Key observation:} from OLS's perspective $\Xmat_\phi$ is just another design matrix. The normal equation is unchanged:
  \[
    \cthh = (\Xmat_\phi^\top \Xmat_\phi)^{-1}\, \Xmat_\phi^\top \cy.
  \]
  \vfill
  In code: \texttt{from sklearn.preprocessing import PolynomialFeatures}. For multiple features, the same idea also generates \emph{interaction} terms like $x_1 x_2$, $x_1^2 x_2$.
\end{frame}

\begin{frame}{What polynomial regression teaches us}
  \begin{enumerate}
    \item \textbf{Linearity is in $\thetav$, not in $\xv$.} Almost every model in classical ML (kernel methods, basis splines, fixed-feature neural networks) is ``linear regression on engineered features''.
    \item \textbf{More features = more flexibility = more overfitting risk.} The U-shape is the bias-variance tradeoff (formalized in L04).
    \item \textbf{We need a way to penalize complexity.} If we can't drop features by hand, can we penalize their size? $\to$ \textbf{regularization} (L03).
    \item \textbf{We need a way to evaluate honestly.} Training error is not enough. $\to$ \textbf{hold-out + cross-validation} (L04, L05).
  \end{enumerate}
\end{frame}

\begin{frame}{Wrap-up}
  \begin{itemize}\small
    \item Pad inputs with a 1 so the intercept lives inside $\cth$. One formula everywhere.
    \item Design matrix $\cX$: rows are observations, columns are features. Predictions vectorize as $\cyh = \cX \cth$.
    \item Normal equation $(\cX^\top \cX)\, \cthh = \cX^\top \cy$ comes from $\nabla \risket = 0$. Geometrically: residual perpendicular to every feature column.
    \item OLS fails when $\cX^\top \cX$ is singular. Fixes: drop a column, pseudo-inverse, ridge, or gradient descent.
    \item Polynomial regression is OLS with $\phi(x) = (1, x, x^2, \dots, x^d)$. Same algebra, more flexibility - and more overfitting risk.
  \end{itemize}
