\section{Backpropagation}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Backpropagation}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      The chain rule, applied backward once, with every shared piece reused.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The idea: chain rule, backward, reuse}
  \begin{columns}[T]
    \column{0.54\textwidth}
    The loss is a \textbf{composition}: input $\to$ hidden $\to$ output $\to$ loss. The chain
    rule turns $\partial L/\partial w$ into a product of local derivatives along the path from
    $w$ to $L$.
    \vskip 6pt
    Compute these products \textbf{from the output backward}, and every layer reuses the
    gradient already computed for the layer above it. That reuse is the whole trick -- one
    backward sweep, not one per weight.
    \vskip 6pt
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      \textbf{Two sweeps per step:} a \textbf{forward} pass caches activations; a
      \textbf{backward} pass turns them into gradients.}}
    \column{0.44\textwidth}
    \begin{center}
      \begin{tikzpicture}[>=Latex, font=\scriptsize,
        n/.style={circle,draw,minimum size=0.6cm,inner sep=0}]
        \node[n,draw=gray,fill=gray!8] (x) at (0,0){$\xv$};
        \node[n,draw=armblue,fill=armblue!12] (z) at (1.8,0){$\mathbf{z}$};
        \node[n,draw=armorange,fill=armorange!16] (f) at (3.6,0){$f$};
        \node (L) at (5.1,0){$L$};
        \draw[armblue,->,thick] (x) to[bend left=35] (z);
        \draw[armblue,->,thick] (z) to[bend left=35] (f);
        \draw[armblue,->,thick] (f) to[bend left=35] (L);
        \draw[armred,->,thick] (L) to[bend left=35] (f);
        \draw[armred,->,thick] (f) to[bend left=35] (z);
        \draw[armred,->,thick] (z) to[bend left=35] (x);
        \node[armblue,font=\tiny] at (2.7,1.05){forward: activations};
        \node[armred,font=\tiny] at (2.4,-1.05){backward: gradients};
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Forward pass: cache what the backward pass will need}
  \begin{columns}[T]
    \column{0.42\textwidth}
    Reuse the Lecture-14 net: $2$ inputs, $2$ sigmoid hidden, $1$ sigmoid output. Input
    $\xv=(2,-1)$, target $y=1$.
    \vskip 3pt
    \footnotesize
    \begin{align*}
      \mathbf{w}_1&=(0.5,-0.5),\,b_1=0\\
      \mathbf{w}_2&=(-0.3,0.8),\,b_2=0.1\\
      \mathbf{u}&=(1.0,-1.0),\,c=0.2
    \end{align*}
    \begin{center}
      \begin{tikzpicture}[>=Latex, font=\tiny,
        n/.style={circle,draw,minimum size=0.4cm,inner sep=0}]
        \node[n,draw=gray,fill=gray!8] (i1) at (0,0.7){$x_1$};
        \node[n,draw=gray,fill=gray!8] (i2) at (0,-0.7){$x_2$};
        \node[n,draw=armblue,fill=armblue!12] (z1) at (1.5,0.7){$z_1$};
        \node[n,draw=armblue,fill=armblue!12] (z2) at (1.5,-0.7){$z_2$};
        \node[n,draw=armorange,fill=armorange!16] (o) at (3.0,0){$f$};
        \foreach \a in {i1,i2}\foreach \b in {z1,z2}\draw[gray!55,->](\a)--(\b);
        \foreach \b in {z1,z2}\draw[armblue!60,->](\b)--(o);
      \end{tikzpicture}
    \end{center}
    \column{0.56\textwidth}
    \footnotesize Cache every intermediate (we need them going back):
    \begin{align*}
      z_{1,\text{in}} &= 0.5(2){+}(-0.5)(-1){+}0 = 1.5,\; & z_1 &= \sigma(1.5)=0.82\\
      z_{2,\text{in}} &= -0.3(2){+}0.8(-1){+}0.1 = -1.3,\; & z_2 &= \sigma(-1.3)=0.21\\
      f_{\text{in}} &= 1.0(0.82){-}1.0(0.21){+}0.2 = 0.80,\; & f &= \sigma(0.80)=0.69
    \end{align*}
    \textbf{Cross-entropy loss} (target $y=1$):
    \[
      L = -\big(y\log f + (1{-}y)\log(1{-}f)\big) = -\log(0.69) = 0.37
    \]
  \end{columns}
\end{frame}

\begin{frame}{The output gradient is just $f-y$}
  Before the numbers, one clean fact. With a \textbf{sigmoid output} and \textbf{cross-entropy}
  loss, the gradient at the output pre-activation collapses:
  \vfill
  \[
    \frac{\partial L}{\partial f} = -\frac{y}{f} + \frac{1-y}{1-f},
    \qquad \frac{\partial f}{\partial f_{\text{in}}} = f(1-f)
  \]
  \[
    \frac{\partial L}{\partial f_{\text{in}}}
    = \left(-\frac{y}{f} + \frac{1-y}{1-f}\right) f(1-f)
    = -(y(1-f) - (1-y)f) = \boxed{\,f - y\,}
  \]
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{The same ``residual'' you already know.} Linear regression, logistic regression,
    and now a neural net: the output-layer gradient is \textbf{prediction $-$ target}. Here
    $f-y = 0.69 - 1 = -0.31$.}}
\end{frame}

\begin{frame}{Backward pass I: the output-layer weights}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Call the output signal $\delta_{\text{out}} = \dfrac{\partial L}{\partial f_{\text{in}}} =
    f - y = -0.31$.
    \vskip 4pt
    Each output weight's gradient is $\delta_{\text{out}}$ times the input on that edge
    ($z_j$ from the forward cache):
    \begin{align*}
      \tfrac{\partial L}{\partial u_1} &= \delta_{\text{out}}\, z_1 = -0.31(0.82) = -0.25\\
      \tfrac{\partial L}{\partial u_2} &= \delta_{\text{out}}\, z_2 = -0.31(0.21) = -0.07\\
      \tfrac{\partial L}{\partial c} &= \delta_{\text{out}} = -0.31
    \end{align*}
    \column{0.5\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      \textbf{Pattern.} Every weight gradient has the form
      \[
        \frac{\partial L}{\partial (\text{weight})} = (\text{signal at its output end})
        \times (\text{activation at its input end}).
      \]
      The forward cache supplies the second factor; the backward pass supplies the first.}}
    \vskip 4pt
    \footnotesize Negative gradients $\Rightarrow$ gradient descent will \emph{increase}
    $u_1, u_2, c$, pushing $f$ up toward the target $y=1$. Good.
  \end{columns}
\end{frame}

\begin{frame}{Backward pass II: reach the hidden-layer weights}
  \begin{columns}[T]
    \column{0.52\textwidth}
    Push the signal back through the output weight and the hidden sigmoid. For hidden neuron
    $j$, with $\sigma'(z_{\text{in}}) = z(1-z)$:
    \[
      \delta_j = \delta_{\text{out}}\, u_j\, \underbrace{z_j(1-z_j)}_{\sigma'(z_{j,\text{in}})}
    \]
    \footnotesize
    \begin{align*}
      \delta_1 &= -0.31(1.0)(0.82)(0.18) = -0.046\\
      \delta_2 &= -0.31(-1.0)(0.21)(0.79) = +0.052
    \end{align*}
    Then, same pattern as before ($\delta_j \times$ input $x_i$):
    \begin{align*}
      \tfrac{\partial L}{\partial \mathbf{w}_1} &= \delta_1\,\xv = (-0.092,\, 0.046)\\
      \tfrac{\partial L}{\partial \mathbf{w}_2} &= \delta_2\,\xv = (0.104,\, -0.052)
    \end{align*}
    \column{0.48\textwidth}
    \begin{center}
      \begin{tikzpicture}[>=Latex, font=\tiny,
        n/.style={circle,draw,minimum size=0.5cm,inner sep=0}]
        \node[n,draw=gray,fill=gray!8] (i1) at (0,0.9){$x_1$};
        \node[n,draw=gray,fill=gray!8] (i2) at (0,-0.9){$x_2$};
        \node[n,draw=armblue,fill=armblue!12] (z1) at (1.7,0.9){$z_1$};
        \node[n,draw=armblue,fill=armblue!12] (z2) at (1.7,-0.9){$z_2$};
        \node[n,draw=armorange,fill=armorange!16] (o) at (3.4,0){$f$};
        \foreach \a in {i1,i2}\foreach \b in {z1,z2}\draw[gray!45,-](\a)--(\b);
        \foreach \b in {z1,z2}\draw[gray!45,-](\b)--(o);
        \draw[armred,->,very thick] (o) to[bend left=25] (z1);
        \draw[armred,->,very thick] (o) to[bend right=25] (z2);
        \draw[armred,->,very thick] (z1) to[bend left=25] (i1);
        \draw[armred,->,very thick] (z2) to[bend right=25] (i2);
        \node[armred,font=\tiny] at (2.55,1.25){$\delta_{\text{out}}$};
      \end{tikzpicture}\\[2pt]
      \scriptsize The signal flows right to left; each node scales it by a local derivative.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Why backprop is cheap: each $\delta$ computed once}
  \begin{columns}[T]
    \column{0.54\textwidth}
    Look again at the hidden-layer gradients. \emph{Both} weights feeding neuron $1$ reused
    the \textbf{same} signal $\delta_1$:
    \[
      \frac{\partial L}{\partial \mathbf{w}_1} = \delta_1\,\xv
      = \big(\underbrace{\delta_1 x_1}_{\partial L/\partial w_{11}},\;
             \underbrace{\delta_1 x_2}_{\partial L/\partial w_{12}}\big)
    \]
    And $\delta_1$ was itself built from $\delta_{\text{out}}$ -- the signal one layer above.
    \vskip 4pt
    So each layer computes its $\delta$ \textbf{once} from the layer above, then every weight
    into that layer is just $\delta \times \text{input}$. Nothing is recomputed.
    \column{0.46\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      \textbf{Cost, side by side ($5$ weights here):}
      \begin{itemize}\scriptsize
        \item \textbf{Naive:} nudge each weight, re-run forward $\to$ $5$ forward passes.
        \item \textbf{Backprop:} one backward sweep reuses the shared $\delta$'s $\to$ \emph{one}
          pass gives all $5$ gradients.
      \end{itemize}}}
    \vskip 4pt
    \footnotesize On the MNIST net that is $109{,}386$ forward passes vs.\ \textbf{one}. The
    reused $\delta$ is the entire reason training is affordable.
  \end{columns}
\end{frame}

\begin{frame}{One step of gradient descent, and the loss drops}
  We now have a gradient for every weight. Update each by $w \leftarrow w - \eta\,\partial
  L/\partial w$ with learning rate $\eta = 0.5$:
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \begin{align*}
      u_1 &: 1.0 - 0.5(-0.25) = 1.13\\
      u_2 &: -1.0 - 0.5(-0.07) = -0.97\\
      c &: 0.2 - 0.5(-0.31) = 0.35\\
      \mathbf{w}_1 &: (0.5,-0.5) - 0.5(-0.092,0.046)\\
      \mathbf{w}_2 &: (-0.3,0.8) - 0.5(0.104,-0.052)
    \end{align*}
    \column{0.5\textwidth}
    Re-run the forward pass with the updated weights:
    \[
      f: 0.69 \;\to\; 0.75, \qquad L: 0.37 \;\to\; 0.28
    \]
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      The prediction moved \textbf{toward the target} ($y=1$) and the loss \textbf{fell}. That
      is one training iteration. Real training does this for \emph{thousands} of mini-batches.}}
  \end{columns}
\end{frame}

\begin{frame}{Sanity check: is your backprop even correct?}
  Backprop is easy to get subtly wrong. Check it against a \textbf{numerical gradient} from
  the definition of a derivative:
  \[
    \frac{\partial L}{\partial w} \;\approx\; \frac{L(w+\varepsilon) - L(w-\varepsilon)}{2\varepsilon},
    \qquad \varepsilon \approx 10^{-5}
  \]
  \begin{itemize}\small
    \item Nudge \emph{one} weight up and down, re-run the forward pass, take the difference.
    \item Compare to your analytic (backprop) gradient -- they should agree to $\sim 10^{-7}$.
    \item If they do not, your backward pass has a bug.
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Debug only.} This costs \emph{two} forward passes \emph{per weight} -- the very
    cost backprop avoids. Use it on a tiny net to validate the code, never as the trainer.}}
\end{frame}

\begin{frame}{Backprop, vectorised: the same rule for a whole network}
  By hand for $5$ weights; as matrices for a million. Let $\delta^{(l)}$ be the signal at
  layer $l$'s pre-activations. Two rules, applied from the last layer backward:
  \vfill
  \[
    \delta^{(L)} = \nabla_{\!a} L \odot \varphi'\!\big(\mathbf{z}^{(L)}\big),
    \qquad
    \delta^{(l)} = \big(\mathbf{W}^{(l+1)\top}\delta^{(l+1)}\big) \odot \varphi'\!\big(\mathbf{z}^{(l)}\big)
  \]
  \[
    \frac{\partial L}{\partial \mathbf{W}^{(l)}} = \delta^{(l)}\,\big(\mathbf{a}^{(l-1)}\big)^{\!\top},
    \qquad
    \frac{\partial L}{\partial \mathbf{b}^{(l)}} = \delta^{(l)}
  \]
  \vfill
  \begin{itemize}\small
    \item $\odot$ is elementwise; $\varphi'$ is the activation's derivative (e.g.\ $z(1-z)$ for
      sigmoid).
    \item Exactly the scalar steps you just did -- ``signal $\times$ local derivative'',
      ``signal $\times$ input'' -- now one matrix multiply per layer.
  \end{itemize}
\end{frame}

\begin{frame}{Why deep sigmoid nets barely learn}
  The recursion $\delta^{(l)} = \big(\mathbf{W}^{(l+1)\top}\delta^{(l+1)}\big)\odot
  \varphi'(\mathbf{z}^{(l)})$ multiplies by $\varphi'$ at \emph{every} layer on the way back.
  \vfill
  \begin{columns}[T]
    \column{0.56\textwidth}
    \begin{itemize}\small
      \item Sigmoid: $\varphi' \le 0.25$. After $k$ layers the gradient is scaled by
        $\le 0.25^{k}$ -- it \textbf{vanishes exponentially} with depth. Early layers get
        almost no signal.
      \item ReLU: $\varphi' = 1$ for $z>0$ -- the gradient passes through undiminished.
    \end{itemize}
    \vskip 4pt
    This is the \textbf{vanishing-gradient} problem we flagged in L14, now visible in the math.
    Three fixes, all coming up: \textbf{ReLU}, careful \textbf{initialization}, and
    \textbf{BatchNorm}.
    \column{0.44\textwidth}
    \fcolorbox{armred}{armred!8}{\parbox{0.95\linewidth}{\small
      $0.25^{10} \approx 10^{-6}$. A $10$-layer sigmoid net shrinks the gradient a
      \emph{millionfold} before it reaches the first layer -- so the first layer effectively
      stops learning.}}
  \end{columns}
\end{frame}

\begin{frame}{Backprop is the chain rule on a computational graph}
  \begin{columns}[T]
    \column{0.55\textwidth}
    Everything from input to loss is one \textbf{computational graph}: each intermediate
    ($z_{\text{in}}$, $z$, $f_{\text{in}}$, $f$, $L$) is a \textbf{node} -- a function of its
    parents. The forward pass just evaluates the nodes top to bottom.
    \vskip 6pt
    A weight $w$ affects $L$ only \emph{along the path} from $w$ to $L$. The chain rule says:
    multiply the \textbf{local derivative} on each edge of that path.
    \[
      \frac{\partial L}{\partial w}
      = \frac{\partial L}{\partial f}\,
        \frac{\partial f}{\partial f_{\text{in}}}\,
        \frac{\partial f_{\text{in}}}{\partial z}\,
        \frac{\partial z}{\partial z_{\text{in}}}\,
        \frac{\partial z_{\text{in}}}{\partial w}
    \]
    \footnotesize That is exactly the product you formed by hand -- now named.
    \column{0.45\textwidth}
    \begin{center}
      \begin{tikzpicture}[>=Latex, font=\scriptsize, node distance=0.62cm,
        v/.style={draw, rounded corners, minimum width=1.4cm, minimum height=0.46cm, inner sep=1pt}]
        \node[v, draw=gray, fill=gray!8] (w) {$w$};
        \node[v, draw=armblue, fill=armblue!12, below=of w] (zin) {$z_{\text{in}}$};
        \node[v, draw=armblue, fill=armblue!12, below=of zin] (z) {$z$};
        \node[v, draw=armorange, fill=armorange!14, below=of z] (fin) {$f_{\text{in}}$};
        \node[v, draw=armorange, fill=armorange!14, below=of fin] (f) {$f$};
        \node[v, below=of f] (L) {$L$};
        \draw[->] (w)--(zin);
        \draw[->] (zin)--(z) node[midway,right,font=\tiny]{$\varphi$};
        \draw[->] (z)--(fin);
        \draw[->] (fin)--(f) node[midway,right,font=\tiny]{$\varphi$};
        \draw[->] (f)--(L);
      \end{tikzpicture}\\[2pt]
      \scriptsize forward evaluates down; the chain rule multiplies edge-derivatives back up.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Reverse mode: one sweep, any architecture}
  \begin{columns}[T]
    \column{0.54\textwidth}
    When a node feeds \textbf{several} children (a hidden unit into many outputs), its gradient
    is the \textbf{sum over all outgoing paths} -- the multivariate chain rule:
    \[
      \frac{\partial L}{\partial z} = \sum_{k}
        \frac{\partial L}{\partial f_{\text{in},k}}\,
        \frac{\partial f_{\text{in},k}}{\partial z}
    \]
    \textbf{Reverse-mode autodiff} visits each node \emph{once}, from $L$ backward, storing
    $\partial L/\partial(\text{node})$ as it goes -- exactly the cached $\delta$'s, now on the
    whole graph.
    \column{0.46\textwidth}
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      \textbf{Why this matters.} The rule is stated on the \emph{graph}, not on ``layers''. So
      the same engine differentiates \textbf{convolutions, attention, residual connections} --
      any composition of differentiable ops -- with no new math. That is what makes deep-learning
      frameworks general.}}
    \vskip 4pt
    \footnotesize You built the graph for one net by hand; autograd builds it for \emph{any} net
    automatically. Next slide.
  \end{columns}
\end{frame}

\begin{frame}[fragile]{You will never write this by hand: autograd}
  Frameworks record every operation of the forward pass into a graph, then replay it backward
  -- running exactly the two rules from two slides ago -- when you call \texttt{.backward()}.
  \vfill
{\lstset{basicstyle=\ttfamily\small}
\begin{lstlisting}
loss = loss_fn(model(x), y)   # forward: build the graph
loss.backward()               # backward: backprop
\end{lstlisting}}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    You still need to \emph{understand} backprop -- to read training curves, debug dead
    layers, and know why ReLU and good init matter -- but you will call
    \texttt{loss.backward()}, not differentiate by hand.}}
\end{frame}

\section{The training loop}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{The training loop}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Mini-batches, an optimizer, and the one hyperparameter you always touch.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Batch, stochastic, mini-batch}
  Gradient descent needs the gradient of the loss over \emph{some} data. How much?
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item \textbf{Batch GD:} use all $n$ examples per step. Accurate gradient, but one slow
        step; infeasible for large data.
      \item \textbf{Stochastic GD:} one example per step. Cheap, very noisy.
      \item \textbf{Mini-batch GD:} a small batch (e.g.\ $32$--$256$) per step. The standard --
        good gradient estimate, hardware-friendly, and the noise even helps escape bad spots.
    \end{itemize}
    \column{0.5\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      One \textbf{epoch} $=$ one full pass over the training set $=$ many mini-batch steps.
      You train for many epochs. Same gradient-descent idea as regression -- just fed data in
      batches.}}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{The training loop, in full}
  Everything in this lecture is these few lines. The rest is choosing what goes in them.
  \vfill
{\lstset{basicstyle=\ttfamily\small}
\begin{lstlisting}
for epoch in range(n_epochs):
    for xb, yb in loader:        # mini-batches
        optimizer.zero_grad()    # reset grads
        out = model(xb)          # forward pass
        loss = loss_fn(out, yb)
        loss.backward()          # backprop
        optimizer.step()         # update weights
\end{lstlisting}}
  \vfill
  \footnotesize The full runnable version -- data loading, evaluation, saving -- lives in the
  practical, not here.
\end{frame}

\begin{frame}{Which optimizer? Reuse the optimization module}
  \texttt{optimizer.step()} hides a real choice. Plain mini-batch \textbf{SGD} works, but two
  refinements are near-universal:
  \vfill
  \begin{itemize}\small
    \item \textbf{Momentum:} accumulate a running ``velocity'' of past gradients -- rolls
      through small bumps and noise instead of zig-zagging.
    \item \textbf{Adam:} keep a \emph{per-parameter} adaptive step size (roughly, momentum $+$
      per-weight scaling). The common default.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{This is a topic you have already covered.} For how momentum and Adam actually work,
    see the course Optimization module -- \emph{``10 Optim: Momentum $+$ First Order Methods''}
    (and \emph{``09 Optim: Prerequisites and Gradient Descent''}). \textbf{Practical default:
    start with Adam.}}}
\end{frame}

\begin{frame}{The one knob you always touch: learning rate}
  \begin{columns}[T]
    \column{0.48\textwidth}
    The learning rate $\eta$ scales every step.
    \begin{itemize}\small
      \item \textbf{Too high:} steps overshoot; the loss oscillates or diverges.
      \item \textbf{Too low:} steps are tiny; training crawls and may stall.
      \item \textbf{Right:} steady, fast decrease.
    \end{itemize}
    \vskip 4pt
    \textbf{Predict:} which of the three curves is too high, too low, and just right?
    \onslide<2->{\vskip 2pt
    {\footnotesize \textbf{Answer:} lr $=1.0$ (red) stalls high -- steps overshoot; lr $=0.002$
    (blue) barely moves; lr $=0.1$ (orange) is the steady, fast drop. Learning rate is the
    first thing to tune.}}
    \column{0.52\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/lr_curves.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Beyond a fixed rate: learning-rate schedules}
  A single fixed $\eta$ is a compromise: big enough to move early, small enough to settle late.
  \textbf{Schedules} vary $\eta$ over training to get both.
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item \textbf{Decay:} start larger, shrink over time (step, linear, or exponential) --
        fast progress first, fine-tuning later. The standard.
      \item \textbf{Warmup:} start \emph{tiny} for a few hundred steps, then ramp up -- stops a
        fresh, unstable net from diverging on the first big step.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item \textbf{Cyclical / cosine:} let $\eta$ rise and fall in cycles -- the bumps can kick
        the optimizer out of saddles and poor minima.
    \end{itemize}
    \vskip 4pt
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      \textbf{Practical:} a fixed rate with Adam is a fine start; add \emph{decay} (or warmup
      $+$ cosine, the modern default for big models) once you are tuning for the last few
      points.}}
  \end{columns}
\end{frame}

\begin{frame}{A word on the loss landscape}
  With $100$k weights the loss surface is wildly \textbf{non-convex} -- countless local minima
  and saddle points. Classical optimization would despair.
  \vfill
  \begin{itemize}\small
    \item In practice, mini-batch SGD/Adam finds \emph{good enough} minima reliably; in high
      dimensions most are about equally good, and the noise helps escape bad ones.
    \item \emph{Why} this works so well is still an active research question -- we take it as an
      empirical fact and move on.
  \end{itemize}
  \vfill
  \footnotesize Contrast with linear/logistic regression (convex: one global minimum, L11).
  Depth buys power at the cost of a messy landscape.
\end{frame}

\begin{frame}{Saddles and cliffs -- and how we cope}
  Two obstacles appear once the landscape is non-convex and high-dimensional:
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Saddle points dominate.} At a flat spot, a true minimum needs \emph{every}
    curvature direction to bend \textbf{up}. With thousands of dimensions that is exponentially
    unlikely, so almost every flat spot is a \textbf{saddle} (up in some directions, down in
    others), not a trap.
    \vskip 3pt
    \footnotesize Good news: the \textbf{noise} in mini-batch SGD nudges you off saddles that a
    noiseless method would sit on forever.
    \column{0.5\textwidth}
    \textbf{Cliffs / exploding gradients.} Deep and recurrent nets can have near-vertical walls
    in the loss surface; one step there flings the weights far and undoes training.
    \vskip 3pt
    \textbf{Fix: gradient clipping} -- cap the gradient's norm at a threshold $h$:
    \[
      \text{if } \|\nabla\| > h:\quad \nabla \leftarrow \frac{h}{\|\nabla\|}\,\nabla
    \]
  \end{columns}
  \vfill
  \footnotesize None of this existed in the convex world of linear/logistic regression (L11) --
  the price of a powerful, non-convex model.
\end{frame}

\section{Regularizing neural networks}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Regularizing neural networks}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      A $100$k-parameter model overfits eagerly. Five ways to hold it back.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Nets overfit eagerly}
  A network has more than enough capacity to \textbf{memorise} the training set -- driving
  train loss to zero while validation loss climbs. Everything in this section fights that gap.
  \vfill
  \begin{itemize}\small
    \item Same \textbf{bias-variance} story as the rest of the course (L01d / [06]) -- nothing
      new conceptually, just a very high-variance model.
    \item The tools: \textbf{dropout}, \textbf{weight decay}, \textbf{data augmentation},
      good \textbf{initialization + BatchNorm}, and \textbf{early stopping}.
  \end{itemize}
  \vfill
  \footnotesize We will read the tell-tale train-vs-validation curves at the end of the section.
\end{frame}

\begin{frame}{Dropout: train a crowd of subnetworks}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Predict:} you delete a random $50\%$ of the hidden neurons on \emph{every}
    mini-batch. Does that help or hurt?
    \pause
    \vskip 4pt
    \footnotesize
    It \textbf{helps}. Each step trains a different random subnetwork, so no neuron can rely on
    a specific partner -- they must learn robust, redundant features. At test time you use all
    neurons: an implicit \textbf{ensemble average}
    over exponentially many subnetworks.
    \vskip 4pt
    Callback: this is \textbf{bagging} (ch4) done inside one network. Typical rate $0.2$--$0.5$.
    \vskip 3pt
    {\scriptsize \textbf{In practice:} PyTorch uses \emph{inverse} dropout -- it scales the
    surviving activations by $1/(1{-}p)$ \emph{during training}, so nothing changes at test
    time.}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/dropout_test_error.pdf}
    \end{center}
    \vskip -2pt
    \scriptsize No dropout ($p{=}0$) overfits; a little dropout lowers test error.
  \end{columns}
\end{frame}

\begin{frame}{Weight decay: you already know this one}
  Add a penalty on the size of the weights to the loss -- exactly \textbf{Ridge} from L01b,
  now called weight decay:
  \[
    L_{\text{reg}} = L + \frac{\lambda}{2}\,\|\mathbf{W}\|_2^2
    \quad\Longrightarrow\quad
    w \leftarrow w - \eta\big(\tfrac{\partial L}{\partial w} + \lambda w\big)
  \]
  \begin{itemize}\small
    \item Every step shrinks each weight a little toward zero -- hence ``decay''.
    \item Smaller weights $\Rightarrow$ a smoother function $\Rightarrow$ less overfitting.
    \item In PyTorch it is one argument: \texttt{optimizer = Adam(..., weight\_decay=1e-4)}.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Same L2 penalty as Ridge regression -- the network is still ``a stack of regressions'', so
    the regressions' regularizer still applies.}}
\end{frame}

\begin{frame}{Data augmentation: free training data}
  \begin{columns}[T]
    \column{0.58\textwidth}
    Overfitting is a ratio: model capacity over amount of data. You cannot easily shrink a net,
    but you can \textbf{grow the data} -- cheaply and for free.
    \vskip 4pt
    \textbf{Idea:} apply \emph{label-preserving} transformations to each example, so the model
    sees many variants of the same thing.
    \begin{itemize}\small
      \item Images: flips, small rotations, crops, brightness, noise.
      \item Audio: time-stretch, pitch shift, background noise.
    \end{itemize}
    \column{0.42\textwidth}
    \fcolorbox{armred}{armred!8}{\parbox{0.95\linewidth}{\small
      \textbf{Label-preserving is the catch.} Rotate a handwritten $6$ too far and it becomes a
      $9$ -- now the label is wrong. Augmentations must respect what the label means.}}
    \vskip 4pt
    \footnotesize Augmentation is strongest exactly where nets are strong: raw perceptual data
    with known invariances.
  \end{columns}
\end{frame}

\begin{frame}{Initialization: give gradients a fighting start}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Where you \emph{start} the weights decides whether signal survives depth. Too large and
    activations explode; too small and they vanish -- before any learning happens.
    \vskip 4pt
    \textbf{Fix: scale the initial weights by the fan-in} so variance is preserved layer to
    layer.
    \begin{itemize}\small
      \item \textbf{He} init (ReLU): $\;w \sim \mathcal{N}\!\big(0,\, 2/n_{\text{in}}\big)$.
      \item \textbf{Xavier/Glorot} (tanh/sigmoid):
        $\;w \sim \mathcal{U}\!\big(\pm\sqrt{6/(n_{\text{in}}+n_{\text{out}})}\big)$.
    \end{itemize}
    \footnotesize Frameworks default to sensible init, but this is the first thing to check when
    a deep net will not train.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/init_variance.pdf}
    \end{center}
    \vskip -2pt
    \scriptsize Bad init: activation scale blows up with depth. He init: stable.
  \end{columns}
\end{frame}

\begin{frame}{BatchNorm: normalize activations mid-network}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Even with good init, activation statistics drift as training proceeds. \textbf{Batch
    normalization} standardizes each layer's pre-activations $z$ over the mini-batch $B$, then
    rescales with two \emph{learned} parameters:
    \[
      \hat z = \frac{z - \mu_B}{\sqrt{\sigma_B^2 + \varepsilon}},\qquad
      \text{out} = \gamma\,\hat z + \beta
    \]
    \begin{itemize}\small
      \item $\mu_B,\sigma_B^2$: the batch mean/variance; keeps signals in range $\Rightarrow$
        faster training, higher learning rates, mild regularization noise.
      \item $\gamma,\beta$ let the net rescale/shift back if it needs to.
      \item One layer in PyTorch: \texttt{nn.BatchNorm1d(width)}.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/batchnorm_curves.pdf}
    \end{center}
    \vskip -2pt
    \scriptsize Same net and optimizer; BatchNorm reaches low loss far sooner.
  \end{columns}
\end{frame}

\begin{frame}{Read your training curves}
  Plot train \emph{and} validation loss every epoch -- the single most useful diagnostic.
  \vfill
  \begin{columns}[T]
    \column{0.52\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/training_curves_4cases.pdf}
    \end{center}
    \column{0.48\textwidth}
    \footnotesize
    \begin{itemize}\small
      \item \textbf{Good:} both fall, validation a little above train.
      \item \textbf{Overfitting:} train keeps falling, validation turns \emph{up} -- stop at
        the valley (\textbf{early stopping}).
      \item \textbf{Underfitting:} both stay high -- too small / too much regularization / too
        few epochs.
      \item \textbf{Val $<$ train:} usually heavy dropout (train loss is measured with noise on)
        -- or a leak. Investigate.
    \end{itemize}
  \end{columns}
\end{frame}

    \begin{minipage}{0.82\textwidth}\centering\small
      Which knob for which symptom -- and what is genuinely new here.
    \end{minipage}
  \end{center}

\begin{frame}{Hyperparameters: symptom $\to$ fix}
  \begin{center}\small
    \begin{tabular}{p{3.3cm} p{4.1cm} p{4.1cm}}
      \textbf{Knob} & \textbf{If underfitting} & \textbf{If overfitting} \\ \hline
      Model size (depth/width) & increase capacity & decrease, or add regularization \\[2pt]
      Learning rate & raise (or use Adam) & usually not the cause \\[2pt]
      Epochs & train longer & fewer / early stopping \\[2pt]
      Dropout & lower it & raise it ($0.2$--$0.5$) \\[2pt]
      Weight decay & lower it & raise it \\[2pt]
      Data augmentation & (not the issue) & add more \\
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize First diagnose \emph{under- vs over-fitting} from the curves, \emph{then} reach
  for the matching knob. Do not turn all of them at once.
\end{frame}

\begin{frame}{Tuning machinery is not new}
  Choosing these hyperparameters is a search problem you already have tools for:
  \begin{itemize}\small
    \item \textbf{Grid / random search}, and \textbf{Optuna} for smarter search -- all from
      lecture [08]. Nothing NN-specific about the search itself.
    \item \textbf{Early stopping} doubles as the cheapest tuner: it picks the epoch count for
      you, for free, every run.
    \item Neural nets just have \emph{more} knobs and cost \emph{more} per trial -- so lean on
      early stopping and sensible defaults before launching a big sweep.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Sensible starting recipe:} Adam, ReLU, a little dropout, weight decay $10^{-4}$,
    early stopping on validation loss. Tune from there.}}
\end{frame}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Backprop} $=$ the chain rule applied backward once, reusing shared terms:
      ``signal $\times$ local derivative'' per layer. Output signal is $f-y$, the same residual
      as regression.
    \item Vectorised, it is two matrix rules per layer; \texttt{loss.backward()} runs them for
      you. Sigmoid's small $\varphi'$ is why deep nets need \textbf{ReLU / good init / BatchNorm}.
    \item \textbf{Training loop:} mini-batch SGD, an optimizer (\textbf{start with Adam};
      details in the Optim module), and the \textbf{learning rate} as the key knob.
    \item \textbf{Regularize:} dropout (ensemble), weight decay ($=$ Ridge), augmentation,
      init$+$BatchNorm, early stopping. \textbf{Read the train-vs-val curves} to know which.
    \item \textbf{Tuning} reuses grid/random/Optuna ([08]); nets just have more knobs.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Next:} the practical -- train an MLP end to end in PyTorch. Later, the deep-learning
    chapter (CNNs, RNNs, transformers) builds on exactly this training machinery.}}
\end{frame}
