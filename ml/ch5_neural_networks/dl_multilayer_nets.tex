% --- OUR SPRINKLE (improving LMU's abstract notation deck): track the shapes ---
\begin{frame}{Improving the notation: track the shapes}
  LMU's notation is compact but abstract -- pure symbols. The one thing to hold onto is that
  \textbf{the shapes must line up}: a whole batch of $n$ inputs flows through the layer as
  matrix multiplies.
  \vfill
  \begin{center}
  \begin{tikzpicture}[font=\footnotesize, >=Latex,
    box/.style={draw, thick, minimum height=1.1cm, rounded corners=2pt, inner sep=4pt}]
    \node[box, fill=gray!12, draw=gray, minimum width=1.3cm] (X) {$X$};
    \node[box, fill=armblue!14, draw=armblue, minimum width=0.9cm, right=1.2cm of X] (W) {$W$};
    \node[box, fill=armorange!16, draw=armorange, minimum width=1.3cm, right=2.0cm of W] (Z) {$Z$};
    \node[below=2pt of X, text=gray]    {$n\times \mathbf{p}$};
    \node[below=2pt of W, text=armblue] {$\mathbf{p}\times m$};
    \node[below=2pt of Z, text=orange1] {$n\times m$};
    \draw[->] (X) -- node[above,font=\scriptsize]{$\times$} (W);
    \draw[->] (W) -- node[above,font=\scriptsize]{$+\,B,\ \sigma$} (Z);
  \end{tikzpicture}
  \end{center}
  \vfill
  \begin{itemize}\small
    \item Hidden layer: $Z=\sigma(XW+B)$ has shape $(n\times m)$ -- the \textbf{inner} $p$'s must
      match ($X$'s columns $=$ $W$'s rows), and they cancel.
    \item Output layer: $f=\tau(Zu+C)$ has shape $(n\times 1)$ -- one prediction per input.
  \end{itemize}
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Whenever the algebra feels abstract, \textbf{write the shape under each symbol}. Two adjacent
    inner dimensions that don't match $=$ an illegal multiply -- the fastest bug-catch in deep
    learning.}}
\end{frame}

% --- OUR SPRINKLE (after the depth deck): why depth is exponentially expressive ---
\begin{frame}{Why depth helps: it folds space}
  LMU's slides said depth gives an \emph{exponential} advantage (Mont\'ufar et al., 2014). Here
  is the picture behind that claim:
  \vfill
  \begin{columns}[T]
    \column{0.58\textwidth}
    \begin{center}\includegraphics[width=\linewidth]{fig/folding_tent.pdf}\end{center}
    \vskip 2pt
    \footnotesize A ReLU layer \textbf{folds} the input along its kinks. Each fold doubles the
    pieces: a depth-$k$ net expresses $2^k$ linear regions from $O(k)$ neurons -- a shallow net
    would need $2^k$ neurons to match.
    \column{0.4\textwidth}
    \begin{center}\includegraphics[width=\linewidth]{fig/folding_space.pdf}\end{center}
    \vskip 2pt
    \scriptsize The same folding in 2D: a small depth-$3$ net creases the plane into $170{+}$
    regions.
  \end{columns}
\end{frame}

\begin{frame}{Depth $=$ a hierarchy of features}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Folding explains \emph{how many} regions depth buys. Here is \emph{what} those extra layers
    tend to represent: each layer composes the one below into something more abstract.
    \vskip 6pt
    On images, a trained deep net learns roughly this progression \textbf{on its own} -- nobody
    programs the edges or the parts. That is \textbf{representation learning}.
    \column{0.5\textwidth}
    \begin{center}

\begin{frame}{Softmax, as a picture}
  LMU's example ran a forward pass to the output scores $f_{in}=(0.57,\,2.95,\,-0.92)$. Softmax
  turns those raw scores into a \textbf{probability distribution over the classes}:
  \vfill
  \begin{center}\includegraphics[width=0.95\linewidth]{fig/softmax_bars.pdf}\end{center}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Exponentiate, then normalise so the bars sum to $1$. It is a \textbf{smooth argmax}: the
    largest score dominates (class 2, $0.90$), but every class keeps a nonzero probability --
    the same softmax as multinomial logistic regression.}}
\end{frame}

% --- OUR SPRINKLE (right after the theorem, before LMU's examples): WHY it works ---
\begin{frame}{Why it works: any curve is a sum of bumps}
  The theorem promises an approximation \emph{exists} -- but not \emph{why}. The intuition:
  each hidden neuron contributes one localized \textbf{bump}, and their sum can trace any
  continuous curve. More neurons $\Rightarrow$ closer fit.
  \vfill
  \begin{center}\includegraphics[width=0.98\linewidth]{fig/uat_bumps.pdf}\end{center}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Each sigmoid pair makes one bump (left). Summing $6$ already follows the target; $20$ nearly
    nails it (right). ``$m\to\infty$'' in the theorem is just \textbf{``add more bumps''} -- and
    the LMU examples on the next slides show exactly this, as the hidden-layer size grows.}}
\end{frame}

% --- OUR SPRINKLE (after the theorem): what it does NOT promise ---
\begin{frame}{Universal approximation: read the small print}
  The theorem is reassuring but easy to over-read. Three cautions:
  \vfill
  \begin{itemize}\small
    \item \textbf{Existence, not construction.} It says good weights \emph{exist} -- not that
      gradient descent will \emph{find} them. Training can still get stuck.
    \item \textbf{One layer can be absurdly wide.} ``Some $m$'' may mean an astronomically large
      hidden layer; \textbf{depth} reaches the same functions far more cheaply (previous
      sprinkle).
    \item \textbf{Approximating $\ne$ generalizing.} The LMU example showed it: a size-$5$ net
      nailed the training data but its \emph{test} error exploded (CV mse $\approx 20$).
      Capacity you cannot control overfits.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Takeaway.} Universal approximation is a \emph{floor} on what a big-enough net
    \emph{could} represent -- not a promise about training or test performance. In practice: a
    \textbf{reasonably sized} hidden layer, plus everything in the training chunk.}}
\end{frame}

\begin{frame}{Recap: the model is now fully specified}
  \begin{itemize}\small
    \item \textbf{Matrix notation:} a layer is $\mathbf{z}=\sigma(\mathbf{W}^\top\mathbf{x}+
      \mathbf{b})$; a whole batch is $Z=\sigma(XW+B)$ -- one matrix multiply per layer.
    \item \textbf{Depth:} stack $l$ layers into a chain
      $f=\tau\circ\ldots\circ\sigma^{(1)}$. Each layer builds more abstract features; depth is
      \textbf{exponentially} more expressive than width (folding).
    \item \textbf{Many classes:} a \textbf{softmax} output layer turns scores into a probability
      distribution -- the same softmax as multinomial logistic regression.
    \item \textbf{Universal approximation:} one hidden layer can approximate any continuous
      function in principle -- but trainability and generalization are separate questions.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    Everything is still ``a stack of affine maps and nonlinearities''. \textbf{One thing is
    still missing: how to learn the weights.}}}
\end{frame}

% ---------------------------------------------------------------
\begin{frame}{Next: learning the weights}
  Every network so far had its weights \textbf{given} to us. With the model fully specified, the
  last piece is \textbf{training} -- finding good weights automatically from data.
  \vfill
  \begin{itemize}\small
    \item \textbf{Gradient descent:} nudge every weight downhill on the loss.
    \item \textbf{Backpropagation:} get the gradient for \emph{all} weights in one backward
      pass -- what makes training a million-weight net feasible.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    That is the next chunk (LMU's \emph{backpropagation} week). It turns these static
    architectures into models that actually learn.}}
\end{frame}
