\section{Why metrics matter}

\begin{frame}{Why even bother choosing a metric?}
  A spam filter sees 100 emails: 99 legit, 1 spam.
  \vfill
  A lazy model labels \emph{everything} ``legit'':
  \begin{itemize}
    \item \textbf{Accuracy: 99\%.} Looks excellent.
    \item Spam caught: \textbf{zero}. The model is useless.
  \end{itemize}
  \vfill
  The model didn't change - only the \emph{number we chose to look at}. Accuracy quietly hid the one mistake that mattered.
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.96\textwidth}{\centering
    A metric is a \textbf{definition of success}. Pick the wrong one and a worthless model looks great.}}
\end{frame}

\begin{frame}{Same trap in regression}
  Two delivery-time models, evaluated on the same orders. \emph{Identical average error.}
  \vfill
  \begin{itemize}
    \item \textbf{Model A:} every estimate is off by about 10 minutes.
    \item \textbf{Model B:} usually spot-on, but once in a while off by 3 hours.
  \end{itemize}
  \vfill
  Same mean error, completely different customer experience. If the \emph{worst case} is what burns you, you care about the \textbf{maximum absolute deviation}, not the average.
  \vfill
  A pricing model is the mirror image: being off by 50k AMD is a disaster on a 200k flat ($25\%$) and a rounding error on a 5M flat ($1\%$). There, \textbf{percentage error} is the honest metric.
  \vfill
  \footnotesize \textbf{Takeaway:} there is no universal ``accuracy.'' The metric encodes \emph{which mistake you cannot afford}.
\end{frame}

\begin{frame}{Loss: what the model optimizes}
  \textbf{Loss} is the function minimized \emph{while the model is being fit}.
  \vfill
  \begin{itemize}
    \item It drives the fitting algorithm, so for gradient-based methods it must be \textbf{differentiable}.
    \item Examples: \textbf{MSE} for ordinary least squares, \textbf{log loss} for logistic regression.
  \end{itemize}
  \vfill
  You rarely look at the loss value itself. It exists to \emph{steer the optimizer}, not to be reported.
\end{frame}

\begin{frame}{Metric: what you judge the model by}
  \textbf{Metric} is the number you use to \emph{evaluate} a trained model. It need not be differentiable.
  \vfill
  A metric does \textbf{two jobs}:
  \begin{enumerate}
    \item \textbf{Reporting} - the number you hand to stakeholders (``typical error is 20 kAMD'').
    \item \textbf{Selecting} - the score your cross-validation and hyperparameter search optimize. \emph{The metric you tune against is the model you end up with.}
  \end{enumerate}
  \vfill
  Loss and metric \emph{can} coincide (often do in regression), but needn't: you might \textbf{train} on MSE for its smooth math, then \textbf{tune and report} MAE because stakeholders think in ``typical error.''
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.96\textwidth}{\small
    \textbf{The question every metric answers:} what kind of mistake hurts you most? One big error or many small ones? Errors on cheap items or expensive ones? Over- or under-prediction?}}
\end{frame}

\section{Magnitude metrics}
\sectiontransition{Magnitude metrics}{How big are the errors, in the units you actually care about? Every metric here answers that - they only disagree about how much a single big miss should count.}

\begin{frame}{MSE, RMSE, MAE: how the penalty shapes differ}
  \begin{columns}[T]
    \column{0.46\textwidth}
    \small
    \textbf{MSE} (Mean Squared Error), \textbf{RMSE} (Root Mean Squared Error):
    \[ \text{MSE} = \tfrac{1}{n}\sum (y_i-\yh_i)^2,\quad \text{RMSE}=\sqrt{\text{MSE}} \]
    \textbf{MAE} (Mean Absolute Error):
    \[ \text{MAE} = \tfrac{1}{n}\sum |y_i-\yh_i| \]
    \begin{itemize}
      \item \textbf{Squared} penalty: one error of 10 costs 100; ten errors of 1 cost 10 total. \emph{Big errors dominate.}
      \item \textbf{Absolute} penalty: those two cases cost the same. \emph{Robust} to outliers.
      \item Report \textbf{RMSE} (units of $y$); minimize \textbf{MSE} internally (smoother).
    \end{itemize}
    \column{0.54\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=5.2cm,
                     axis lines=left, axis line style={gray!55},
                     xlabel={residual $e=y-\hat y$}, ylabel={penalty},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     xmin=-3.3, xmax=3.3, ymin=0, ymax=9.6,
                     xtick={-3,-2,-1,1,2,3}, ytick={2,4,6,8},
                     tick label style={font=\scriptsize},
                     legend pos=north west, legend style={font=\small, draw=gray!40}]
          \addplot[armred, very thick, domain=-3:3, samples=120] {x^2};
          \addlegendentry{$e^2$ (MSE)}
          \addplot[armblue, very thick, domain=-3:3, samples=120] {abs(x)};
          \addlegendentry{$|e|$ (MAE)}
        \end{axis}
      \end{tikzpicture}\\[2pt]
      \scriptsize The squared penalty curves upward fast; the absolute penalty grows at a constant rate.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{MAE and MedAE: robust variants}
  \[
    \text{MAE} = \frac{1}{n} \sum_{i=1}^n |y_i - \yh_i|
    \qquad
    \text{MedAE} = \operatorname{median}_i\, |y_i - \yh_i|
  \]
  \vfill
  \begin{itemize}
    \item \textbf{MAE} (Mean Absolute Error). Units of $y$. Linear penalty - an outlier of 100 adds 100, not 10\,000.
    \item \textbf{MedAE} (Median Absolute Error). The middle residual. Even more robust: half your predictions are at least this good.
    \item Where each shines: \emph{MAE} for outlier-heavy targets (real estate, finance). \emph{MedAE} for ``typical case'' reporting: ``half my predictions land within 30 kAMD.''
  \end{itemize}
\end{frame}

\begin{frame}{Predict first: which metric notices the outlier?}
  10 rent predictions. Nine are off by 1 kAMD; one is off by 100 kAMD (a luxury flat the model missed).
  \vfill
  Before looking: rank MSE, RMSE, MAE, MedAE from most to least disturbed by that one outlier.
  \pause
  \vfill
  \begin{center}\small
    \begin{tabular}{lcl}
      Metric & Value & Interpretation \\ \hline
      MSE   & $\dfrac{9 \cdot 1^2 + 1 \cdot 100^2}{10} = 1000.9$ & dominated by the outlier \\[4pt]
      RMSE  & $\sqrt{1000.9} \approx 31.6$ & ``typical error 32 kAMD'' (misleading) \\[2pt]
      MAE   & $\dfrac{9 \cdot 1 + 100}{10} = 10.9$ & visible but moderate \\[4pt]
      MedAE & $\operatorname{median} = 1$ & outlier completely invisible \\
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize \textbf{Same model, four different stories.} The metric you pick \emph{is} the conclusion you report.
\end{frame}

\section{Relative metrics}
\sectiontransition{Relative metrics}{A raw error of ``20'' means nothing on its own - 20 kAMD or 20 EUR? These metrics normalize, so the number travels across problems and scales.}

\begin{frame}{$R^2$: fraction of variance explained}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \[
      R^2 = 1 - \frac{\sum_i (y_i - \yh_i)^2}{\sum_i (y_i - \bar{y})^2}
    \]
    \small
    \begin{itemize}
      \item Compares your model (blue) to the dumbest baseline (orange dashed): \emph{always predict the mean} $\bar{y}$.
      \item $R^2 = 1$: perfect. $R^2 = 0$: no better than $\bar{y}$.
      \item \textbf{Scale-free} - no units, so it reads the same whether $y$ is in AMD or EUR (unlike MAE/RMSE).
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4.6cm,
                     axis lines=left, axis line style={gray!55},
                     xlabel={$x$}, ylabel={$y$},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     xmin=0, xmax=10, ymin=0, ymax=10,
                     xtick={0,2,4,6,8,10}, ytick={0,2,4,6,8,10},
                     tick label style={font=\tiny}]
          \addplot[only marks, mark=*, mark size=1.3pt, armred] coordinates {
            (0.5,1.2)(1.2,1.0)(1.9,2.6)(2.6,2.1)(3.3,3.9)(4.0,3.4)(4.7,5.2)
            (5.4,4.6)(6.1,6.3)(6.8,5.9)(7.5,7.6)(8.2,7.0)(8.9,8.8)(9.5,8.3)};
          \addplot[armorange, thick, dashed, domain=0:10, samples=2] {4.85};
          \node[font=\tiny, armorange, anchor=south east] at (axis cs:9.9,4.95) {baseline: predict $\bar y$};
          \addplot[armblue, very thick, domain=0:10, samples=2] {0.85*x + 0.6};
          \node[font=\small, anchor=north west] at (axis cs:0.2,9.8) {$R^2 \approx 0.94$};
        \end{axis}
      \end{tikzpicture}\\[2pt]
      \scriptsize Tight scatter around the line $\Rightarrow$ high $R^2$.
    \end{center}
  \end{columns}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.96\textwidth}{\small
    \textbf{$R^2$ can go negative on test data.} It just means your model does worse than predicting the training mean. Not a bug - a warning. Investigate.}}
\end{frame}

\begin{frame}{Better $R^2 \neq$ better fit}
  Same model, same MSE, two views of the data. $R^2$ depends on the \emph{range} of $y$; MSE depends on the \emph{magnitude} of residuals.
  \vfill
  \begin{center}
    \begin{tikzpicture}
      \begin{groupplot}[
        group style={group size=2 by 1, horizontal sep=26pt},
        width=0.42\textwidth, height=4cm,
        axis lines=left, axis line style={gray!55},
        xmin=0, xmax=1, ymin=-0.3, ymax=1.4,
        xtick={0,0.5,1}, ytick={0,0.5,1},
        tick label style={font=\tiny},
        title style={font=\footnotesize},
      ]
        \nextgroupplot[title={half range: $R^2 = 0.74$, MSE $= 0.02$}]
          \addplot[only marks, mark=*, mark size=1.2pt, armred] coordinates {
            (0.05, 0.12) (0.12, 0.08) (0.18, 0.30) (0.25, 0.16) (0.32, 0.42)
            (0.40, 0.34) (0.45, 0.58) (0.48, 0.46)};
          \addplot[armblue, very thick, domain=0:0.55, samples=2] {1.1*x + 0.03};
        \nextgroupplot[title={full range: $R^2 = 0.83$, MSE $= 0.02$}]
          \addplot[only marks, mark=*, mark size=1.2pt, armred] coordinates {
            (0.05, 0.12) (0.12, 0.08) (0.18, 0.30) (0.25, 0.16) (0.32, 0.42)
            (0.40, 0.34) (0.45, 0.58) (0.48, 0.46) (0.58, 0.55) (0.65, 0.85)
            (0.72, 0.72) (0.80, 0.95) (0.88, 1.05) (0.95, 1.18)};
          \addplot[armblue, very thick, domain=0:1, samples=2] {1.1*x + 0.03};
      \end{groupplot}
    \end{tikzpicture}
  \end{center}
  \vfill
  \footnotesize \textbf{Why?} $R^2 = 1 - \text{SSE}/\text{SST}$. Extending the range inflates SST much faster than SSE $\Rightarrow$ the ratio shrinks $\Rightarrow$ $R^2$ rises, even though the fit didn't improve. \textbf{Don't compare $R^2$ across datasets with different $y$-ranges.} Pair it with RMSE/MAE.
\end{frame}

\begin{frame}{Adjusted and generalized $R^2$}
  \textbf{Adjusted $R^2$} - penalizes useless features:
  \[
    R^2_\text{adj} = 1 - (1 - R^2)\, \frac{n-1}{n-p-1}
  \]
  Plain $R^2$ \emph{never} drops when you add a feature, even pure noise. Adjusted $R^2$ \emph{can} drop, so it won't reward bloating the model with junk.
  \vfill
  \textbf{Generalized $R^2$} - same skeleton, any loss, any baseline:
  \[
    R^2_\text{gen} \;=\; 1 - \frac{\text{Loss}_\text{model}}{\text{Loss}_\text{baseline}}
  \]
  Classical $R^2$ is the special case (loss $=$ SSE, baseline $=$ constant mean). Swap in MAE, or a smarter baseline, and the same ``fraction of the gap closed'' reading still holds.
\end{frame}

\begin{frame}{MAPE: the popular but flawed default}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \[
      \text{MAPE} = \frac{100\%}{n} \sum_{i=1}^n \frac{|y_i - \yh_i|}{|y_i|}
    \]
    \textbf{MAPE} = Mean Absolute Percentage Error. Popular because it reports as a percentage; stakeholders love it.
    \vskip 4pt
    \footnotesize Three known problems:
    \begin{enumerate}\footnotesize
      \item \textbf{Undefined when $y_i = 0$.} Any zero target $\Rightarrow$ division by zero $\Rightarrow$ infinite MAPE.
      \item \textbf{Asymmetric.} Under-prediction capped at 100\%; over-prediction unbounded (plot, fixed truth $y=10$).
      \item \textbf{Biases models toward under-prediction} when used as a loss.
    \end{enumerate}
    \column{0.45\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4cm,
                     xlabel={prediction $\hat{y}$ (truth $=10$)},
                     ylabel={MAPE (\%)},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     axis line style={gray!60},
                     xmin=0, xmax=50, ymin=0, ymax=420,
                     ytick={0, 100, 200, 300, 400},
                     yticklabel style={font=\tiny},
                     xtick={0, 10, 20, 30, 40, 50},
                     xticklabel style={font=\tiny},
                     grid=major, grid style={dashed,gray!30}]
          \addplot[armred, very thick, domain=0:10, samples=2] {100*(10-x)/10};
          \addplot[armred, very thick, domain=10:50, samples=2] {100*(x-10)/10};
          \draw[gray!50, dashed] (axis cs:10,0) -- (axis cs:10,420);
          \node[font=\tiny, gray] at (axis cs:5, 200) {capped};
          \node[font=\tiny, gray] at (axis cs:35, 380) {unbounded};
        \end{axis}
      \end{tikzpicture}\\[2pt]
      \scriptsize Asymmetric around $y=10$: under- vs over-prediction are penalized unequally.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{SMAPE: the symmetric fix}
  \[
    \text{SMAPE} = \frac{100\%}{n} \sum_{i=1}^n \frac{2\,|y_i - \yh_i|}{|y_i| + |\yh_i|}
  \]
  \textbf{SMAPE} = Symmetric Mean Absolute Percentage Error.
  \vfill
  \begin{itemize}
    \item Symmetric: a 50\% over-prediction and a 50\% under-prediction get the same score.
    \item Bounded in $[0\%, 200\%]$ - no infinities.
    \item Defined when $y_i = 0$ (as long as $\yh_i \neq 0$).
    \item Still quirky (a missed zero pins it near 200\%), but better-behaved than MAPE.
  \end{itemize}
  \vfill
  \footnotesize \textbf{MASE} (Mean Absolute Scaled Error): MAE of the model divided by the MAE of a naive ``predict last value'' forecast. Scale-free, handles zeros, the standard for time-series competitions (M4, M5).
\end{frame}

\section{Rank and association}
\sectiontransition{Rank and association}{Sometimes you don't need the exact value, only the right order - which listing is pricier, not by how much. These measures score the ordering.}

\begin{frame}{Pearson correlation: do predictions move with the truth?}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \[
      r = \frac{\operatorname{cov}(y,\yh)}{\sigma_y\,\sigma_{\yh}} \in [-1,1]
    \]
    \small
    \begin{itemize}
      \item Measures \textbf{linear co-movement}: when $y$ goes up, does $\yh$ go up by a proportional amount?
      \item \textbf{Bounded} in $[-1,1]$ and scale-free - which is exactly what makes it usable: unlike MAE/RMSE, $0.9$ reads as ``strong'' on its own, no units to look up. A clean \emph{screening} tool for which raw feature tracks the target.
      \item $r = 1$: perfect straight-line agreement. $0$: none. $-1$: perfectly inverted.
    \end{itemize}
    \column{0.48\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4.4cm,
                     axis lines=left, axis line style={gray!55},
                     xlabel={$y$}, ylabel={$\hat y$},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     xmin=0, xmax=10, ymin=0, ymax=10,
                     xtick={0,2,4,6,8,10}, ytick={0,2,4,6,8,10},
                     tick label style={font=\tiny}]
          \addplot[only marks, mark=*, mark size=1.3pt, armblue] coordinates {
            (1,1.2)(2,1.8)(3,2.9)(4,4.1)(5,5.0)(6,5.9)(7,7.1)(8,7.8)(9,9.1)};
          \node[font=\small, anchor=north west] at (axis cs:0.2,9.8) {$r \approx 0.99$};
        \end{axis}
      \end{tikzpicture}\\[2pt]
      \scriptsize Points hug a straight line $\Rightarrow$ $r$ near 1.
    \end{center}
  \end{columns}
  \vfill
  \footnotesize \textbf{Catch:} $r$ only sees \emph{straight-line} structure. If $\yh$ tracks $y$ through a bent-but-always-increasing curve, $r$ drops below 1 even though the ranking is perfect.
\end{frame}

\begin{frame}{Spearman: correlation on ranks}
  \textbf{Spearman's $\rho$} is just Pearson computed on the \emph{ranks} of $y$ and $\yh$ instead of their raw values.
  \vfill
  \begin{itemize}
    \item Sees any \textbf{monotonic} relationship, not only linear ones - a curved-but-increasing fit scores $\rho = 1$.
    \item Robust to outliers and to monotone transforms (log, sqrt) of either variable.
    \item Use it when you care about \textbf{getting the order right} (ranking listings, ordinal targets), not the exact values.
  \end{itemize}
  \vfill
  \footnotesize Kendall's $\tau$ is a third option (fraction of correctly-ordered pairs minus incorrectly-ordered ones); same spirit, more robust to ties, slower to compute.
\end{frame}

\begin{frame}{Predict first: when correlation lies}
  Your model's predictions on a 3-row test set:
  \vskip 4pt
  \begin{center}\small
    \begin{tabular}{lccc}
      $y_\text{true}$ & 100 & 200 & 300 \\
      $\yh$           & 1100 & 1200 & 1300 \\
    \end{tabular}
  \end{center}
  \vfill
  Predict: Pearson $r$ between $y$ and $\yh$? MAE? Would you deploy this?
  \pause
  \vfill
  \begin{itemize}
    \item \textbf{Pearson $r = 1.0$.} Predictions are perfectly linearly related to truth - just shifted by $+1000$.
    \item \textbf{MAE $= 1000$.} Predictions are uniformly terrible.
    \item No.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.96\textwidth}{\small
    \textbf{Correlation measures association, not accuracy.} Use correlations for screening features. Use MAE/RMSE for evaluating models.}}
\end{frame}

\section{Information criteria (AIC, BIC)}

\begin{frame}{AIC and BIC: model comparison, not evaluation}
  \[
    \text{AIC} = 2k - 2 \ln(\hat{L})
    \qquad
    \text{BIC} = k \ln(n) - 2 \ln(\hat{L})
  \]
  \textbf{AIC} = Akaike Information Criterion, \textbf{BIC} = Bayesian Information Criterion; $k$ = number of parameters, $n$ = sample size, $\hat{L}$ = maximized likelihood.
  \vfill
  \begin{itemize}
    \item Both compare \emph{candidate models on the same data}. Lower is better.
    \item They balance \textbf{fit quality} ($-2 \ln \hat{L}$) against \textbf{model complexity} ($k$).
    \item Require a probabilistic model so $\hat{L}$ is defined - available from \texttt{statsmodels} OLS, not directly from sklearn estimators.
    \item These are \emph{model selection} criteria, not accuracy metrics. They tell you which model to pick; they do not tell you how accurate it is on unseen data.
  \end{itemize}
\end{frame}

\begin{frame}{AIC vs.\ BIC: when each wins}
  \begin{columns}[T]
    \column{0.55\textwidth}
    Both penalize complexity, but \textbf{BIC penalizes harder} once $n \geq 8$ (because $\ln n > 2$).
    \vskip 6pt
    \textbf{AIC} ($2k$ per param):
    \begin{itemize}\small
      \item Better for \emph{prediction} - keeps useful features.
      \item Asymptotically equivalent to leave-one-out CV.
    \end{itemize}
    \textbf{BIC} ($\ln(n)\cdot k$ per param):
    \begin{itemize}\small
      \item Better for \emph{model identification} - conservative.
      \item Consistent: as $n \to \infty$, BIC picks the true model.
    \end{itemize}
    \textbf{Practical:} report both. When they disagree, BIC is more conservative. Pair with CV for honest accuracy.
    \column{0.45\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4.5cm,
                     xlabel={sample size $n$ (log scale)},
                     ylabel={penalty per parameter},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     axis line style={gray!60},
                     xmin=2, xmax=1000, ymin=0, ymax=8,
                     xmode=log,
                     xtick={2, 8, 100, 1000},
                     xticklabel style={font=\tiny},
                     yticklabel style={font=\tiny},
                     grid=major, grid style={dashed,gray!30},
                     legend pos=north west, legend style={font=\scriptsize}]
          \addplot[armred, very thick, domain=2:1000, samples=2] {2};
          \addlegendentry{AIC ($=2$)}
          \addplot[armblue, very thick, domain=2:1000, samples=100] {ln(x)};
          \addlegendentry{BIC ($=\ln n$)}
          \draw[gray!60, dashed] (axis cs:7.4, 0) -- (axis cs:7.4, 8);
          \node[gray, font=\tiny, anchor=south] at (axis cs:7.4, 0) {$n{=}e^2$};
        \end{axis}
      \end{tikzpicture}\\
      \scriptsize They cross at $n = e^2 \approx 7.4$. Beyond that, BIC's complexity tax is harsher.
    \end{center}
  \end{columns}
\end{frame}

\section{Residual diagnostics}
\sectiontransition{Residual diagnostics}{A metric collapses every mistake into one number. A plot shows you the \emph{shape} of those mistakes - and points at what to fix.}

\begin{frame}{The five diagnostic plots}
  We'll use five standard plots, in the order you'd actually look at them:
  \vfill
  \begin{itemize}
    \item \textbf{Predicted vs.\ actual.} The intuitive overview - where and how predictions drift.
    \item \textbf{Residuals vs.\ fitted.} Are residuals patterned? (missed nonlinearity, heteroscedasticity)
    \item \textbf{Scale-Location.} Does the residual \emph{spread} depend on the prediction?
    \item \textbf{Normal QQ.} Are residuals roughly Gaussian? (matters for OLS inference)
    \item \textbf{Leverage / Cook's distance.} Which observations have outsized influence?
  \end{itemize}
  \vfill
  Look at all of them. A model with $R^2 = 0.97$ can still be broken in ways only these plots reveal.
\end{frame}

\begin{frame}{Predicted vs.\ actual: the first plot to draw}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Plot $\yh_i$ against $y_i$ with the reference line $\yh = y$. Perfect predictions sit exactly on it.
    \vfill
    \begin{itemize}\small
      \item Points \textbf{above} the line: over-prediction. \textbf{Below}: under-prediction.
      \item A \textbf{bend} away from the line at one end = systematic bias there (often the model compresses the extremes).
      \item A \textbf{widening funnel} = error grows with the target (heteroscedasticity).
    \end{itemize}
    \vfill
    \footnotesize The single most intuitive diagnostic, and the easiest to show a non-technical stakeholder.
    \column{0.54\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=5cm,
                     axis lines=left, axis line style={gray!55},
                     xlabel={actual $y$}, ylabel={predicted $\hat y$},
                     xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
                     xmin=0, xmax=10, ymin=0, ymax=10,
                     xtick={0,2,4,6,8,10}, ytick={0,2,4,6,8,10},
                     tick label style={font=\tiny}]

\begin{frame}{Normal QQ plot: is the error Gaussian?}
  A \textbf{quantile} is a ``what fraction falls below'' cut: the 0.9 quantile is the value 90\% of the data sits under.
  \vfill
  \textbf{How the plot is built:}
  \begin{enumerate}
    \item Sort the residuals from smallest to largest.
    \item For each, ask: if these were truly Normal, where \emph{should} this ranked value land? That's its theoretical quantile.
    \item Plot actual (sorted residual) on one axis vs.\ theoretical (Normal quantile) on the other.
  \end{enumerate}
  \vfill
  If the residuals really are Normal, every point lands on the \textbf{45$^\circ$ diagonal}: the value they have equals the value Normal predicts. Departures from the line are departures from Normality - and the \emph{shape} of the departure tells you which way.
\end{frame}

\begin{frame}{Leverage and Cook's distance}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \footnotesize
    \textbf{Leverage $h_i$.} How unusual is $\xv^{(i)}$? Diagonal of the hat matrix $\Xmat(\Xmat^\top\Xmat)^{-1}\Xmat^\top$. High leverage $\Rightarrow$ this point can pull the fit strongly toward itself.
    \vskip 4pt

      D_i \;=\; \frac{r_i^2}{p \cdot \text{MSE}} \cdot \frac{h_i}{(1 - h_i)^2}
    \]

\section{Prediction intervals}

\begin{frame}{Point predictions aren't enough}

\section{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item A metric is a \textbf{definition of success} - it decides which model wins your tuning, not just what you report.
    \item \textbf{Magnitude} (MSE, RMSE, MAE, MedAE): how big are the errors? They differ entirely in how they treat outliers.
    \item \textbf{Relative} ($R^2$, adjusted, MAPE, SMAPE): scale-free. $R^2$ can go negative on test; MAPE blows up near zero; SMAPE tames it.
    \item \textbf{Correlation} measures association, not accuracy - a perfect Pearson can hide a constant offset of 1000.
    \item \textbf{AIC / BIC}: compare candidate models, don't measure accuracy. Pair with CV.
    \item \textbf{Diagnostics} (predicted-vs-actual, residuals-vs-fitted, scale-location, QQ, leverage/Cook's) show \emph{what kind} of wrong - one number can't.
    \item \textbf{Prediction intervals} turn a point guess into an honest range.
  \end{itemize}
  \vfill
  \emph{One model, many metrics. Report - and tune on - the ones that match what the decision actually costs.}
\end{frame}
