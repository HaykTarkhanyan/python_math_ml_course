% OUR BRIDGE FRAME: convolution -> cross-correlation (makes L16's footnote precise)
\begin{frame}{From convolution to cross-correlation}
  You have just seen \emph{true} convolution: flip the kernel, then slide. What every
  deep-learning framework actually computes drops the flip -- that is
  \textbf{cross-correlation}.
  \vfill
  \begin{itemize}\small
    \item Same slide-and-sum, \emph{without} reversing the kernel.
    \item This is L16's footnote made precise: \texttt{nn.Conv2d} is cross-correlation
      wearing convolution's name.
    \item It changes nothing in practice -- the kernel is \textbf{learned}, so the network
      would just learn the flipped weights either way.
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.95\linewidth}{\small
    The next stretch is LMU's formal cross-correlation: definition, its own click-through
    animation, and the final side-by-side comparison.}}
\end{frame}

\begin{frame}{2D convolution: blurring an image}
  The moving average of the LMU pages, in two dimensions: slide a small \textbf{kernel} of
  weights over the image and average each neighborhood.
  \vfill
  \begin{itemize}\small
    \item A $3\times3$ \textbf{box kernel} - every weight $\tfrac19$ - replaces each pixel
      with the mean of its $3\times3$ patch. The image blurs.
    \item Colors are vectors, so the average runs \textbf{per channel} (R, G, B).
    \item Pick different weights and you get sharpening or edge detection instead - a CNN
      \emph{learns} those weights (L16).
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    The next two stills show the $\tfrac19$ kernel marching across the image, then the
    weighted sum it computes at one stop.}}
\end{frame}

\begin{frame}{You have been convolving since school}
  Multiply two polynomials and watch the coefficients:
  \[
    (1 + 2x + 3x^2)(4 + 5x + 6x^2) = 4 + 13x + 28x^2 + 27x^3 + 18x^4
  \]
  Each output coefficient collects the pairwise products whose indices add up - the
  diagonals of the product table:
  \vskip 4pt
  \begin{center}\small
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{c|ccc}
       & $1$ & $2x$ & $3x^2$\\
      \hline
      $4$    & $4$    & $8x$    & $12x^2$\\
      $5x$   & $5x$   & $10x^2$ & $15x^3$\\
      $6x^2$ & $6x^2$ & $12x^3$ & $18x^4$\\
    \end{tabular}
  \end{center}
  \vskip 4pt
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Expanding a polynomial and collecting like terms \emph{is} a convolution of the two
    coefficient lists: $(1,2,3) \ast (4,5,6) = (4,13,28,27,18)$.}}
\end{frame}

\begin{frame}{Predict: how fast can we convolve?}
  Two lists of \textbf{100{,}000} random numbers. \texttt{numpy.convolve} averages
  \textbf{4.87 s}. \textbf{Predict} how long \texttt{scipy.signal.fftconvolve} - the same
  result, a cleverer algorithm - takes.
  \pause
  \vskip 8pt
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{4.3 ms} - roughly \textbf{1000$\times$ faster}, identical output. Naive
    slide-and-sum fills an $n\times n$ table of pairwise products: $O(n^2)$. The fast route
    is $O(n\log n)$.}}
\end{frame}

\begin{frame}{The trick: multiply in frequency}
  The convolution theorem (proved a few pages back) turns convolving into
  \emph{multiplying} in the frequency domain:
  \[
    f \ast g \;=\; \mathcal{F}^{-1}\!\big\{\,\mathcal{F}\{f\}\cdot\mathcal{F}\{g\}\,\big\}.
  \]
  \begin{itemize}\small
    \item Treat each list as polynomial coefficients; \textbf{sample} both at the same
      points and the convolution becomes a cheap \textbf{pointwise} product.
    \item Sample at the \textbf{roots of unity} - evenly spaced on the unit circle. That
      choice is the \textbf{discrete Fourier transform}, and the \textbf{FFT} both evaluates
      it and inverts it in $O(n\log n)$.
  \end{itemize}
  \vskip 4pt
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    FFT both lists $\to$ multiply pointwise $\to$ inverse FFT. That is what
    \texttt{fftconvolve} does - and why two $n$-digit numbers can be multiplied in
    $O(n\log n)$, not the $O(n^2)$ you learned in school.}}
\end{frame}

% OUR CLOSING FRAME: back to the main thread
\begin{frame}{Back to L16}
  \begin{itemize}\small
    \item \textbf{Convolution} is a flip-and-slide dot product; it is commutative and has a
      clean identity in frequency (the convolution theorem).
    \item \textbf{CNNs compute cross-correlation} -- the same operation without the flip --
      because the kernel is learned, so the flip is irrelevant.
    \item The frequency view is why a kernel is a \textbf{frequency filter}: box blur $=$
      low-pass, Sobel $=$ high-pass (L16, section 2).
    \item It also makes convolution \textbf{fast}: FFT both lists, multiply pointwise,
      invert - $O(n\log n)$, the algorithm behind \texttt{fftconvolve}.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    None of this changes how you build or train a CNN. It is the mathematics that justifies the
    one line L16 asked you to take on faith. Back to the main thread: \textbf{L17, the
    architectures.}}}
\end{frame}
