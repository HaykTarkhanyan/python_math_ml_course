\section{Classification basics}

\begin{frame}{Binary and multiclass}
  Classification = assign each observation to one of a \textbf{finite set of unordered
  classes}. The target is categorical, not numeric.
  \vfill
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Binary} (2 classes)
    \begin{itemize}\small
      \item cheese: good / \textbf{bad}
      \item email: ham / spam
      \item loan: repay / default
    \end{itemize}
    \column{0.48\textwidth}
    \textbf{Multiclass} ($> 2$)
    \begin{itemize}\small
      \item digit: 0--9
      \item iris: setosa / versicolor / virginica
      \item sentiment: neg / neutral / pos
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{This deck:} binary logistic regression first, then \textbf{multiclass}
    (one-vs-rest, softmax) later on.}}
\end{frame}

\begin{frame}{Probabilistic classifiers and the threshold}
  \begin{columns}[T]
    \column{0.5\textwidth}
    A good classifier outputs a \textbf{score} $\hat p(\xv) \in [0,1]$, not just a label.
    Threshold it into a decision:
    \[
      \hat y = \begin{cases} 1 \ (\text{bad}) & \hat p \ge c \\ 0 \ (\text{good}) & \text{else} \end{cases}
    \]
    Default $c = 0.5$, but $c$ is a \textbf{choice}. Missing a bad batch hurts more than
    a re-check, so for cheese you may slide $c$ \emph{left} (catch more bad). How to
    pick it: the classification-metrics lecture.
    \column{0.48\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \draw[->, gray] (0,0) -- (5.3,0);
        \foreach \x in {0,2.5,5} \draw[gray] (\x,0.06) -- (\x,-0.06);
        \node[font=\scriptsize] at (0,-0.32){0};
        \node[font=\scriptsize] at (2.5,-0.32){0.5};
        \node[font=\scriptsize] at (5,-0.32){1};
        \node[font=\scriptsize, gray] at (2.7,-0.7){predicted score $\hat p$};
        \foreach \x in {0.4,0.7,1.0,1.3,1.7,2.1,2.9} \fill[armblue] (\x,0.4) circle (1.7pt);
        \foreach \x in {2.3,3.1,3.6,4.0,4.4,4.7} \fill[armred] (\x,0.4) circle (1.7pt);
        \draw[armorange, very thick] (2.5,-0.2) -- (2.5,1.0);
        \node[armorange!80!black, font=\scriptsize] at (2.5,1.2){threshold $c$};
        \node[font=\tiny, armblue] at (1.1,0.72){good batches};
        \node[font=\tiny, armred] at (3.9,0.72){bad batches};
        \draw[->, armorange, thick] (2.4,0.55) -- (1.8,0.55);
      \end{tikzpicture}\\[3pt]
      \scriptsize Right of $c$ $\Rightarrow$ predict bad. Sliding $c$ left catches more
      bad (with more false alarms).
    \end{center}
  \end{columns}
  \vskip 4pt
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Score vs probability.} The number is a \emph{score} that behaves like a
    probability; whether it is a \emph{trustworthy} probability is a separate question
    (\emph{calibration}). For now we use it to rank and threshold.}}
\end{frame}

\section{From odds to the sigmoid}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Building the model: odds $\to$ log-odds $\to$ sigmoid}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      A line outputs \emph{any} real number, but a probability must live in $[0,1]$.
      Three steps --- odds, log-odds, then the sigmoid --- bridge the two.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Odds}
  \textbf{Odds} $= \dfrac{p}{1-p}$ \quad($p$ = probability of the positive class).
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{center}\small
      \renewcommand{\arraystretch}{1.3}
      \begin{tabular}{c c c}
        prob.\ $p$ & odds & read as \\
        \hline
        $0.5$ & $1$ & 1 : 1 (even) \\
        $0.75$ & $3$ & 3 : 1 \\
        $0.9$ & $9$ & 9 : 1 \\
        $0.1$ & $1/9$ & 1 : 9 \\
        \hline
      \end{tabular}
    \end{center}
    \column{0.48\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=4cm,
          xlabel={$p$}, ylabel={odds $= p/(1-p)$},
          xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
          xmin=0, xmax=1, ymin=0, ymax=10, xtick={0,0.5,1}, ytick={0,5,10},
          axis line style={gray!60}]
          \addplot[armblue, very thick, domain=0:0.91, samples=60] {x/(1-x)};
          \draw[dashed, gray] (axis cs:0.5,0) -- (axis cs:0.5,1);
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
  \vskip 2pt
  \footnotesize \textbf{This is betting odds.} Bookmakers think in odds, not
  probabilities -- ``3 to 1'' is the same $p/(1-p)$ (they quote odds \emph{against},
  $(1-p)/p$, shaded so the house wins). Probabilities live in $[0,1]$; odds in
  $(0,\infty)$ -- they \emph{explode} as $p\to 1$. Not yet a free real number.
\end{frame}

\begin{frame}{Log-odds (the logit)}
  Take the log of the odds:
  \[
    \mathrm{logit}(p) = \log\!\frac{p}{1-p} \ \in \ (-\infty, +\infty).
  \]
  \begin{columns}[T]
    \column{0.54\textwidth}
    Now the value is a \textbf{free real number} -- exactly what a linear model
    outputs.
    \vskip 4pt
    \textbf{Why take the log?} A linear model \emph{adds} effects up. Logs turn
    \emph{multiplying} the odds into \emph{adding} a fixed amount, so each coefficient
    becomes one constant, readable number (an odds ratio) -- more later.
    \vskip 4pt
    \footnotesize\emph{Symmetry:} $\mathrm{logit}(p)$ and $\mathrm{logit}(1-p)$ are
    negatives of each other -- $p=0.5$ sits at $0$.
    \column{0.44\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[
          width=\textwidth, height=4.2cm,
          xlabel={$p$}, ylabel={$\mathrm{logit}(p)$},
          xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
          xmin=0, xmax=1, ymin=-4.5, ymax=4.5,
          xtick={0,0.5,1}, ytick={-4,0,4}, axis line style={gray!60},
        ]
          \addplot[armblue, very thick, domain=0.02:0.98, samples=80] {ln(x/(1-x))};
          \draw[dashed, gray] (axis cs:0.5,-4.5) -- (axis cs:0.5,4.5);
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Why the log-odds? A fair, unbounded scale}
  \begin{columns}[T]
    \column{0.46\textwidth}
    \small
    A linear model $\thx$ can output \emph{any} real number, so we want a target on the
    same scale:
    \begin{itemize}\small
      \item $p \in [0,1]$ --- \textbf{capped}; a line shoots past it (slide 2).
      \item odds $\frac{p}{1-p} \in (0,\infty)$ --- removes the \emph{top} cap, but floored at $0$ and \textbf{lopsided}.
      \item log-odds $\in (-\infty,\infty)$ --- \textbf{free both ways}: the right target.
    \end{itemize}
    \vskip 2pt
    \textbf{Fair to both directions.} ``$4\times$ more likely'' and ``$4\times$ less likely''
    should be equal-and-opposite. On the odds scale they are \emph{not} ($4$ vs $0.25$);
    on the log scale they are exactly $\pm\log 4$.
    \column{0.5\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize, scale=0.9]
        % odds axis
        \draw[->, gray] (-0.2,0) -- (5.9,0); \node[gray, anchor=west] at (5.9,0){odds};
        \foreach \x in {1,4,5}\draw[gray](\x,0.08)--(\x,-0.08) node[below=1pt,black]{\x};
        \fill[armred](0.2,0)circle(1.6pt); \fill[armred](0.25,0)circle(1.6pt);

        \fill[black](1,0)circle(1.1pt);

        \node[armblue, anchor=south] at (4.5,0.5){$\times4,\times5$ spread};
        % log-odds axis

          \draw[->, gray] (-2.7,0)--(2.7,0); \node[gray, anchor=west] at (2.7,0){log-odds};
          \draw[gray](0,0.08)--(0,-0.08) node[below=1pt,black]{0};
          \foreach \x in {-1.61,-1.39,1.39,1.61}\draw[gray](\x,0.08)--(\x,-0.08);

\end{frame}

\begin{frame}{Model the log-odds with a line}
  The whole idea of logistic regression in one line: assume the \textbf{log-odds is
  linear} in the features.
  \[
    \log\!\frac{p(\xv)}{1-p(\xv)} \;=\; z \;=\; \theta_0 + \theta_1 x_1 + \dots + \theta_p x_p \;=\; \thx .
  \]
  \vfill
  We reuse everything from linear regression (a linear score $\thx$), but apply it to
  the log-odds instead of to $y$ directly.
  \vfill
  To turn this score back into a probability $p(\xv) \in (0,1)$, we \textbf{invert} it -- next.
\end{frame}

\begin{frame}{Invert: derive the sigmoid}
  Start from the log-odds, \textbf{exponentiate} both sides, then solve for $p$:
  \begin{align*}
    \log\tfrac{p}{1-p} = z
      &\;\Rightarrow\; \tfrac{p}{1-p} = e^{z}
      \;\Rightarrow\; p = (1-p)\,e^{z} \\
      &\;\Rightarrow\; p\,(1+e^{z}) = e^{z}
      \;\Rightarrow\; p = \tfrac{e^z}{1+e^z} = \tfrac{1}{1+e^{-z}} =: \sigma(z).
  \end{align*}
  \vfill
  The \textbf{sigmoid} $\sigma$ squashes any real score $z = \thx$ into a probability in $(0,1)$:
  \[
    z \to +\infty \Rightarrow p \to 1; \qquad z = 0 \Rightarrow p = 0.5; \qquad z \to -\infty \Rightarrow p \to 0 .
  \]
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Logistic regression's hypothesis: $\;\hat p(\xv) = \sigma(\thx)$ --- a linear score, squashed into a probability.}}
\end{frame}

\begin{frame}{The sigmoid (logistic) curve}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \small
    The score $z = \thx$ runs along the $x$-axis; the sigmoid bends it into a probability.
    \begin{itemize}\small
      \item \textbf{Steepest at $z=0$} ($p=0.5$) --- the decision point.
      \item \textbf{Flattens} toward $0$ and $1$ --- the confident regions.
    \end{itemize}
    \vskip 6pt
    \footnotesize \emph{Same curve, ``logistic growth'':} cumulative COVID cases in a wave

\section{Logistic regression as a model}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{What model did we just build?}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      We have $\hat p = \sigma(\thx)$. Now: what is it as a classifier --- its decision
      boundary, and what each weight does to the curve.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Logistic regression is a linear classifier}
  \begin{columns}[T]
    \column{0.54\textwidth}
    Prediction: $\hat p(\xv) = \sigma(\thx)$. Decide ``bad'' when $\hat p \ge 0.5$,
    i.e.\ when $\thx \ge 0$.
    \vskip 4pt
    So the \textbf{decision boundary} $\thx = 0$ is a straight line (a hyperplane). The
    sigmoid only bends the \emph{probability}, not the boundary.
    \vfill
    \fcolorbox{armblue}{armblue!8}{\parbox{0.96\linewidth}{\scriptsize
      \textbf{Why is it called ``regression''?} It \emph{regresses} a continuous
      quantity -- the log-odds -- on the features. The \emph{fit} is regression-like;
      the \emph{task} is classification.}}
    \column{0.44\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/logreg_boundary_2d.pdf}\\[-3pt]
      \scriptsize Real fit (illustrative, balanced). Straight boundary ($\thx{=}0$);
      ``bad'' upper-right.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{What the parameters do}
  \begin{columns}[T]
    \column{0.43\textwidth}
    \begin{itemize}\small
      \item \textbf{Intercept} $\theta_0$ shifts the S-curve left/right (the baseline level).
      \item \textbf{Weight} $\theta_1$: \emph{magnitude} $=$ steepness; \emph{sign} $=$ direction (a negative weight flips the curve).
      \item Boundary: $\sigma(\thx) = 0.5 \iff \thx = 0$.
    \end{itemize}
    \column{0.57\textwidth}
    \begin{center}
      \begin{tikzpicture}

          \addplot[armred, domain=-6:6, samples=60]{1/(1+exp(-(x-2)))}; \addlegendentry{${-}2$}
        \end{axis}
      \end{tikzpicture}\\[3pt]
      \begin{tikzpicture}
        \begin{axis}[width=\textwidth, height=2.7cm, xmin=-6, xmax=6, ymin=0, ymax=1,

  \end{columns}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{$0.5$ is the \emph{default} threshold, not a law.} It is just where
    $\thx = 0$. With asymmetric costs (missing bad cheese!) you will move it --
    the metrics lecture shows how. The \emph{model} gives the score; \emph{you} pick the cutoff.}}
\end{frame}

\begin{frame}{Worked numbers (one batch, by hand)}
  Fitted $\theta_0 = -4,\ \theta_1 = 0.5$ (one feature, aging temp $x$); take a batch at
  $x = 10^\circ$C:
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{align*}

    \vskip 2pt
    Slope as an \textbf{odds ratio}: $e^{\theta_1} = e^{0.5} \approx \mathbf{1.65}$ --
    each $+1^\circ$C multiplies the odds of ``bad'' by $1.65$ ($+65\%$).

    \begin{center}

\end{frame}

\section{The loss: how we fit it}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{How do we fit it?}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      We have the hypothesis $\hat p = \sigma(\thx)$ but no weights yet. What should we
      minimize --- and why not the squared loss we already know?
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Predict-first: why not squared loss?}
  We have a model $\hat p = \sigma(\thx)$. Could we fit it by minimizing
  $\sum_i (y^{(i)} - \hat p^{(i)})^2$, like linear regression?
  \begin{columns}[T]
    \column{0.5\textwidth}
    Two problems:
    \begin{itemize}\small
      \item \textbf{Non-convex.} Squared loss $\circ$ sigmoid is not convex in $\thetav$
        --- gradient descent can get stuck in local minima.
      \item \textbf{Weak penalty.} It is \emph{bounded}: even a confidently wrong $\hat p$
        costs at most $\approx 1$. Confident mistakes should hurt far more.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}\scriptsize
    \textbf{True $y=1$: what each loss charges}\\[2pt]
    \renewcommand{\arraystretch}{1.25}
    \begin{tabular}{c c c}
      model $\hat p$ & squared $(1-\hat p)^2$ & log $-\log\hat p$ \\
      \hline
      $0.5$ & $0.25$ & $0.69$ \\
      $0.1$ & $0.81$ & $2.30$ \\
      $0.01$ & $0.98$ & $\mathbf{4.61}$ \\
      $0.001$ & $\approx 1.00$ & $\mathbf{6.91}$ \\
      \hline
    \end{tabular}\\[3pt]
    Squared \textbf{saturates near $1$}; log keeps growing $\to \infty$.
    \end{center}
  \end{columns}
  \vfill
  \footnotesize We need a loss built for probabilities --- next.
\end{frame}

\begin{frame}{Why not squared loss: the loss surface}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Fit a single weight $w$ and watch each loss as $w$ varies:
    \begin{itemize}\small
      \item \textbf{Squared loss} on the sigmoid is \textbf{non-convex} --- a bump (barrier)
        can trap gradient descent in a shallow \emph{local} min, away from the real one.
      \item \textbf{Log-loss} is a single \textbf{convex bowl} --- one minimum, so GD always
        reaches it.
    \end{itemize}
    \vskip 2pt
    \footnotesize Convexity is exactly why we can fit logistic regression by gradient descent
    with confidence (next slides).
    \column{0.54\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/logreg_loss_convexity.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Log-loss (cross-entropy)}
  \begin{columns}[T]
    \column{0.52\textwidth}
    Penalize by how much probability you gave the \emph{true} label:
    \begin{itemize}\small
      \item true label $y=1$: penalty $= -\log \hat p$
      \item true label $y=0$: penalty $= -\log(1-\hat p)$
    \end{itemize}
    Both blow up to $\infty$ as you become confidently wrong, and go to $0$ when you are
    confidently right.
    \vskip 4pt
    One formula, where $y$ picks the branch:
    \[
      L(y,\hat p) = -\big[\, y \log \hat p + (1-y)\log(1-\hat p)\,\big].
    \]
    \footnotesize This is \textbf{cross-entropy} / Bernoulli loss -- the same loss
    reappears in neural nets and boosting.
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[
          width=\textwidth, height=4.4cm,
          xlabel={$\hat p$ (predicted prob.\ of $y{=}1$)}, ylabel={penalty},
          xlabel style={font=\scriptsize}, ylabel style={font=\scriptsize},
          xmin=0, xmax=1, ymin=0, ymax=4.5,
          xtick={0,0.5,1}, ytick=\empty, axis line style={gray!60},
          legend style={font=\scriptsize, at={(0.5,0.97)}, anchor=north}, legend cell align=left,
        ]
          \addplot[armred, very thick, domain=0.02:1, samples=80] {-ln(x)};
          \addlegendentry{$-\log \hat p$ (true $y{=}1$)}
          \addplot[armblue, very thick, domain=0:0.98, samples=80] {-ln(1-x)};
          \addlegendentry{$-\log(1-\hat p)$ (true $y{=}0$)}
        \end{axis}
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Where log-loss comes from: one formula, step by step}
  \small
  The \textbf{likelihood} is the probability the model assigns to the labels we actually saw.
  \begin{enumerate}\small
    \item One example, two cases: \quad $P(y{=}1)=\hat p$, \qquad $P(y{=}0)=1-\hat p$.
    \item Fold into \textbf{one} expression -- the exponent switches the right term on:
      \[ P(y)=\hat p^{\,y}\,(1-\hat p)^{\,1-y}
         \qquad {\footnotesize \big(y{=}1:\ \hat p^1(1{-}\hat p)^0=\hat p;\quad y{=}0:\ \hat p^0(1{-}\hat p)^1=1{-}\hat p\big).} \]
    \item All $n$ rows are \textbf{independent} $\Rightarrow$ multiply:
      \[ \mathcal{L}(\thetav)=\prod_{i=1}^{n}\hat p_i^{\,y_i}(1-\hat p_i)^{\,1-y_i}. \]
    \item \textbf{Log} (turns the product into a sum), then \textbf{negate} (so we minimize):
      \[ -\log\mathcal{L}=-\sumin\big[\,y_i\log\hat p_i+(1-y_i)\log(1-\hat p_i)\,\big]=\ \text{log-loss}. \]
  \end{enumerate}
  \vskip 1pt
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Two names, one quantity: this is \textbf{maximum likelihood} (statistics) and the
    \textbf{cross-entropy} between the labels and the model's predictions (information theory).}}
\end{frame}

\begin{frame}{Fitting logistic regression}
  We minimize the empirical risk (average log-loss):
  \[
    \thetah = \argmin_{\thetav} \; \frac1n \sumin -\big[\,\yi \log \sigma(\thx^{(i)}) + (1-\yi)\log(1-\sigma(\thx^{(i)}))\,\big].
  \]
  \begin{itemize}
    \item \textbf{Convex} in $\thetav$ -- no bad local minima.
    \item But \textbf{no closed-form} solution (unlike OLS) -- fit \textbf{numerically}
      with an iterative optimizer (next slide). Convex $\Rightarrow$ it reaches the
      global optimum.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Separable-data trap.} If a line perfectly splits the classes, the weights
    grow without bound (push $\hat p \to 0/1$ for more reward). The fix is the same as
    in regression: \textbf{regularization} (Ridge/Lasso) -- in sklearn this is the
    \texttt{C} parameter.}}
\end{frame}

\begin{frame}{The gradient: same shape as linear regression}
  Gradient descent needs the gradient of the log-loss. The sigmoid's tidy derivative
  $\sigma'(z) = \sigma(z)\big(1-\sigma(z)\big)$ makes the chain rule collapse:
  \[
    \nabla_{\thetav}\, \frac1n \sumin L\big(\yi, \hat p^{(i)}\big)
      \;=\; \frac1n \sumin \big(\,\underbrace{\sigma(\thx^{(i)})}_{\hat p^{(i)}} - \yi\,\big)\,\xv^{(i)} .
  \]
  \begin{columns}[T]
    \column{0.5\textwidth}
    \small
    Read it as \textbf{(prediction $-$ truth) $\times$ feature}, averaged over the data:
    \begin{itemize}\small
      \item points we already predict well ($\hat p \approx y$) contribute $\approx 0$;
      \item confident mistakes pull the hardest.
    \end{itemize}
    \column{0.5\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.96\linewidth}{\small
      \textbf{Identical to linear regression!} There the gradient is
      $\frac1n\sum(\thx^{(i)}-\yi)\,\xv^{(i)}$ -- same formula, only the prediction changes
      from $\hat y = \thx$ to $\hat p = \sigma(\thx)$. The same GD machinery (Ch.\,1) and
      $L_2$ weight decay carry straight over.}}
  \end{columns}
  \vskip 2pt
  \footnotesize Plain gradient descent steps downhill along this -- the next slide reaches the optimum in fewer steps.
\end{frame}

\begin{frame}{How is it actually optimized?}
  The log-loss is \textbf{convex and smooth} -- we get a gradient \emph{and} curvature
  to exploit.
  \begin{itemize}\small
    \item Plain \textbf{gradient descent} works (Chapter 1), but needs many steps.
    \item sklearn's default is \textbf{L-BFGS} -- a \emph{quasi-Newton} method: it uses
      curvature via an \emph{approximate} inverse Hessian built from recent gradients
      (never the full $p\times p$ matrix), so it converges in far fewer iterations and
      stays memory-light. Convex $+$ smooth is exactly where Newton-type methods shine.
  \end{itemize}
  \vfill
  \begin{center}\small
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{l l}
      \textbf{solver} & \textbf{when to use} \\
      \hline
      \texttt{lbfgs} (default) & medium, dense data; L2 \\
      \texttt{liblinear} & small data; L1 --- via \emph{coordinate descent} \\
      \texttt{saga} & large / sparse; L1 / elastic-net \\
      \texttt{newton-cholesky} & $n \gg$ \#features (exact Hessian) \\
      \hline
    \end{tabular}
  \end{center}
  \vskip 2pt
  \footnotesize \emph{Why not plain GD?} On a convex, smooth loss, curvature
  (second-order) info reaches the optimum in a handful of iterations, not hundreds.
  \emph{Coordinate descent} -- the Lasso optimizer -- is also available; it is what
  \texttt{liblinear} uses, not the default here.
\end{frame}

\begin{frame}{Strengths, limits, and what's next}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Strengths}
    \begin{itemize}\small
      \item Interpretable (odds ratios).
      \item Outputs a probability/score.
      \item Fast; convex; great \emph{baseline}.
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Limits}
    \begin{itemize}\small
      \item \textbf{Linear} decision boundary only.
      \item Assumes log-odds linear in features.
      \item Needs feature engineering (e.g.\ polynomial terms) for curved boundaries.
    \end{itemize}
  \end{columns}

  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small

    \textbf{classification-metrics} lecture).}}
\end{frame}

\begin{frame}{Wait -- how do we know it's any good?}
  We can \emph{build} a classifier now. But how do we \textbf{score} it?
  \vfill
  \textbf{Predict-first:} cheese is $3\%$ bad. A lazy model that calls \emph{every} batch
  ``good'' scores\dots
  \;\textbf{97\% accuracy} -- while catching \textbf{zero} bad batches.
  \vfill
  Accuracy is \textbf{misleading} on imbalanced data, and it ignores the asymmetric cost
  (a missed bad batch $\gg$ a needless re-check). Classification needs metrics that
  separate \emph{``catches the bad ones''} from \emph{``doesn't cry wolf''}: precision,
  recall, F1, ROC-AUC, PR-AUC -- and the threshold $c$ itself.
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    That is the whole next lecture -- \textbf{classification metrics}, on the same
    cheese factory.}}
\end{frame}

\section{Multiclass}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Beyond two classes}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      We lumped every defect as ``bad''. Really a wheel is \emph{good}, \emph{moldy}, or
      \emph{too dry} -- now $K=3$ classes, not $2$. One sigmoid gives one probability;
      we need $K$ of them that sum to $1$.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Two ways to reach $K$ classes}
  The target is now one of $K$ \textbf{mutually exclusive} classes (a wheel is exactly one
  type): cheese \{good, moldy, dry\}, digits 0--9, iris species.
  \vfill
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{1. Reduce to binary} -- reuse what we have:
    \begin{itemize}\small
      \item \textbf{One-vs-Rest (OvR)}: $K$ ``class $k$ vs.\ the rest'' classifiers.
      \item \textbf{One-vs-One (OvO)}: one classifier per \emph{pair}, then vote.
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{2. Generalize the model} -- one joint fit:
    \begin{itemize}\small
      \item \textbf{Softmax} (multinomial logistic regression): $K$ scores, normalized to
        probabilities that sum to $1$.
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Both keep the model \textbf{linear}. We look at each, then pick a default.}}
\end{frame}

\begin{frame}{One-vs-Rest (and One-vs-One)}
  Train $K$ \emph{binary} logistic classifiers, each ``class $k$ vs.\ the rest,'' then
  predict the highest score: $\;\hat y = \argmax_k \, \sigma(\thetav_k^\top\xv)$.
  \begin{center}
    \begin{tikzpicture}[font=\scriptsize, >=stealth,
      box/.style={draw=armblue, fill=armblue!8, rounded corners=2pt, inner sep=3pt,
        minimum width=2cm, minimum height=0.5cm}]
      \node (x) at (0,0) {$\xv$};
      \node[box] (c1) at (2.4,1.05) {good vs rest};
      \node[box] (c2) at (2.4,0) {moldy vs rest};
      \node[box] (c3) at (2.4,-1.05) {dry vs rest};
      \node[draw, rounded corners, fill=armorange!18, minimum height=1cm] (am) at (5.2,0) {argmax};
      \node (y) at (6.6,0) {$\hat y$};
      \foreach \c in {c1,c2,c3} \draw[->] (x) -- (\c);
      \foreach \c in {c1,c2,c3} \draw[->] (\c) -- (am);
      \draw[->] (am) -- (y);
    \end{tikzpicture}
  \end{center}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item \textcolor{paramgreen}{\ding{51}} dead simple; reuses binary; trivially parallel.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item \textcolor{armred}{\ding{55}} $K$ fits; scores not jointly calibrated; each sees imbalance.
    \end{itemize}
  \end{columns}
  \vskip 1pt
  \footnotesize \textbf{One-vs-One:} instead train $\binom{K}{2}$ pairwise classifiers and
  vote -- more models, each on less data (the default for SVMs).
\end{frame}

\begin{frame}{Softmax: a soft \texttt{argmax}}
  Give each class its own weights $\thetav_k$, score $z_k = \thetav_k^\top\xv$, then
  \textbf{normalize} the scores into probabilities:
  \[
    \boxed{\; P(y=k \mid \xv) \;=\; \frac{e^{z_k}}{\sum_{j=1}^{K} e^{z_j}} \;}
    \qquad (\text{the \textbf{softmax}}).
  \]
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}\small
      \item Each $P \ge 0$ and they \textbf{sum to $1$} -- a distribution over classes; predict $\hat y = \argmax_k z_k$.
      \item \textbf{Soft version of hardmax.} A true \texttt{argmax} returns a one-hot spike --- e.g.\ $(1,0,0)$ --- a flat-then-jump function that is \emph{not differentiable}.
      \item Softmax \textbf{smooths} it into probabilities leaning toward the largest score --- differentiable, so we can train by gradient descent. Scale the scores up and softmax $\to$ the hard one-hot.
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}
      \begin{tikzpicture}
        \begin{axis}[ybar, width=\textwidth, height=3.7cm, ymin=0, ymax=0.8,
          symbolic x coords={good,moldy,dry}, xtick=data,
          nodes near coords, nodes near coords style={font=\tiny},
          ylabel={$P(y=k)$}, ylabel style={font=\scriptsize}, tick label style={font=\scriptsize},
          bar width=13pt, axis line style={gray!60}]
          \addplot[fill=armblue!60, draw=armblue] coordinates {(good,0.66) (moldy,0.24) (dry,0.10)};
        \end{axis}
      \end{tikzpicture}\\[-3pt]
      \scriptsize $z=(2,1,0.1)$: softmax $\to (.66,.24,.10)$; hardmax $\to (1,0,0)$.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Softmax generalizes the sigmoid}
  Write the sigmoid's $1$ as $e^{0}$:
  \[
    \sigma(z) \;=\; \frac{1}{1+e^{-z}} \;=\; \frac{e^{z}}{e^{z}+1} \;=\; \frac{e^{z}}{e^{z}+e^{0}}.
  \]
  \vfill
  That is exactly \textbf{softmax over two scores} $\{\,z_1 = z,\ z_0 = 0\,\}$: class~1 carries the
  score $z$, class~0 is pinned at $0$. The sigmoid is the $K=2$ softmax with one score fixed.
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{Binary logistic regression is the two-class softmax.} The log-loss, the
    $(\hat p - y)\,\xv$ gradient, the linear boundary --- all are the $K=2$ special case of
    what we just built. One model, $K$ classes.}}
\end{frame}

\begin{frame}{Softmax: loss, gradient, and boundaries}
  \begin{columns}[T]
    \column{0.54\textwidth}
    \small
    \textbf{Loss} -- categorical cross-entropy (same MLE story, $K$ branches):
    \[ -\sumin \log P\big(y=y_i \mid \xv^{(i)}\big). \]
    \textbf{Gradient} -- it keeps the same shape:
    \[ \nabla_{\thetav_k} = \sumin \big(P(k\mid\xv^{(i)}) - 1[\,y_i = k\,]\big)\,\xv^{(i)}. \]
    \fcolorbox{armblue}{armblue!8}{\parbox{0.96\linewidth}{\scriptsize
      Still \textbf{(prediction $-$ truth) $\times$ feature}, where the truth $1[\,y_i{=}k\,]$ is the
      one-hot label ($1$ for the true class, else $0$). The binary gradient, one per class.}}
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize]
        \draw[gray!50] (-2.1,-1.6) rectangle (2.1,1.9);
        \coordinate (c) at (0,0.2);
        \draw[gray, thick] (c) -- (0,1.9);
        \draw[gray, thick] (c) -- (-1.95,-1.6);
        \draw[gray, thick] (c) -- (1.95,-1.6);
        \foreach \p in {(-1.3,1.2),(-0.9,0.9),(-1.5,0.7),(-1.0,1.45)} \fill[armblue] \p circle (1.3pt);
        \foreach \p in {(1.3,1.2),(0.9,0.9),(1.5,0.7),(1.0,1.45)} \fill[armred] \p circle (1.3pt);
        \foreach \p in {(-0.4,-1.0),(0.4,-1.05),(0,-0.6),(-0.15,-1.3)} \fill[armorange] \p circle (1.3pt);
        \node[armblue] at (-1.3,1.75){good};
        \node[armred] at (1.3,1.75){moldy};
        \node[armorange] at (0,-1.45){dry};
      \end{tikzpicture}\\[-2pt]
      \scriptsize Boundaries ($z_i=z_j$) are \textbf{lines}: $K$ convex regions.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Which one? And in \texttt{sklearn}}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Softmax} -- the default choice:
    \begin{itemize}\small
      \item classes mutually exclusive;
      \item one joint fit, coherent probabilities (sum to $1$).
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{One-vs-Rest} -- reach for when:
    \begin{itemize}\small
      \item you want a simple, reusable binary model;
      \item labels can overlap / need independent scores.
    \end{itemize}
  \end{columns}
  \vskip 4pt
\begin{lstlisting}
from sklearn.linear_model import LogisticRegression
clf = LogisticRegression(max_iter=1000)   # softmax automatic for K >= 3
clf.fit(X_train, y_train)                  # y in {0, 1, ..., K-1}
proba = clf.predict_proba(X_test)          # shape (n, K); each row sums to 1

from sklearn.multiclass import OneVsRestClassifier   # explicit OvR, if needed
ovr = OneVsRestClassifier(LogisticRegression()).fit(X_train, y_train)
\end{lstlisting}
  \vskip 1pt
  \footnotesize Since sklearn 1.7 the \texttt{multi\_class} argument is gone:
  \texttt{LogisticRegression} does softmax for $K\ge 3$ (solvers \texttt{lbfgs}/\texttt{saga}/\dots);
  \texttt{liblinear} is binary-only, so wrap it in \texttt{OneVsRestClassifier}.
\end{frame}

\section*{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Classification} predicts a category; a good classifier outputs a
      \textbf{score} in $[0,1]$, then you \textbf{threshold}.
    \item \textbf{Odds} $=p/(1-p)$; \textbf{log-odds} maps $(0,1)\to\mathbb{R}$.
    \item \textbf{Logistic regression} = a linear model on the log-odds, squashed to a
      probability by the \textbf{sigmoid} $\sigma(\thx)$. Linear boundary.
    \item Fit by minimizing \textbf{log-loss} (cross-entropy) with gradient descent;
      regularize via \texttt{C}.
    \item Coefficients read as \textbf{odds ratios} ($e^{\theta_j}$).
    \item \textbf{Multiclass}: \textbf{softmax} (multinomial) or one-vs-rest -- softmax
      generalizes the sigmoid; the $K$ probabilities sum to $1$.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Workflow:} scale features $\to$ fit logistic regression as your baseline
    $\to$ read the odds ratios $\to$ evaluate with the right metric (next lecture) $\to$ tune the
    threshold for your costs.}}
\end{frame}
