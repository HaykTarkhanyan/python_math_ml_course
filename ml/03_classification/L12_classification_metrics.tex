\section{Accuracy and its problem}

\begin{frame}{The cheese factory (our running example)}
  A factory makes 1000 cheese wheels a day; about \textbf{3\% are bad} (contaminated or
  spoiled). A bad wheel reaching customers means recalls, illness, lawsuits.
  \vfill
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Positive} (rare, $\sim 3\%$): a \emph{bad} batch.\\[2pt]
    \textbf{Negative} (common, $\sim 97\%$): a \emph{clean} batch.
    \column{0.48\textwidth}
    The model outputs a \textbf{score} (risk), not a hard label --- more useful, as we'll see.
  \end{columns}
  \vfill
  \footnotesize Same factory as the logistic-regression lecture; now we ask: \emph{how do we
  know the detector is any good?}
\end{frame}

\begin{frame}{What we need from a metric}
  \begin{enumerate}
    \item A number to \textbf{compare} candidate models.
    \item A way to pick a \textbf{threshold} that turns scores into decisions.
    \item Respect for the \textbf{asymmetric cost} --- missing bad cheese $\gg$ over-inspecting clean.
  \end{enumerate}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{The cascade} (each one fixes the previous one's flaw):
    accuracy $\to$ precision/recall $\to$ ROC AUC $\to$ PR AUC.}}
\end{frame}

\begin{frame}{Accuracy: the obvious starting point}
  \begin{columns}[T]
    \column{0.6\textwidth}
    \[ \text{Accuracy} = \frac{\text{correct}}{\text{total}} = \frac{TP+TN}{\text{all}} \]
    \begin{itemize}\small
      \item Simple to compute, simple to explain to stakeholders.
      \item ``What fraction of batches did we classify correctly?''
      \item The default metric in many tutorials.
    \end{itemize}
    \vskip 4pt
    \footnotesize Sounds reasonable --- let's see when it stops being useful.
    \column{0.4\textwidth}
    \begin{center}
      \begin{tikzpicture}[scale=0.55, font=\footnotesize]
        \fill[paramgreen!25] (0,1) rectangle (1,2);
        \fill[paramgreen!25] (1,0) rectangle (2,1);
        \cmcells
      \end{tikzpicture}\\[2pt]
      \scriptsize accuracy $=$ \textcolor{paramgreen!60!black}{shaded diagonal} $/$ all
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Predict-first: where accuracy breaks}
  Your factory has 1000 daily batches: 970 clean, 30 contaminated. You ship a model that
  predicts ``clean'' for \emph{every single batch} (no real model --- just the constant).
  \vfill
  \textbf{Predict:} its accuracy? its recall on contaminated batches? would you ship it?
  \vfill
  \begin{itemize}
    \item \textbf{Accuracy: $970/1000 = 97\%$.} Looks excellent.
    \item \textbf{Recall on positives: $0\%$.} Catches none of the contaminated batches.
    \item \textbf{Ship it?} Customers get sick. Lawsuits. Recalls cost millions.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{The fundamental problem with accuracy:} it weights every error equally. When
    classes are imbalanced, the rare class becomes invisible. \emph{We need a metric that
    focuses on the positive class.}}}
\end{frame}

\section{Precision, recall, F1}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Precision, recall, F1}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Accuracy weights every error equally, so the rare $3\%$ that matters vanishes.
      Fix: stop counting all correctness and \textbf{focus on the positive class}.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The confusion matrix}
  All classification metrics derive from four counts. Lay them on a $2 \times 2$:
  \vfill
  \begin{center}
    \begin{tikzpicture}[every node/.style={font=\small}]
      \node at (1.5, 2.0) {\textbf{Pred.\ Positive}};
      \node at (4.5, 2.0) {\textbf{Pred.\ Negative}};
      \node at (-1.5, 1.0) {\textbf{Actual Pos.}};
      \node at (-1.5, -0.5) {\textbf{Actual Neg.}};
      \draw[armred, thick, fill=armred!10] (0, 0.4) rectangle (3, 1.6);
      \node[armred] at (1.5, 1.1) {\textbf{TP}};
      \node[armred, font=\scriptsize] at (1.5, 0.7) {bad cheese flagged};
      \draw[armorange, thick, fill=armorange!10] (3, 0.4) rectangle (6, 1.6);
      \node[armorange!80!black] at (4.5, 1.1) {\textbf{FN}};
      \node[armorange!80!black, font=\scriptsize] at (4.5, 0.7) {bad cheese missed!};
      \draw[armorange, thick, fill=armorange!10] (0, -1.1) rectangle (3, 0.1);
      \node[armorange!80!black] at (1.5, -0.4) {\textbf{FP}};
      \node[armorange!80!black, font=\scriptsize] at (1.5, -0.8) {wasted inspection};
      \draw[paramgreen, thick, fill=paramgreen!10] (3, -1.1) rectangle (6, 0.1);
      \node[paramgreen] at (4.5, -0.4) {\textbf{TN}};
      \node[paramgreen, font=\scriptsize] at (4.5, -0.8) {clean cheese passed};
    \end{tikzpicture}
  \end{center}
  \vfill
  \footnotesize In the cheese factory the costs are \textbf{asymmetric}: an \textbf{FN}
  (contaminated cheese reaches customers $\to$ recall/illness/lawsuit) is the expensive
  mistake; an \textbf{FP} (a clean batch re-inspected) is annoying but cheap.
  \vskip 2pt
  \scriptsize \emph{Heads-up:} sklearn's \texttt{confusion\_matrix} is $[TN, FP; FN, TP]$
  (negative class first); pass \texttt{labels=[1,0]} for the positive-first layout above.
\end{frame}

\begin{frame}{The same errors statisticians call Type I \& Type II}
  \begin{columns}[T]
    \column{0.5\textwidth}
    The two mistakes are the classic errors from \textbf{hypothesis testing}
    (here $H_0 =$ ``the batch is clean''):
    \begin{itemize}\small
      \item \textbf{Type I} $=$ \textbf{FP}: reject a true $H_0$ --- flag a clean batch (false alarm).
      \item \textbf{Type II} $=$ \textbf{FN}: fail to reject a false $H_0$ --- pass a bad batch (a miss).
    \end{itemize}
    \vskip 4pt
    \footnotesize Same $2\times2$, two vocabularies. We introduced these in the
    \textbf{Hypothesis Testing} lecture (\texttt{math/25\_stat\_hypothesis\_testing.qmd}).
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=0.92\textwidth]{fig/error_types.jpg}\\[1pt]
      \scriptsize The classic ``pregnant?'' picture of Type I vs.\ Type II.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Precision and recall: of what, out of what?}
  \begin{center}
    \includegraphics[width=0.84\textwidth]{fig/precision_recall.png}
  \end{center}
  \vskip 2pt
  \footnotesize Cheese mapping: \textbf{relevant} $=$ bad batches; \textbf{selected} $=$ what we flag.
  \quad \textbf{Precision} $=\frac{TP}{TP+FP}$ --- of what we flagged, how many are bad (efficiency).
  \quad \textbf{Recall} $=\frac{TP}{TP+FN}$ --- of all bad, how many we flagged (safety).
\end{frame}

\begin{frame}{F1: when you need a single number}
  \textbf{F1} is the \emph{harmonic} mean of precision and recall:
  \[
    F_1 = 2 \cdot \frac{\text{Precision} \cdot \text{Recall}}{\text{Precision} + \text{Recall}}
  \]
  \vfill
  \textbf{Why harmonic, not arithmetic?} The harmonic mean is dominated by the smaller value:
  \begin{center}\small
    \begin{tabular}{lccc}
      Precision & Recall & Arithmetic mean & Harmonic mean (F1) \\ \hline
      1.0 & 0.0 & 0.5 & 0.0 \\
      0.8 & 0.8 & 0.8 & 0.8 \\
      0.5 & 0.9 & 0.7 & 0.64 \\
      0.9 & 0.5 & 0.7 & 0.64
    \end{tabular}
  \end{center}
  \vfill
  Arithmetic mean lets a $1.0$ in one column hide a $0.0$ in the other; the harmonic mean refuses --- so F1 can't be gamed by an extreme threshold.
  \vfill
  \scriptsize \textbf{$F_\beta$ generalization:} $F_\beta = (1 + \beta^2)\,\dfrac{P \cdot R}{\beta^2 P + R}$.
  Higher $\beta$ weights recall more; $\beta = 2$ fits the cheese factory (a miss $\gg$ a false alarm).
\end{frame}

\begin{frame}{Worked example: cheese batch inspection}
  50 batches inspected; 10 actually bad; the model flags 13.
  \vfill
  \begin{columns}[T]
    \column{0.46\textwidth}
    \begin{center}\small
    \begin{tabular}{lcc}
       & \textbf{Pred Bad} & \textbf{Pred Good} \\ \hline
      Actual Bad (10) & TP $=8$ & FN $=2$ \\
      Actual Good (40) & FP $=5$ & TN $=35$ \\
    \end{tabular}
    \end{center}
    \column{0.54\textwidth}
    \begin{center}\small
    \begin{tabular}{lll}
      \textbf{Metric} & \textbf{Calc.} & \textbf{Value} \\ \hline
      Accuracy & $(8{+}35)/50$ & $86\%$ \\
      Recall & $8/(8{+}2)$ & $80\%$ \\
      Precision & $8/(8{+}5)$ & $62\%$ \\
      F1 & $2{\cdot}.62{\cdot}.80/(.62{+}.80)$ & $70\%$ \\
    \end{tabular}
    \end{center}
  \end{columns}
  \vfill
  \footnotesize Accuracy ($86\%$) hides the 2 missed bad batches; recall ($80\%$) is the
  operational truth; precision ($62\%$) says $38\%$ of inspections were wasted; F1 ($70\%$)
  punishes the gap between the two.
\end{frame}

\section{Threshold dependence and the ROC fix}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Thresholds and the ROC curve}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Precision, recall and F1 all shift as you move the cutoff $c$. Fix: summarize a
      model across \emph{every} threshold at once.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Precision and recall come from a threshold}
  ``Flag more $\to$ recall up, precision down'' --- that tension \textbf{is} the threshold knob.
  Most classifiers output a \emph{score}, not a label:
  \[ \text{predict positive} \iff \text{score}(\xv) > c. \]
  \vfill
  \begin{itemize}\small
    \item \textbf{Lower $c$} ($\to 0$): flag more $\to$ higher recall, lower precision (more false alarms).
    \item \textbf{Higher $c$} ($\to 1$): flag fewer $\to$ higher precision, lower recall (more misses).
    \item \textbf{Default $c = 0.5$ is just one point on a curve} --- often not the right one.
  \end{itemize}
  \vfill
  \footnotesize So instead of one $(P,R)$ pair, summarize the model over \emph{all} thresholds.
\end{frame}

\begin{frame}{Predict-first: P and R at extreme thresholds}
  Two extreme cutoffs --- what are precision and recall at each?
  \begin{itemize}
    \item $c = 0$: predict positive for \emph{every} batch.
    \item $c = 1$: predict negative for \emph{every} batch.
  \end{itemize}
  \vfill
  \begin{itemize}\small
    \item \textbf{$c = 0$:} $FN = 0 \Rightarrow$ \textbf{recall $=100\%$}; $FP =$ all negatives $\Rightarrow$ \textbf{precision $=$ base rate} ($\sim 3\%$).
    \item \textbf{$c = 1$:} $TP = 0 \Rightarrow$ \textbf{recall $=0\%$}; precision $= 0/0$ (sklearn returns $0$).
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    Each of P and R is \textbf{trivially gamed} by an extreme threshold. F1 stops that but
    \emph{still depends on the threshold you picked}. We want a metric that summarizes
    performance \emph{across all thresholds at once}.}}
\end{frame}

\begin{frame}{The ROC curve: TPR vs.\ FPR}
  Two ratios: \textbf{TPR} $= \text{recall} = \dfrac{TP}{TP+FN}$ \quad and \quad
  \textbf{FPR} $= \dfrac{FP}{FP+TN}$ (the false-alarm rate).
  \vfill
  \textbf{Sweep} the cutoff from $c = 1$ (nothing flagged) down to $c = 0$ (all flagged); at
  each step plot the point $(\text{FPR}, \text{TPR})$. The trail is the \textbf{ROC curve}.
  \vfill
  \begin{itemize}\small
    \item Top-left corner $=$ perfect (TPR $=1$ at FPR $=0$).
    \item Diagonal $=$ random guessing; below it $=$ worse than random (flip the labels).
  \end{itemize}
  \vfill
  \footnotesize Let's build one, point by point (next).
\end{frame}

    \vskip 4pt
    The staircase joining $(0,0)$ to $(1,1)$ is the \textbf{ROC curve} --- one curve summarizing
    the model at \emph{every} threshold.
    \vskip 6pt

  \end{columns}
\end{frame}

\begin{frame}{ROC AUC}
  Summarize the whole ROC curve as one number: the \textbf{area under the curve}.
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \begin{itemize}\footnotesize
      \item AUC $= 1.0$: perfect (curve hits top-left).
      \item AUC $= 0.5$: random (curve $=$ diagonal).
      \item AUC $= 0.0$: perfectly wrong (flip labels).
    \end{itemize}
    \vskip 6pt
    \textbf{Probabilistic meaning.} AUC $=$ the probability that a random positive scores
    higher than a random negative (the Mann-Whitney $U$ statistic) --- a real probability,
    not just an area.
    \vskip 6pt
    \textbf{Threshold-free.} Compares models on ranking quality without picking a cutoff.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth,height=3.7cm,keepaspectratio]{fig/cm_roc.pdf}\\[-2pt]
      \scriptsize Real cheese model: shaded area $=$ AUC.
    \end{center}
  \end{columns}
  \vfill
  \footnotesize \textbf{Footgun:} feed AUC the \emph{scores} (\texttt{predict\_proba}), not the
  predicted labels --- labels collapse it toward accuracy.
\end{frame}

\section{When ROC fails: PR and lift}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{When ROC lies: PR and lift}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      On rare-positive data the huge negative class hides real differences in ROC.
      Fix: measure precision directly --- the PR curve.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Predict-first: ROC's blind spot on imbalanced data}
  Two models, same imbalanced cheese data: 100 bad batches, 10{,}000 clean.
  \vfill
  \begin{center}\small
    \begin{tabular}{lcccc}
      & \textbf{TP} & \textbf{FP} & \textbf{FN} & \textbf{TN} \\ \hline
      Model A & 91 & 10 & 9 & 9{,}990 \\
      Model B & 91 & 100 & 9 & 9{,}900 \\
    \end{tabular}
  \end{center}
  \vfill
  \textbf{Predict:} recall, FPR, precision for each --- which would you ship?
  \vfill
  \begin{center}\small
    \begin{tabular}{lccc}
      & \textbf{Recall} & \textbf{FPR} & \textbf{Precision} \\ \hline
      Model A & 91\% & $10/10{,}000 = \mathbf{0.10\%}$ & $91/101 = \mathbf{90\%}$ \\
      Model B & 91\% & $100/10{,}000 = \mathbf{1.0\%}$ & $91/191 = \mathbf{48\%}$ \\
    \end{tabular}
  \end{center}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{ROC's blind spot:} FPR barely moved ($0.1\% \to 1\%$) --- tiny in ROC space ---
    yet precision \emph{crashed} $90\% \to 48\%$ (half of B's inspections wasted). The huge
    negative class dominates the FPR denominator and hides the operational difference.}}
\end{frame}

\begin{frame}{The PR curve}
  Same threshold sweep, but plot \textbf{precision vs.\ recall} instead of TPR vs.\ FPR.
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \begin{itemize}\footnotesize
      \item Top-right $=$ perfect (precision $=1$, recall $=1$).
      \item Top-left $=$ strict threshold (few but high-quality alarms).
      \item Bottom-right $=$ loose threshold (many alarms, many false).
      \item Baseline: ``always positive'' has precision $=$ base rate.
    \end{itemize}
    \vskip 6pt
    \textbf{Doesn't depend on TN} --- the TN count never enters precision or recall. That is
    exactly what makes PR informative when positives are rare.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{fig/cm_pr.pdf}\\[-2pt]
      \scriptsize Real PR curve (note the zig-zag); dashed baseline $=$ base rate.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{PR AUC (Average Precision)}
  Summarize the PR curve as one number --- \textbf{average precision}, the area under it:
  \[ \text{AP} = \sum_n (R_n - R_{n-1}) \cdot P_n \]
  ($n$ indexes the distinct recall steps as you sort by score).
  \vfill
  \begin{itemize}
    \item AP $= 1.0$: perfect ranking.
    \item AP $=$ \textbf{base rate}: random ranking. \emph{Not 0.5!}
    \item Higher AP $=$ better ranking, weighted toward high-precision regions.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Sensitive to imbalance, by design.} In the A-vs-B example ROC AUC barely changes
    but PR AUC drops sharply (precision dropped sharply). For rare-class problems, PR AUC
    matches operational reality.}}
\end{frame}

\begin{frame}{Lift and gain charts: the ``top-$k$\%'' question}
  Operational constraint: the factory can inspect only the top 20\% most-suspicious batches
  per day. \emph{Which to inspect, and how much better than random?}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \footnotesize
    \textbf{Cumulative gain.} Rank by score; $x =$ fraction inspected, $y =$ fraction of bad
    caught. Random $=$ diagonal; perfect $=$ climbs to $100\%$ by the base rate.
    \vskip 6pt
    \textbf{Lift} $= \dfrac{\text{\% bad caught in top }k\%}{k\%}$. Random $=1$; lift $5$ at the
    top decile $=$ $5\times$ more bad caught than random for the same budget.
    \vskip 6pt
    \textbf{Why it matters.} AUCs summarize the whole curve; lift answers the question
    production actually asks: \emph{given my budget, how much better than random?}
    \column{0.48\textwidth}
    \begin{center}
      \includegraphics[width=\textwidth]{fig/cm_lift.pdf}\\[-2pt]
      \scriptsize Real cheese model; annotation reads off the lift at the top 20\%.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{ROC vs.\ PR side-by-side on imbalanced data}
  Same imbalanced dataset (1\% positive), same two models. Two views:
  \begin{center}
    \includegraphics[width=0.8\textwidth]{fig/cm_roc_vs_pr.pdf}
  \end{center}
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Same models, two stories.} ROC: both near top-left (AUC $0.96$--$0.98$). PR: A
    holds precision (AP $0.72$), B collapses toward baseline (AP $0.35$). On imbalanced data
    \emph{ROC overstates and PR clarifies} --- reach for PR when the positive class is rare.}}
\end{frame}

\section{Practical: choosing the metric}

\begin{frame}{The metric zoo: one matrix, many names}
  Every metric is a ratio of $TP, FP, FN, TN$ --- they split by \emph{what you condition on}:
  \vskip 4pt
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Given the actual class} (row-wise):
    \begin{itemize}\small
      \item \textbf{TPR} $=$ recall $=$ sensitivity $= \frac{TP}{TP+FN}$
      \item \textbf{FNR} $=$ miss rate $= \frac{FN}{TP+FN} = 1-\text{TPR}$
      \item \textbf{TNR} $=$ specificity $= \frac{TN}{TN+FP}$
      \item \textbf{FPR} $=$ fall-out $= \frac{FP}{FP+TN} = 1-\text{TNR}$
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Given the prediction} (column-wise):
    \begin{itemize}\small
      \item \textbf{PPV} $=$ precision $= \frac{TP}{TP+FP}$
      \item \textbf{FDR} $= \frac{FP}{TP+FP} = 1-\text{PPV}$
      \item \textbf{NPV} $= \frac{TN}{TN+FN}$
      \item \textbf{FOR} $= \frac{FN}{TN+FN} = 1-\text{NPV}$
    \end{itemize}
  \end{columns}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.985\linewidth}{\small
    \textbf{Overall:} accuracy $=\frac{TP+TN}{\text{all}}$; \textbf{balanced accuracy}
    $=\frac{\text{TPR}+\text{TNR}}{2}$ (imbalance-robust); \textbf{F1} (harmonic mean of
    precision \& recall); \textbf{MCC} $\in[-1,1]$ (correlation of prediction \& truth, $0=$
    random). ROC plots TPR vs.\ FPR; PR plots precision vs.\ recall.}}
\end{frame}

\begin{frame}{Which metric for which problem?}
  \begin{center}\small
    \begin{tabular}{p{4.5cm} p{4.3cm} p{4cm}}
      \textbf{Situation} & \textbf{Use} & \textbf{Why} \\ \hline
      Balanced classes, single threshold & F1 (or accuracy) & Single number; symmetric. \\
      Balanced, threshold-free & ROC AUC & Compares ranking quality. \\
      \textbf{Imbalanced, threshold-free} & PR AUC (avg.\ precision) & ROC overstates; PR matches reality. \\
      Bounded inspection budget & Lift / gain at top-$k$\% & Capacity is the real constraint. \\
      FN $\gg$ FP (cheese, fraud, screening) & Recall, with a precision floor & Asymmetric cost. \\
      FP $\gg$ FN (spam, churn calls) & Precision, with a recall floor & Asymmetric cost, other way. \\
      One imbalance-robust number & MCC (Matthews CC) & Symmetric; $\in[-1,1]$; $0=$ random. \\
    \end{tabular}
  \end{center}
  \vfill
  \scriptsize \textbf{Default:} report 3--4 metrics --- one threshold-free (ROC or PR) plus
  precision/recall at a chosen threshold. \emph{Calibration is its own lecture.}
\end{frame}

\section{Multi-class evaluation}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Beyond two classes}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      With $K$ classes the confusion matrix grows to $K\times K$, and the binary metrics need
      an averaging rule. (Cheese defects: good / moldy / dry.)
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The $K\times K$ confusion matrix}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \includegraphics[width=\textwidth]{fig/cm_multiclass.pdf}
    \column{0.48\textwidth}
    \small
    \begin{itemize}\small
      \item Diagonal $=$ correct; off-diagonal shows \emph{which classes get confused}.
      \item Here \textbf{moldy} and \textbf{dry} bleed into each other ($30$ and $49$); \textbf{good} is clean.
      \item Per-class precision/recall come from \textbf{one-vs-rest}: treat ``class $k$ vs.\ the rest'' as a binary problem and read its row/column.
    \end{itemize}
    \vskip 4pt
    \footnotesize One matrix, $K$ binary views.
  \end{columns}
\end{frame}

\begin{frame}{Averaging: macro vs.\ micro vs.\ weighted}
  Compute the binary metric one-vs-rest per class, then average. Three ways:
  \vfill
  \begin{itemize}
    \item \textbf{Macro:} average per-class, \emph{equal weight per class}. Best when every class
      matters regardless of size (``how well on each cheese type?'').
    \item \textbf{Micro:} pool all $TP/FP/FN$ across classes, then compute. \emph{Equals accuracy}
      for single-label problems. Best for overall correctness.
    \item \textbf{Weighted:} per-class, weighted by class support. Reflects the data distribution
      (the common default in reports).
  \end{itemize}
  \vfill
  \footnotesize Multi-class AUC: average per-class one-vs-rest (macro-OVR is standard).
  \vskip 2pt
  \scriptsize \emph{Caveat:} with few samples in a class, its per-class F1 is noisy and macro F1
  inherits that noise --- always report \texttt{support} (samples per class).
\end{frame}

\section{Threshold tuning}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Picking the operating point}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      AUC and PR rank models \emph{without} a threshold. To deploy you must choose one
      cutoff $c$ --- and on imbalanced, asymmetric-cost problems $0.5$ is almost never it.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Every metric is a function of the threshold}
  \begin{columns}[T]
    \column{0.46\textwidth}
    The model outputs scores; the \textbf{cutoff $c$} turns them into decisions. Sweep it:
    \begin{itemize}\small
      \item low $c$ $\Rightarrow$ flag more $\Rightarrow$ recall up, precision down;
      \item high $c$ $\Rightarrow$ flag fewer $\Rightarrow$ precision up, recall down.
    \end{itemize}
    \vskip 4pt
    The \textbf{default $0.5$ is just one point}. Here the F1-best cutoff is $\approx 0.08$
    --- the $3\%$ imbalance pushes it far left of $0.5$.
    \column{0.54\textwidth}
    \includegraphics[width=\textwidth]{fig/cm_threshold_metrics.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Cost-sensitive thresholding}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Choose $c$ to minimize \emph{expected cost}, not a symmetric metric:
    \[ c^* = \argmin_c \; C_{FN}\,FN(c) + C_{FP}\,FP(c). \]
    With a miss $10\times$ a false alarm, $c^* \approx 0.08$ here --- far below $0.5$ (real
    cheese ratios are larger, pushing it lower still).
    \vskip 6pt
    \fcolorbox{armred}{armred!8}{\parbox{0.95\linewidth}{\small
      \textbf{Tune on a validation set}, never on the test set you report --- otherwise the
      chosen threshold is optimistically biased too.}}
    \column{0.5\textwidth}
    \includegraphics[width=\textwidth]{fig/cm_cost_curve.pdf}
  \end{columns}
  \vskip 3pt
  \footnotesize In sklearn, \texttt{TunedThresholdClassifierCV} ($\ge 1.5$) picks $c$ by
  cross-validation for any scorer --- including a business-cost scorer.
\end{frame}

  \vfill
  \footnotesize \textbf{Always report multiple metrics.} A single number hides the trade-off; the trade-off is the story.
\end{frame}

\begin{frame}{Recap: the cascade}
  \textbf{Each metric fixes the previous one's problem:}
  \begin{enumerate}\small
    \item \textbf{Accuracy} --- simple, fails on imbalance ($97\%$ accuracy, $0\%$ recall).
    \item \textbf{Precision \& recall} --- focus on the positive class. \emph{But threshold-dependent.}
    \item \textbf{F1} --- one number that punishes a P/R gap. \emph{Still threshold-dependent.}
    \item \textbf{ROC AUC} --- threshold-free ranking quality. \emph{Overstates on imbalance.}
    \item \textbf{PR AUC} --- threshold-free \emph{and} imbalance-sensitive: the rare-class metric.
    \item \textbf{Lift / gain} --- the operational ``top-$k$\%'' question under a budget.
  \end{enumerate}
  \vskip 4pt
  \textbf{Plus:} multi-class averaging (macro/micro/weighted) and \textbf{threshold tuning} (the
  cost-optimal cutoff rarely equals $0.5$).
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{One-line takeaway:} report multiple metrics --- always one that matches your data's
    imbalance and your problem's cost structure. \textbf{Next:} calibration.}}
\end{frame}
