\begin{frame}{When the model says 0.9, is it bad 90\% of the time?}
  Last lecture measured how well our cheese detector \emph{ranks} batches (ROC, PR). But it
  also outputs a \textbf{number} --- $\hat p = 0.9$. If we tell the factory ``$90\%$ chance
  this batch is bad,'' is that \emph{true}?
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    A model can \textbf{rank} perfectly (high AUC) and still output probabilities that are
    systematically wrong --- saying $0.9$ when only $70\%$ of such batches are actually bad.
    That gap is \textbf{(mis)calibration}.}}
  \vfill
  This deck: what calibration is, how to measure it, and how to fix it.
\end{frame}

\section{What calibration is}

\begin{frame}{What ``calibrated'' means}
  A classifier is \textbf{calibrated} if its probabilities match reality:
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Among all batches it scores $\hat p \approx 0.8$, about \textbf{80\%} are actually bad.}}
  \vfill
  \begin{itemize}
    \item A \textbf{score} ranks; a \textbf{calibrated probability} can be read as a real chance.
    \item Calibration is about the \emph{value} of $\hat p$, not the \emph{order} --- a separate
      property from accuracy or AUC.
    \item We check it by bucketing predictions and comparing predicted vs.\ observed frequency.
  \end{itemize}
\end{frame}

\begin{frame}{Building the reliability diagram}
  Take the model's predictions on held-out data and:
  \begin{enumerate}\small
    \item \textbf{Bin} the predicted probabilities $\hat p$ --- say 10 equal-width bins
      $[0,0.1), [0.1,0.2), \dots, [0.9,1]$.
    \item In each bin compute two numbers: the \textbf{mean predicted} $\hat p$ ($\to x$) and the
      \textbf{observed fraction of positives} ($\to y$). \emph{Of the batches this bin scored
      ``$\approx 0.8$'', how many were actually bad?}
    \item \textbf{Plot} each bin as a point $(x,y)$ and join them.
  \end{enumerate}
  \vskip 2pt
  A perfectly calibrated model has $y = x$ in every bin --- the points land on the
  \textbf{45$^\circ$ diagonal}.
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Not a CDF.} Both axes are probabilities and it rises left-to-right, so it \emph{looks}
    like one --- but it is \textbf{not cumulative}. Each point is a \emph{local} average inside one
    bin, and the reference is the diagonal $y=x$ (not a CDF's $0\!\to\!1$ ramp). The same model can
    sit \emph{above} the diagonal in some bins and \emph{below} it in others.}}
\end{frame}

\begin{frame}{Reading the reliability diagram}
  \begin{columns}[T]
    \column{0.48\textwidth}
    Compare each bin's point to the diagonal:
    \begin{itemize}\small
      \item \textbf{on} the diagonal $=$ perfectly calibrated;
      \item \textbf{below} $=$ \textbf{over-confident} (predicted higher than reality);
      \item \textbf{above} $=$ \textbf{under-confident} (predicted lower).
    \end{itemize}
    \vskip 4pt
    Logistic regression tracks the diagonal; naive Bayes bows away from it, pushing $\hat p$ toward
    $0$ and $1$.
    \vskip 4pt
    Its \textbf{ECE} (\emph{expected calibration error}) --- the bin-size-weighted average of those
    vertical gaps --- is $0.055$, vs.\ $0.016$ for the calibrated model. We compute one by hand
    next.
    \column{0.52\textwidth}
    \includegraphics[width=\textwidth]{fig/cal_reliability.pdf}
  \end{columns}
\end{frame}

\begin{frame}{Worked example: computing ECE by hand}
  Bin the predictions; in each bin compare the \textbf{average prediction} $\hat p_b$ to the
  \textbf{observed frequency} of bad batches, then average the gaps \emph{weighted by bin size}.
  \begin{columns}[T]
    \column{0.54\textwidth}
    \begin{center}\small
      \renewcommand{\arraystretch}{1.25}
      \begin{tabular}{c c c c}
        bin & $n_b$ & $\hat p_b$ / obs. & $|\mathrm{gap}|$ \\
        \hline
        $[0.0,0.2)$ & $50$ & $0.10 / 0.10$ & $0.00$ \\
        $[0.2,0.4)$ & $20$ & $0.30 / 0.25$ & $0.05$ \\
        $[0.6,0.8)$ & $20$ & $0.70 / 0.50$ & $0.20$ \\
        $[0.8,1.0]$ & $10$ & $0.90 / 0.60$ & $0.30$ \\
        \hline
      \end{tabular}\\[3pt]
      \scriptsize $n=100$ predictions; empty bins contribute $0$.
    \end{center}
    \column{0.46\textwidth}
    \small Weight each bin's gap by its share of all predictions:
    \[ \mathrm{ECE} = \sum_b \frac{n_b}{n}\,\big|\mathrm{obs}_b - \hat p_b\big| \]
    \begin{align*}
      &= \tfrac{50}{100}(0) + \tfrac{20}{100}(.05) \\
      &\quad + \tfrac{20}{100}(.20) + \tfrac{10}{100}(.30) \\[2pt]
      &= .01 + .04 + .03 = \mathbf{0.08}.
    \end{align*}
  \end{columns}
  \vskip 2pt
  \footnotesize The two high-score bins are \textbf{over-confident} (say $0.7$/$0.9$, only
  $0.5$/$0.6$ actually bad) and drive almost all the error. \emph{ECE is just a weighted average
  of the reliability diagram's vertical gaps.}
\end{frame}

\begin{frame}{Over-confidence, seen another way}
  \begin{columns}[T]
    \column{0.48\textwidth}
    Same two models, histogram of their predicted probabilities.
    \begin{itemize}\small
      \item \textbf{Over-confident} models pile scores near $0$ and $1$ --- they rarely say ``maybe''.
      \item Calibrated models spread mass across $[0,1]$, matching real uncertainty.
    \end{itemize}
    \vskip 4pt
    Piling at the extremes is the fingerprint of miscalibration --- and exactly what breaks any
    decision that reads $\hat p$ as a chance.
    \column{0.52\textwidth}
    \includegraphics[width=\textwidth]{fig/cal_hist.pdf}
  \end{columns}
\end{frame}

\section{Why it matters}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Calibration $\neq$ ranking}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      A model can order batches perfectly yet report wrong probabilities. AUC won't catch it
      --- and any cost decision that uses $\hat p$ will be off.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Predict-first: perfect ranking, honest probabilities?}
  A cheese detector ranks \textbf{perfectly} --- every bad batch scores above every clean one,
  so \textbf{ROC AUC $= 1.0$}. Now \textbf{halve every predicted probability}
  ($\hat p \to \hat p / 2$).
  \vfill
  \textbf{Predict:} what happens to the AUC? to the calibration?
  \vfill
  \begin{itemize}\small
    \item \textbf{AUC: unchanged} ($=1.0$). Halving preserves the \emph{order}, and AUC sees
      only order.
    \item \textbf{Calibration: wrecked.} Batches now scored $0.45$ are still bad $90\%$ of the
      time --- the numbers stopped meaning anything.
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Ranking and calibration are different properties.} A model can order perfectly and
    still lie about probabilities --- and AUC will never catch it.}}
\end{frame}

\begin{frame}{Calibration is separate from discrimination}
  \begin{itemize}
    \item \textbf{Discrimination} (ROC AUC): does the score \emph{order} positives above
      negatives? Cares only about ranking.
    \item \textbf{Calibration}: are the score \emph{values} true frequencies?
  \end{itemize}
  \vfill
  As the last slide showed, a model can reach \textbf{AUC $=1$} (perfect ranking) and still be
  badly miscalibrated --- ranking and probability values are different properties.
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Why it matters.} The cost-optimal threshold $c^* = \frac{C_{FP}}{C_{FN}+C_{FP}}$
    and \emph{any} expected-value decision assume $\hat p$ is a true probability. Reporting
    ``$90\%$ risk'' to the factory (or a doctor) only means something if it is calibrated.}}
\end{frame}

\begin{frame}{When calibration matters}
  \textbf{The deciding question: does your decision read the \emph{value} of $\hat p$ as a chance?}
  If yes, calibrate.
  \vfill
  \begin{itemize}\small
    \item \textbf{You report the probability to someone who acts on the number} --- ``$30\%$ chance
      malignant'' to a doctor, ``$90\%$ bad'' to the factory, ``$5\%$ default'' to a lender. A
      wrong $0.9$ is a wrong decision.
    \item \textbf{Expected-value / cost decisions.} The cost-optimal threshold
      $c^* = \frac{C_{FP}}{C_{FN}+C_{FP}}$ (L13) plugs $\hat p$ into an expected-cost calculation
      --- feed it garbage probabilities and $c^*$ is wrong.
    \item \textbf{You combine probabilities} --- averaging or ensembling several models, multiplying
      independent ones, or feeding $\hat p$ into a downstream probabilistic model. Only calibrated
      numbers combine correctly.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Rule: \textbf{if the number itself is consumed}, it has to be true.}}
\end{frame}

\begin{frame}{When you can skip it}
  \textbf{If you only ever \emph{compare} scores --- never read one as a chance --- calibration
  changes nothing.} A monotonic distortion leaves the order intact.
  \vfill
  \begin{itemize}\small
    \item \textbf{You only need the ranking:} ROC-AUC, PR ranking, top-$k\%$ / lift, ``flag the
      worst 100 batches'' --- all depend on \emph{order}, which miscalibration doesn't touch.
    \item \textbf{You tune the threshold directly} on your target metric (F1, cost) on a validation
      set --- the threshold \textbf{absorbs} the miscalibration, so $\hat p$'s exact value never
      matters.
  \end{itemize}
  \vskip 2pt
  \footnotesize Even then, calibrated probabilities can still help \emph{interpretability and
  monitoring} --- they just won't change these particular decisions.
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{Golden rule (whenever you calibrate):} fit on data the base model never saw, and don't
    re-tune the threshold on that same split --- both bias the result optimistically.}}
\end{frame}

\begin{frame}{Who is miscalibrated?}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Usually fine}
    \begin{itemize}\small
      \item \textbf{Logistic regression} --- it minimizes log-loss, a \emph{proper scoring rule}
        (one minimized only by reporting the true probabilities, so honesty is rewarded).
    \end{itemize}
    \vskip 2pt
    \textbf{Over-confident} (scores too extreme)
    \begin{itemize}\small
      \item Naive Bayes (independence assumption)
      \item Boosted trees, deep neural nets
      \item SVMs (margins, not probabilities)
    \end{itemize}
    \column{0.5\textwidth}
    \textbf{Under-confident} (scores too timid)
    \begin{itemize}\small
      \item Random forests (averaging pulls toward $0.5$)
    \end{itemize}
    \vskip 6pt
    \fcolorbox{armorange}{armorange!12}{\parbox{0.95\linewidth}{\scriptsize
      \textbf{Watch out:} \texttt{class\_weight} and resampling (for imbalance) also distort
      probabilities --- they change the base rate the model trains on, so re-calibrate
      afterwards.}}
  \end{columns}
\end{frame}

\begin{frame}{Measuring calibration}
  \begin{itemize}
    \item \textbf{Reliability diagram} --- the picture (qualitative).
    \item \textbf{ECE} (expected calibration error): bin predictions, average
      $|\text{observed} - \text{predicted}|$ weighted by bin size. The headline number; $0$ is
      perfect. \emph{Caveat: depends on the binning.}
    \item \textbf{Brier score} $=\tfrac1n\sum(\hat p_i - y_i)^2$ --- a proper scoring rule, but
      it \emph{mixes} calibration and ranking, so a sharp model can score well while miscalibrated.
    \item \textbf{Log-loss} --- also proper; punishes confident mistakes hardest.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    For a \emph{pure} calibration number use \textbf{ECE} (or read the reliability diagram);
    Brier and log-loss are good overall scores but conflate calibration with discrimination.}}
\end{frame}

\section{Fixing it}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Fixing calibration}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Don't retrain --- learn a small post-hoc map from the model's raw scores to true
      probabilities, fit on held-out data.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Post-hoc calibration: the common idea}
  \textbf{Don't retrain.} Freeze the fitted model, then learn a one-dimensional map
  \[ g:\ \text{raw score } s \ \longmapsto\ \text{calibrated probability } \hat p \]
  from a \textbf{held-out calibration set} of (score, label) pairs --- data the base model never
  trained on.
  \vfill
  \begin{itemize}\small
    \item Two standard choices, differing only in how flexible $g$ is: \textbf{Platt scaling}
      (a rigid sigmoid) and \textbf{isotonic regression} (a free monotonic curve).
    \item Both are \textbf{monotonic} maps, so they never reorder the scores --- the
      \textbf{ranking (and AUC) is unchanged}; only the \emph{values} move onto the diagonal.
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Calibration is a cheap \emph{post-processing} step: one extra fit on a held-out split, and the
    model itself is untouched.}}
\end{frame}

\begin{frame}{Platt scaling: fit a sigmoid on the scores}
  \begin{columns}[T]
    \column{0.56\textwidth}
    Push the raw score $s$ through a \textbf{logistic sigmoid}:
    \[ \hat p = \sigma(a\,s + b) = \frac{1}{1 + e^{-(a s + b)}}. \]
    \begin{itemize}\small
      \item Fit just \textbf{two numbers} $a, b$ by minimizing log-loss on the held-out
        (score, label) pairs --- literally a 1-feature logistic regression \emph{on the scores}.
      \item \textbf{Data-thrifty:} two parameters work with only a few hundred calibration points.
      \item \textbf{Assumption:} the distortion is a smooth \textbf{S-shape}. If the real
        miscalibration isn't sigmoidal, Platt can't remove it.
    \end{itemize}
    \column{0.44\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      Originally \textbf{Platt (1999)}, to turn SVM margins into probabilities.}}
    \vskip 8pt
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      \textbf{Neural nets:} \textbf{temperature scaling} is the 1-parameter version
      $\hat p = \sigma(z/T)$ (fit only $T$ on the logits $z$) --- the standard fix (Guo et al.,
      2017).}}
  \end{columns}
\end{frame}

\begin{frame}{Isotonic regression: a free monotonic staircase}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \textbf{On what:} the held-out pairs $(s_i, y_i)$ --- raw score and true label
    $y_i \in \{0,1\}$ (same data as Platt).\\[2pt]
    \textbf{How:} fit the \textbf{non-decreasing} map $m$ that matches those 0/1 labels in least
    squares:
    \[ m = \argmin_{g\ \text{non-decr.}}\ \sum_i \big(g(s_i) - y_i\big)^2. \]
    \begin{itemize}\small
      \item \textbf{Pool-adjacent-violators:} sort by score; wherever the fit would \emph{drop},
        \textbf{pool} those points into their \textbf{average label} (the local positive rate);
        repeat until monotone $\Rightarrow$ a piecewise-constant \textbf{staircase}.
    \end{itemize}
    \column{0.45\textwidth}
    \includegraphics[width=\textwidth]{fig/cal_before_after.pdf}\\[3pt]
    \footnotesize Snaps naive Bayes onto the diagonal: ECE $0.055 \to 0.012$.
  \end{columns}
  \vskip 3pt
  \footnotesize \textbf{No shape assumption} $\Rightarrow$ fixes any monotonic distortion, but
  \textbf{needs more data} ($\gtrsim$ a few thousand) or the steps overfit noise.
\end{frame}

\begin{frame}[fragile]{Calibration in \texttt{sklearn}}
\begin{lstlisting}
from sklearn.calibration import CalibratedClassifierCV

# wrap any classifier; calibrate by cross-validation on the training data
cal = CalibratedClassifierCV(
    GaussianNB(),
    method="isotonic",     # or "sigmoid" (Platt) for small data
    cv=5,
)
cal.fit(X_train, y_train)
cal.predict_proba(X_test)[:, 1]    # now-calibrated probabilities
\end{lstlisting}
  \vskip 2pt
  \begin{itemize}\small
    \item \texttt{method="sigmoid"} (Platt) for small data; \texttt{"isotonic"} for larger
      ($\gtrsim$ a few thousand).
    \item Calibrate on data the base model did \textbf{not} train on --- \texttt{cv} handles this.
    \item \texttt{CalibrationDisplay.from\_estimator(...)} draws the reliability diagram.
  \end{itemize}
\end{frame}

\section*{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Calibration} $=$ predicted probabilities match observed frequencies (read it
      off the \textbf{reliability diagram}).
    \item It is \textbf{separate from ranking}: high AUC does not imply good probabilities.
    \item Measure with \textbf{ECE} (pure calibration) or Brier / log-loss (overall, proper scores).
    \item Logistic regression is usually calibrated; naive Bayes, boosted trees and nets are
      over-confident; random forests under-confident; imbalance fixes distort probabilities.
    \item Fix post-hoc with \textbf{Platt} (sigmoid, small data) or \textbf{isotonic} (flexible,
      more data) via \texttt{CalibratedClassifierCV} --- always on held-out data.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Next:} imbalanced learning --- class weights and resampling when the positive class
    is rare (and why they then need re-calibration).}}
\end{frame}
