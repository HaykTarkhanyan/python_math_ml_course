\section{What SHAP is}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{What SHAP is}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      A fair split of the prediction among the features.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The idea: a fair payout}
  \begin{columns}[T]
    \column{0.48\textwidth}
    Treat the prediction as a \textbf{payout to split fairly} among the features --- like a team
    bonus among players who contributed differently.
    \vskip 6pt
    \textbf{SHAP} gives each feature its \textbf{fair share}: its contribution \emph{averaged over
    all the orders} in which features could be added to the model.
    \vskip 4pt
    \footnotesize That averaging is the whole game-theory part. \textbf{Next:} how that fair share is
    \emph{actually computed} --- then we read the plots.
    \column{0.52\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/shapley_1.png}\\[-2pt]
      \scriptsize Source: LMU IML (CC BY 4.0).
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{By hand: three friends split the payout}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Adi}, \textbf{Budi} and \textbf{Gogi} run a stall together. Working in different groups,
    they would earn different amounts --- the group's \textbf{payout} $v(S)$:
    \vskip 6pt
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      Together they earn \textbf{90}. \textbf{What is each one's fair share} of that 90?}}
    \vskip 6pt
    {\footnotesize \textit{Names: qarakusi.am}}
    \column{0.48\textwidth}
    \centering\small
    \renewcommand{\arraystretch}{1.12}
    \begin{tabular}{l r}
      \textbf{who works together} $S$ & $v(S)$ \\
      \hline
      nobody & 0 \\
      Adi & 10 \\
      Budi & 20 \\
      Gogi & 30 \\
      Adi $+$ Budi & 40 \\
      Adi $+$ Gogi & 50 \\
      Budi $+$ Gogi & 60 \\
      \textbf{all three} & \textbf{90} \\
      \hline
    \end{tabular}
  \end{columns}
\end{frame}

\begin{frame}{Fair share $=$ average marginal contribution}
  \begin{columns}[T]
    \column{0.54\textwidth}
    \textbf{Adi}'s \emph{extra} contribution depends on who is \textbf{already there}:
    \vskip 3pt
    {\footnotesize
    \renewcommand{\arraystretch}{1.15}
    \begin{tabular}{l c c}
      who's before Adi & $\Delta_{\text{Adi}}$ & \#orders \\
      \hline
      nobody (Adi first) & $10-0 = 10$ & 2 \\
      Budi & $40-20 = 20$ & 1 \\
      Gogi & $50-30 = 20$ & 1 \\
      Budi \& Gogi & $90-60 = 30$ & 2 \\
      \hline
    \end{tabular}}
    \vskip 4pt
    Average over all \textbf{6 orders}:
    \[ \phi_{\text{Adi}} = \tfrac{2(10)+20+20+2(30)}{6} = \mathbf{20}. \]
    \column{0.46\textwidth}
    Same for each friend:
    \[ \phi_{\text{Adi}}{=}20,\ \ \phi_{\text{Budi}}{=}30,\ \ \phi_{\text{Gogi}}{=}40 \]
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      They \textbf{sum to 90} --- the whole payout is split, nothing lost or invented
      (\emph{efficiency}).}}
    \vskip 4pt
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      \textbf{In a model:} the players are \textbf{features}, the payout is
      $\hat f(\xv) - \mathbb{E}[\hat f]$, and each fair share is that feature's \textbf{SHAP value}
      $\phi_j$.}}
  \end{columns}
\end{frame}

\begin{frame}{One prediction = base value + the SHAP values}
  Every SHAP explanation is one simple sum:
  \[
    \underbrace{\hat f(\xv)}_{\text{this prediction}} \;=\;
    \underbrace{\mathbb{E}[\hat f(X)]}_{\text{base value (avg.\ prediction)}}
    \;+\; \sum_j \phi_j ,
  \]
  where $\phi_j$ is feature $j$'s SHAP value.
  \begin{itemize}\small
    \item $\phi_j > 0$ pushes this prediction \textbf{up}; $\phi_j < 0$ pushes it \textbf{down}.
    \item \textbf{Every SHAP plot visualizes this sum} --- for one row, or aggregated over many.
  \end{itemize}
\end{frame}

\begin{frame}{Exact SHAP explodes --- TreeSHAP \& KernelSHAP to the rescue}
  \begin{columns}[T]
    \column{0.44\textwidth}
    3 features $\to$ $3! = 6$ orders ($2^3 = 8$ coalitions): easy by hand. But it \textbf{explodes}
    with the number of features:
    \vskip 4pt
    \centering\small
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{r r}
      \textbf{features} & \textbf{orders} $n!$ \\
      \hline
      3  & 6 \\
      10 & $3.6$ million \\
      20 & $2.4\times10^{18}$ \\
      \hline
    \end{tabular}
    \vskip 4pt
    \raggedright You \textbf{cannot} enumerate them all $\Rightarrow$ we \textbf{approximate}.
    \column{0.56\textwidth}
    \fcolorbox{armblue}{armblue!8}{\parbox{0.95\linewidth}{\small
      \textbf{KernelSHAP:} \emph{sample} a handful of coalitions and fit a \textbf{weighted linear
      model} to recover the $\phi_j$. Works on \textbf{any} model --- but approximate and slow.}}
    \vskip 8pt
    \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.95\linewidth}{\small
      \textbf{TreeSHAP:} exploits the \textbf{tree structure} to get \textbf{exact} SHAP in
      \emph{polynomial} time --- fast $+$ exact for trees/boosting $\Rightarrow$ the \textbf{default}.}}
  \end{columns}
\end{frame}

\section{The SHAP plots (how to read them)}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{The SHAP plots --- how to read them}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      One local plot, three global ones. Same SHAP values, different views.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Waterfall: one prediction, fully explained}
  \begin{columns}[T]
    \column{0.42\textwidth}
    Start at the \textbf{base value} ($\sim\!4502$ rentals) and add each feature's SHAP value until
    you reach \textbf{this day's prediction} ($4311$).
    \vskip 4pt
    \textbf{Read it:} \texttt{yr} ($+860$) and \texttt{temp} ($+771$) push it \textbf{up}, but a
    very humid day (\texttt{hum} $-1590$) and poor \texttt{weathersit} ($-418$) push it
    \textbf{down} --- landing just below average.
    \column{0.58\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/shap_waterfall_bike.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Bar: global importance (mean $|$SHAP$|$)}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Average the \textbf{magnitude} of each feature's SHAP over all rows:
    \[ \text{importance}_j = \tfrac1n\sum_i |\phi_j^{(i)}| \]
    \begin{itemize}\small
      \item One bar per feature --- \textbf{reads like the PFI bar} from deck 2.
      \item \texttt{yr}, \texttt{temp} dominate --- roughly agreeing with PFI.
    \end{itemize}
    \column{0.54\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/shap_bar.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Beeswarm: the signature SHAP plot}
  \begin{columns}[T]
    \column{0.42\textwidth}
    \textbf{One dot per row, per feature.} \quad $x$ = SHAP value; \textbf{colour = feature value}.
    \vskip 4pt
    \textbf{Read it:}
    \begin{itemize}\small
      \item \textbf{Spread} = how much the feature matters.
      \item \textbf{Colour pattern} = direction: high \texttt{temp} (red) $\to$ positive SHAP
        (more rentals); low (blue) $\to$ negative.
    \end{itemize}
    \column{0.58\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/shap_beeswarm.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{Dependence: the effect shape (\& interactions)}
  \begin{columns}[T]
    \column{0.42\textwidth}
    SHAP value of a feature \textbf{vs.\ its value} --- one dot per row.
    \vskip 4pt
    \begin{itemize}\small
      \item Shows the \textbf{effect shape}: rentals \textbf{rise sharply} with \texttt{temp}, peak,
        then \textbf{dip at extreme heat} ($\approx$ a PDP).
      \item Colour by a second feature (here \texttt{yr}) reveals \textbf{interactions}.
      \item Per-instance --- \emph{like ICE did for PDP}, it shows the spread the average hides.
    \end{itemize}
    \column{0.58\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/shap_dependence.pdf}
    \end{center}
  \end{columns}
\end{frame}

\section{SHAP in practice}

\begin{frame}{How SHAP ties the methods together}
  \begin{itemize}\small
    \item \textbf{TreeSHAP} (previous slide) makes SHAP the \textbf{default} for trees/boosting.
    \item \textbf{SHAP unifies what we've seen:}
      \begin{itemize}\small
        \item the \textbf{bar} $\approx$ PFI (global importance),
        \item the \textbf{dependence} plot $\approx$ PDP (effect shape),
        \item \textbf{plus} a \emph{local} view (the waterfall) the others can't give --- with a
          consistency guarantee (use a feature more $\Rightarrow$ its SHAP can't fall).
      \end{itemize}
  \end{itemize}
  \vfill
  \fcolorbox{armred}{armred!8}{\parbox{0.97\linewidth}{\small
    \textbf{But like every method here, correlation muddies SHAP too} --- the \texttt{temp}/\texttt{atemp}
    credit split stays ambiguous. SHAP also explains the model's \emph{score}: if that score isn't
    \textbf{calibrated} (L13), SHAP faithfully explains a \emph{miscalibrated} number. Still
    \textbf{model behaviour, not causation} (deck 2's ice cream), and rich plots are \textbf{easy to over-read}.}}
\end{frame}

\begin{frame}[fragile]{SHAP in code}
\begin{lstlisting}
import shap
explainer = shap.TreeExplainer(model)      # fast & exact for trees/boosting
sv = explainer(X)                          # SHAP values for every row

shap.plots.waterfall(sv[i])                # one prediction
shap.plots.bar(sv)                         # global: mean |SHAP|
shap.plots.beeswarm(sv)                    # global: dot per row, colour = value
shap.plots.scatter(sv[:, "temp"])          # dependence: effect shape
\end{lstlisting}
  \vskip 2pt
  \footnotesize \texttt{shap.Explainer(model, X)} auto-picks the right method (TreeSHAP here).
\end{frame}

\section{LIME}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{LIME: the lighter local cousin}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      A simple, model-agnostic local explanation --- with sharper edges.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{LIME: fit a simple model, locally}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{LIME} explains one prediction with a \textbf{local surrogate}: jitter the inputs around
    the instance, weight by \textbf{closeness}, and fit a \textbf{simple sparse linear model} ---
    its weights say ``these few features pushed this prediction.'' Works on tabular, text, \& images.
    \vskip 6pt
    \textbf{Predict-first:} a husky-vs-wolf classifier is $\sim\!90\%$ accurate. What did LIME
    reveal it actually keys on?
    \vskip 2pt
    \pause
    \textbf{The \emph{snow} in the background} --- not the animal. A \emph{clever-Hans} model (right
    answer, wrong reason), caught by a local explanation.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/lime_husky.png}\\[-2pt]
      \scriptsize Husky vs.\ wolf: LIME highlights snow pixels. LMU IML (CC BY 4.0).
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{LIME's proof: the explanation \emph{is} the snow}
  \begin{center}
    \includegraphics[width=0.73\linewidth]{fig/lime_husky_full.png}
  \end{center}
  \vskip 2pt
  \footnotesize In each pair: the photo (left) and \textbf{LIME's explanation} (right --- the kept
  superpixels). The model's ``reason'' is the \textbf{snow}, not the animal, so the snowy
  \textbf{husky} (bottom-left) is misread as a \textbf{wolf}. \hfill LMU IML (CC BY 4.0) / Ribeiro et al.\ 2016.
\end{frame}

\begin{frame}{LIME caveats, and SHAP vs.\ LIME}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Caveats}
    \begin{itemize}\small
      \item \textbf{Unstable}: re-run and the explanation can change.
      \item The \textbf{neighbourhood} (how far to jitter, how to weight) is an \emph{arbitrary}
        choice that moves the answer.
    \end{itemize}
    \vskip 4pt
    \fcolorbox{armblue}{armblue!8}{\parbox{0.96\linewidth}{\small
      \textbf{SHAP vs.\ LIME:} SHAP has theory (consistency) + richer plots, but is slower; LIME is
      simpler/faster but less stable. \textbf{SHAP is the modern default.}}}
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=0.86\linewidth]{fig/lime_robustness.png}\\[-2pt]
      \scriptsize Repeated LIME fits near one point \textbf{disagree} --- unstable. LMU IML.
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{LIME's global sibling: the surrogate model}
  \begin{columns}[T]
    \column{0.54\textwidth}
    LIME fits a simple model \emph{locally}, around one prediction. Its \textbf{global} counterpart
    does the same for the \emph{whole} model:
    \vskip 4pt
    \fcolorbox{armblue}{armblue!8}{\parbox{0.96\linewidth}{\small
      \textbf{Global surrogate:} train an interpretable model --- a shallow tree, a linear model, a
      GAM --- to \textbf{mimic the black box's predictions} everywhere, then read the surrogate
      instead of the black box.}}
    \vskip 6pt
    \begin{itemize}\small
      \item Check its \textbf{fidelity} first ($R^2$ of surrogate vs.\ black-box output): a poor
        mimic is a misleading explanation.
      \item In effect it \textbf{distills the forest back into deck 1's glass box}.
    \end{itemize}
    \column{0.46\textwidth}
    \begin{center}
      \begin{tikzpicture}[font=\scriptsize, >=stealth]
        \node[draw, rounded corners, fill=armred!12, minimum width=3.2cm, minimum height=0.8cm] (bb)
          at (0,0) {black box $\hat f$ (forest, NN)};
        \node[draw, rounded corners, fill=armblue!12, minimum width=3.2cm, minimum height=0.8cm] (s)
          at (0,-1.9) {surrogate: tree / linear / GAM};
        \draw[->, thick] (bb.south) -- node[right, font=\tiny, align=left]
          {fit to $\hat f$'s\\ predictions} (s.north);
        \node[font=\tiny, armblue] at (0,-2.7) {read this instead};
      \end{tikzpicture}
    \end{center}
  \end{columns}
\end{frame}

\section{Counterfactual explanations}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Counterfactual explanations}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Not \emph{why} the model decided --- but \emph{what would change} the outcome.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{The smallest change that flips the prediction}
  SHAP and LIME answer ``\emph{why} this prediction?'' A \textbf{counterfactual} answers a
  different, often more useful question:
  \vskip 4pt
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    \textbf{What is the smallest change to the input that would flip the prediction?}}}
  \vskip 6pt
  \begin{itemize}\small
    \item ``Flip'' needs a \textbf{decision}, so we switch from the (regression) bike model to
      the \textbf{ch.\,3 bank-marketing} classifier --- \emph{will this client subscribe?}
      (realistic model, the leaky \texttt{duration} feature removed).
    \item A counterfactual is \textbf{actionable}: it tells you \emph{what to change}, not just
      what mattered.
  \end{itemize}
  \vfill
  \footnotesize \textit{Example:} a client predicted \textbf{won't subscribe} --- the
  counterfactual says that \emph{had the bank called in a different month}, the model would flip
  to \textbf{will subscribe}.
\end{frame}

\begin{frame}{What makes a good counterfactual?}
  \begin{columns}[T]
    \column{0.5\textwidth}
    Five things we want at once:
    \begin{itemize}\small
      \item \textbf{Validity} --- it actually flips the prediction.
      \item \textbf{Proximity} --- close to the original (a small change).
      \item \textbf{Sparsity} --- few features changed (easy to act on, easy to explain).
      \item \textbf{Plausibility} --- stays in realistic data, no impossible rows (the
        extrapolation trap, deck 2).
      \item \textbf{Actionability} --- change only what the user controls (raise a balance, not
        lower an age; never a protected attribute). The \emph{recourse} angle.
    \end{itemize}
    \vskip 2pt
    \footnotesize Usually \textbf{several} valid counterfactuals exist --- prefer the sparse,
    actionable ones.
    \column{0.5\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/cf_flip_bank.pdf}
    \end{center}
    \vskip 2pt
    \footnotesize Real DiCE counterfactual (bank): \texttt{age}, \texttt{job}, \texttt{balance},
    \texttt{poutcome} \textbf{frozen} (actionability); only the \textbf{call month} moved --- one
    change, and the model crosses the decision line.
  \end{columns}
\end{frame}

\begin{frame}[fragile]{Counterfactuals in code}
\begin{lstlisting}
import dice_ml
d   = dice_ml.Data(dataframe=df, continuous_features=num, outcome_name="y")
m   = dice_ml.Model(model=clf, backend="sklearn")   # any fitted classifier
exp = dice_ml.Dice(d, m, method="random")

cf = exp.generate_counterfactuals(
        client, total_CFs=3, desired_class="opposite",
        features_to_vary=["contact", "month", "campaign"])  # actionable only
cf.visualize_as_dataframe()
\end{lstlisting}
  \vskip 2pt
  \footnotesize Libraries: \textbf{DiCE} (diverse counterfactuals), \textbf{Alibi}. Constrain
  \texttt{features\_to\_vary} so the advice stays \emph{actionable}.
\end{frame}

\section*{Wrap-up}

\begin{frame}{Same data, four rankings}
  One bike dataset, (mostly) one model --- four ``importance'' methods, four different top-3s:
  \vskip 4pt
  \begin{center}\small
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{l l}
      \textbf{Method} & \textbf{Top 3 features} \\
      \hline
      Linear coefficients (deck 1) & \texttt{yr}, \texttt{atemp}, \texttt{season} \\
      Tree impurity (deck 1) & \texttt{temp}, \texttt{yr}, \texttt{atemp} \\
      Permutation / PFI (deck 2) & \texttt{yr}, \texttt{atemp}, \texttt{temp} \\
      SHAP (deck 3) & \texttt{yr}, \texttt{temp}, \texttt{hum} \\
      \hline
    \end{tabular}
  \end{center}
  \vskip 4pt
  \footnotesize They mostly agree on \texttt{yr}/temperature, but disagree on the rest
  (\texttt{season}? \texttt{hum}? \texttt{temp} vs \texttt{atemp}?). \textbf{That} is what
  ``importance is method-dependent'' means --- always say \emph{which} importance, on \emph{what} data.
\end{frame}

\begin{frame}{Which method when? (the whole chapter)}
  \begin{center}\small
    \renewcommand{\arraystretch}{1.35}
    \begin{tabular}{p{5.2cm} p{7.2cm}}
      \textbf{You want\dots} & \textbf{Reach for\dots} \\
      \hline
      to read the model itself & a \textbf{glass-box model} --- coefficients / a tree / rules (deck 1) \\
      global ``which features matter'' & \textbf{PFI} or \textbf{SHAP bar} (decks 2--3) \\
      ``how does a feature act'' & \textbf{PDP / ICE} or \textbf{SHAP dependence} (decks 2--3) \\
      ``why this one prediction'' & \textbf{SHAP waterfall} or \textbf{LIME} (deck 3) \\
      ``what would change the outcome'' & \textbf{counterfactuals} --- DiCE / Alibi (deck 3) \\
      \hline
    \end{tabular}
  \end{center}
  \vfill
  \footnotesize Whatever you pick: \textbf{(1)} correlated features mislead them all, \textbf{(2)}
  different methods rank differently (previous slide), \textbf{(3)} none of it is \emph{causation}.
\end{frame}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{SHAP} fairly splits a prediction among features: $\hat f(\xv) = \text{base} + \sum_j \phi_j$.
    \item Read the four plots: \textbf{waterfall} (one prediction), \textbf{bar} (global importance),
      \textbf{beeswarm} (global, colour = value), \textbf{dependence} (effect shape + interaction).
    \item \textbf{TreeSHAP} makes it the default for trees/boosting; SHAP \textbf{unifies} PFI \& PDP
      and adds a local view.
    \item \textbf{LIME} is the simpler local cousin --- handy but unstable.
    \item \textbf{Counterfactuals} answer ``what would change the outcome'' --- the smallest
      \textbf{valid, sparse, actionable} edit that flips the prediction (DiCE).
    \item Everything explains the \textbf{model}, not causation --- and plots are easy to over-read.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{That's interpretability:} read the model when you can, explain it when you can't ---
    importance, effects, and per-prediction attributions, each answering a different question.}}
\end{frame}
