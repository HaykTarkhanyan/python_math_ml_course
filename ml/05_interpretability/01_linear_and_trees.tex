\begin{frame}{Predict-first: which one can you actually read?}
  A linear model and a random forest reach the \textbf{same accuracy} on the bike-rental data.
  Which can you \emph{read} --- and how?
  \vfill
  \pause
  \begin{itemize}\small
    \item \textbf{Linear model:} yes --- each feature has one coefficient.
    \item \textbf{A single tree:} yes --- follow the splits from root to leaf (a rule).
    \item \textbf{A random forest:} \textbf{no} --- hundreds of trees, no readable whole. But it
      still hands you a built-in \emph{importance} number (which, we'll see, is biased).
  \end{itemize}
  \vfill
  \footnotesize Running example: \textbf{bike sharing} --- predict daily rental count from weather
  \& calendar features.
\end{frame}

\section{Reading a linear model}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Reading a linear model}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      The most transparent model: one number per feature.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{Coefficients as importance}
  A linear model $\hat y = \theta_0 + \sum_j \theta_j x_j$ tells you everything directly:
  \begin{itemize}\small
    \item \textbf{Sign} of $\theta_j$ = direction; \textbf{magnitude} = effect per $+1$ unit of $x_j$.
    \item \textbf{Standardize first} to compare --- a raw coef of $5000$ on \texttt{temp} (range
      $0$--$1$) vs $2$ on \texttt{hum} is \emph{not} ``$2500\times$ more important''; that's units
      (L12). Standardized, $|\theta_j|$ is fair: on bike \texttt{yr} \& \texttt{atemp} top it.
    \item Classification analog: $e^{\theta_j}$ = \textbf{odds ratio} (L11).
  \end{itemize}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    No extra method needed: a (standardized) coefficient \emph{is} a built-in importance and
    effect, in one number.}}
\end{frame}

\begin{frame}{Linear models: the caveats}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Only additive, linear effects}
    \begin{itemize}\small
      \item One straight-line effect per feature; no interactions or curves unless you engineer them.
    \end{itemize}
    \vskip 4pt
    \textbf{Correlated features split the credit}
    \begin{itemize}\small
      \item \texttt{temp} and \texttt{atemp} (feels-like) are nearly the same; the model may give
        each half the coefficient --- or unstable, even opposite, signs.
    \end{itemize}
    \column{0.5\textwidth}
    \fcolorbox{armred}{armred!8}{\parbox{0.96\linewidth}{\small
      A coefficient answers ``how does the prediction move with this feature, \emph{holding the
      others fixed}'' --- which is exactly what breaks when features move \emph{together}.}}
  \end{columns}
\end{frame}

\begin{frame}{Lasso: the model selects its own features}
  \textbf{L1 (Lasso) regularization} shrinks coefficients toward zero --- and drives many to
  \textbf{exactly zero} (callback: Ch.\,2 regularization).
  \vfill
  \begin{itemize}\small
    \item Non-zero coefficients $=$ the features the model \emph{kept} --- a built-in,
      sparsity-based \textbf{feature selection}.
    \item Stronger penalty $\Rightarrow$ fewer survivors. \textbf{On bike (scaled):} the
      CV-tuned Lasso keeps $10/11$; turn the penalty up and survivors drop $10 \to 4 \to 1$ ---
      \textbf{sparsity is a knob you set}.
  \end{itemize}
  \vfill
  \fcolorbox{armorange}{armorange!12}{\parbox{0.97\linewidth}{\small
    \textbf{Caveat:} among correlated features, Lasso keeps \emph{one} and zeros the rest
    \emph{~arbitrarily} --- ``not selected'' does not mean ``unimportant.''}}
\end{frame}

\section{Reading trees}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Reading trees}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      One tree you can read; a forest you cannot --- but it still scores features.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{A single tree is a glass box}
  \begin{columns}[T]
    \column{0.4\textwidth}
    Each \textbf{root-to-leaf path is a rule}:
    \vskip 2pt
    \emph{warm day (\texttt{temp}$>0.43$), year 2, good weather $\to$ predict $\sim 6726$ rentals.}
    \vskip 6pt
    \begin{itemize}\small
      \item Fully transparent --- if it stays shallow.
      \item Depth $\uparrow$ $\Rightarrow$ accuracy $\uparrow$ but readability $\downarrow$.
    \end{itemize}
    \column{0.6\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/tree_bike.pdf}
    \end{center}
  \end{columns}
\end{frame}

\begin{frame}{How a split is chosen: reduce impurity}
  A tree picks the split that most reduces \textbf{impurity}: \quad
  reduction $=$ parent impurity $-$ \emph{weighted} child impurity. Two impurity measures:
  \vskip 4pt
  \begin{columns}[T]
    \column{0.5\textwidth}
    \textbf{Classification: information gain} (entropy $H$) \\[2pt]
    \footnotesize Toy: parent $5/5 \Rightarrow H=1.0$. A $4/1$ child:
    $-\tfrac45\log_2\tfrac45-\tfrac15\log_2\tfrac15 = 0.72$.
    \[ H_{\text{child}} = \tfrac12(0.72)+\tfrac12(0.72)=0.72,\ \ \text{gain}=1.0-0.72=\mathbf{0.28} \]
    \column{0.5\textwidth}
    \textbf{Regression: variance reduction} (bike root split) \\[2pt]
    \footnotesize Split \texttt{temp} $\le 0.43$ ($n=731$):
    \[ \mathrm{Var}_{\text{parent}} \approx 3.75\text{M} \]
    \[ \mathrm{Var}_{\text{child}} = 0.40(2.24\text{M}) + 0.60(2.31\text{M}) = 2.28\text{M} \]
    \[ \text{reduction} \approx \mathbf{1.46\text{M}}\ (\approx 39\%) \]
  \end{columns}
  \vskip 4pt
  \footnotesize Same idea, two rulers --- the split with the biggest reduction wins.
\end{frame}

\begin{frame}{Forests: unreadable, but they score features}
  A random forest is \textbf{hundreds of trees} --- no readable whole. Yet it gives a built-in
  \textbf{feature importance}:
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    Sum each feature's impurity reduction (info gain / variance reduction) over \emph{all} its
    splits, averaged across the trees $\Rightarrow$ one importance per feature.}}
  \vfill
  \textbf{But it is biased:}
  \begin{itemize}\small
    \item Favors \textbf{high-cardinality / continuous} features (more places to split).
    \item \textbf{Dilutes} across correlated features (credit shared between \texttt{temp},
      \texttt{atemp}).
    \item Computed on \textbf{training} data $\Rightarrow$ can reward overfitting.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Reading these in \texttt{sklearn}}
\begin{lstlisting}
# linear: standardized coefficients as importance
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LinearRegression, LassoCV
from sklearn.pipeline import make_pipeline
Xs = StandardScaler().fit_transform(X)
coef = LinearRegression().fit(Xs, y).coef_           # |coef| = importance
sel  = make_pipeline(StandardScaler(), LassoCV())    # SCALE, then L1 (Lasso is scale-sensitive)
kept = sel.fit(X, y)[-1].coef_ != 0                  # features Lasso keeps (alpha tuned by CV)

# trees: built-in (impurity) importance + read one tree
from sklearn.ensemble import RandomForestRegressor
from sklearn.tree import DecisionTreeRegressor, plot_tree
RandomForestRegressor().fit(X, y).feature_importances_
plot_tree(DecisionTreeRegressor(max_depth=3).fit(X, y))
\end{lstlisting}
\end{frame}

\section{Rule-based models}

\begin{frame}[plain]
  \vfill
  \begin{center}
    {\Large\color{popblue}\textbf{Rule-based models}}\\[0.9em]
    \begin{minipage}{0.82\textwidth}\centering\small
      Rules you can read --- a tree's logic with a linear model's sparsity.
    \end{minipage}
  \end{center}
  \vfill
\end{frame}

\begin{frame}{RuleFit: turn rules into features}
  A single tree is readable but brittle; a forest is accurate but unreadable. \textbf{RuleFit}
  goes for both, in three steps:
  \begin{enumerate}\small
    \item \textbf{Mine rules} from a tree ensemble: every root-to-node path becomes a binary
      rule --- e.g.\ \texttt{(temp$>$0.35 and yr$=$1)}, which is $1$ when the row falls in that
      region and $0$ otherwise.
    \item \textbf{Fit a sparse linear model} (Lasso) over \emph{both} the original features and
      all these new rule-features.
    \item \textbf{Keep the non-zero terms} --- a short, weighted list of rules (plus a few
      surviving linear terms).
  \end{enumerate}
  \vfill
  \fcolorbox{armblue}{armblue!8}{\parbox{0.97\linewidth}{\small
    RuleFit $=$ \textbf{rules from trees} $+$ \textbf{Lasso selection} --- the two glass boxes
    from this deck, combined into one readable model.}}
\end{frame}

\begin{frame}{Reading a RuleFit model: a scorecard}
  \begin{columns}[T]
    \column{0.58\textwidth}
    On bike (baseline $\approx$ the average, $\sim\!4500$ rentals), read it like a
    \textbf{scorecard}: start at the baseline, then add the weight of every rule that
    \emph{fires} for this day.
    \vskip 6pt
    \footnotesize
    \renewcommand{\arraystretch}{1.3}
    \begin{tabular}{p{5.0cm} r}
      \textbf{rule (if it fires, add the weight)} & \textbf{weight} \\
      \hline
      \texttt{temp$>$0.35 and yr$=$1 and hum$\le$0.88} & $+551$ \\
      \texttt{atemp$>$0.29 and yr$=$1} & $+266$ \\
      \texttt{temp$\le$0.43} & $-256$ \\
      \hline
    \end{tabular}
    \vskip 6pt
    \footnotesize A warm 2012 day fires the first two ($+817$); a cold day fires the third
    ($-256$). This $13$-rule model scores $R^2\approx0.65$.
    \column{0.42\textwidth}
    \fcolorbox{armorange}{armorange!12}{\parbox{0.96\linewidth}{\small
      \textbf{Watch out:}
      \begin{itemize}\small
        \item More rules $\Rightarrow$ more accurate but less readable --- the tree's
          depth/readability dial again (a denser RuleFit climbs toward the forest's $\sim\!0.87$).
        \item Overlapping rules \textbf{share credit}: one weight is not a feature's full effect.
      \end{itemize}}}
  \end{columns}
\end{frame}

\section{So which features matter?}

\begin{frame}{Predict-first: do the two methods agree?}
  \begin{columns}[T]
    \column{0.46\textwidth}
    Same bike data, same goal. \textbf{Predict:} will the linear model and the random forest pick
    the \emph{same} top features?
    \vskip 4pt
    \pause
    \textbf{They don't:}
    \begin{itemize}\small
      \item Linear: \texttt{yr}, \texttt{atemp}, \texttt{season} on top.
      \item Forest: \texttt{temp}, \texttt{yr}, \texttt{atemp} on top.
    \end{itemize}
    \vskip 4pt
    Part of the gap is the \textbf{impurity bias} we just saw; the rest is that they genuinely
    \textbf{measure different things} (a linear effect vs.\ impurity reduction), splitting the
    \texttt{temp}/\texttt{atemp} correlation differently.
    \column{0.54\textwidth}
    \begin{center}
      \includegraphics[width=\linewidth]{fig/imp_methods_disagree.pdf}
    \end{center}
  \end{columns}
  \vskip 2pt
  \footnotesize \textbf{Importance is method-dependent} --- always say \emph{which} importance you mean.
\end{frame}

\section*{Wrap-up}

\begin{frame}{Recap}
  \begin{itemize}\small
    \item \textbf{Linear models} are read directly: (standardized) \textbf{coefficients} =
      effect + importance; $e^{\theta_j}$ = odds ratio; \textbf{Lasso} selects features by zeroing them.
    \item \textbf{A single tree} is a glass box (paths = rules); splits are chosen by
      \textbf{impurity reduction} (information gain / variance reduction).
    \item \textbf{RuleFit} mines rules from an ensemble, then Lasso-selects a short weighted
      list you read as a \textbf{scorecard} --- accuracy vs readability is a dial you set.
    \item \textbf{Forests} are black boxes but emit a built-in \textbf{impurity importance} ---
      fast, but \textbf{biased} (cardinality, correlation, train-based).
    \item Different methods give \textbf{different rankings} --- importance is method-dependent,
      and \emph{none of it is causation}.
  \end{itemize}
  \vfill
  \fcolorbox{paramgreen}{paramgreen!8}{\parbox{0.97\linewidth}{\small
    \textbf{Next:} model-agnostic interpretability --- \textbf{permutation importance} (fixes the
    cardinality \& train-based biases --- though correlation still bites) and \textbf{feature
    effects} (PDP / ICE), for \emph{any} model.}}
\end{frame}
